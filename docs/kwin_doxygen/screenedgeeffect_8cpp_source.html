70void ScreenEdgeEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
71{
72 effects->paintScreen(mask, region, data);
73 for (auto &[border, glow] : m_borders) {
74 const qreal opacity = glow->strength;
75 if (opacity == 0.0) {
76 continue;
77 }
78 if (effects->isOpenGLCompositing()) {
79 GLTexture *texture = glow->texture.get();
80 glEnable(GL_BLEND);
81 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
82 texture->bind();
83 ShaderBinder binder(ShaderTrait::MapTexture | ShaderTrait::Modulate);
84 const QVector4D constant(opacity, opacity, opacity, opacity);
85 binder.shader()->setUniform(GLShader::ModulationConstant, constant);
86 const auto scale = effects->renderTargetScale();
87 QMatrix4x4 mvp = data.projectionMatrix();
88 mvp.translate(glow->geometry.x() * scale, glow->geometry.y() * scale);
89 binder.shader()->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
90 texture->render(glow->geometry.size(), scale);
91 texture->unbind();
92 glDisable(GL_BLEND);
93 } else if (effects->compositingType() == QPainterCompositing) {
94 QImage tmp(glow->image->size(), QImage::Format_ARGB32_Premultiplied);
95 tmp.fill(Qt::transparent);
96 QPainter p(&tmp);
97 p.drawImage(0, 0, *glow->image.get());
98 QColor color(Qt::transparent);
99 color.setAlphaF(opacity);
100 p.setCompositionMode(QPainter::CompositionMode_DestinationIn);
101 p.fillRect(QRect(QPoint(0, 0), tmp.size()), color);
102 p.end();
103
104 QPainter *painter = effects->scenePainter();
105 const QRect &rect = glow->geometry;
106 const QSize &size = glow->pictureSize;
107 int x = rect.x();
108 int y = rect.y();
109 switch (glow->border) {
110 case ElectricTopRight:
111 x = rect.x() + rect.width() - size.width();
112 break;
113 case ElectricBottomRight:
114 x = rect.x() + rect.width() - size.width();
115 y = rect.y() + rect.height() - size.height();
116 break;
117 case ElectricBottomLeft:
118 y = rect.y() + rect.height() - size.height();
119 break;
120 default:
121 // nothing
122 break;
123 }
124 painter->drawImage(QPoint(x, y), tmp);
125 }
126 }
127}

129void ScreenEdgeEffect::edgeApproaching(ElectricBorder border, qreal factor, const QRect &geometry)
130{
131 auto it = m_borders.find(border);
132 if (it != m_borders.end()) {
133 Glow *glow = it->second.get();
134 // need to update
135 effects->addRepaint(glow->geometry);
136 glow->strength = factor;
137 if (glow->geometry != geometry) {
138 glow->geometry = geometry;
139 effects->addRepaint(glow->geometry);
140 if (border == ElectricLeft || border == ElectricRight || border == ElectricTop || border == ElectricBottom) {
141 if (effects->isOpenGLCompositing()) {
142 glow->texture.reset(createEdgeGlow<GLTexture>(border, geometry.size()));
143 } else if (effects->compositingType() == QPainterCompositing) {
144 glow->image.reset(createEdgeGlow<QImage>(border, geometry.size()));
145 }
146 }
147 }
148 if (factor == 0.0) {
149 m_cleanupTimer->start();
150 } else {
151 m_cleanupTimer->stop();
152 }
153 } else if (factor != 0.0) {
154 // need to generate new Glow
155 std::unique_ptr<Glow> glow = createGlow(border, factor, geometry);
156 if (glow) {
157 effects->addRepaint(glow->geometry);
158 m_borders[border] = std::move(glow);
159 }
160 }
161}

163std::unique_ptr<Glow> ScreenEdgeEffect::createGlow(ElectricBorder border, qreal factor, const QRect &geometry)
164{
165 auto glow = std::make_unique<Glow>();
166 glow->border = border;
167 glow->strength = factor;
168 glow->geometry = geometry;
169
170 // render the glow image
171 if (effects->isOpenGLCompositing()) {
172 effects->makeOpenGLContextCurrent();
173 if (border == ElectricTopLeft || border == ElectricTopRight || border == ElectricBottomRight || border == ElectricBottomLeft) {
174 glow->texture.reset(createCornerGlow<GLTexture>(border));
175 } else {
176 glow->texture.reset(createEdgeGlow<GLTexture>(border, geometry.size()));
177 }
178 if (glow->texture) {
179 glow->texture->setWrapMode(GL_CLAMP_TO_EDGE);
180 }
181 if (!glow->texture) {
182 return nullptr;
183 }
184 } else if (effects->compositingType() == QPainterCompositing) {
185 if (border == ElectricTopLeft || border == ElectricTopRight || border == ElectricBottomRight || border == ElectricBottomLeft) {
186 glow->image.reset(createCornerGlow<QImage>(border));
187 glow->pictureSize = cornerGlowSize(border);
188 } else {
189 glow->image.reset(createEdgeGlow<QImage>(border, geometry.size()));
190 glow->pictureSize = geometry.size();
191 }
192 if (!glow->image) {
193 return nullptr;
194 }
195 }
196
197 return glow;
198}

237template<typename T>
238T *ScreenEdgeEffect::createEdgeGlow(ElectricBorder border, const QSize &size)
239{
240 ensureGlowSvg();
241
242 const bool stretchBorder = m_glow->hasElement(QStringLiteral("hint-stretch-borders"));
243
244 QPoint pixmapPosition(0, 0);
245 QPixmap l, r, c;
246 switch (border) {
247 case ElectricTop:
248 l = m_glow->pixmap(QStringLiteral("bottomleft"));
249 r = m_glow->pixmap(QStringLiteral("bottomright"));
250 c = m_glow->pixmap(QStringLiteral("bottom"));
251 break;
252 case ElectricBottom:
253 l = m_glow->pixmap(QStringLiteral("topleft"));
254 r = m_glow->pixmap(QStringLiteral("topright"));
255 c = m_glow->pixmap(QStringLiteral("top"));
256 pixmapPosition = QPoint(0, size.height() - c.height());
257 break;
258 case ElectricLeft:
259 l = m_glow->pixmap(QStringLiteral("topright"));
260 r = m_glow->pixmap(QStringLiteral("bottomright"));
261 c = m_glow->pixmap(QStringLiteral("right"));
262 break;
263 case ElectricRight:
264 l = m_glow->pixmap(QStringLiteral("topleft"));
265 r = m_glow->pixmap(QStringLiteral("bottomleft"));
266 c = m_glow->pixmap(QStringLiteral("left"));
267 pixmapPosition = QPoint(size.width() - c.width(), 0);
268 break;
269 default:
270 return nullptr;
271 }
272 QPixmap image(size);
273 image.fill(Qt::transparent);
274 QPainter p;
275 p.begin(&image);
276 if (border == ElectricBottom || border == ElectricTop) {
277 p.drawPixmap(pixmapPosition, l);
278 const QRect cRect(l.width(), pixmapPosition.y(), size.width() - l.width() - r.width(), c.height());
279 if (stretchBorder) {
280 p.drawPixmap(cRect, c);
281 } else {
282 p.drawTiledPixmap(cRect, c);
283 }
284 p.drawPixmap(QPoint(size.width() - r.width(), pixmapPosition.y()), r);
285 } else {
286 p.drawPixmap(pixmapPosition, l);
287 const QRect cRect(pixmapPosition.x(), l.height(), c.width(), size.height() - l.height() - r.height());
288 if (stretchBorder) {
289 p.drawPixmap(cRect, c);
290 } else {
291 p.drawTiledPixmap(cRect, c);
292 }
293 p.drawPixmap(QPoint(pixmapPosition.x(), size.height() - r.height()), r);
294 }
295 p.end();
296 return new T(image.toImage());
297}
