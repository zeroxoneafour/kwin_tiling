48Dnd::Dnd(xcb_atom_t atom, QObject *parent)
49 : Selection(atom, parent)

80void Dnd::doHandleXfixesNotify(xcb_xfixes_selection_notify_event_t *event)
81{
82 if (qobject_cast<XToWlDrag *>(m_currentDrag)) {
83 // X drag is in progress, rogue X client took over the selection.
84 return;
85 }
86 if (m_currentDrag) {
87 // Wl drag is in progress - don't overwrite by rogue X client,
88 // get it back instead!
89 ownSelection(true);
90 return;
91 }
92 createX11Source(nullptr);
93 const auto *seat = waylandServer()->seat();
94 auto *originSurface = seat->focusedPointerSurface();
95 if (!originSurface) {
96 return;
97 }
98 if (originSurface->client() != waylandServer()->xWaylandConnection()) {
99 // focused surface client is not Xwayland - do not allow drag to start
100 // TODO: can we make this stronger (window id comparison)?
101 return;
102 }
103 if (!seat->isPointerButtonPressed(Qt::LeftButton)) {
104 // we only allow drags to be started on (left) pointer button being
105 // pressed for now
106 return;
107 }
108 createX11Source(event);
109 X11Source *source = x11Source();
110 if (!source) {
111 return;
112 }
113 m_currentDrag = new XToWlDrag(source, this);
114}

120bool Dnd::handleClientMessage(xcb_client_message_event_t *event)
121{
122 for (Drag *drag : std::as_const(m_oldDrags)) {
123 if (drag->handleClientMessage(event)) {
124 return true;
125 }
126 }
127 if (m_currentDrag && m_currentDrag->handleClientMessage(event)) {
128 return true;
129 }
130 return false;
131}

139void Dnd::startDrag()
140{
141 auto dragSource = waylandServer()->seat()->dragSource();
142 if (qobject_cast<XwlDataSource *>(dragSource)) {
143 return;
144 }
145
146 // There can only ever be one Wl native drag at the same time.
147 Q_ASSERT(!m_currentDrag);
148
149 // New Wl to X drag, init drag and Wl source.
150 m_currentDrag = new WlToXDrag(this);
151 auto source = new WlSource(this);
152 source->setDataSourceIface(dragSource);
153 connect(dragSource, &KWaylandServer::AbstractDataSource::aboutToBeDestroyed, this, [this, source] {
154 if (source == wlSource()) {
155 setWlSource(nullptr);
156 }
157 });
158 setWlSource(source);
159 ownSelection(true);
160}

162void Dnd::endDrag()
163{
164 Q_ASSERT(m_currentDrag);
165
166 connect(m_currentDrag, &Drag::finish, this, &Dnd::clearOldDrag);
167 m_oldDrags << m_currentDrag;
168
169 m_currentDrag = nullptr;

181DnDAction Dnd::atomToClientAction(xcb_atom_t atom)
182{
183 if (atom == atoms->xdnd_action_copy) {
184 return DnDAction::Copy;
185 } else if (atom == atoms->xdnd_action_move) {
186 return DnDAction::Move;
187 } else if (atom == atoms->xdnd_action_ask) {
188 // we currently do not support it - need some test client first
189 return DnDAction::None;
190 // return DnDAction::Ask;
191 }
192 return DnDAction::None;
193}
