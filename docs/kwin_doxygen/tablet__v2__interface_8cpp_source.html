228void TabletToolV2Interface::setCurrentSurface(SurfaceInterface *surface)
229{
230 if (d->m_surface == surface)
231 return;
232
233 TabletV2Interface *const lastTablet = d->m_lastTablet;
234 if (d->m_surface && d->resourceMap().contains(*d->m_surface->client())) {
235 sendProximityOut();
236 sendFrame(0);
237 }
238
239 d->m_surface = surface;
240
241 if (lastTablet && lastTablet->d->resourceForSurface(surface)) {
242 sendProximityIn(lastTablet);
243 } else {
244 d->m_lastTablet = lastTablet;
245 }
246
247 Q_EMIT cursorChanged(d->m_cursors.value(d->targetResource()));
248}

589void TabletPadV2Interface::setCurrentSurface(SurfaceInterface *surface, TabletV2Interface *tablet)
590{
591 if (surface == d->m_currentSurface) {
592 return;
593 }
594
595 if (d->m_currentSurface) {
596 d->send_leave(d->resourceForSurface(d->m_currentSurface), d->m_display->nextSerial(), d->m_currentSurface->resource());
597 }
598
599 d->m_currentSurface = surface;
600 if (surface) {
601 wl_resource *tabletResource = tablet->d->resourceForSurface(surface);
602
603 d->send_enter(d->resourceForSurface(surface), d->m_display->nextSerial(), tabletResource, surface->resource());
604 d->m_padGroup->sendModeSwitch(0, d->m_display->nextSerial(), d->m_padGroup->d->m_currentMode);
605 }

644
645 void sendToolAdded(Resource *resource, TabletToolV2Interface *tool)
646 {
647 wl_resource *toolResource = tool->d->add(resource->client(), resource->version())->handle;
648 send_tool_added(resource->handle, toolResource);
649
650 tool->d->send_type(toolResource, tool->d->m_type);
651 tool->d->send_hardware_serial(toolResource, tool->d->m_hardwareSerialHigh, tool->d->m_hardwareSerialLow);
652 tool->d->send_hardware_id_wacom(toolResource, tool->d->m_hardwareIdHigh, tool->d->m_hardwareIdLow);
653 for (uint32_t cap : std::as_const(tool->d->m_capabilities)) {
654 tool->d->send_capability(toolResource, cap);
655 }
656 tool->d->send_done(toolResource);
657 }
658 void sendTabletAdded(Resource *resource, TabletV2Interface *tablet)
659 {
660 wl_resource *tabletResource = tablet->d->add(resource->client(), resource->version())->handle;
661 send_tablet_added(resource->handle, tabletResource);
662
663 tablet->d->send_name(tabletResource, tablet->d->m_name);
664 if (tablet->d->m_vendorId && tablet->d->m_productId) {
665 tablet->d->send_id(tabletResource, tablet->d->m_vendorId, tablet->d->m_productId);
666 }
667 for (const QString &path : std::as_const(tablet->d->m_paths)) {
668 tablet->d->send_path(tabletResource, path);
669 }
670 tablet->d->send_done(tabletResource);
671 }
672
673 void sendPadAdded(Resource *resource, TabletPadV2Interface *pad)
674 {
675 wl_resource *tabletResource = pad->d->add(resource->client(), resource->version())->handle;
676 send_pad_added(resource->handle, tabletResource);
677
678 pad->d->send_buttons(tabletResource, pad->d->m_buttons.size());
679 pad->d->send_path(tabletResource, pad->d->m_path);
680
681 auto groupResource = pad->d->m_padGroup->d->add(resource->client(), resource->version());
682 pad->d->send_group(tabletResource, groupResource->handle);
683 pad->d->m_padGroup->d->send_modes(groupResource->handle, pad->d->m_modes);
684
685 pad->d->m_padGroup->d->send_buttons(
686 groupResource->handle,
687 QByteArray::fromRawData(reinterpret_cast<const char *>(pad->d->m_buttons.data()), pad->d->m_buttons.size() * sizeof(quint32)));
688
689 for (auto ring : std::as_const(pad->d->m_rings)) {
690 auto ringResource = ring->d->add(resource->client(), resource->version());
691 pad->d->m_padGroup->d->send_ring(groupResource->handle, ringResource->handle);
692 }
693
694 for (auto strip : std::as_const(pad->d->m_strips)) {
695 auto stripResource = strip->d->add(resource->client(), resource->version());
696 pad->d->m_padGroup->d->send_strip(groupResource->handle, stripResource->handle);
697 }
698 pad->d->m_padGroup->d->send_done(groupResource->handle);
699 pad->d->send_done(tabletResource);
700 }
