167static QString buttonsToString(Qt::MouseButtons buttons)
168{
169 QString ret;
170 for (uint i = 1; i < Qt::ExtraButton24; i = i << 1) {
171 if (buttons & i) {
172 ret.append(buttonToString(Qt::MouseButton(uint(buttons) & i)));

191void DebugConsoleFilter::pointerEvent(MouseEvent *event)
192{
193 QString text = s_hr;
194 const QString timestamp = timestampRow(event->timestamp());
195
196 text.append(s_tableStart);
197 switch (event->type()) {
198 case QEvent::MouseMove: {
199 text.append(tableHeaderRow(i18nc("A mouse pointer motion event", "Pointer Motion")));
200 text.append(deviceRow(event->device()));
201 text.append(timestamp);
202 text.append(timestampRowUsec(event->timestamp()));
203 if (!event->delta().isNull()) {
204 text.append(tableRow(i18nc("The relative mouse movement", "Delta"),
205 QStringLiteral("%1/%2").arg(event->delta().x()).arg(event->delta().y())));
206 }
207 if (!event->deltaUnaccelerated().isNull()) {
208 text.append(tableRow(i18nc("The relative mouse movement", "Delta (not accelerated)"),
209 QStringLiteral("%1/%2").arg(event->deltaUnaccelerated().x()).arg(event->deltaUnaccelerated().y())));
210 }
211 text.append(tableRow(i18nc("The global mouse pointer position", "Global Position"), QStringLiteral("%1/%2").arg(event->pos().x()).arg(event->pos().y())));
212 break;
213 }
214 case QEvent::MouseButtonPress:
215 text.append(tableHeaderRow(i18nc("A mouse pointer button press event", "Pointer Button Press")));
216 text.append(deviceRow(event->device()));
217 text.append(timestamp);
218 text.append(tableRow(i18nc("A button in a mouse press/release event", "Button"), buttonToString(event->button())));
219 text.append(tableRow(i18nc("A button in a mouse press/release event", "Native Button code"), event->nativeButton()));
220 text.append(tableRow(i18nc("All currently pressed buttons in a mouse press/release event", "Pressed Buttons"), buttonsToString(event->buttons())));
221 break;
222 case QEvent::MouseButtonRelease:
223 text.append(tableHeaderRow(i18nc("A mouse pointer button release event", "Pointer Button Release")));
224 text.append(deviceRow(event->device()));
225 text.append(timestamp);
226 text.append(tableRow(i18nc("A button in a mouse press/release event", "Button"), buttonToString(event->button())));
227 text.append(tableRow(i18nc("A button in a mouse press/release event", "Native Button code"), event->nativeButton()));
228 text.append(tableRow(i18nc("All currently pressed buttons in a mouse press/release event", "Pressed Buttons"), buttonsToString(event->buttons())));
229 break;
230 default:
231 break;
232 }
233 text.append(s_tableEnd);
234
235 m_textEdit->insertHtml(text);
236 m_textEdit->ensureCursorVisible();
237}

258void DebugConsoleFilter::keyEvent(KeyEvent *event)
259{
260 QString text = s_hr;
261 text.append(s_tableStart);
262
263 switch (event->type()) {
264 case QEvent::KeyPress:
265 text.append(tableHeaderRow(i18nc("A key press event", "Key Press")));

277 ret.append(i18nc("A keyboard modifier", "Shift"));
278 ret.append(QStringLiteral(" "));
279 }
280 if (event->modifiers().testFlag(Qt::ControlModifier)) {
281 ret.append(i18nc("A keyboard modifier", "Control"));
282 ret.append(QStringLiteral(" "));
283 }
284 if (event->modifiers().testFlag(Qt::AltModifier)) {
285 ret.append(i18nc("A keyboard modifier", "Alt"));
286 ret.append(QStringLiteral(" "));
287 }
288 if (event->modifiers().testFlag(Qt::MetaModifier)) {
289 ret.append(i18nc("A keyboard modifier", "Meta"));
290 ret.append(QStringLiteral(" "));
291 }
292 if (event->modifiers().testFlag(Qt::KeypadModifier)) {
293 ret.append(i18nc("A keyboard modifier", "Keypad"));
294 ret.append(QStringLiteral(" "));
295 }
296 if (event->modifiers().testFlag(Qt::GroupSwitchModifier)) {
297 ret.append(i18nc("A keyboard modifier", "Group-switch"));
298 ret.append(QStringLiteral(" "));
299 }
300 return ret;
301 };
302 text.append(timestampRow(event->timestamp()));
303 text.append(tableRow(i18nc("Whether the event is an automatic key repeat", "Repeat"), event->isAutoRepeat()));
304
305 const auto keyMetaObject = Qt::qt_getEnumMetaObject(Qt::Key());
306 const auto enumerator = keyMetaObject->enumerator(keyMetaObject->indexOfEnumerator("Key"));
307 text.append(tableRow(i18nc("The code as read from the input device", "Scan code"), event->nativeScanCode()));
308 text.append(tableRow(i18nc("Key according to Qt", "Qt::Key code"),
309 enumerator.valueToKey(event->key())));
310 text.append(tableRow(i18nc("The translated code to an Xkb symbol", "Xkb symbol"), event->nativeVirtualKey()));
311 text.append(tableRow(i18nc("The translated code interpreted as text", "Utf8"), event->text()));
312 text.append(tableRow(i18nc("The currently active modifiers", "Modifiers"), modifiersToString()));
313
314 text.append(s_tableEnd);
315
316 m_textEdit->insertHtml(text);
317 m_textEdit->ensureCursorVisible();
318}

579static QString sourceString(const KWaylandServer::AbstractDataSource *const source)
580{
581 if (!source) {
582 return QString();
583 }
584
585 if (!source->client()) {
586 return QStringLiteral("XWayland source");
587 }
588
589 const QString executable = waylandServer()->display()->getConnection(source->client())->executablePath();
590
591 if (auto dataSource = qobject_cast<const KWaylandServer::DataSourceInterface *const>(source)) {
592 return QStringLiteral("wl_data_source@%1 of %2").arg(wl_resource_get_id(dataSource->resource())).arg(executable);
593 } else if (qobject_cast<const KWaylandServer::PrimarySelectionSourceV1Interface *const>(source)) {
594 return QStringLiteral("zwp_primary_selection_source_v1 of %2").arg(executable);
595 } else if (qobject_cast<const KWaylandServer::DataControlSourceV1Interface *const>(source)) {
596 return QStringLiteral("data control by %1").arg(executable);
597 }
598 return QStringLiteral("unknown source of").arg(executable);
599}
