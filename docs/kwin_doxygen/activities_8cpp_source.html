82void Activities::toggleWindowOnActivity(Window *window, const QString &activity, bool dont_activate)
83{
84 // int old_desktop = window->desktop();
85 bool was_on_activity = window->isOnActivity(activity);
86 bool was_on_all = window->isOnAllActivities();
87 // note: all activities === no activities
88 bool enable = was_on_all || !was_on_activity;
89 window->setOnActivity(activity, enable);
90 if (window->isOnActivity(activity) == was_on_activity && window->isOnAllActivities() == was_on_all) { // No change
91 return;
92 }
93
94 Workspace *ws = Workspace::self();
95 if (window->isOnCurrentActivity()) {
96 if (window->wantsTabFocus() && options->focusPolicyIsReasonable() && !was_on_activity && // for stickyness changes
97 // FIXME not sure if the line above refers to the correct activity
98 !dont_activate) {
99 ws->requestFocus(window);
100 } else {
101 ws->restackWindowUnderActive(window);
102 }
103 } else {
104 ws->raiseWindow(window);
105 }
106
107 // notifyWindowDesktopChanged( c, old_desktop );
108
109 const auto transients_stacking_order = ws->ensureStackingOrder(window->transients());
110 for (auto *const window : transients_stacking_order) {
111 if (!window) {
112 continue;
113 }
114 toggleWindowOnActivity(window, activity, dont_activate);
115 }
116 ws->updateClientArea();
117}

119bool Activities::start(const QString &id)
120{
121 Workspace *ws = Workspace::self();
122 if (ws->sessionManager()->state() == SessionState::Saving) {
123 return false; // ksmserver doesn't queue requests (yet)
124 }
125
126 if (!all().contains(id)) {
127 return false; // bogus id
128 }
129
130 ws->sessionManager()->loadSubSessionInfo(id);
131
132 QDBusInterface ksmserver("org.kde.ksmserver", "/KSMServer", "org.kde.KSMServerInterface");
133 if (ksmserver.isValid()) {
134 ksmserver.asyncCall("restoreSubSession", id);
135 } else {
136 qCDebug(KWIN_CORE) << "couldn't get ksmserver interface";
137 return false;
138 }
139 return true;
140}

159void Activities::reallyStop(const QString &id)
160{
161 Workspace *ws = Workspace::self();
162 if (ws->sessionManager()->state() == SessionState::Saving) {
163 return; // ksmserver doesn't queue requests (yet)
164 }
165
166 qCDebug(KWIN_CORE) << id;
167
168 QSet<QByteArray> saveSessionIds;
169 QSet<QByteArray> dontCloseSessionIds;
170 const auto windows = ws->allClientList();
171 for (auto *const window : windows) {
172 if (window->isDesktop()) {
173 continue;
174 }
175 const QByteArray sessionId = window->sessionId();
176 if (sessionId.isEmpty()) {
177 continue; // TODO support old wm_command apps too?
178 }
179
180 // qDebug() << sessionId;
181
182 // if it's on the activity that's closing, it needs saving
183 // but if a process is on some other open activity, I don't wanna close it yet
184 // this is, of course, complicated by a process having many windows.
185 if (window->isOnAllActivities()) {
186 dontCloseSessionIds << sessionId;
187 continue;
188 }
189
190 const QStringList activities = window->activities();
191 for (const QString &activityId : activities) {
192 if (activityId == id) {
193 saveSessionIds << sessionId;
194 } else if (running().contains(activityId)) {
195 dontCloseSessionIds << sessionId;
196 }
197 }
198 }
199
200 ws->sessionManager()->storeSubSession(id, saveSessionIds);
201
202 QStringList saveAndClose;
203 QStringList saveOnly;
204 for (const QByteArray &sessionId : std::as_const(saveSessionIds)) {
205 if (dontCloseSessionIds.contains(sessionId)) {
206 saveOnly << sessionId;
207 } else {
208 saveAndClose << sessionId;
209 }
210 }
211
212 qCDebug(KWIN_CORE) << "saveActivity" << id << saveAndClose << saveOnly;
213
214 // pass off to ksmserver
215 QDBusInterface ksmserver("org.kde.ksmserver", "/KSMServer", "org.kde.KSMServerInterface");
216 if (ksmserver.isValid()) {
217 ksmserver.asyncCall("saveSubSession", id, saveAndClose, saveOnly);
218 } else {
219 qCDebug(KWIN_CORE) << "couldn't get ksmserver interface";
220 }
221}
