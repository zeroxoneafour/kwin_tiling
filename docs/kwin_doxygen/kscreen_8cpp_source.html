93void KscreenEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
94{
95 if (isScreenActive(data.screen)) {
96 auto &state = !effects->waylandDisplay() ? m_xcbState : m_waylandStates[data.screen];
97 m_currentScreen = data.screen;
98
99 if (state.m_state == StateFadingIn || state.m_state == StateFadingOut) {
100 state.m_timeLine.advance(presentTime);
101 if (state.m_timeLine.done()) {
102 switchState(state);
103 if (state.m_state == StateNormal) {
104 m_waylandStates.remove(data.screen);
105 }
106 }
107 }
108 }
109
110 effects->prePaintScreen(data, presentTime);
111}

113void KscreenEffect::postPaintScreen()

136void KscreenEffect::paintWindow(EffectWindow *w, int mask, QRegion region, WindowPaintData &data)
137{
138 auto screen = w->screen();
139 if (isScreenActive(screen)) {
140 auto &state = !effects->waylandDisplay() ? m_xcbState : m_waylandStates[screen];
141 // fade to black and fully opaque
142 switch (state.m_state) {
143 case StateFadingOut:
144 data.setOpacity(data.opacity() + (1.0 - data.opacity()) * state.m_timeLine.value());
145 data.multiplyBrightness(1.0 - state.m_timeLine.value());
146 break;
147 case StateFadedOut:
148 data.multiplyOpacity(0.0);
149 data.multiplyBrightness(0.0);
150 break;
151 case StateFadingIn:
152 data.setOpacity(data.opacity() + (1.0 - data.opacity()) * (1.0 - state.m_timeLine.value()));
153 data.multiplyBrightness(state.m_timeLine.value());
154 break;
155 default:
156 // no adjustment
157 break;
158 }
159 }
160 effects->paintWindow(w, mask, region, data);
161}

174void KscreenEffect::propertyNotify(EffectWindow *window, long int atom)
175{
176 if (window || atom != m_atom || m_atom == XCB_ATOM_NONE) {
177 return;
178 }
179
180 const QByteArray byteData = effects->readRootProperty(m_atom, XCB_ATOM_CARDINAL, 32);
181 const uint32_t *data = byteData.isEmpty() ? nullptr : reinterpret_cast<const uint32_t *>(byteData.data());
182 if (!data || data[0] >= LastState) { // Property was deleted
183 if (data) {
184 qCDebug(KWIN_KSCREEN) << "Incorrect Property state, immediate stop: " << data[0];
185 }
186 setState(m_xcbState, StateNormal);
187 return;
188 }
189
190 setState(m_xcbState, FadeOutState(data[0]));
191}

193void KscreenEffect::switchState(ScreenState &state)
194{
195 long value = -1l;
196 if (state.m_state == StateFadingOut) {
197 state.m_state = StateFadedOut;
198 value = 2l;
199 } else if (state.m_state == StateFadingIn) {
200 state.m_state = StateNormal;
201 value = 0l;
202 }
203 if (value != -1l && m_atom != XCB_ATOM_NONE) {
204 xcb_change_property(xcbConnection(), XCB_PROP_MODE_REPLACE, x11RootWindow(), m_atom, XCB_ATOM_CARDINAL, 32, 1, &value);
205 }
206}
