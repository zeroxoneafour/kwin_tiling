63void FakeInputInterfacePrivate::org_kde_kwin_fake_input_bind_resource(Resource *resource)
64{
65 auto device = new FakeInputDevice(q, resource->handle);
66 devices[resource->handle] = std::unique_ptr<FakeInputDevice>(device);
67 Q_EMIT q->deviceCreated(device);
68}
69
70void FakeInputInterfacePrivate::org_kde_kwin_fake_input_destroy_resource(Resource *resource)
71{
72 auto it = devices.find(resource->handle);
73 if (it != devices.end()) {
74 const auto [resource, device] = std::move(*it);
75 devices.erase(it);
76 Q_EMIT q->deviceDestroyed(device.get());
77 }
78}

103void FakeInputInterfacePrivate::org_kde_kwin_fake_input_button(Resource *resource, uint32_t button, uint32_t state)
104{
105 FakeInputDevice *d = device(resource->handle);
106 if (!d || !d->isAuthenticated()) {
107 return;
108 }
109 switch (state) {
110 case WL_POINTER_BUTTON_STATE_PRESSED:
111 Q_EMIT d->pointerButtonPressRequested(button);
112 break;
113 case WL_POINTER_BUTTON_STATE_RELEASED:
114 Q_EMIT d->pointerButtonReleaseRequested(button);
115 break;
116 default:
117 // nothing
118 break;
119 }

121void FakeInputInterfacePrivate::org_kde_kwin_fake_input_axis(Resource *resource, uint32_t axis, wl_fixed_t value)
122{
123 FakeInputDevice *d = device(resource->handle);
124 if (!d || !d->isAuthenticated()) {
125 return;
126 }
127 Qt::Orientation orientation;
128 switch (axis) {
129 case WL_POINTER_AXIS_HORIZONTAL_SCROLL:
130 orientation = Qt::Horizontal;
131 break;
132 case WL_POINTER_AXIS_VERTICAL_SCROLL:
133 orientation = Qt::Vertical;
134 break;
135 default:
136 // invalid
137 return;
138 }
139 Q_EMIT d->pointerAxisRequested(orientation, wl_fixed_to_double(value));
140}

142void FakeInputInterfacePrivate::org_kde_kwin_fake_input_touch_down(Resource *resource, uint32_t id, wl_fixed_t x, wl_fixed_t y)
143{
