98void DimInactiveEffect::paintWindow(EffectWindow *w, int mask, QRegion region, WindowPaintData &data)
99{
100 auto transitionIt = m_transitions.constFind(w);
101 if (transitionIt != m_transitions.constEnd()) {
102 const qreal transitionProgress = (*transitionIt).value();
103 dimWindow(data, m_dimStrength * transitionProgress);
104 effects->paintWindow(w, mask, region, data);
105 return;
106 }
107
108 auto forceIt = m_forceDim.constFind(w);
109 if (forceIt != m_forceDim.constEnd()) {
110 const qreal forcedStrength = *forceIt;
111 dimWindow(data, forcedStrength);
112 effects->paintWindow(w, mask, region, data);
113 return;
114 }
115
116 if (canDimWindow(w)) {
117 dimWindow(data, m_dimStrength);
118 }
119
120 effects->paintWindow(w, mask, region, data);
121}

123void DimInactiveEffect::postPaintScreen()
124{
125 if (m_fullScreenTransition.active) {
126 if (m_fullScreenTransition.timeLine.done()) {
127 m_fullScreenTransition.active = false;
128 }
129 effects->addRepaintFull();
130 }
131
132 auto transitionIt = m_transitions.begin();
133 while (transitionIt != m_transitions.end()) {
134 EffectWindow *w = transitionIt.key();
135 if ((*transitionIt).done()) {
136 transitionIt = m_transitions.erase(transitionIt);
137 } else {
138 ++transitionIt;
139 }
140 w->addRepaintFull();
141 }
142
143 effects->postPaintScreen();
144}

146void DimInactiveEffect::dimWindow(WindowPaintData &data, qreal strength)
147{
148 qreal dimFactor;
149 if (m_fullScreenTransition.active) {
150 dimFactor = 1.0 - m_fullScreenTransition.timeLine.value();
151 } else if (effects->activeFullScreenEffect()) {
152 dimFactor = 0.0;
153 } else {
154 dimFactor = 1.0;
155 }
156
157 data.multiplyBrightness(1.0 - strength * dimFactor);
158 data.multiplySaturation(1.0 - strength * dimFactor);
159}

161bool DimInactiveEffect::canDimWindow(const EffectWindow *w) const
162{
163 if (m_activeWindow == w) {
164 return false;
165 }
166
167 if (m_dimByGroup && belongToSameGroup(m_activeWindow, w)) {
168 return false;
169 }
170
171 if (w->isDock() && !m_dimPanels) {
172 return false;
173 }
174
175 if (w->isDesktop() && !m_dimDesktop) {
176 return false;
177 }
178
179 if (w->keepAbove() && !m_dimKeepAbove) {
180 return false;
181 }
182
183 if (w->isFullScreen() && !m_dimFullScreen) {
184 return false;
185 }
186
187 if (w->isPopupWindow()) {
188 return false;
189 }
190
191 if (w->isX11Client() && !w->isManaged()) {
192 return false;
193 }
194
195 return w->isNormalWindow()
196 || w->isDialog()
197 || w->isUtility()
198 || w->isDock()
199 || w->isDesktop();
200}

202void DimInactiveEffect::scheduleInTransition(EffectWindow *w)
203{
204 TimeLine &timeLine = m_transitions[w];
205 timeLine.setDuration(
206 std::chrono::milliseconds(static_cast<int>(animationTime(160))));
207 if (timeLine.done()) {
208 // If the Out animation is still active, then we're trucating
209 // duration of the timeline(from 250ms to 160ms). If the timeline
210 // is about to be finished with the old duration, then after
211 // changing duration it will be in the "done" state. Thus, we
212 // have to reset the timeline, otherwise it won't update progress.
213 timeLine.reset();
214 }
215 timeLine.setDirection(TimeLine::Backward);
216 timeLine.setEasingCurve(QEasingCurve::InOutSine);
217}

285void DimInactiveEffect::windowActivated(EffectWindow *w)
286{
287 if (!w) {
288 return;
289 }
290
291 if (m_activeWindow == w) {
292 return;
293 }
294
295 if (m_dimByGroup && belongToSameGroup(m_activeWindow, w)) {
296 m_activeWindow = w;
297 return;
298 }
299
300 // WORKAROUND: Deleted windows do not belong to any of window groups.
301 // So, if one of windows in a window group is closed, the In transition
302 // will be false-triggered for the rest of the window group. In addition
303 // to the active window, keep track of active window group so we can
304 // tell whether "focus" moved from a closed window to some other window
305 // in a window group.
306 if (m_dimByGroup && w->group() && w->group() == m_activeWindowGroup) {
307 m_activeWindow = w;
308 return;
309 }
310
311 EffectWindow *previousActiveWindow = m_activeWindow;
312 m_activeWindow = canDimWindow(w) ? w : nullptr;
313
314 m_activeWindowGroup = (m_dimByGroup && m_activeWindow)
315 ? m_activeWindow->group()
316 : nullptr;
317
318 if (previousActiveWindow) {
319 scheduleGroupOutTransition(previousActiveWindow);
320 scheduleRepaint(previousActiveWindow);
321 }
322
323 if (m_activeWindow) {
324 scheduleGroupInTransition(m_activeWindow);
325 scheduleRepaint(m_activeWindow);
326 }
327}

329void DimInactiveEffect::windowClosed(EffectWindow *w)
330{
331 // When a window is closed, we should force current dim strength that
332 // is applied to it to avoid flickering when some effect animates
333 // the disappearing of the window. If there is no such effect then
334 // it won't be dimmed.
335 qreal forcedStrength = 0.0;
336 bool shouldForceDim = false;
337
338 auto transitionIt = m_transitions.find(w);
339 if (transitionIt != m_transitions.end()) {
340 forcedStrength = m_dimStrength * (*transitionIt).value();
341 shouldForceDim = true;
342 m_transitions.erase(transitionIt);
343 } else if (m_activeWindow == w) {
344 forcedStrength = 0.0;
345 shouldForceDim = true;
346 } else if (m_dimByGroup && belongToSameGroup(m_activeWindow, w)) {
347 forcedStrength = 0.0;
348 shouldForceDim = true;
349 } else if (canDimWindow(w)) {
350 forcedStrength = m_dimStrength;
351 shouldForceDim = true;
352 }
353
354 if (shouldForceDim) {
355 m_forceDim.insert(w, forcedStrength);
356 }
357
358 if (m_activeWindow == w) {
359 m_activeWindow = nullptr;
360 }
361}

363void DimInactiveEffect::windowDeleted(EffectWindow *w)
364{
365 m_forceDim.remove(w);
366
367 // FIXME: Sometimes we can miss the window close signal because KWin
368 // can activate a window that is not ready for painting and the window
369 // gets destroyed immediately. So, we have to remove active transitions
370 // for that window here, otherwise we'll crash in postPaintScreen.
371 m_transitions.remove(w);
372 if (m_activeWindow == w) {
373 m_activeWindow = nullptr;
374 }
375}

391void DimInactiveEffect::updateActiveWindow(EffectWindow *w)
392{
393 if (effects->activeWindow() == nullptr) {
394 return;
395 }
396
397 if (effects->activeWindow() != w) {
398 return;
399 }
400
401 // Need to reset m_activeWindow because canDimWindow depends on it.
402 m_activeWindow = nullptr;
403
404 m_activeWindow = canDimWindow(w) ? w : nullptr;
