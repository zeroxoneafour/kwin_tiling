37void RenderLoopPrivate::scheduleRepaint()
38{
39 if (kwinApp()->isTerminating() || (compositeTimer.isActive() && !allowTearing)) {
40 return;
41 }
42 if (vrrPolicy == RenderLoop::VrrPolicy::Always || (vrrPolicy == RenderLoop::VrrPolicy::Automatic && fullscreenItem != nullptr)) {
43 presentMode = allowTearing ? SyncMode::AdaptiveAsync : SyncMode::Adaptive;
44 } else {
45 presentMode = allowTearing ? SyncMode::Async : SyncMode::Fixed;
46 }
47 const std::chrono::nanoseconds vblankInterval(1'000'000'000'000ull / refreshRate);
48 const std::chrono::nanoseconds currentTime(std::chrono::steady_clock::now().time_since_epoch());
49
50 // Estimate when the next presentation will occur. Note that this is a prediction.
51 nextPresentationTimestamp = lastPresentationTimestamp + vblankInterval;
52 if (nextPresentationTimestamp < currentTime && presentMode == SyncMode::Fixed) {
53 nextPresentationTimestamp = lastPresentationTimestamp
54 + alignTimestamp(currentTime - lastPresentationTimestamp, vblankInterval);
55 }
56
57 // Estimate when it's a good time to perform the next compositing cycle.
58 const std::chrono::nanoseconds safetyMargin = std::chrono::milliseconds(3);
59
60 std::chrono::nanoseconds renderTime;
61 switch (q->latencyPolicy()) {
62 case LatencyExtremelyLow:
63 renderTime = std::chrono::nanoseconds(long(vblankInterval.count() * 0.1));
64 break;
65 case LatencyLow:
66 renderTime = std::chrono::nanoseconds(long(vblankInterval.count() * 0.25));
67 break;
68 case LatencyMedium:
69 renderTime = std::chrono::nanoseconds(long(vblankInterval.count() * 0.5));
70 break;
71 case LatencyHigh:
72 renderTime = std::chrono::nanoseconds(long(vblankInterval.count() * 0.75));
73 break;
74 case LatencyExtremelyHigh:
75 renderTime = std::chrono::nanoseconds(long(vblankInterval.count() * 0.9));
76 break;
77 }
78
79 switch (options->renderTimeEstimator()) {
80 case RenderTimeEstimatorMinimum:
81 renderTime = std::max(renderTime, renderJournal.minimum());
82 break;
83 case RenderTimeEstimatorMaximum:
84 renderTime = std::max(renderTime, renderJournal.maximum());
85 break;
86 case RenderTimeEstimatorAverage:
87 renderTime = std::max(renderTime, renderJournal.average());
88 break;
89 }
90
91 std::chrono::nanoseconds nextRenderTimestamp = nextPresentationTimestamp - renderTime - safetyMargin;
92
93 // If we can't render the frame before the deadline, start compositing immediately.
94 if (nextRenderTimestamp < currentTime) {
95 nextRenderTimestamp = currentTime;
96 }
97
98 if (presentMode == SyncMode::Async || presentMode == SyncMode::AdaptiveAsync) {
99 compositeTimer.start(0);
100 } else {
101 const std::chrono::nanoseconds waitInterval = nextRenderTimestamp - currentTime;
102 compositeTimer.start(std::chrono::duration_cast<std::chrono::milliseconds>(waitInterval));
103 }
104}

129void RenderLoopPrivate::notifyFrameCompleted(std::chrono::nanoseconds timestamp)
130{
131 Q_ASSERT(pendingFrameCount > 0);
132 pendingFrameCount--;
133
134 if (lastPresentationTimestamp <= timestamp) {
135 lastPresentationTimestamp = timestamp;
136 } else {
137 qCDebug(KWIN_CORE,
138 "Got invalid presentation timestamp: %lld (current %lld)",
139 static_cast<long long>(timestamp.count()),
140 static_cast<long long>(lastPresentationTimestamp.count()));
141 lastPresentationTimestamp = std::chrono::steady_clock::now().time_since_epoch();
142 }
143
144 if (!inhibitCount) {
145 maybeScheduleRepaint();
146 }
147
148 Q_EMIT q->framePresented(q, timestamp);
149}

151void RenderLoopPrivate::dispatch()
152{
153 // On X11, we want to ignore repaints that are scheduled by windows right before
154 // the Compositor starts repainting.
155 pendingRepaint = true;
156
157 Q_EMIT q->frameRequested(q);
158
159 // The Compositor may decide to not repaint when the frameRequested() signal is
160 // emitted, in which case the pending repaint flag has to be reset manually.
161 pendingRepaint = false;
162}

225void RenderLoop::scheduleRepaint(Item *item)
226{
227 if (d->pendingRepaint || (d->fullscreenItem != nullptr && item != nullptr && item != d->fullscreenItem)) {
228 return;
229 }
230 if (!d->pendingFrameCount && !d->inhibitCount) {
231 d->scheduleRepaint();
232 } else {
233 d->delayScheduleRepaint();
234 }
