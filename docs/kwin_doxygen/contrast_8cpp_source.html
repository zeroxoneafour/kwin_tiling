33{
34 m_shader = std::make_unique<ContrastShader>();
35 m_shader->init();
36
37 // ### Hackish way to announce support.
38 // Should be included in _NET_SUPPORTED instead.
39 if (m_shader && m_shader->isValid()) {
40 if (effects->xcbConnection()) {
41 m_net_wm_contrast_region = effects->announceSupportProperty(s_contrastAtomName, this);
42 }
43 if (effects->waylandDisplay()) {
44 if (!s_contrastManagerRemoveTimer) {
45 s_contrastManagerRemoveTimer = new QTimer(QCoreApplication::instance());
46 s_contrastManagerRemoveTimer->setSingleShot(true);
47 s_contrastManagerRemoveTimer->callOnTimeout([]() {
48 s_contrastManager->remove();

50 });
51 }
52 s_contrastManagerRemoveTimer->stop();
53 if (!s_contrastManager) {
54 s_contrastManager = new KWaylandServer::ContrastManagerInterface(effects->waylandDisplay(), s_contrastManagerRemoveTimer);
55 }
56 }
57 }
58
59 connect(effects, &EffectsHandler::windowAdded, this, &ContrastEffect::slotWindowAdded);
60 connect(effects, &EffectsHandler::windowDeleted, this, &ContrastEffect::slotWindowDeleted);
61 connect(effects, &EffectsHandler::propertyNotify, this, &ContrastEffect::slotPropertyNotify);
62 connect(effects, &EffectsHandler::virtualScreenGeometryChanged, this, &ContrastEffect::slotScreenGeometryChanged);
63 connect(effects, &EffectsHandler::xcbConnectionChanged, this, [this]() {
64 if (m_shader && m_shader->isValid()) {
65 m_net_wm_contrast_region = effects->announceSupportProperty(s_contrastAtomName, this);
66 }
67 });
68
69 // Fetch the contrast regions for all windows
70 const EffectWindowList windowList = effects->stackingOrder();
71 for (EffectWindow *window : windowList) {

98void ContrastEffect::updateContrastRegion(EffectWindow *w)
99{
100 QRegion region;
101 QMatrix4x4 matrix;
102 float colorTransform[16];
103 bool valid = false;
104
105 if (m_net_wm_contrast_region != XCB_ATOM_NONE) {
106 const QByteArray value = w->readProperty(m_net_wm_contrast_region, m_net_wm_contrast_region, 32);
107
108 if (value.size() > 0 && !((value.size() - (16 * sizeof(uint32_t))) % ((4 * sizeof(uint32_t))))) {
109 const uint32_t *cardinals = reinterpret_cast<const uint32_t *>(value.constData());
110 const float *floatCardinals = reinterpret_cast<const float *>(value.constData());
111 unsigned int i = 0;
112 for (; i < ((value.size() - (16 * sizeof(uint32_t)))) / sizeof(uint32_t);) {
113 int x = cardinals[i++];
114 int y = cardinals[i++];
115 int w = cardinals[i++];
116 int h = cardinals[i++];
117 region += Xcb::fromXNative(QRect(x, y, w, h)).toRect();
118 }
119
120 for (unsigned int j = 0; j < 16; ++j) {
121 colorTransform[j] = floatCardinals[i + j];
122 }
123
124 matrix = QMatrix4x4(colorTransform);
125 }
126
127 valid = !value.isNull();
128 }
129
130 KWaylandServer::SurfaceInterface *surf = w->surface();
131
132 if (surf && surf->contrast()) {
133 region = surf->contrast()->region();
134 matrix = colorMatrix(surf->contrast()->contrast(), surf->contrast()->intensity(), surf->contrast()->saturation());
135 valid = true;
136 }
137
138 if (auto internal = w->internalWindow()) {
139 const auto property = internal->property("kwin_background_region");
140 if (property.isValid()) {
141 region = property.value<QRegion>();
142 bool ok = false;
143 qreal contrast = internal->property("kwin_background_contrast").toReal(&ok);
144 if (!ok) {
145 contrast = 1.0;
146 }
147 qreal intensity = internal->property("kwin_background_intensity").toReal(&ok);
148 if (!ok) {
149 intensity = 1.0;
150 }
151 qreal saturation = internal->property("kwin_background_saturation").toReal(&ok);
152 if (!ok) {
153 saturation = 1.0;
154 }
155 matrix = colorMatrix(contrast, intensity, saturation);
156 valid = true;
157 }
158 }
159
160 if (valid) {
161 m_windowData[w] = {
162 .colorMatrix = matrix,
163 .contrastRegion = region,
164 };
165 } else {
166 m_windowData.remove(w);
167 }
168}

219QMatrix4x4 ContrastEffect::colorMatrix(qreal contrast, qreal intensity, qreal saturation)
220{
221 QMatrix4x4 satMatrix; // saturation
222 QMatrix4x4 intMatrix; // intensity
223 QMatrix4x4 contMatrix; // contrast
224
225 // Saturation matrix
226 if (!qFuzzyCompare(saturation, 1.0)) {
227 const qreal rval = (1.0 - saturation) * .2126;
228 const qreal gval = (1.0 - saturation) * .7152;
229 const qreal bval = (1.0 - saturation) * .0722;
230
231 satMatrix = QMatrix4x4(rval + saturation, rval, rval, 0.0,
232 gval, gval + saturation, gval, 0.0,
233 bval, bval, bval + saturation, 0.0,
234 0, 0, 0, 1.0);
235 }
236
237 // IntensityMatrix
238 if (!qFuzzyCompare(intensity, 1.0)) {
239 intMatrix.scale(intensity, intensity, intensity);
240 }
241
242 // Contrast Matrix
243 if (!qFuzzyCompare(contrast, 1.0)) {
244 const float transl = (1.0 - contrast) / 2.0;
245
246 contMatrix = QMatrix4x4(contrast, 0, 0, 0.0,
247 0, contrast, 0, 0.0,
248 0, 0, contrast, 0.0,
249 transl, transl, transl, 1.0);
250 }
251
252 QMatrix4x4 colorMatrix = contMatrix * satMatrix * intMatrix;
253 // colorMatrix = colorMatrix.transposed();
254
255 return colorMatrix;
256}

311void ContrastEffect::uploadRegion(QVector2D *&map, const QRegion &region, qreal scale)
312{
313 Q_ASSERT(map);
314 for (const QRect &r : region) {
315 const auto deviceRect = scaledRect(r, scale);
316 const QVector2D topLeft = roundVector(QVector2D(deviceRect.x(), deviceRect.y()));
317 const QVector2D topRight = roundVector(QVector2D(deviceRect.x() + deviceRect.width(), deviceRect.y()));
318 const QVector2D bottomLeft = roundVector(QVector2D(deviceRect.x(), deviceRect.y() + deviceRect.height()));
319 const QVector2D bottomRight = roundVector(QVector2D(deviceRect.x() + deviceRect.width(), deviceRect.y() + deviceRect.height()));
320
321 // First triangle
322 *(map++) = topRight;
323 *(map++) = topLeft;
324 *(map++) = bottomLeft;
325
326 // Second triangle
327 *(map++) = bottomLeft;
328 *(map++) = bottomRight;
329 *(map++) = topRight;
330 }
331}

333bool ContrastEffect::uploadGeometry(GLVertexBuffer *vbo, const QRegion &region, qreal scale)
334{
335 const int vertexCount = region.rectCount() * 6;
336 if (!vertexCount) {
337 return false;
338 }
339
340 QVector2D *map = (QVector2D *)vbo->map(vertexCount * sizeof(QVector2D));
341 if (!map) {
342 return false;
343 }
344 uploadRegion(map, region, scale);
345 vbo->unmap();
346
347 const GLVertexAttrib layout[] = {
348 {VA_Position, 2, GL_FLOAT, 0},
349 {VA_TexCoord, 2, GL_FLOAT, 0}};
350
351 vbo->setAttribLayout(layout, 2, sizeof(QVector2D));
352 return true;
353}

355bool ContrastEffect::shouldContrast(const EffectWindow *w, int mask, const WindowPaintData &data) const
356{
357 if (!m_shader || !m_shader->isValid()) {
358 return false;
359 }
360
361 if (effects->activeFullScreenEffect() && !w->data(WindowForceBackgroundContrastRole).toBool()) {
362 return false;
363 }
364
365 if (w->isDesktop()) {
366 return false;
367 }
368
369 bool scaled = !qFuzzyCompare(data.xScale(), 1.0) && !qFuzzyCompare(data.yScale(), 1.0);
370 bool translated = data.xTranslation() || data.yTranslation();
371
372 if ((scaled || (translated || (mask & PAINT_WINDOW_TRANSFORMED))) && !w->data(WindowForceBackgroundContrastRole).toBool()) {
373 return false;
374 }
375
376 return true;
377}

379void ContrastEffect::drawWindow(EffectWindow *w, int mask, const QRegion &region, WindowPaintData &data)
380{
381 if (shouldContrast(w, mask, data)) {
382 const QRect screen = effects->renderTargetRect();
383 QRegion shape = region & contrastRegion(w).translated(w->pos().toPoint()) & screen;
384
385 // let's do the evil parts - someone wants to blur behind a transformed window
386 const bool translated = data.xTranslation() || data.yTranslation();
387 const bool scaled = data.xScale() != 1 || data.yScale() != 1;
388 if (scaled) {
389 QPoint pt = shape.boundingRect().topLeft();
390 QRegion scaledShape;
391 for (QRect r : shape) {
392 r.moveTo(pt.x() + (r.x() - pt.x()) * data.xScale() + data.xTranslation(),
393 pt.y() + (r.y() - pt.y()) * data.yScale() + data.yTranslation());
394 r.setWidth(std::ceil(r.width() * data.xScale()));
395 r.setHeight(std::ceil(r.height() * data.yScale()));
396 scaledShape |= r;
397 }
398 shape = scaledShape & region;
399
400 // Only translated, not scaled
401 } else if (translated) {
402 shape = shape.translated(data.xTranslation(), data.yTranslation());
403 shape = shape & region;
404 }
405
406 if (!shape.isEmpty()) {
407 doContrast(w, shape, screen, data.opacity(), data.projectionMatrix());
408 }
409 }
410
411 // Draw the window over the contrast area
412 effects->drawWindow(w, mask, region, data);
413}

415void ContrastEffect::doContrast(EffectWindow *w, const QRegion &shape, const QRect &screen, const float opacity, const QMatrix4x4 &screenProjection)
416{
417 const qreal scale = effects->renderTargetScale();
418 const QRegion actualShape = shape & screen;
419 const QRectF r = scaledRect(actualShape.boundingRect(), scale);
420
421 // Upload geometry for the horizontal and vertical passes
422 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
423 vbo->reset();
424 if (!uploadGeometry(vbo, actualShape, scale)) {
425 return;
426 }
427 vbo->bindArrays();
428
429 // Create a scratch texture and copy the area in the back buffer that we're
430 // going to blur into it
431 GLTexture scratch(GL_RGBA8, r.width(), r.height());
432 scratch.setFilter(GL_LINEAR);
433 scratch.setWrapMode(GL_CLAMP_TO_EDGE);
434 scratch.bind();
435
436 const QRectF sg = scaledRect(effects->renderTargetRect(), scale);
437 glCopyTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, (r.x() - sg.x()), (sg.height() - (r.y() - sg.y() + r.height())),
438 scratch.width(), scratch.height());
439
440 // Draw the texture on the offscreen framebuffer object, while blurring it horizontally
441
442 m_shader->setColorMatrix(m_windowData.value(w).colorMatrix);
443 m_shader->bind();
444
445 m_shader->setOpacity(opacity);
446 // Set up the texture matrix to transform from screen coordinates
447 // to texture coordinates.
448 QMatrix4x4 textureMatrix;
449 textureMatrix.scale(1.0 / r.width(), -1.0 / r.height(), 1);
450 textureMatrix.translate(-r.x(), -r.height() - r.y(), 0);
451 m_shader->setTextureMatrix(textureMatrix);
452 m_shader->setModelViewProjectionMatrix(screenProjection);
453
454 vbo->draw(GL_TRIANGLES, 0, actualShape.rectCount() * 6);
455
456 scratch.unbind();
457 scratch.discard();
458
459 vbo->unbindArrays();
460
461 if (opacity < 1.0) {
462 glDisable(GL_BLEND);
463 }
464
465 m_shader->unbind();
466}
