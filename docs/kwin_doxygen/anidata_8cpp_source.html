61bool AniData::isActive() const
62{
63 if (!timeLine.done()) {
64 return true;
65 }
66
67 if (timeLine.direction() == TimeLine::Backward) {
68 return !(terminationFlags & AnimationEffect::TerminateAtSource);
69 }
70
71 return !(terminationFlags & AnimationEffect::TerminateAtTarget);
72}
