62bool AniData::isActive() const
63{
64 if (!timeLine.done()) {
65 return true;
66 }
67
68 if (timeLine.direction() == TimeLine::Backward) {
69 return !(terminationFlags & AnimationEffect::TerminateAtSource);
70 }
71
72 return !(terminationFlags & AnimationEffect::TerminateAtTarget);
73}
