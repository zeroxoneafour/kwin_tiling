29WorkspaceWrapper::WorkspaceWrapper(QObject *parent)
30 : QObject(parent)
31{
32 KWin::Workspace *ws = KWin::Workspace::self();
33 KWin::VirtualDesktopManager *vds = KWin::VirtualDesktopManager::self();
34 connect(ws, &Workspace::desktopPresenceChanged, this, &WorkspaceWrapper::desktopPresenceChanged);
35 connect(ws, &Workspace::windowAdded, this, &WorkspaceWrapper::clientAdded);
36 connect(ws, &Workspace::windowRemoved, this, &WorkspaceWrapper::clientRemoved);
37 connect(ws, &Workspace::windowActivated, this, &WorkspaceWrapper::clientActivated);
38 connect(vds, &VirtualDesktopManager::desktopCreated, this, &WorkspaceWrapper::desktopsChanged);
39 connect(vds, &VirtualDesktopManager::desktopRemoved, this, &WorkspaceWrapper::desktopsChanged);
40 connect(vds, &VirtualDesktopManager::layoutChanged, this, &WorkspaceWrapper::desktopLayoutChanged);
41 connect(vds, &VirtualDesktopManager::currentChanged, this, &WorkspaceWrapper::currentDesktopChanged);
42 connect(ws, &Workspace::windowDemandsAttentionChanged, this, &WorkspaceWrapper::clientDemandsAttentionChanged);
43#if KWIN_BUILD_ACTIVITIES
44 if (KWin::Activities *activities = ws->activities()) {
45 connect(activities, &Activities::currentChanged, this, &WorkspaceWrapper::currentActivityChanged);
46 connect(activities, &Activities::added, this, &WorkspaceWrapper::activitiesChanged);
47 connect(activities, &Activities::added, this, &WorkspaceWrapper::activityAdded);
48 connect(activities, &Activities::removed, this, &WorkspaceWrapper::activitiesChanged);
49 connect(activities, &Activities::removed, this, &WorkspaceWrapper::activityRemoved);
50 }
51#endif
52 connect(ws, &Workspace::geometryChanged, this, &WorkspaceWrapper::virtualScreenSizeChanged);
53 connect(ws, &Workspace::geometryChanged, this, &WorkspaceWrapper::virtualScreenGeometryChanged);
54 connect(ws, &Workspace::outputsChanged, this, &WorkspaceWrapper::screensChanged);
55 connect(Cursors::self()->mouse(), &Cursor::posChanged, this, &WorkspaceWrapper::cursorPosChanged);
56}
