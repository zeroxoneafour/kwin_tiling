111void BlurShader::setModelViewProjectionMatrix(const QMatrix4x4 &matrix)
112{
113 if (!isValid()) {
114 return;
115 }
116
117 switch (m_activeSampleType) {
118 case CopySampleType:
119 if (matrix == m_matrixCopysample) {
120 return;
121 }
122
123 m_matrixCopysample = matrix;
124 m_shaderCopysample->setUniform(m_mvpMatrixLocationCopysample, matrix);
125 break;
126
127 case UpSampleType:
128 if (matrix == m_matrixUpsample) {
129 return;
130 }
131
132 m_matrixUpsample = matrix;
133 m_shaderUpsample->setUniform(m_mvpMatrixLocationUpsample, matrix);
134 break;
135
136 case DownSampleType:
137 if (matrix == m_matrixDownsample) {
138 return;
139 }
140
141 m_matrixDownsample = matrix;
142 m_shaderDownsample->setUniform(m_mvpMatrixLocationDownsample, matrix);
143 break;
144
145 case NoiseSampleType:
146 if (matrix == m_matrixNoisesample) {
147 return;
148 }
149
150 m_matrixNoisesample = matrix;
151 m_shaderNoisesample->setUniform(m_mvpMatrixLocationNoisesample, matrix);
152 break;
153
154 default:
155 Q_UNREACHABLE();
156 break;
157 }

200void BlurShader::setTargetTextureSize(const QSize &renderTextureSize)
201{
202 if (!isValid()) {
203 return;
204 }
205
206 const QVector2D texSize(renderTextureSize.width(), renderTextureSize.height());
207
208 switch (m_activeSampleType) {
209 case CopySampleType:
210 m_shaderCopysample->setUniform(m_renderTextureSizeLocationCopysample, texSize);
211 break;
212
213 case UpSampleType:
214 m_shaderUpsample->setUniform(m_renderTextureSizeLocationUpsample, texSize);
215 m_shaderUpsample->setUniform(m_halfpixelLocationUpsample, QVector2D(0.5 / texSize.x(), 0.5 / texSize.y()));
216 break;
217
218 case DownSampleType:
219 m_shaderDownsample->setUniform(m_renderTextureSizeLocationDownsample, texSize);
220 m_shaderDownsample->setUniform(m_halfpixelLocationDownsample, QVector2D(0.5 / texSize.x(), 0.5 / texSize.y()));
221 break;
222
223 case NoiseSampleType:
224 m_shaderNoisesample->setUniform(m_renderTextureSizeLocationNoisesample, texSize);
225 m_shaderNoisesample->setUniform(m_halfpixelLocationNoisesample, QVector2D(0.5 / texSize.x(), 0.5 / texSize.y()));
226 break;
227
228 default:
229 Q_UNREACHABLE();
230 break;
231 }
