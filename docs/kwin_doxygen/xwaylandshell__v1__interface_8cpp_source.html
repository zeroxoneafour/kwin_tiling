68void XwaylandShellV1InterfacePrivate::xwayland_shell_v1_get_xwayland_surface(Resource *resource, uint32_t id, ::wl_resource *surface_resource)
69{
70 SurfaceInterface *surface = SurfaceInterface::get(surface_resource);
71 if (auto role = SurfaceRole::get(surface)) {
72 wl_resource_post_error(resource->handle, error_role, "the surface already has a role assigned %s", role->name().constData());
73 return;
74 }
75
76 auto xwaylandSurface = new XwaylandSurfaceV1Interface(q, surface, resource->client(), id, resource->version());
77 m_surfaces.append(xwaylandSurface);
78 QObject::connect(xwaylandSurface, &QObject::destroyed, q, [this, xwaylandSurface]() {
79 m_surfaces.removeOne(xwaylandSurface);
80 });
81}

104void XwaylandSurfaceV1InterfacePrivate::xwayland_surface_v1_set_serial(Resource *resource, uint32_t serial_lo, uint32_t serial_hi)
105{
106 const uint64_t serial = (uint64_t(serial_hi) << 32) | serial_lo;
107 if (!serial) {
108 wl_resource_post_error(resource->handle, error_invalid_serial, "given serial is 0");
109 return;
110 }
111
112 if (current.serial.has_value()) {
113 wl_resource_post_error(resource->handle, error_already_associated,
114 "xwayland_surface_v1 already has a serial assigned to it: %" PRIu64, current.serial.value());
115 return;
116 }
117
118 pending.serial = serial;
119}
