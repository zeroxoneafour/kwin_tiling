49void ServerSideDecorationManagerInterfacePrivate::org_kde_kwin_server_decoration_manager_create(Resource *resource, uint32_t id, wl_resource *surface)
50{
51 SurfaceInterface *s = SurfaceInterface::get(surface);
52 if (!s) {
53 wl_resource_post_error(resource->handle, 0, "Invalid surface");
54 return;
55 }
56
57 wl_resource *decorationResource = wl_resource_create(resource->client(), &org_kde_kwin_server_decoration_interface, resource->version(), id);
58 if (!decorationResource) {
59 wl_client_post_no_memory(resource->client());
60 return;
61 }
62 auto decoration = new ServerSideDecorationInterface(q, s, decorationResource);
63 decoration->setMode(defaultMode);
64 Q_EMIT q->decorationCreated(decoration);
65}

67void ServerSideDecorationManagerInterfacePrivate::setDefaultMode(ServerSideDecorationManagerInterface::Mode mode)
68{
69 defaultMode = mode;
70 const uint32_t wlMode = modeWayland(mode);
71
72 const auto clientResources = resourceMap();
73 for (Resource *resource : clientResources) {
74 send_default_mode(resource->handle, wlMode);
75 }
76}

136void ServerSideDecorationInterfacePrivate::org_kde_kwin_server_decoration_request_mode(Resource *resource, uint32_t mode)
137{
138 ServerSideDecorationManagerInterface::Mode m = ServerSideDecorationManagerInterface::Mode::None;
139 switch (mode) {
140 case ServerSideDecorationManagerInterfacePrivate::mode_None:
141 m = ServerSideDecorationManagerInterface::Mode::None;
142 break;
143 case ServerSideDecorationManagerInterfacePrivate::mode_Client:
144 m = ServerSideDecorationManagerInterface::Mode::Client;
145 break;
146 case ServerSideDecorationManagerInterfacePrivate::mode_Server:
147 m = ServerSideDecorationManagerInterface::Mode::Server;
148 break;
149 default:
150 // invalid mode
151 qCWarning(KWIN_CORE) << "Invalid mode:" << mode;
152 return;
153 }
154 preferredMode = m;
155 Q_EMIT q->preferredModeChanged();
156}
