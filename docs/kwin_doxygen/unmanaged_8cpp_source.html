52Unmanaged::Unmanaged()
53 : Window()
54{
55 switch (kwinApp()->operationMode()) {
56 case Application::OperationModeXwayland:
57 // The wayland surface is associated with the override-redirect window asynchronously.
58 connect(this, &Window::surfaceChanged, this, &Unmanaged::associate);
59 break;
60 case Application::OperationModeX11:
61 // We have no way knowing whether the override-redirect window can be painted. Mark it
62 // as ready for painting after synthetic 50ms delay.
63 QTimer::singleShot(50, this, &Unmanaged::initialize);
64 break;
65 case Application::OperationModeWaylandOnly:
66 Q_UNREACHABLE();
67 }
68}

79void Unmanaged::associate()
80{
81 if (surface()->isMapped()) {
82 initialize();
83 } else {
84 // Queued connection because we want to mark the window ready for painting after
85 // the associated surface item has processed the new surface state.
86 connect(surface(), &SurfaceInterface::mapped, this, &Unmanaged::initialize, Qt::QueuedConnection);
87 }

95bool Unmanaged::track(xcb_window_t w)
96{
97 XServerGrabber xserverGrabber;
98 Xcb::WindowAttributes attr(w);
99 Xcb::WindowGeometry geo(w);
100 if (attr.isNull() || attr->map_state != XCB_MAP_STATE_VIEWABLE) {
101 return false;
102 }
103 if (attr->_class == XCB_WINDOW_CLASS_INPUT_ONLY) {
104 return false;
105 }
106 if (geo.isNull()) {
107 return false;
108 }
109
110 setWindowHandles(w); // the window is also the frame
111 Xcb::selectInput(w, attr->your_event_mask | XCB_EVENT_MASK_STRUCTURE_NOTIFY | XCB_EVENT_MASK_PROPERTY_CHANGE);
112 m_bufferGeometry = geo.rect();
113 m_frameGeometry = geo.rect();
114 m_clientGeometry = geo.rect();
115 checkOutput();
116 m_visual = attr->visual;
117 bit_depth = geo->depth;
118 info = new NETWinInfo(kwinApp()->x11Connection(), w, kwinApp()->x11RootWindow(),
119 NET::WMWindowType | NET::WMPid,
120 NET::WM2Opacity | NET::WM2WindowRole | NET::WM2WindowClass | NET::WM2OpaqueRegion);
121 setOpacity(info->opacityF());
122 getResourceClass();
123 getWmClientLeader();
124 getWmClientMachine();
125 if (Xcb::Extensions::self()->isShapeAvailable()) {
126 xcb_shape_select_input(kwinApp()->x11Connection(), w, true);
127 }
128 detectShape(w);
129 getWmOpaqueRegion();
130 getSkipCloseAnimation();
131 setupCompositing();
132 if (QWindow *internalWindow = findInternalWindow()) {
133 m_outline = internalWindow->property("__kwin_outline").toBool();
134 }
135 if (effects) {
136 static_cast<EffectsHandlerImpl *>(effects)->checkInputWindowStacking();
137 }
138 return true;
139}

141void Unmanaged::release(ReleaseReason releaseReason)
142{
143 Deleted *del = nullptr;
144 if (releaseReason != ReleaseReason::KWinShutsDown) {
145 del = Deleted::create(this);
146 }
147 Q_EMIT windowClosed(this, del);
148 finishCompositing(releaseReason);
149 if (!QWidget::find(window()) && releaseReason != ReleaseReason::Destroyed) { // don't affect our own windows
150 if (Xcb::Extensions::self()->isShapeAvailable()) {
151 xcb_shape_select_input(kwinApp()->x11Connection(), window(), false);
152 }
153 Xcb::selectInput(window(), XCB_EVENT_MASK_NO_EVENT);
154 }
155 workspace()->removeUnmanaged(this);
156 if (releaseReason != ReleaseReason::KWinShutsDown) {
157 disownDataPassedToDeleted();
158 del->unrefWindow();
159 }
160 deleteUnmanaged(this);
161}

193NET::WindowType Unmanaged::windowType(bool direct, int supportedTypes) const
194{
195 // for unmanaged windows the direct does not make any difference
