253uint NightColorDBusInterface::inhibit()
254{
255 const QString serviceName = QDBusContext::message().service();
256
257 if (!m_inhibitors.contains(serviceName)) {
258 m_inhibitorWatcher->addWatchedService(serviceName);
259 }
260
261 m_inhibitors.insert(serviceName, ++m_lastInhibitionCookie);
262
263 m_manager->inhibit();
264
265 return m_lastInhibitionCookie;
266}

275void NightColorDBusInterface::uninhibit(const QString &serviceName, uint cookie)
276{
277 const int removedCount = m_inhibitors.remove(serviceName, cookie);
278 if (!removedCount) {
279 return;
280 }
281
282 if (!m_inhibitors.contains(serviceName)) {
283 m_inhibitorWatcher->removeWatchedService(serviceName);
284 }
285
286 m_manager->uninhibit();
287}
288
289void NightColorDBusInterface::removeInhibitorService(const QString &serviceName)
290{
291 const auto cookies = m_inhibitors.values(serviceName);
292 for (const uint &cookie : cookies) {
293 uninhibit(serviceName, cookie);
294 }
295}
