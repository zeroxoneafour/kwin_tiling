40{
41 uint32_t ret = 0;

434void OutputDeviceV2Interface::updateModes()
435{
436 const auto oldModes = d->m_modes;
437 d->m_modes.clear();
438 d->m_currentMode = nullptr;
439
440 const auto clientResources = d->resourceMap();
441 const auto nativeModes = d->m_handle->modes();
442
443 for (const std::shared_ptr<OutputMode> &mode : nativeModes) {
444 OutputDeviceModeV2Interface *deviceMode = new OutputDeviceModeV2Interface(mode, this);
445 d->m_modes.append(deviceMode);
446
447 if (d->m_handle->currentMode() == mode) {
448 d->m_currentMode = deviceMode;
449 }
450
451 for (auto resource : clientResources) {
452 d->sendNewMode(resource, deviceMode);
453 }
454 }
455
456 for (auto resource : clientResources) {
457 d->sendCurrentMode(resource);
458 }
459
460 qDeleteAll(oldModes.crbegin(), oldModes.crend());
461
462 for (auto resource : clientResources) {
463 d->sendDone(resource);
464 }
465}

467void OutputDeviceV2Interface::updateCurrentMode()
468{
469 for (OutputDeviceModeV2Interface *mode : std::as_const(d->m_modes)) {
470 if (mode->handle().lock() == d->m_handle->currentMode()) {
471 if (d->m_currentMode != mode) {
472 d->m_currentMode = mode;
473 const auto clientResources = d->resourceMap();
474 for (auto resource : clientResources) {
475 d->sendCurrentMode(resource);
476 d->sendDone(resource);
477 }
478 updateGeometry();
479 }
480 return;
481 }
482 }
