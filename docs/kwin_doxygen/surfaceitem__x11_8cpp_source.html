16SurfaceItemX11::SurfaceItemX11(Window *window, Scene *scene, Item *parent)
17 : SurfaceItem(scene, parent)
18 , m_window(window)
19{
20 connect(window, &Window::bufferGeometryChanged,
21 this, &SurfaceItemX11::handleBufferGeometryChanged);
22 connect(window, &Window::geometryShapeChanged,
23 this, &SurfaceItemX11::handleGeometryShapeChanged);
24 connect(window, &Window::windowClosed,
25 this, &SurfaceItemX11::handleWindowClosed);
26
27 m_damageHandle = xcb_generate_id(kwinApp()->x11Connection());
28 xcb_damage_create(kwinApp()->x11Connection(), m_damageHandle, window->frameId(),
29 XCB_DAMAGE_REPORT_LEVEL_NON_EMPTY);
30
31 // With unmanaged windows there is a race condition between the client painting the window
32 // and us setting up damage tracking. If the client wins we won't get a damage event even
33 // though the window has been painted. To avoid this we mark the whole window as damaged
34 // immediately after creating the damage object.
35 if (window->isUnmanaged()) {
36 m_isDamaged = true;
37 }
38
39 setSize(window->bufferGeometry().size());
40}

74bool SurfaceItemX11::fetchDamage()
75{
76 if (!m_isDamaged) {
77 return false;
78 }
79
80 if (m_damageHandle == XCB_NONE) {
81 return true;
82 }
83
84 xcb_xfixes_region_t region = xcb_generate_id(kwinApp()->x11Connection());
85 xcb_xfixes_create_region(kwinApp()->x11Connection(), region, 0, nullptr);
86 xcb_damage_subtract(kwinApp()->x11Connection(), m_damageHandle, 0, region);
87
88 m_damageCookie = xcb_xfixes_fetch_region_unchecked(kwinApp()->x11Connection(), region);
89 xcb_xfixes_destroy_region(kwinApp()->x11Connection(), region);
90
91 m_havePendingDamageRegion = true;
92
93 return true;
94}

96void SurfaceItemX11::waitForDamage()
97{
98 if (!m_havePendingDamageRegion) {
99 return;
100 }
101 m_havePendingDamageRegion = false;
102
103 xcb_xfixes_fetch_region_reply_t *reply =
104 xcb_xfixes_fetch_region_reply(kwinApp()->x11Connection(), m_damageCookie, nullptr);
105 if (!reply) {
106 qCDebug(KWIN_CORE) << "Failed to check damage region";
107 return;
108 }
109
110 const int rectCount = xcb_xfixes_fetch_region_rectangles_length(reply);
111 QRegion region;
112
113 if (rectCount > 1 && rectCount < 16) {
114 xcb_rectangle_t *rects = xcb_xfixes_fetch_region_rectangles(reply);
115
116 QVector<QRect> qtRects;
117 qtRects.reserve(rectCount);
118
119 for (int i = 0; i < rectCount; ++i) {
120 qtRects << QRect(rects[i].x, rects[i].y, rects[i].width, rects[i].height);
121 }
122 region.setRects(qtRects.constData(), rectCount);
123 } else {
124 region = QRect(reply->extents.x, reply->extents.y, reply->extents.width, reply->extents.height);
125 }
126 free(reply);
127
128 addDamage(region);
129 m_isDamaged = false;
130}

155QVector<QRectF> SurfaceItemX11::shape() const
156{
157 const QRectF clipRect = m_window->clientGeometry().translated(-m_window->bufferGeometry().topLeft());
158 QVector<QRectF> shape = m_window->shapeRegion();
159 // bounded to clipRect
160 for (QRectF &shapePart : shape) {
161 shapePart = shapePart.intersected(clipRect);
162 }
163 return shape;

213void SurfacePixmapX11::create()
214{
215 const Window *window = m_item->window();
216 if (window->isDeleted()) {
217 return;
218 }
219
220 XServerGrabber grabber;
221 xcb_connection_t *connection = kwinApp()->x11Connection();
222 xcb_window_t frame = window->frameId();
223 xcb_pixmap_t pixmap = xcb_generate_id(connection);
224 xcb_void_cookie_t namePixmapCookie = xcb_composite_name_window_pixmap_checked(connection,
225 frame,
226 pixmap);
227 Xcb::WindowAttributes windowAttributes(frame);
228 Xcb::WindowGeometry windowGeometry(frame);
229 if (xcb_generic_error_t *error = xcb_request_check(connection, namePixmapCookie)) {
230 qCDebug(KWIN_CORE, "Failed to create window pixmap for window 0x%x (error code %d)",
231 window->window(), error->error_code);
232 free(error);
233 return;
234 }
235 // check that the received pixmap is valid and actually matches what we
236 // know about the window (i.e. size)
237 if (!windowAttributes || windowAttributes->map_state != XCB_MAP_STATE_VIEWABLE) {
238 qCDebug(KWIN_CORE, "Failed to create window pixmap for window 0x%x (not viewable)",
239 window->window());
240 xcb_free_pixmap(connection, pixmap);
241 return;
242 }
243 const QRectF bufferGeometry = window->bufferGeometry();
244 if (windowGeometry.size() != bufferGeometry.size()) {
245 qCDebug(KWIN_CORE, "Failed to create window pixmap for window 0x%x: window size (%fx%f) != buffer size (%fx%f)", window->window(),
246 windowGeometry.size().width(), windowGeometry.size().height(), bufferGeometry.width(), bufferGeometry.height());
247 xcb_free_pixmap(connection, pixmap);
248 return;
249 }
250
251 m_pixmap = pixmap;
252 m_hasAlphaChannel = window->hasAlpha();
253 // this class is only used on X11 where the logical size and
254 // device pixel size is guaranteed to be the same and we can convert safely
255 m_size = bufferGeometry.size().toSize();
256}
