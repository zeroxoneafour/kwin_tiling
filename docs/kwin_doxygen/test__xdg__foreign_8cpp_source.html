80void TestForeign::init()
81{
82 using namespace KWaylandServer;
83 delete m_display;
84 m_display = new KWaylandServer::Display(this);
85 m_display->addSocketName(s_socketName);
86 m_display->start();
87 QVERIFY(m_display->isRunning());
88
89 qRegisterMetaType<SurfaceInterface *>();
90 // setup connection
91 m_connection = new KWayland::Client::ConnectionThread;
92 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
93 m_connection->setSocketName(s_socketName);
94
95 m_thread = new QThread(this);
96 m_connection->moveToThread(m_thread);
97 m_thread->start();
98
99 m_connection->initConnection();
100 QVERIFY(connectedSpy.wait());
101
102 m_queue = new KWayland::Client::EventQueue(this);
103 QVERIFY(!m_queue->isValid());
104 m_queue->setup(m_connection);
105 QVERIFY(m_queue->isValid());
106
107 KWayland::Client::Registry registry;
108 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
109
110 QSignalSpy exporterSpy(&registry, &KWayland::Client::Registry::exporterUnstableV2Announced);
111
112 QSignalSpy importerSpy(&registry, &KWayland::Client::Registry::importerUnstableV2Announced);
113
114 QVERIFY(!registry.eventQueue());
115 registry.setEventQueue(m_queue);
116 QCOMPARE(registry.eventQueue(), m_queue);
117 registry.create(m_connection->display());
118 QVERIFY(registry.isValid());
119 registry.setup();
120
121 m_compositorInterface = new CompositorInterface(m_display, m_display);
122 QVERIFY(compositorSpy.wait());
123 m_compositor = registry.createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
124
125 m_foreignInterface = new XdgForeignV2Interface(m_display, m_display);
126
127 QVERIFY(exporterSpy.wait());
128 // Both importer and exporter should have been triggered by now
129 QCOMPARE(exporterSpy.count(), 1);
130 QCOMPARE(importerSpy.count(), 1);
131
132 m_exporter = registry.createXdgExporter(exporterSpy.first().first().value<quint32>(), exporterSpy.first().last().value<quint32>(), this);
133 m_importer = registry.createXdgImporter(importerSpy.first().first().value<quint32>(), importerSpy.first().last().value<quint32>(), this);
134}

166void TestForeign::doExport()
167{
168 QSignalSpy serverSurfaceCreated(m_compositorInterface.data(), &KWaylandServer::CompositorInterface::surfaceCreated);
169
170 m_exportedSurface = m_compositor->createSurface();
171 QVERIFY(serverSurfaceCreated.wait());
172
173 m_exportedSurfaceInterface = serverSurfaceCreated.first().first().value<KWaylandServer::SurfaceInterface *>();
174
175 // Export a window
176 m_exported = m_exporter->exportTopLevel(m_exportedSurface);
177 QVERIFY(m_exported->handle().isEmpty());
178 QSignalSpy doneSpy(m_exported.data(), &KWayland::Client::XdgExported::done);
179 QVERIFY(doneSpy.wait());
180 QVERIFY(!m_exported->handle().isEmpty());
181
182 QSignalSpy transientSpy(m_foreignInterface, &KWaylandServer::XdgForeignV2Interface::transientChanged);
183
184 // Import the just exported window
185 m_imported = m_importer->importTopLevel(m_exported->handle());
186 QVERIFY(m_imported->isValid());
187
188 QSignalSpy childSurfaceInterfaceCreated(m_compositorInterface.data(), &KWaylandServer::CompositorInterface::surfaceCreated);
189 m_childSurface = m_compositor->createSurface();
190 QVERIFY(childSurfaceInterfaceCreated.wait());
191 m_childSurfaceInterface = childSurfaceInterfaceCreated.first().first().value<KWaylandServer::SurfaceInterface *>();
192 m_childSurface->commit(KWayland::Client::Surface::CommitFlag::None);
193
194 m_imported->setParentOf(m_childSurface);
195 QVERIFY(transientSpy.wait());
196
197 QCOMPARE(transientSpy.first().first().value<KWaylandServer::SurfaceInterface *>(), m_childSurfaceInterface.data());
198 QCOMPARE(transientSpy.first().at(1).value<KWaylandServer::SurfaceInterface *>(), m_exportedSurfaceInterface.data());
199
200 // transientFor api
201 QCOMPARE(m_foreignInterface->transientFor(m_childSurfaceInterface), m_exportedSurfaceInterface.data());
202}

272void TestForeign::testExportTwoTimes()
273{
274 doExport();
275
276 // Export second window
277 KWayland::Client::XdgExported *exported2 = m_exporter->exportTopLevel(m_exportedSurface);
278 QVERIFY(exported2->handle().isEmpty());
279 QSignalSpy doneSpy(exported2, &KWayland::Client::XdgExported::done);
280 QVERIFY(doneSpy.wait());
281 QVERIFY(!exported2->handle().isEmpty());
282
283 QSignalSpy transientSpy(m_foreignInterface, &KWaylandServer::XdgForeignV2Interface::transientChanged);
284
285 // Import the just exported window
286 KWayland::Client::XdgImported *imported2 = m_importer->importTopLevel(exported2->handle());
287 QVERIFY(imported2->isValid());
288
289 // create a second child surface
290 QSignalSpy serverSurfaceCreated(m_compositorInterface.data(), &KWaylandServer::CompositorInterface::surfaceCreated);
291
292 KWayland::Client::Surface *childSurface2 = m_compositor->createSurface();
293 QVERIFY(serverSurfaceCreated.wait());
294
295 KWaylandServer::SurfaceInterface *childSurface2Interface = serverSurfaceCreated.first().first().value<KWaylandServer::SurfaceInterface *>();
296
297 imported2->setParentOf(childSurface2);
298 QVERIFY(transientSpy.wait());
299
300 QCOMPARE(transientSpy.first().first().value<KWaylandServer::SurfaceInterface *>(), childSurface2Interface);
301 QCOMPARE(transientSpy.first().at(1).value<KWaylandServer::SurfaceInterface *>(), m_exportedSurfaceInterface.data());
302
303 // transientFor api
304 // check the old relationship is still here
305 QCOMPARE(m_foreignInterface->transientFor(m_childSurfaceInterface), m_exportedSurfaceInterface.data());
306 // check the new relationship
307 QCOMPARE(m_foreignInterface->transientFor(childSurface2Interface), m_exportedSurfaceInterface.data());
308}
