4158template<typename T>
4159void Motion<T>::calculate(const int msec)
4160{
4161 if (m_value == m_target && m_velocity == T()) { // At target and not moving
4162 return;
4163 }
4164
4165 // Poor man's time independent calculation
4166 int steps = std::max(1, msec / 5);
4167 for (int i = 0; i < steps; i++) {
4168 T diff = m_target - m_value;
4169 T strength = diff * m_strength;
4170 m_velocity = (m_smoothness * m_velocity + strength) / (m_smoothness + 1.0);
4171 m_value += m_velocity;
4172 }
4173}
