90 : m_edges(parent)

92 , m_action(ElectricActionNone)

94 , m_approaching(false)

96 , m_blocked(false)
97 , m_pushBackBlocked(false)
98 , m_client(nullptr)
99 , m_output(nullptr)
100 , m_gesture(std::make_unique<SwipeGesture>())
101{
102 m_gesture->setMinimumFingerCount(1);
103 m_gesture->setMaximumFingerCount(1);
104 connect(
105 m_gesture.get(), &Gesture::triggered, this, [this]() {
106 stopApproaching();
107 if (m_client) {
108 m_client->showOnScreenEdge();
109 unreserve();
110 return;
111 }
112 handleTouchAction();
113 handleTouchCallback();
114 },
115 Qt::QueuedConnection);
116 connect(m_gesture.get(), &SwipeGesture::started, this, &Edge::startApproaching);
117 connect(m_gesture.get(), &SwipeGesture::cancelled, this, &Edge::stopApproaching);
118 connect(m_gesture.get(), &SwipeGesture::cancelled, this, [this]() {
119 if (!m_touchCallbacks.isEmpty() && m_touchCallbacks.constFirst().hasProgressCallback()) {
120 handleTouchCallback();
121 }
122 });
123 connect(m_gesture.get(), &SwipeGesture::progress, this, [this](qreal progress) {
124 int factor = progress * 256.0f;
125 if (m_lastApproachingFactor != factor) {
126 m_lastApproachingFactor = factor;
127 Q_EMIT approaching(border(), m_lastApproachingFactor / 256.0f, m_approachGeometry);
128 }
129 });
130 connect(m_gesture.get(), &SwipeGesture::deltaProgress, this, [this](const QPointF &progressDelta) {
131 if (!m_touchCallbacks.isEmpty()) {
132 m_touchCallbacks.constFirst().progressCallback(border(), progressDelta, m_output);
133 }
134 });
135 connect(this, &Edge::activatesForTouchGestureChanged, this, [this]() {
136 if (isReserved()) {
137 if (activatesForTouchGesture()) {
138 m_edges->gestureRecognizer()->registerSwipeGesture(m_gesture.get());
139 } else {
140 m_edges->gestureRecognizer()->unregisterSwipeGesture(m_gesture.get());
141 }
142 }
143 });

261bool Edge::triggersFor(const QPoint &cursorPos) const
262{
263 if (isBlocked()) {
264 return false;
265 }
266 if (!activatesForPointer()) {
267 return false;
268 }
269 if (!m_geometry.contains(cursorPos)) {
270 return false;
271 }
272 if (isLeft() && cursorPos.x() != m_geometry.x()) {
273 return false;
274 }
275 if (isRight() && cursorPos.x() != (m_geometry.x() + m_geometry.width() - 1)) {

282 return false;
283 }

287bool Edge::check(const QPoint &cursorPos, const QDateTime &triggerTime, bool forceNoPushBack)
288{
289 if (!triggersFor(cursorPos)) {
290 return false;
291 }
292 if (m_lastTrigger.isValid() && // still in cooldown
293 m_lastTrigger.msecsTo(triggerTime) < edges()->reActivationThreshold() - edges()->timeThreshold()) {
294 // Reset the time, so the user has to actually keep the mouse still for this long to retrigger
295 m_lastTrigger = triggerTime;
296 return false;
297 }
298 // no pushback so we have to activate at once
299 bool directActivate = forceNoPushBack || edges()->cursorPushBackDistance().isNull() || m_client;
300 if (directActivate || canActivate(cursorPos, triggerTime)) {
301 markAsTriggered(cursorPos, triggerTime);
302 handle(cursorPos);
303 return true;
304 } else {
305 pushCursorBack(cursorPos);
306 m_triggeredPoint = cursorPos;
307 }
308 return false;
309}

318bool Edge::canActivate(const QPoint &cursorPos, const QDateTime &triggerTime)
319{
320 // we check whether either the timer has explicitly been invalidated (successful trigger) or is
321 // bigger than the reactivation threshold (activation "aborted", usually due to moving away the cursor
322 // from the corner after successful activation)
323 // either condition means that "this is the first event in a new attempt"
324 if (!m_lastReset.isValid() || m_lastReset.msecsTo(triggerTime) > edges()->reActivationThreshold()) {
325 m_lastReset = triggerTime;
326 return false;
327 }
328 if (m_lastTrigger.isValid() && m_lastTrigger.msecsTo(triggerTime) < edges()->reActivationThreshold() - edges()->timeThreshold()) {
329 return false;
330 }
331 if (m_lastReset.msecsTo(triggerTime) < edges()->timeThreshold()) {
332 return false;
333 }
334 // does the check on position make any sense at all?
335 if ((cursorPos - m_triggeredPoint).manhattanLength() > DISTANCE_RESET) {
336 return false;
337 }
338 return true;
339}

341void Edge::handle(const QPoint &cursorPos)
342{
343 Window *movingClient = Workspace::self()->moveResizeWindow();
344 if ((edges()->isDesktopSwitchingMovingClients() && movingClient && !movingClient->isInteractiveResize()) || (edges()->isDesktopSwitching() && isScreenEdge())) {
345 // always switch desktops in case:
346 // moving a Client and option for switch on client move is enabled
347 // or switch on screen edge is enabled
348 switchDesktop(cursorPos);
349 return;
350 }
351 if (movingClient) {
352 // if we are moving a window we don't want to trigger the actions. This just results in
353 // problems, e.g. Desktop Grid activated or screen locker activated which just cannot
354 // work as we hold a grab.
355 return;
356 }
357
358 if (m_client) {
359 pushCursorBack(cursorPos);
360 m_client->showOnScreenEdge();
361 unreserve();
362 return;
363 }
364
365 if (handlePointerAction() || handleByCallback()) {
366 pushCursorBack(cursorPos);
367 return;
368 }
369 if (edges()->isDesktopSwitching() && isCorner()) {
370 // try again desktop switching for the corner
371 switchDesktop(cursorPos);
372 }
373}

375bool Edge::handleAction(ElectricBorderAction action)
376{
377 switch (action) {
378 case ElectricActionShowDesktop: {
379 Workspace::self()->setShowingDesktop(!Workspace::self()->showingDesktop());
380 return true;
381 }
382 case ElectricActionLockScreen: { // Lock the screen
383#if KWIN_BUILD_SCREENLOCKER
384 OrgFreedesktopScreenSaverInterface interface(QStringLiteral("org.freedesktop.ScreenSaver"),
385 QStringLiteral("/ScreenSaver"),
386 QDBusConnection::sessionBus());
387 if (interface.isValid()) {
388 interface.Lock();
389 }
390 return true;
391#else
392 return false;
393#endif
394 }
395 case ElectricActionKRunner: { // open krunner
396 QDBusConnection::sessionBus().asyncCall(
397 QDBusMessage::createMethodCall(QStringLiteral("org.kde.krunner"),
398 QStringLiteral("/App"),
399 QStringLiteral("org.kde.krunner.App"),
400 QStringLiteral("display")));
401 return true;
402 }
403 case ElectricActionActivityManager: { // open activity manager
404 QDBusConnection::sessionBus().asyncCall(
405 QDBusMessage::createMethodCall(QStringLiteral("org.kde.plasmashell"),
406 QStringLiteral("/PlasmaShell"),
407 QStringLiteral("org.kde.PlasmaShell"),
408 QStringLiteral("toggleActivityManager")));
409 return true;
410 }
411 case ElectricActionApplicationLauncher: {
412 QDBusConnection::sessionBus().asyncCall(
413 QDBusMessage::createMethodCall(QStringLiteral("org.kde.plasmashell"),
414 QStringLiteral("/PlasmaShell"),
415 QStringLiteral("org.kde.PlasmaShell"),
416 QStringLiteral("activateLauncherMenu")));
417 return true;
418 }
419 default:
420 return false;
421 }
422}

446void Edge::switchDesktop(const QPoint &cursorPos)
447{
448 QPoint pos(cursorPos);
449 VirtualDesktopManager *vds = VirtualDesktopManager::self();
450 VirtualDesktop *oldDesktop = vds->currentDesktop();
451 VirtualDesktop *desktop = oldDesktop;
452 const int OFFSET = 2;
453 if (isLeft()) {
454 const VirtualDesktop *interimDesktop = desktop;
455 desktop = vds->toLeft(desktop, vds->isNavigationWrappingAround());
456 if (desktop != interimDesktop) {
457 pos.setX(workspace()->geometry().width() - 1 - OFFSET);
458 }
459 } else if (isRight()) {
460 const VirtualDesktop *interimDesktop = desktop;
461 desktop = vds->toRight(desktop, vds->isNavigationWrappingAround());
462 if (desktop != interimDesktop) {
463 pos.setX(OFFSET);
464 }
465 }
466 if (isTop()) {
467 const VirtualDesktop *interimDesktop = desktop;
468 desktop = vds->above(desktop, vds->isNavigationWrappingAround());
469 if (desktop != interimDesktop) {
470 pos.setY(workspace()->geometry().height() - 1 - OFFSET);
471 }
472 } else if (isBottom()) {
473 const VirtualDesktop *interimDesktop = desktop;
474 desktop = vds->below(desktop, vds->isNavigationWrappingAround());
475 if (desktop != interimDesktop) {
476 pos.setY(OFFSET);
477 }
478 }
479 if (Window *c = Workspace::self()->moveResizeWindow()) {
480 const QVector<VirtualDesktop *> desktops{desktop};
481 if (c->rules()->checkDesktops(desktops) != desktops) {
482 // user attempts to move a client to another desktop where it is ruleforced to not be
483 return;
484 }
485 }
486 vds->setCurrent(desktop);
487 if (vds->currentDesktop() != oldDesktop) {
488 m_pushBackBlocked = true;
489 Cursors::self()->mouse()->setPos(pos);
490 QMetaObject::Connection *me = new QMetaObject::Connection();
491 *me = QObject::connect(QCoreApplication::eventDispatcher(), &QAbstractEventDispatcher::aboutToBlock, this, [this, me]() {
492 QObject::disconnect(*me);
493 delete me;
494 m_pushBackBlocked = false;
495 });
496 }
497}

499void Edge::pushCursorBack(const QPoint &cursorPos)
500{
501 if (m_pushBackBlocked) {
502 return;
503 }
504 int x = cursorPos.x();
505 int y = cursorPos.y();
506 const QSize &distance = edges()->cursorPushBackDistance();
507 if (isLeft()) {
508 x += distance.width();
509 }
510 if (isRight()) {
511 x -= distance.width();
512 }
513 if (isTop()) {
514 y += distance.height();
515 }
516 if (isBottom()) {
517 y -= distance.height();
518 }
519 Cursors::self()->mouse()->setPos(QPoint(x, y));
520}

522void Edge::setGeometry(const QRect &geometry)
523{
524 if (m_geometry == geometry) {
525 return;
526 }
527 m_geometry = geometry;
528 int x = m_geometry.x();
529 int y = m_geometry.y();
530 int width = m_geometry.width();
531 int height = m_geometry.height();
532 const int offset = m_edges->cornerOffset();
533 if (isCorner()) {
534 if (isRight()) {
535 x = x + width - offset;
536 }
537 if (isBottom()) {
538 y = y + height - offset;
539 }
540 width = offset;
541 height = offset;
542 } else {
543 if (isLeft()) {
544 y += offset;
545 width = offset;
546 height = height - offset * 2;
547 } else if (isRight()) {
548 x = x + width - offset;
549 y += offset;
550 width = offset;
551 height = height - offset * 2;
552 } else if (isTop()) {
553 x += offset;
554 width = width - offset * 2;
555 height = offset;
556 } else if (isBottom()) {
557 x += offset;
558 y = y + height - offset;
559 width = width - offset * 2;
560 height = offset;
561 }
562 }
563 m_approachGeometry = QRect(x, y, width, height);
564 doGeometryUpdate();
565
566 if (isScreenEdge()) {
567 const Output *output = workspace()->outputAt(m_geometry.center());
568 m_gesture->setStartGeometry(m_geometry);
569 m_gesture->setMinimumDelta(QPointF(MINIMUM_DELTA, MINIMUM_DELTA) / output->scale());
570 }
571}

651void Edge::updateApproaching(const QPointF &point)
652{
653 if (exclusiveContains(approachGeometry(), point)) {
654 int factor = 0;
655 const int edgeDistance = m_edges->cornerOffset();
656 auto cornerDistance = [=](const QPointF &corner) {
657 return std::max(std::abs(corner.x() - point.x()), std::abs(corner.y() - point.y()));
658 };
659 constexpr double factorScale = 256;
660 switch (border()) {
661 case ElectricTopLeft:
662 factor = (cornerDistance(approachGeometry().topLeft()) * factorScale) / edgeDistance;
663 break;
664 case ElectricTopRight:
665 factor = (cornerDistance(approachGeometry().topRight()) * factorScale) / edgeDistance;
666 break;
667 case ElectricBottomRight:
668 factor = (cornerDistance(approachGeometry().bottomRight()) * factorScale) / edgeDistance;
669 break;
670 case ElectricBottomLeft:
671 factor = (cornerDistance(approachGeometry().bottomLeft()) * factorScale) / edgeDistance;
672 break;
673 case ElectricTop:
674 factor = (std::abs(point.y() - approachGeometry().y()) * factorScale) / edgeDistance;
675 break;
676 case ElectricRight:
677 factor = (std::abs(point.x() - approachGeometry().right()) * factorScale) / edgeDistance;
678 break;
679 case ElectricBottom:
680 factor = (std::abs(point.y() - approachGeometry().bottom()) * factorScale) / edgeDistance;
681 break;
682 case ElectricLeft:
683 factor = (std::abs(point.x() - approachGeometry().x()) * factorScale) / edgeDistance;
684 break;
685 default:
686 break;
687 }
688 factor = factorScale - factor;
689 if (m_lastApproachingFactor != factor) {
690 m_lastApproachingFactor = factor;
691 Q_EMIT approaching(border(), m_lastApproachingFactor / factorScale, m_approachGeometry);
692 }
693 } else {
694 stopApproaching();
695 }
696}

789static ElectricBorderAction electricBorderAction(const QString &name)
790{
791 QString lowerName = name.toLower();
792 if (lowerName == QStringLiteral("showdesktop")) {
793 return ElectricActionShowDesktop;
794 } else if (lowerName == QStringLiteral("lockscreen")) {
795 return ElectricActionLockScreen;
796 } else if (lowerName == QLatin1String("krunner")) {
797 return ElectricActionKRunner;
798 } else if (lowerName == QLatin1String("activitymanager")) {

951 return true;
952 }
953 if (screen.x() == fullArea.x()) {

955 }

957 for (const Output *output : outputs) {
958 const QRect otherGeo = output->geometry();
959 if (otherGeo == screen) {
960 // that's our screen to test
961 continue;
962 }

964 && screen.y() < otherGeo.y() + otherGeo.height()
965 && screen.y() + screen.height() > otherGeo.y()) {
966 // There is a screen to the left

968 }
969 }
970 // No screen exists to the left, so this is a left screen
971 return true;

1060void ScreenEdges::recreateEdges()
1061{
1062 std::vector<std::unique_ptr<Edge>> oldEdges = std::move(m_edges);
1063 m_edges.clear();
1064 const QRect fullArea = workspace()->geometry();
1065 QRegion processedRegion;
1066
1067 const auto outputs = workspace()->outputs();
1068 for (Output *output : outputs) {
1069 const QRegion screen = QRegion(output->geometry()).subtracted(processedRegion);
1070 processedRegion += screen;
1071 for (const QRect &screenPart : screen) {
1072 if (isLeftScreen(screenPart, fullArea)) {
1073 // left most screen
1074 createVerticalEdge(ElectricLeft, screenPart, fullArea, output);
1075 }
1076 if (isRightScreen(screenPart, fullArea)) {
1077 // right most screen
1078 createVerticalEdge(ElectricRight, screenPart, fullArea, output);
1079 }
1080 if (isTopScreen(screenPart, fullArea)) {
1081 // top most screen
1082 createHorizontalEdge(ElectricTop, screenPart, fullArea, output);
1083 }
1084 if (isBottomScreen(screenPart, fullArea)) {
1085 // bottom most screen
1086 createHorizontalEdge(ElectricBottom, screenPart, fullArea, output);
1087 }
1088 }
1089 }
1090 // copy over the effect/script reservations from the old edges
1091 for (const auto &edge : m_edges) {
1092 for (const auto &oldEdge : oldEdges) {
1093 if (oldEdge->client()) {
1094 // show the client again and don't recreate the edge
1095 oldEdge->client()->showOnScreenEdge();
1096 continue;
1097 }
1098 if (oldEdge->border() != edge->border()) {
1099 continue;
1100 }
1101 const QHash<QObject *, QByteArray> &callbacks = oldEdge->callBacks();
1102 for (auto callback = callbacks.begin(); callback != callbacks.end(); callback++) {
1103 edge->reserve(callback.key(), callback.value().constData());
1104 }
1105 const auto touchCallBacks = oldEdge->touchCallBacks();
1106 for (auto c : touchCallBacks) {
1107 edge->reserveTouchCallBack(c);
1108 }
1109 }
1110 }
1111}

1113void ScreenEdges::createVerticalEdge(ElectricBorder border, const QRect &screen, const QRect &fullArea, Output *output)
1114{
1115 if (border != ElectricRight && border != KWin::ElectricLeft) {
1116 return;
1117 }
1118 int y = screen.y();
1119 int height = screen.height();
1120 const int x = (border == ElectricLeft) ? screen.x() : screen.x() + screen.width() - TOUCH_TARGET;
1121 if (isTopScreen(screen, fullArea)) {
1122 // also top most screen
1123 height -= m_cornerOffset;
1124 y += m_cornerOffset;
1125 // create top left/right edge
1126 const ElectricBorder edge = (border == ElectricLeft) ? ElectricTopLeft : ElectricTopRight;
1127 m_edges.push_back(createEdge(edge, x, screen.y(), TOUCH_TARGET, TOUCH_TARGET, output));
1128 }
1129 if (isBottomScreen(screen, fullArea)) {
1130 // also bottom most screen
1131 height -= m_cornerOffset;
1132 // create bottom left/right edge
1133 const ElectricBorder edge = (border == ElectricLeft) ? ElectricBottomLeft : ElectricBottomRight;
1134 m_edges.push_back(createEdge(edge, x, screen.y() + screen.height() - TOUCH_TARGET, TOUCH_TARGET, TOUCH_TARGET, output));
1135 }
1136 if (height <= m_cornerOffset) {
1137 // An overlap with another output is near complete. We ignore this border.
1138 return;
1139 }
1140 m_edges.push_back(createEdge(border, x, y, TOUCH_TARGET, height, output));
1141}
1142
1143void ScreenEdges::createHorizontalEdge(ElectricBorder border, const QRect &screen, const QRect &fullArea, Output *output)
1144{
1145 if (border != ElectricTop && border != ElectricBottom) {
1146 return;
1147 }
1148 int x = screen.x();
1149 int width = screen.width();
1150 if (isLeftScreen(screen, fullArea)) {
1151 // also left most - adjust only x and width
1152 x += m_cornerOffset;
1153 width -= m_cornerOffset;
1154 }
1155 if (isRightScreen(screen, fullArea)) {
1156 // also right most edge
1157 width -= m_cornerOffset;
1158 }
1159 if (width <= m_cornerOffset) {
1160 // An overlap with another output is near complete. We ignore this border.
1161 return;
1162 }
1163 const int y = (border == ElectricTop) ? screen.y() : screen.y() + screen.height() - TOUCH_TARGET;
1164 m_edges.push_back(createEdge(border, x, y, width, TOUCH_TARGET, output));
1165}

1167std::unique_ptr<Edge> ScreenEdges::createEdge(ElectricBorder border, int x, int y, int width, int height, Output *output, bool createAction)
1168{
1169 std::unique_ptr<Edge> edge = kwinApp()->createScreenEdge(this);
1170 // Edges can not have negative size.
1171 Q_ASSERT(width >= 0);
1172 Q_ASSERT(height >= 0);
1173
1174 edge->setBorder(border);
1175 edge->setGeometry(QRect(x, y, width, height));
1176 edge->setOutput(output);
1177 if (createAction) {
1178 const ElectricBorderAction action = actionForEdge(edge.get());
1179 if (action != KWin::ElectricActionNone) {
1180 edge->reserve();
1181 edge->setAction(action);
1182 }
1183 const ElectricBorderAction touchAction = actionForTouchEdge(edge.get());
1184 if (touchAction != KWin::ElectricActionNone) {
1185 edge->reserve();
1186 edge->setTouchAction(touchAction);
1187 }
1188 }
1189 if (isDesktopSwitching()) {
1190 if (edge->isCorner()) {
1191 edge->reserve();
1192 } else {
1193 if ((m_virtualDesktopLayout & Qt::Horizontal) && (edge->isLeft() || edge->isRight())) {
1194 edge->reserve();
1195 }
1196 if ((m_virtualDesktopLayout & Qt::Vertical) && (edge->isTop() || edge->isBottom())) {
1197 edge->reserve();
1198 }
1199 }
1200 }
1201 connect(edge.get(), &Edge::approaching, this, &ScreenEdges::approaching);
1202 connect(this, &ScreenEdges::checkBlocking, edge.get(), &Edge::checkBlocking);
1203 return edge;
1204}

1318void ScreenEdges::createEdgeForClient(Window *client, ElectricBorder border)
1319{
1320 int y = 0;
1321 int x = 0;
1322 int width = 0;
1323 int height = 0;
1324 const QRect geo = client->frameGeometry().toRect();
1325 const QRect fullArea = workspace()->geometry();
1326
1327 const auto outputs = workspace()->outputs();
1328 Output *foundOutput = nullptr;
1329 for (Output *output : outputs) {
1330 foundOutput = output;
1331 const QRect screen = output->geometry();
1332 if (!screen.contains(geo)) {
1333 // ignoring Clients having a geometry overlapping with multiple screens
1334 // this would make the code more complex. If it's needed in future it can be added
1335 continue;
1336 }
1337 const bool bordersTop = (screen.y() == geo.y());
1338 const bool bordersLeft = (screen.x() == geo.x());
1339 const bool bordersBottom = (screen.y() + screen.height() == geo.y() + geo.height());
1340 const bool bordersRight = (screen.x() + screen.width() == geo.x() + geo.width());
1341
1342 if (bordersTop && border == ElectricTop) {
1343 if (!isTopScreen(screen, fullArea)) {
1344 continue;
1345 }
1346 y = geo.y();
1347 x = geo.x();
1348 height = 1;
1349 width = geo.width();
1350 break;
1351 }
1352 if (bordersBottom && border == ElectricBottom) {
1353 if (!isBottomScreen(screen, fullArea)) {
1354 continue;
1355 }
1356 y = geo.y() + geo.height() - 1;
1357 x = geo.x();
1358 height = 1;
1359 width = geo.width();
1360 break;
1361 }
1362 if (bordersLeft && border == ElectricLeft) {
1363 if (!isLeftScreen(screen, fullArea)) {
1364 continue;
1365 }
1366 x = geo.x();
1367 y = geo.y();
1368 width = 1;
1369 height = geo.height();
1370 break;
1371 }
1372 if (bordersRight && border == ElectricRight) {
1373 if (!isRightScreen(screen, fullArea)) {
1374 continue;
1375 }
1376 x = geo.x() + geo.width() - 1;
1377 y = geo.y();
1378 width = 1;
1379 height = geo.height();
1380 break;
1381 }
1382 }
1383
1384 if (width > 0 && height > 0) {
1385 m_edges.push_back(createEdge(border, x, y, width, height, foundOutput, false));
1386 Edge *edge = m_edges.back().get();
1387 edge->setClient(client);
1388 edge->reserve();
1389 } else {
1390 // we could not create an edge window, so don't allow the window to hide
1391 client->showOnScreenEdge();
1392 }
1393}

1403void ScreenEdges::check(const QPoint &pos, const QDateTime &now, bool forceNoPushBack)
1404{
1405 bool activatedForClient = false;
1406 for (const auto &edge : m_edges) {
1407 if (!edge->isReserved() || edge->isBlocked()) {
1408 continue;
1409 }
1410 if (!edge->activatesForPointer()) {
1411 continue;
1412 }
1413 if (edge->approachGeometry().contains(pos)) {
1414 edge->startApproaching();
1415 }
1416 if (edge->client() != nullptr && activatedForClient) {
1417 edge->markAsTriggered(pos, now);
1418 continue;
1419 }
1420 if (edge->check(pos, now, forceNoPushBack)) {
1421 if (edge->client()) {
1422 activatedForClient = true;
1423 }
1424 }
1425 }

1428bool ScreenEdges::isEntered(QMouseEvent *event)
1429{
1430 if (event->type() != QEvent::MouseMove) {
1431 return false;
1432 }
1433 bool activated = false;
1434 bool activatedForClient = false;
1435 for (const auto &edge : m_edges) {
1436 if (!edge->isReserved() || edge->isBlocked()) {
1437 continue;
1438 }
1439 if (!edge->activatesForPointer()) {
1440 continue;
1441 }
1442 if (edge->approachGeometry().contains(event->globalPos())) {
1443 if (!edge->isApproaching()) {
1444 edge->startApproaching();
1445 } else {
1446 edge->updateApproaching(event->globalPos());
1447 }
1448 } else {
1449 if (edge->isApproaching()) {
1450 edge->stopApproaching();
1451 }
1452 }
1453 if (edge->geometry().contains(event->globalPos())) {
1454 if (edge->check(event->globalPos(), QDateTime::fromMSecsSinceEpoch(event->timestamp(), Qt::UTC))) {
1455 if (edge->client()) {
1456 activatedForClient = true;
1457 }
1458 }
1459 }
1460 }
1461 if (activatedForClient) {
1462 for (const auto &edge : m_edges) {
1463 if (edge) {
1464 edge->markAsTriggered(event->globalPos(), QDateTime::fromMSecsSinceEpoch(event->timestamp(), Qt::UTC));
1465 }
1466 }
1467 }
1468 return activated;
1469}

1471bool ScreenEdges::handleEnterNotifiy(xcb_window_t window, const QPoint &point, const QDateTime &timestamp)
1472{
1473 bool activated = false;
1474 bool activatedForClient = false;
1475 for (const auto &edge : m_edges) {
1476 if (!edge || edge->window() == XCB_WINDOW_NONE) {
1477 continue;
1478 }
1479 if (!edge->isReserved() || edge->isBlocked()) {
1480 continue;
1481 }
1482 if (!edge->activatesForPointer()) {
1483 continue;
1484 }
1485 if (edge->window() == window) {
1486 if (edge->check(point, timestamp)) {
1487 if (edge->client()) {
1488 activatedForClient = true;
1489 }
1490 }
1491 activated = true;
1492 break;
1493 }
1494 if (edge->approachWindow() == window) {
1495 edge->startApproaching();
1496 // TODO: if it's a corner, it should also trigger for other windows
1497 return true;
1498 }
