40std::optional<OutputLayerBeginFrameInfo> VirtualEglLayer::beginFrame()
41{
42 m_backend->makeCurrent();
43
44 const QSize nativeSize = m_output->geometry().size() * m_output->scale();
45 if (!m_texture || m_texture->size() != nativeSize) {
46 m_fbo.reset();
47 m_texture = std::make_unique<GLTexture>(GL_RGB8, nativeSize);
48 m_fbo = std::make_unique<GLFramebuffer>(m_texture.get());
49 }
50
51 return OutputLayerBeginFrameInfo{
52 .renderTarget = RenderTarget(m_fbo.get()),
53 .repaint = infiniteRegion(),
54 };
55}

78bool VirtualEglBackend::initializeEgl()
79{
80 initClientExtensions();
81 EGLDisplay display = m_backend->sceneEglDisplay();
82
83 // Use eglGetPlatformDisplayEXT() to get the display pointer
84 // if the implementation supports it.
85 if (display == EGL_NO_DISPLAY) {
86 // first try surfaceless
87 if (hasClientExtension(QByteArrayLiteral("EGL_MESA_platform_surfaceless"))) {
88 display = eglGetPlatformDisplayEXT(EGL_PLATFORM_SURFACELESS_MESA, EGL_DEFAULT_DISPLAY, nullptr);
89 } else {
90 qCWarning(KWIN_VIRTUAL) << "Extension EGL_MESA_platform_surfaceless not available";
91 }
92 }
93
94 if (display == EGL_NO_DISPLAY) {
95 return false;
96 }
97 setEglDisplay(display);
98 return initEglAPI();
99}

101void VirtualEglBackend::init()
102{
103 if (!initializeEgl()) {
104 setFailed("Could not initialize egl");
105 return;
106 }
107 if (!initRenderingContext()) {
108 setFailed("Could not initialize rendering context");
109 return;
110 }
111
112 initKWinGL();
113 if (checkGLError("Init")) {
114 setFailed("Error during init of EglGbmBackend");
115 return;
116 }
117
118 setSupportsBufferAge(false);
119 initWayland();
120
121 const auto outputs = m_backend->outputs();
122 for (Output *output : outputs) {
123 addOutput(output);
124 }
125
126 connect(m_backend, &VirtualBackend::outputAdded, this, &VirtualEglBackend::addOutput);
127 connect(m_backend, &VirtualBackend::outputRemoved, this, &VirtualEglBackend::removeOutput);
128}
