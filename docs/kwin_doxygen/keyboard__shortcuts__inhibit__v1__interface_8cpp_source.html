137void KeyboardShortcutsInhibitManagerV1InterfacePrivate::zwp_keyboard_shortcuts_inhibit_manager_v1_inhibit_shortcuts(Resource *resource,
138 uint32_t id,
139 wl_resource *surface_resource,
140 wl_resource *seat_resource)
141{
142 SeatInterface *seat = SeatInterface::get(seat_resource);
143 SurfaceInterface *surface = SurfaceInterface::get(surface_resource);
144 if (m_inhibitors.contains({surface, seat})) {
145 wl_resource_post_error(resource->handle, error::error_already_inhibited, "the shortcuts are already inhibited for this surface and seat");
146 return;
147 }
148
149 wl_resource *inhibitorResource = wl_resource_create(resource->client(), &zwp_keyboard_shortcuts_inhibitor_v1_interface, resource->version(), id);
150 auto inhibitor = new KeyboardShortcutsInhibitorV1Interface(surface, seat, q, inhibitorResource);
151 m_inhibitors[{surface, seat}] = inhibitor;
152 Q_EMIT q->inhibitorCreated(inhibitor);
153 inhibitor->setActive(true);
154}
