57void ExpoArea::update()
58{
59 if (!m_screen) {
60 return;
61 }
62 const QRectF oldRect = m_rect;
63
64 m_rect = effects->clientArea(MaximizeArea, m_screen, effects->currentDesktop());
65
66 // Map the area to the output local coordinates.
67 m_rect.translate(-m_screen->geometry().topLeft());
68
69 if (oldRect.x() != m_rect.x()) {
70 Q_EMIT xChanged();
71 }
72 if (oldRect.y() != m_rect.y()) {
73 Q_EMIT yChanged();
74 }
75 if (oldRect.width() != m_rect.width()) {
76 Q_EMIT widthChanged();
77 }
78 if (oldRect.height() != m_rect.height()) {
79 Q_EMIT heightChanged();
