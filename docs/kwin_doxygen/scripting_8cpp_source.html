159void KWin::Script::run()
160{
161 if (running() || m_starting) {
162 return;
163 }
164
165 if (calledFromDBus()) {
166 m_invocationContext = message();
167 setDelayedReply(true);
168 }
169
170 m_starting = true;
171 QFutureWatcher<QByteArray> *watcher = new QFutureWatcher<QByteArray>(this);
172 connect(watcher, &QFutureWatcherBase::finished, this, &Script::slotScriptLoadedFromFile);
173#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
174 watcher->setFuture(QtConcurrent::run(this, &KWin::Script::loadScriptFromFile, fileName()));
175#else
176 watcher->setFuture(QtConcurrent::run(&KWin::Script::loadScriptFromFile, this, fileName()));
177#endif
178}

190void KWin::Script::slotScriptLoadedFromFile()
191{
192 QFutureWatcher<QByteArray> *watcher = dynamic_cast<QFutureWatcher<QByteArray> *>(sender());
193 if (!watcher) {
194 // not invoked from a QFutureWatcher
195 return;
196 }
197 if (watcher->result().isNull()) {
198 // do not load empty script
199 deleteLater();
200 watcher->deleteLater();
201
202 if (m_invocationContext.type() == QDBusMessage::MethodCallMessage) {
203 auto reply = m_invocationContext.createErrorReply("org.kde.kwin.Scripting.FileError", QString("Could not open %1").arg(fileName()));
204 QDBusConnection::sessionBus().send(reply);
205 m_invocationContext = QDBusMessage();
206 }
207
208 return;
209 }
210
211 // Install console functions (e.g. console.assert(), console.log(), etc).
212 m_engine->installExtensions(QJSEngine::ConsoleExtension);
213
214 // Make the timer visible to QJSEngine.
215 QJSValue timerMetaObject = m_engine->newQMetaObject(&ScriptTimer::staticMetaObject);
216 m_engine->globalObject().setProperty("QTimer", timerMetaObject);
217
218 // Expose enums.
219 m_engine->globalObject().setProperty(QStringLiteral("KWin"), m_engine->newQMetaObject(&QtScriptWorkspaceWrapper::staticMetaObject));
220
221 // Make the options object visible to QJSEngine.
222 QJSValue optionsObject = m_engine->newQObject(options);
223 QQmlEngine::setObjectOwnership(options, QQmlEngine::CppOwnership);
224 m_engine->globalObject().setProperty(QStringLiteral("options"), optionsObject);
225
226 // Make the workspace visible to QJSEngine.
227 QJSValue workspaceObject = m_engine->newQObject(Scripting::self()->workspaceWrapper());
228 QQmlEngine::setObjectOwnership(Scripting::self()->workspaceWrapper(), QQmlEngine::CppOwnership);
229 m_engine->globalObject().setProperty(QStringLiteral("workspace"), workspaceObject);
230
231 QJSValue self = m_engine->newQObject(this);
232 QQmlEngine::setObjectOwnership(this, QQmlEngine::CppOwnership);

234 static const QStringList globalProperties{
235 QStringLiteral("readConfig"),
236 QStringLiteral("callDBus"),
237
238 QStringLiteral("registerShortcut"),
239 QStringLiteral("registerScreenEdge"),
240 QStringLiteral("unregisterScreenEdge"),
241 QStringLiteral("registerTouchScreenEdge"),
242 QStringLiteral("unregisterTouchScreenEdge"),
243 QStringLiteral("registerUserActionsMenu"),
244 };
245
246 for (const QString &propertyName : globalProperties) {
247 m_engine->globalObject().setProperty(propertyName, self.property(propertyName));
248 }
249
250 // Inject assertion functions. It would be better to create a module with all
251 // this assert functions or just deprecate them in favor of console.assert().
252 QJSValue result = m_engine->evaluate(QStringLiteral(R"(
253 function assert(condition, message) {
254 console.assert(condition, message || 'Assertion failed');
255 }
256 function assertTrue(condition, message) {
257 console.assert(condition, message || 'Assertion failed');
258 }
259 function assertFalse(condition, message) {
260 console.assert(!condition, message || 'Assertion failed');
261 }
262 function assertNull(value, message) {
263 console.assert(value === null, message || 'Assertion failed');
264 }
265 function assertNotNull(value, message) {
266 console.assert(value !== null, message || 'Assertion failed');
267 }
268 function assertEquals(expected, actual, message) {
269 console.assert(expected === actual, message || 'Assertion failed');
270 }
271 )"));
272 Q_ASSERT(!result.isError());
273
274 result = m_engine->evaluate(QString::fromUtf8(watcher->result()), fileName());
275 if (result.isError()) {
276 qCWarning(KWIN_SCRIPTING, "%s:%d: error: %s", qPrintable(fileName()),
277 result.property(QStringLiteral("lineNumber")).toInt(),
278 qPrintable(result.property(QStringLiteral("message")).toString()));
279 deleteLater();
280 }
281
282 if (m_invocationContext.type() == QDBusMessage::MethodCallMessage) {
283 auto reply = m_invocationContext.createReply();
284 QDBusConnection::sessionBus().send(reply);
285 m_invocationContext = QDBusMessage();
286 }
287
288 watcher->deleteLater();
289 setRunning(true);
290 m_starting = false;
291}

298void KWin::Script::callDBus(const QString &service, const QString &path, const QString &interface,
299 const QString &method, const QJSValue &arg1, const QJSValue &arg2,
300 const QJSValue &arg3, const QJSValue &arg4, const QJSValue &arg5,
301 const QJSValue &arg6, const QJSValue &arg7, const QJSValue &arg8,
302 const QJSValue &arg9)
303{
304 QJSValueList jsArguments;
305 jsArguments.reserve(9);
306
307 if (!arg1.isUndefined()) {
308 jsArguments << arg1;
309 }
310 if (!arg2.isUndefined()) {
311 jsArguments << arg2;
312 }
313 if (!arg3.isUndefined()) {
314 jsArguments << arg3;
315 }
316 if (!arg4.isUndefined()) {
317 jsArguments << arg4;
318 }
319 if (!arg5.isUndefined()) {
320 jsArguments << arg5;
321 }
322 if (!arg6.isUndefined()) {
323 jsArguments << arg6;
324 }
325 if (!arg7.isUndefined()) {
326 jsArguments << arg7;
327 }
328 if (!arg8.isUndefined()) {
329 jsArguments << arg8;
330 }
331 if (!arg9.isUndefined()) {
332 jsArguments << arg9;
333 }
334
335 QJSValue callback;
336 if (!jsArguments.isEmpty() && jsArguments.last().isCallable()) {
337 callback = jsArguments.takeLast();
338 }
339
340 QVariantList dbusArguments;
341 dbusArguments.reserve(jsArguments.count());
342 for (const QJSValue &jsArgument : std::as_const(jsArguments)) {
343 dbusArguments << jsArgument.toVariant();
344 }
345
346 QDBusMessage message = QDBusMessage::createMethodCall(service, path, interface, method);
347 message.setArguments(dbusArguments);
348
349 const QDBusPendingCall call = QDBusConnection::sessionBus().asyncCall(message);
350 if (callback.isUndefined()) {
351 return;
352 }
353
354 QDBusPendingCallWatcher *watcher = new QDBusPendingCallWatcher(call, this);
355 connect(watcher, &QDBusPendingCallWatcher::finished, this, [this, callback](QDBusPendingCallWatcher *self) {
356 self->deleteLater();
357
358 if (self->isError()) {
359 qCWarning(KWIN_SCRIPTING) << "Received D-Bus message is error:" << self->error().message();
360 return;
361 }
362
363 QJSValueList arguments;
364 const QVariantList reply = self->reply().arguments();
365 for (const QVariant &variant : reply) {
366 arguments << m_engine->toScriptValue(dbusToVariant(variant));
367 }
368
369 QJSValue(callback).call(arguments);
370 });
371}

467QList<QAction *> KWin::Script::actionsForUserActionMenu(KWin::Window *client, QMenu *parent)
468{
469 QList<QAction *> actions;
470 actions.reserve(m_userActionsMenuCallbacks.count());
471
472 for (QJSValue callback : std::as_const(m_userActionsMenuCallbacks)) {
473 const QJSValue result = callback.call({m_engine->toScriptValue(client)});
474 if (result.isError()) {
475 continue;
476 }
477 if (!result.isObject()) {
478 continue;
479 }
480 if (QAction *action = scriptValueToAction(result, parent)) {
481 actions << action;
482 }
483 }
484
485 return actions;

500QAction *KWin::Script::scriptValueToAction(const QJSValue &value, QMenu *parent)
501{
502 const QString title = value.property(QStringLiteral("text")).toString();
503 if (title.isEmpty()) {
504 return nullptr;
505 }
506
507 // Either a menu or a menu item.
508 const QJSValue itemsValue = value.property(QStringLiteral("items"));
509 if (!itemsValue.isUndefined()) {
510 return createMenu(title, itemsValue, parent);
511 }
512
513 return createAction(title, value, parent);

516QAction *KWin::Script::createAction(const QString &title, const QJSValue &item, QMenu *parent)
517{
518 const QJSValue callback = item.property(QStringLiteral("triggered"));
519 if (!callback.isCallable()) {
520 return nullptr;
521 }
522
523 const bool checkable = item.property(QStringLiteral("checkable")).toBool();
524 const bool checked = item.property(QStringLiteral("checked")).toBool();
525
526 QAction *action = new QAction(title, parent);
527 action->setCheckable(checkable);
528 action->setChecked(checked);
529
530 connect(action, &QAction::triggered, this, [this, action, callback]() {
531 QJSValue(callback).call({m_engine->toScriptValue(action)});
532 });
533
534 return action;
535}

537QAction *KWin::Script::createMenu(const QString &title, const QJSValue &items, QMenu *parent)
538{
539 if (!items.isArray()) {
540 return nullptr;
541 }
542
543 const int length = items.property(QStringLiteral("length")).toInt();
544 if (!length) {
545 return nullptr;
546 }
547
548 QMenu *menu = new QMenu(title, parent);
549 for (int i = 0; i < length; ++i) {
550 const QJSValue value = items.property(QString::number(i));
551 if (!value.isObject()) {
552 continue;
553 }
554 if (QAction *action = scriptValueToAction(value, menu)) {
555 menu->addAction(action);
556 }
557 }
558
559 return menu->menuAction();
560}

727LoadScriptList KWin::Scripting::queryScriptsToLoad()
728{
729 KSharedConfig::Ptr _config = kwinApp()->config();
730 static bool s_started = false;
731 if (s_started) {
732 _config->reparseConfiguration();
733 } else {
734 s_started = true;
735 }
736 QMap<QString, QString> pluginStates = KConfigGroup(_config, "Plugins").entryMap();
737 const QString scriptFolder = QStringLiteral("kwin/scripts/");
738 const auto offers = KPackage::PackageLoader::self()->listPackages(QStringLiteral("KWin/Script"), scriptFolder);
739
740 LoadScriptList scriptsToLoad;
741
742 for (const KPluginMetaData &service : offers) {
743 const QString value = pluginStates.value(service.pluginId() + QLatin1String("Enabled"), QString());
744 const bool enabled = value.isNull() ? service.isEnabledByDefault() : QVariant(value).toBool();
745 const bool javaScript = service.value(QStringLiteral("X-Plasma-API")) == QLatin1String("javascript");
746 const bool declarativeScript = service.value(QStringLiteral("X-Plasma-API")) == QLatin1String("declarativescript");
747 if (!javaScript && !declarativeScript) {
748 continue;
749 }
750
751 if (!enabled) {
752 if (isScriptLoaded(service.pluginId())) {
753 // unload the script
754 unloadScript(service.pluginId());
755 }
756 continue;
757 }
758 const QString pluginName = service.pluginId();
759 const QString scriptName = service.value(QStringLiteral("X-Plasma-MainScript"));
760 const QString file = QStandardPaths::locate(QStandardPaths::GenericDataLocation, scriptFolder + pluginName + QLatin1String("/contents/") + scriptName);
761 if (file.isNull()) {
762 qCDebug(KWIN_SCRIPTING) << "Could not find script file for " << pluginName;
763 continue;
764 }
765 scriptsToLoad << qMakePair(javaScript, qMakePair(file, pluginName));
766 }
767 return scriptsToLoad;
768}

779 for (LoadScriptList::const_iterator it = scriptsToLoad.constBegin();
780 it != scriptsToLoad.constEnd();
781 ++it) {

784 } else {
785 loadDeclarativeScript(it->second.first, it->second.second);
786 }
787 }
788
789 runScripts();
790 watcher->deleteLater();
