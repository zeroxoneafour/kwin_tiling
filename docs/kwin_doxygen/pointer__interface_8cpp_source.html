61void PointerInterfacePrivate::pointer_set_cursor(Resource *resource, uint32_t serial, ::wl_resource *surface_resource, int32_t hotspot_x, int32_t hotspot_y)
62{
63 SurfaceInterface *surface = nullptr;
64
65 if (!focusedSurface) {
66 return;
67 }
68 if (focusedSurface->client()->client() != resource->client()) {
69 qCDebug(KWIN_CORE, "Denied set_cursor request from unfocused client");
70 return;
71 }
72
73 if (surface_resource) {
74 surface = SurfaceInterface::get(surface_resource);
75 if (!surface) {
76 wl_resource_post_error(resource->handle, WL_DISPLAY_ERROR_INVALID_OBJECT, "invalid surface");
77 return;
78 }
79
80 const SurfaceRole *surfaceRole = SurfaceRole::get(surface);
81 if (surfaceRole) {
82 wl_resource_post_error(resource->handle, error_role, "the wl_surface already has a role assigned %s", surfaceRole->name().constData());
83 return;
84 }
85 }
86
87 if (!cursor) { // TODO: Assign the cursor surface role.
88 cursor = new Cursor(q);
89 cursor->d->update(surface, serial, QPointF(hotspot_x, hotspot_y) / focusedSurface->client()->scaleOverride());
90 QObject::connect(cursor, &Cursor::changed, q, &PointerInterface::cursorChanged);
91 Q_EMIT q->cursorChanged();
92 } else {
93 cursor->d->update(surface, serial, QPointF(hotspot_x, hotspot_y) / focusedSurface->client()->scaleOverride());
94 }
95}

155void PointerInterface::sendEnter(SurfaceInterface *surface, const QPointF &position, quint32 serial)
156{
157 if (d->focusedSurface == surface) {
158 return;
159 }
160
161 if (d->focusedSurface) {
162 d->sendLeave(serial);
163 if (d->focusedSurface->client() != surface->client()) {
164 d->sendFrame();
165 }
166 disconnect(d->destroyConnection);
167 }
168
169 d->focusedSurface = surface;
170 d->destroyConnection = connect(d->focusedSurface, &SurfaceInterface::aboutToBeDestroyed, this, [this]() {
171 d->sendLeave(d->seat->display()->nextSerial());
172 d->sendFrame();
173 d->focusedSurface = nullptr;
174 Q_EMIT focusedSurfaceChanged();
175 });
176
177 d->sendEnter(d->focusedSurface->toSurfaceLocal(position), serial);
178 d->sendFrame();
179 d->lastPosition = position;
180
181 Q_EMIT focusedSurfaceChanged();
182}

211static bool shouldResetAccumulator(int accumulator, int delta)
212{
213 // Reset the accumulator if the delta has opposite sign.
214 return accumulator && (accumulator < 0 != delta < 0);
215}

217static void updateAccumulators(Qt::Orientation orientation, qreal delta, qint32 deltaV120, PointerInterfacePrivate *d, qint32 &valueAxisLowRes, qint32 &valueDiscrete)
218{
219 qint32 *accumulatorV120;
220 qreal *accumulatorAxis;
221
222 if (orientation == Qt::Horizontal) {
223 accumulatorV120 = &d->accumulatorV120.rx();
224 accumulatorAxis = &d->accumulatorAxis.rx();
225 } else {
226 accumulatorV120 = &d->accumulatorV120.ry();
227 accumulatorAxis = &d->accumulatorAxis.ry();
228 }
229
230 if (shouldResetAccumulator(*accumulatorV120, deltaV120)) {
231 *accumulatorV120 = 0;
232 *accumulatorAxis = 0;
233 }
234
235 *accumulatorAxis += delta;
236 *accumulatorV120 += deltaV120;
237
238 // ±120 is a "wheel click"
239 valueDiscrete = *accumulatorV120 / 120;
240 *accumulatorV120 -= valueDiscrete * 120;
241
242 if (valueDiscrete) {
243 // Accumulate the axis values to send to low-res clients
244 valueAxisLowRes = *accumulatorAxis;
245 *accumulatorAxis = 0;
246 }
247}

249void PointerInterface::sendAxis(Qt::Orientation orientation, qreal delta, qint32 deltaV120, PointerAxisSource source)
250{
251 if (!d->focusedSurface) {
252 return;
253 }
254
255 qint32 valueAxisLowRes = 0;
256 qint32 valueDiscrete = 0;
257
258 if (deltaV120) {
259 updateAccumulators(orientation, delta, deltaV120, d.get(),
260 valueAxisLowRes, valueDiscrete);
261 } else {
262 valueAxisLowRes = delta;
263 }
264
265 const auto pointerResources = d->pointersForClient(d->focusedSurface->client());
266 for (PointerInterfacePrivate::Resource *resource : pointerResources) {
267 const quint32 version = resource->version();
268
269 // Don't send anything if the client doesn't support high-res scrolling and
270 // we haven't accumulated a wheel click's worth of events.
271 if (version < WL_POINTER_AXIS_VALUE120_SINCE_VERSION && deltaV120 && !valueDiscrete) {
272 continue;
273 }
274
275 if (source != PointerAxisSource::Unknown && version >= WL_POINTER_AXIS_SOURCE_SINCE_VERSION) {
276 PointerInterfacePrivate::axis_source wlSource;
277 switch (source) {
278 case PointerAxisSource::Wheel:
279 wlSource = PointerInterfacePrivate::axis_source_wheel;
280 break;
281 case PointerAxisSource::Finger:
282 wlSource = PointerInterfacePrivate::axis_source_finger;
283 break;
284 case PointerAxisSource::Continuous:
285 wlSource = PointerInterfacePrivate::axis_source_continuous;
286 break;
287 case PointerAxisSource::WheelTilt:
288 wlSource = PointerInterfacePrivate::axis_source_wheel_tilt;
289 break;
290 default:
291 Q_UNREACHABLE();
292 break;
293 }
294 d->send_axis_source(resource->handle, wlSource);
295 }
296
297 const auto wlOrientation =
298 (orientation == Qt::Vertical) ? PointerInterfacePrivate::axis_vertical_scroll : PointerInterfacePrivate::axis_horizontal_scroll;
299
300 if (delta) {
301 if (deltaV120) {
302 if (version >= WL_POINTER_AXIS_VALUE120_SINCE_VERSION) {
303 // Send high resolution scroll events if client supports them
304 d->send_axis_value120(resource->handle, wlOrientation, deltaV120);
305 d->send_axis(resource->handle, d->seat->timestamp().count(), wlOrientation,
306 wl_fixed_from_double(delta));
307 } else {
308 if (version >= WL_POINTER_AXIS_DISCRETE_SINCE_VERSION && valueDiscrete) {
309 // Send discrete scroll events if client supports them.
310 d->send_axis_discrete(resource->handle, wlOrientation,
311 valueDiscrete);
312 }
313 // Send accumulated axis values
314 d->send_axis(resource->handle, d->seat->timestamp().count(), wlOrientation,
315 wl_fixed_from_double(valueAxisLowRes));
316 }
317 } else {
318 // Finger or continuous scroll
319 d->send_axis(resource->handle, d->seat->timestamp().count(), wlOrientation,
320 wl_fixed_from_double(delta));
321 }
322 } else if (version >= WL_POINTER_AXIS_STOP_SINCE_VERSION) {
323 d->send_axis_stop(resource->handle, d->seat->timestamp().count(), wlOrientation);
324 }
325 }

375void CursorPrivate::update(SurfaceInterface *s, quint32 serial, const QPointF &p)
376{
377 bool emitChanged = false;
378 if (enteredSerial != serial) {
379 enteredSerial = serial;
380 emitChanged = true;
381 Q_EMIT q->enteredSerialChanged();
382 }
383 if (hotspot != p) {
384 hotspot = p;
385 emitChanged = true;
386 Q_EMIT q->hotspotChanged();
387 }
388 if (surface != s) {
389 if (!surface.isNull()) {
390 QObject::disconnect(surface.data(), &SurfaceInterface::committed, q, &Cursor::changed);
391 }
392 surface = s;
393 if (!surface.isNull()) {
394 QObject::connect(surface.data(), &SurfaceInterface::committed, q, &Cursor::changed);
395 }
396 emitChanged = true;
397 Q_EMIT q->surfaceChanged();
398 }
399 if (emitChanged) {
400 Q_EMIT q->changed();
401 }
402}
