110 static QByteArray sha256(const QString &fileName)
111 {
112 QFile f(fileName);
113 if (f.open(QFile::ReadOnly)) {
114 QCryptographicHash hash(QCryptographicHash::Sha256);
115 if (hash.addData(&f)) {
116 return hash.result();
117 }
118 }
119 return QByteArray();
120 }
121
122 bool isTrustedOrigin(KWaylandServer::ClientConnection *client) const
123 {
124 const auto fullPathSha = sha256(client->executablePath());
125 const auto localSha = sha256(QLatin1String("/proc/") + QString::number(client->processId()) + QLatin1String("/exe"));
126 const bool trusted = !localSha.isEmpty() && fullPathSha == localSha;
127
128 if (!trusted) {
129 qCWarning(KWIN_CORE) << "Could not trust" << client->executablePath() << "sha" << localSha << fullPathSha;
130 }
131
132 return trusted;
133 }
134
135 QStringList fetchRequestedInterfaces(KWaylandServer::ClientConnection *client) const
136 {
137 return KWin::fetchRequestedInterfaces(client->executablePath());
138 }
139
140 const QSet<QByteArray> interfacesBlackList = {
141 QByteArrayLiteral("org_kde_kwin_remote_access_manager"),
142 QByteArrayLiteral("org_kde_plasma_window_management"),
143 QByteArrayLiteral("org_kde_kwin_fake_input"),
144 QByteArrayLiteral("org_kde_kwin_keystate"),
145 QByteArrayLiteral("zkde_screencast_unstable_v1"),
146 QByteArrayLiteral("org_kde_plasma_activation_feedback"),
147 QByteArrayLiteral("kde_lockscreen_overlay_v1"),
148 };
149
150 const QSet<QByteArray> inputmethodInterfaces = {"zwp_input_panel_v1", "zwp_input_method_v1"};
151 const QSet<QByteArray> xwaylandInterfaces = {
152 QByteArrayLiteral("zwp_xwayland_keyboard_grab_manager_v1"),
153 QByteArrayLiteral("xwayland_shell_v1"),
154 };
155
156 QSet<QString> m_reported;
157
158 bool allowInterface(KWaylandServer::ClientConnection *client, const QByteArray &interfaceName) override
159 {
160 if (client->processId() == getpid()) {
161 return true;
162 }
163
164 if (client != waylandServer()->inputMethodConnection() && inputmethodInterfaces.contains(interfaceName)) {
165 return false;
166 }
167
168 if (client != waylandServer()->xWaylandConnection() && xwaylandInterfaces.contains(interfaceName)) {
169 return false;
170 }
171
172 if (!interfacesBlackList.contains(interfaceName)) {
173 return true;
174 }
175
176 if (client->executablePath().isEmpty()) {
177 qCDebug(KWIN_CORE) << "Could not identify process with pid" << client->processId();
178 return false;
179 }
180
181 static bool permissionCheckDisabled = qEnvironmentVariableIntValue("KWIN_WAYLAND_NO_PERMISSION_CHECKS") == 1;
182 if (!permissionCheckDisabled) {
183 auto requestedInterfaces = client->property("requestedInterfaces");
184 if (requestedInterfaces.isNull()) {
185 requestedInterfaces = fetchRequestedInterfaces(client);
186 client->setProperty("requestedInterfaces", requestedInterfaces);
187 }
188 if (!requestedInterfaces.toStringList().contains(QString::fromUtf8(interfaceName))) {
189 if (KWIN_CORE().isDebugEnabled()) {
190 const QString id = client->executablePath() + QLatin1Char('|') + QString::fromUtf8(interfaceName);
191 if (!m_reported.contains(id)) {
192 m_reported.insert(id);
193 qCDebug(KWIN_CORE) << "Interface" << interfaceName << "not in X-KDE-Wayland-Interfaces of" << client->executablePath();
194 }
195 }
196 return false;
197 }
198 }
199
200 {
201 auto trustedOrigin = client->property("isPrivileged");
202 if (trustedOrigin.isNull()) {
203 trustedOrigin = isTrustedOrigin(client);
204 client->setProperty("isPrivileged", trustedOrigin);
205 }
206
207 if (!trustedOrigin.toBool()) {
208 return false;
209 }
210 }
211 qCDebug(KWIN_CORE) << "authorized" << client->executablePath() << interfaceName;
212 return true;
213 }

542void WaylandServer::initWorkspace()
543{
544 new KeyStateInterface(m_display, m_display);
545
546 VirtualDesktopManager::self()->setVirtualDesktopManagement(m_virtualDesktopManagement);
547
548 if (m_windowManagement) {
549 connect(workspace(), &Workspace::showingDesktopChanged, this, [this](bool set) {
550 using namespace KWaylandServer;
551 m_windowManagement->setShowingDesktopState(set ? PlasmaWindowManagementInterface::ShowingDesktopState::Enabled : PlasmaWindowManagementInterface::ShowingDesktopState::Disabled);
552 });
553
554 connect(workspace(), &Workspace::workspaceInitialized, this, [this] {
555 auto f = [this]() {
556 QVector<quint32> ids;
557 QVector<QString> uuids;
558 for (Window *toplevel : workspace()->stackingOrder()) {
559 if (toplevel->windowManagementInterface()) {
560 ids << toplevel->windowManagementInterface()->internalId();
561 uuids << toplevel->windowManagementInterface()->uuid();
562 }
563 }
564 m_windowManagement->setStackingOrder(ids);
565 m_windowManagement->setStackingOrderUuids(uuids);
566 };
567 f();
568 connect(workspace(), &Workspace::stackingOrderChanged, this, f);
569 });
570 }
571
572 const auto availableOutputs = kwinApp()->outputBackend()->outputs();
573 for (Output *output : availableOutputs) {
574 handleOutputAdded(output);
575 }
576 connect(kwinApp()->outputBackend(), &OutputBackend::outputAdded, this, &WaylandServer::handleOutputAdded);
577 connect(kwinApp()->outputBackend(), &OutputBackend::outputRemoved, this, &WaylandServer::handleOutputRemoved);
578
579 const auto outputs = workspace()->outputs();
580 for (Output *output : outputs) {
581 handleOutputEnabled(output);
582 }
583 connect(workspace(), &Workspace::outputAdded, this, &WaylandServer::handleOutputEnabled);
584 connect(workspace(), &Workspace::outputRemoved, this, &WaylandServer::handleOutputDisabled);
585
586 if (hasScreenLockerIntegration()) {
587 initScreenLocker();
588 }
589
590 if (auto backend = qobject_cast<DrmBackend *>(kwinApp()->outputBackend())) {
591 m_leaseManager = new KWaylandServer::DrmLeaseManagerV1(backend, m_display, m_display);
592 }
593
594 m_outputOrder = new KWaylandServer::OutputOrderV1Interface(m_display, m_display);
595 m_outputOrder->setOutputOrder(workspace()->outputOrder());
596 connect(workspace(), &Workspace::outputOrderChanged, m_outputOrder, [this]() {
597 m_outputOrder->setOutputOrder(workspace()->outputOrder());
598 });
599
600 Q_EMIT initialized();
601}

603void WaylandServer::initScreenLocker()
604{
605#if KWIN_BUILD_SCREENLOCKER
606 auto *screenLockerApp = ScreenLocker::KSldApp::self();
607
608 ScreenLocker::KSldApp::self()->setGreeterEnvironment(kwinApp()->processStartupEnvironment());
609
610 connect(ScreenLocker::KSldApp::self(), &ScreenLocker::KSldApp::aboutToLock, this, [this, screenLockerApp]() {
611 if (m_screenLockerClientConnection) {
612 // Already sent data to KScreenLocker.
613 return;
614 }
615 int clientFd = createScreenLockerConnection();
616 if (clientFd < 0) {
617 return;
618 }
619 ScreenLocker::KSldApp::self()->setWaylandFd(clientFd);
620
621 new LockScreenPresentationWatcher(this);
622
623 const QVector<SeatInterface *> seatIfaces = m_display->seats();
624 for (auto *seat : seatIfaces) {
625 connect(seat, &KWaylandServer::SeatInterface::timestampChanged,
626 screenLockerApp, &ScreenLocker::KSldApp::userActivity);
627 }
628 });
629
630 connect(ScreenLocker::KSldApp::self(), &ScreenLocker::KSldApp::unlocked, this, [this, screenLockerApp]() {
631 if (m_screenLockerClientConnection) {
632 m_screenLockerClientConnection->destroy();
633 delete m_screenLockerClientConnection;
634 m_screenLockerClientConnection = nullptr;
635 }
636
637 const QVector<SeatInterface *> seatIfaces = m_display->seats();
638 for (auto *seat : seatIfaces) {
639 disconnect(seat, &KWaylandServer::SeatInterface::timestampChanged,
640 screenLockerApp, &ScreenLocker::KSldApp::userActivity);
641 }
642 ScreenLocker::KSldApp::self()->setWaylandFd(-1);
643 });
644
645 connect(screenLockerApp, &ScreenLocker::KSldApp::lockStateChanged, this, &WaylandServer::lockStateChanged);
646
647 ScreenLocker::KSldApp::self()->initialize();

656{

663 ret.connection = m_display->createClient(sx[0]);
