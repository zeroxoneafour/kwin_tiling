62void SubSurfaceMonitor::registerSurface(SurfaceInterface *surface)
63{
64 connect(surface, &SurfaceInterface::childSubSurfaceAdded,
65 this, &SubSurfaceMonitor::subSurfaceAdded);
66 connect(surface, &SurfaceInterface::childSubSurfaceRemoved,
67 this, &SubSurfaceMonitor::subSurfaceRemoved);
68 connect(surface, &SurfaceInterface::childSubSurfaceAdded,
69 this, &SubSurfaceMonitor::registerSubSurface);
70 connect(surface, &SurfaceInterface::childSubSurfaceRemoved,
71 this, &SubSurfaceMonitor::unregisterSubSurface);
72
73 const QList<SubSurfaceInterface *> below = surface->below();
74 for (SubSurfaceInterface *childSubSurface : below) {
75 registerSubSurface(childSubSurface);
76 }
77
78 const QList<SubSurfaceInterface *> above = surface->above();
79 for (SubSurfaceInterface *childSubSurface : above) {
80 registerSubSurface(childSubSurface);
81 }
82}
