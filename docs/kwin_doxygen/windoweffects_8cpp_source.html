30QWindow *findWindow(WId win)
31{
32 const auto windows = qApp->allWindows();
33 auto it = std::find_if(windows.begin(), windows.end(), [win](QWindow *w) {
34 return w->winId() == win;
35 });
36 if (it == windows.end()) {
37 return nullptr;
38 }
39 return *it;
40}

43bool WindowEffects::isEffectAvailable(KWindowEffects::Effect effect)
44{
45 if (!effects) {
46 return false;
47 }
48 auto e = static_cast<EffectsHandlerImpl *>(effects);
49 switch (effect) {
50 case KWindowEffects::BackgroundContrast:
51 return e->isEffectLoaded(QStringLiteral("contrast"));
52 case KWindowEffects::BlurBehind:
53 return e->isEffectLoaded(QStringLiteral("blur"));
54 case KWindowEffects::Slide:
55 return e->isEffectLoaded(QStringLiteral("slidingpopups"));
56 default:
57 // plugin does not provide integration for other effects
58 return false;
59 }
60}
