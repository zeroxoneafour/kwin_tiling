58void TestSlide::init()
59{
60 using namespace KWin;
61 delete m_display;
62 m_display = new KWin::Display(this);
63 m_display->addSocketName(s_socketName);
64 m_display->start();
65 QVERIFY(m_display->isRunning());
66
67 // setup connection
68 m_connection = new KWayland::Client::ConnectionThread;
69 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
70 m_connection->setSocketName(s_socketName);
71
72 m_thread = new QThread(this);
73 m_connection->moveToThread(m_thread);
74 m_thread->start();
75
76 m_connection->initConnection();
77 QVERIFY(connectedSpy.wait());
78
79 m_queue = new KWayland::Client::EventQueue(this);
80 QVERIFY(!m_queue->isValid());
81 m_queue->setup(m_connection);
82 QVERIFY(m_queue->isValid());
83
84 KWayland::Client::Registry registry;
85 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
86
87 QSignalSpy slideSpy(&registry, &KWayland::Client::Registry::slideAnnounced);
88
89 QVERIFY(!registry.eventQueue());
90 registry.setEventQueue(m_queue);
91 QCOMPARE(registry.eventQueue(), m_queue);
92 registry.create(m_connection->display());
93 QVERIFY(registry.isValid());
94 registry.setup();
95
96 m_compositorInterface = new CompositorInterface(m_display, m_display);
97 QVERIFY(compositorSpy.wait());
98 m_compositor = registry.createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
99
100 m_slideManagerInterface = new SlideManagerInterface(m_display, m_display);
101
102 QVERIFY(slideSpy.wait());
103 m_slideManager = registry.createSlideManager(slideSpy.first().first().value<quint32>(), slideSpy.first().last().value<quint32>(), this);
104}

133void TestSlide::testCreate()
134{
135 QSignalSpy serverSurfaceCreated(m_compositorInterface, &KWin::CompositorInterface::surfaceCreated);
136 std::unique_ptr<KWayland::Client::Surface> surface(m_compositor->createSurface());
137 QVERIFY(serverSurfaceCreated.wait());
138
139 auto serverSurface = serverSurfaceCreated.first().first().value<KWin::SurfaceInterface *>();
140 QSignalSpy slideChanged(serverSurface, &KWin::SurfaceInterface::slideOnShowHideChanged);
141
142 auto slide = m_slideManager->createSlide(surface.get(), surface.get());
143 slide->setLocation(KWayland::Client::Slide::Location::Top);
144 slide->setOffset(15);
145 slide->commit();
146 surface->commit(KWayland::Client::Surface::CommitFlag::None);
147
148 QVERIFY(slideChanged.wait());
149 QCOMPARE(serverSurface->slideOnShowHide()->location(), KWin::SlideInterface::Location::Top);
150 QCOMPARE(serverSurface->slideOnShowHide()->offset(), 15);
151
152 // and destroy
153 QSignalSpy destroyedSpy(serverSurface->slideOnShowHide(), &QObject::destroyed);
154 delete slide;
155 QVERIFY(destroyedSpy.wait());
156}

158void TestSlide::testSurfaceDestroy()
159{
160 using namespace KWin;
161 QSignalSpy serverSurfaceCreated(m_compositorInterface, &CompositorInterface::surfaceCreated);
162 std::unique_ptr<KWayland::Client::Surface> surface(m_compositor->createSurface());
163 QVERIFY(serverSurfaceCreated.wait());
164
165 auto serverSurface = serverSurfaceCreated.first().first().value<SurfaceInterface *>();
166 QSignalSpy slideChanged(serverSurface, &SurfaceInterface::slideOnShowHideChanged);
167
168 std::unique_ptr<KWayland::Client::Slide> slide(m_slideManager->createSlide(surface.get()));
169 slide->commit();
170 surface->commit(KWayland::Client::Surface::CommitFlag::None);
171 QVERIFY(slideChanged.wait());
172 auto serverSlide = serverSurface->slideOnShowHide();
173 QVERIFY(serverSlide);
174
175 // destroy the parent surface
176 QSignalSpy surfaceDestroyedSpy(serverSurface, &QObject::destroyed);
177 QSignalSpy slideDestroyedSpy(serverSlide, &QObject::destroyed);
178 surface.reset();
179 QVERIFY(surfaceDestroyedSpy.wait());
180 QVERIFY(slideDestroyedSpy.isEmpty());
181 // destroy the slide
182 slide.reset();
183 QVERIFY(slideDestroyedSpy.wait());
184}
