44bool DrmPlane::init()
45{
46 DrmUniquePtr<drmModePlane> p(drmModeGetPlane(gpu()->fd(), id()));
47
48 if (!p) {
49 qCWarning(KWIN_DRM) << "Failed to get kernel plane" << id();
50 return false;
51 }
52
53 m_possibleCrtcs = p->possible_crtcs;
54
55 bool success = initProps();
56 if (success) {
57 if (const auto prop = getProp(PropertyIndex::Rotation)) {
58 m_supportedTransformations = Transformations();
59 auto checkSupport = [this, prop](Transformation t) {
60 if (prop->hasEnum(t)) {
61 m_supportedTransformations |= t;
62 }
63 };
64 checkSupport(Transformation::Rotate0);
65 checkSupport(Transformation::Rotate90);
66 checkSupport(Transformation::Rotate180);
67 checkSupport(Transformation::Rotate270);
68 checkSupport(Transformation::ReflectX);
69 checkSupport(Transformation::ReflectY);
70 } else {
71 m_supportedTransformations = Transformation::Rotate0;
72 }
73
74 // read formats from blob if available and if modifiers are supported, and from the plane object if not
75 if (const auto formatProp = getProp(PropertyIndex::In_Formats); formatProp && formatProp->immutableBlob() && gpu()->addFB2ModifiersSupported()) {
76 drmModeFormatModifierIterator iterator{};
77 while (drmModeFormatModifierBlobIterNext(formatProp->immutableBlob(), &iterator)) {
78 m_supportedFormats[iterator.fmt].push_back(iterator.mod);
79 }
80 } else {
81 for (uint32_t i = 0; i < p->count_formats; i++) {
82 m_supportedFormats.insert(p->formats[i], {DRM_FORMAT_MOD_LINEAR});
83 }
84 }
85 if (m_supportedFormats.isEmpty()) {
86 qCWarning(KWIN_DRM) << "Driver doesn't advertise any formats for this plane. Falling back to XRGB8888 without explicit modifiers";
87 m_supportedFormats.insert(DRM_FORMAT_XRGB8888, {});
88 }
89 }
90 return success;
91}

118void DrmPlane::set(const QPoint &srcPos, const QSize &srcSize, const QRect &dst)
119{
120 // Src* are in 16.16 fixed point format
121 setPending(PropertyIndex::SrcX, srcPos.x() << 16);
122 setPending(PropertyIndex::SrcY, srcPos.y() << 16);
123 setPending(PropertyIndex::SrcW, srcSize.width() << 16);
124 setPending(PropertyIndex::SrcH, srcSize.height() << 16);
125 setPending(PropertyIndex::CrtcX, dst.x());
126 setPending(PropertyIndex::CrtcY, dst.y());
127 setPending(PropertyIndex::CrtcW, dst.width());
128 setPending(PropertyIndex::CrtcH, dst.height());
129}
