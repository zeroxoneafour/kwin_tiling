19TextInputContentHints convertContentHint(uint32_t hint)
20{
21 const auto hints = QtWaylandServer::zwp_text_input_v3::content_hint(hint);
22 TextInputContentHints ret = TextInputContentHint::None;
23
24 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_completion) {
25 ret |= TextInputContentHint::AutoCompletion;
26 }
27 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_auto_capitalization) {
28 ret |= TextInputContentHint::AutoCapitalization;
29 }
30 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_lowercase) {
31 ret |= TextInputContentHint::LowerCase;
32 }
33 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_uppercase) {
34 ret |= TextInputContentHint::UpperCase;
35 }
36 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_titlecase) {
37 ret |= TextInputContentHint::TitleCase;
38 }
39 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_hidden_text) {
40 ret |= TextInputContentHint::HiddenText;
41 }
42 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_sensitive_data) {
43 ret |= TextInputContentHint::SensitiveData;
44 }
45 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_latin) {
46 ret |= TextInputContentHint::Latin;
47 }
48 if (hints & QtWaylandServer::zwp_text_input_v3::content_hint_multiline) {

140void TextInputManagerV3InterfacePrivate::zwp_text_input_manager_v3_get_text_input(Resource *resource, uint32_t id, wl_resource *seat)
141{
142 SeatInterface *s = SeatInterface::get(seat);
143 if (!s) {
144 wl_resource_post_error(resource->handle, 0, "Invalid seat");
145 return;
146 }
147 TextInputV3InterfacePrivate *textInputPrivate = TextInputV3InterfacePrivate::get(s->textInputV3());
148 auto *textInputResource = textInputPrivate->add(resource->client(), id, resource->version());
149 // Send enter to this new text input object if the surface is already focused.
150 if (textInputPrivate->surface && textInputPrivate->surface->client()->client() == resource->client()) {
151 textInputPrivate->send_enter(textInputResource->handle, textInputPrivate->surface->resource());
152 }

185 EnabledEmitter emitter(q);

245void TextInputV3InterfacePrivate::done()
246{
247 if (!surface) {
248 return;
249 }
250 const QList<Resource *> textInputs = enabledTextInputsForClient(surface->client());
251
252 preeditText = pending.preeditText;
253 preeditCursorBegin = pending.preeditCursorBegin;
254 preeditCursorEnd = pending.preeditCursorEnd;
255 defaultPendingPreedit();
256
257 for (auto resource : textInputs) {
258 // zwp_text_input_v3.done takes the serial argument which is equal to number of commit requests issued
259 send_done(resource->handle, serialHash[resource]);
260 }

346void TextInputV3InterfacePrivate::zwp_text_input_v3_commit(Resource *resource)
347{
348 EnabledEmitter emitter(q);
349 serialHash[resource]++;
350
351 auto &resourceEnabled = enabled[resource];
352 const auto oldResourceEnabled = resourceEnabled;
353 if (resourceEnabled != pending.enabled) {
354 resourceEnabled = pending.enabled;
355 }
356
357 if (surroundingTextChangeCause != pending.surroundingTextChangeCause) {
358 surroundingTextChangeCause = pending.surroundingTextChangeCause;
359 pending.surroundingTextChangeCause = TextInputChangeCause::InputMethod;
360 }
361
362 if (contentHints != pending.contentHints || contentPurpose != pending.contentPurpose) {
363 contentHints = pending.contentHints;
364 contentPurpose = pending.contentPurpose;
365 if (resourceEnabled) {
366 Q_EMIT q->contentTypeChanged();
367 }
368 }
369
370 if (cursorRectangle != pending.cursorRectangle) {
371 cursorRectangle = pending.cursorRectangle;
372 if (resourceEnabled) {
373 Q_EMIT q->cursorRectangleChanged(cursorRectangle);
374 }
375 }
376
377 if (surroundingText != pending.surroundingText || surroundingTextCursorPosition != pending.surroundingTextCursorPosition
378 || surroundingTextSelectionAnchor != pending.surroundingTextSelectionAnchor) {
379 surroundingText = pending.surroundingText;
380 surroundingTextCursorPosition = pending.surroundingTextCursorPosition;
381 surroundingTextSelectionAnchor = pending.surroundingTextSelectionAnchor;
382 if (resourceEnabled) {
383 Q_EMIT q->surroundingTextChanged();
384 }
385 }
386
387 Q_EMIT q->stateCommitted(serialHash[resource]);
388
389 // Gtk text input implementation expect done to be sent after every commit to synchronize the serial value between commit() and done().
390 // So we need to send the current preedit text with done().
391 // If current preedit is empty, there is no need to send it.
392 if (!preeditText.isEmpty() || preeditCursorBegin != 0 || preeditCursorEnd != 0) {
393 send_preedit_string(resource->handle, preeditText, preeditCursorBegin, preeditCursorEnd);
394 }
395 send_done(resource->handle, serialHash[resource]);
396
397 if (resourceEnabled && oldResourceEnabled) {
398 Q_EMIT q->enableRequested();
399 }
400}
