25void IdleInterfacePrivate::org_kde_kwin_idle_get_idle_timeout(Resource *resource, uint32_t id, wl_resource *seat, uint32_t timeout)
26{
27 SeatInterface *s = SeatInterface::get(seat);
28 Q_ASSERT(s);
29
30 wl_resource *idleTimoutResource = wl_resource_create(resource->client(), &org_kde_kwin_idle_timeout_interface, resource->version(), id);
31 if (!idleTimoutResource) {
32 wl_client_post_no_memory(resource->client());
33 return;
34 }
35
36 new IdleTimeoutInterface(std::chrono::milliseconds(timeout), idleTimoutResource);
37}
