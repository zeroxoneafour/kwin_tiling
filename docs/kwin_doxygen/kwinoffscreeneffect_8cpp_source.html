90void OffscreenData::maybeRender(EffectWindow *window)
91{
92 QRectF logicalGeometry = window->expandedGeometry();
93 QRectF deviceGeometry = scaledRect(logicalGeometry, effects->renderTargetScale());
94
95 QSize textureSize = deviceGeometry.toAlignedRect().size();
96
97 if (!m_texture || m_texture->size() != textureSize) {
98 m_texture.reset(new GLTexture(GL_RGBA8, textureSize));
99 m_texture->setFilter(GL_LINEAR);
100 m_texture->setWrapMode(GL_CLAMP_TO_EDGE);
101 m_fbo.reset(new GLFramebuffer(m_texture.get()));
102 m_isDirty = true;
103 }
104
105 if (m_isDirty) {
106 GLFramebuffer::pushFramebuffer(m_fbo.get());
107 glClearColor(0.0, 0.0, 0.0, 0.0);
108 glClear(GL_COLOR_BUFFER_BIT);
109
110 QMatrix4x4 projectionMatrix;
111 projectionMatrix.ortho(QRectF(0, 0, deviceGeometry.width(), deviceGeometry.height()));
112
113 WindowPaintData data;
114 data.setXTranslation(-logicalGeometry.x());
115 data.setYTranslation(-logicalGeometry.y());
116 data.setOpacity(1.0);
117 data.setProjectionMatrix(projectionMatrix);
118
119 const int mask = Effect::PAINT_WINDOW_TRANSFORMED | Effect::PAINT_WINDOW_TRANSLUCENT;
120 effects->drawWindow(window, mask, infiniteRegion(), data);
121
122 GLFramebuffer::popFramebuffer();
123 m_isDirty = false;
124 }
125}

146void OffscreenData::paint(EffectWindow *window, const QRegion &region,
147 const WindowPaintData &data, const WindowQuadList &quads)
148{
149 GLShader *shader = m_shader ? m_shader : ShaderManager::instance()->shader(ShaderTrait::MapTexture | ShaderTrait::Modulate | ShaderTrait::AdjustSaturation);
150 ShaderBinder binder(shader);
151
152 const qreal scale = effects->renderTargetScale();
153
154 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
155 vbo->reset();
156 vbo->setAttribLayout(GLVertexBuffer::GLVertex2DLayout, 2, sizeof(GLVertex2D));
157
158 RenderGeometry geometry;
159 geometry.setVertexSnappingMode(m_vertexSnappingMode);
160 for (auto &quad : quads) {
161 geometry.appendWindowQuad(quad, scale);
162 }
163 geometry.postProcessTextureCoordinates(m_texture->matrix(NormalizedCoordinates));
164
165 GLVertex2D *map = static_cast<GLVertex2D *>(vbo->map(geometry.count() * sizeof(GLVertex2D)));
166 geometry.copy(std::span(map, geometry.count()));
167 vbo->unmap();
168
169 vbo->bindArrays();
170
171 const qreal rgb = data.brightness() * data.opacity();
172 const qreal a = data.opacity();
173
174 QMatrix4x4 mvp = data.projectionMatrix();
175 mvp.translate(window->x() * scale, window->y() * scale);
176
177 shader->setUniform(GLShader::ModelViewProjectionMatrix, mvp * data.toMatrix(effects->renderTargetScale()));
178 shader->setUniform(GLShader::ModulationConstant, QVector4D(rgb, rgb, rgb, a));
179 shader->setUniform(GLShader::Saturation, data.saturation());
180 shader->setUniform(GLShader::TextureWidth, m_texture->width());
181 shader->setUniform(GLShader::TextureHeight, m_texture->height());
182
183 const bool clipping = region != infiniteRegion();
184 const QRegion clipRegion = clipping ? effects->mapToRenderTarget(region) : infiniteRegion();
185
186 if (clipping) {
187 glEnable(GL_SCISSOR_TEST);
188 }
189
190 glEnable(GL_BLEND);
191 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
192
193 m_texture->bind();
194 vbo->draw(clipRegion, GL_TRIANGLES, 0, geometry.count(), clipping);
195 m_texture->unbind();
196
197 glDisable(GL_BLEND);
198 if (clipping) {
199 glDisable(GL_SCISSOR_TEST);
200 }
201 vbo->unbindArrays();
202}

204void OffscreenEffect::drawWindow(EffectWindow *window, int mask, const QRegion &region, WindowPaintData &data)
205{
206 const auto it = d->windows.find(window);
207 if (it == d->windows.end()) {
208 effects->drawWindow(window, mask, region, data);
209 return;
210 }
211 OffscreenData *offscreenData = it->second.get();
212
213 const QRectF expandedGeometry = window->expandedGeometry();
214 const QRectF frameGeometry = window->frameGeometry();
215
216 QRectF visibleRect = expandedGeometry;
217 visibleRect.moveTopLeft(expandedGeometry.topLeft() - frameGeometry.topLeft());
218 WindowQuad quad;
219 quad[0] = WindowVertex(visibleRect.topLeft(), QPointF(0, 0));
220 quad[1] = WindowVertex(visibleRect.topRight(), QPointF(1, 0));
221 quad[2] = WindowVertex(visibleRect.bottomRight(), QPointF(1, 1));
222 quad[3] = WindowVertex(visibleRect.bottomLeft(), QPointF(0, 1));
223
224 WindowQuadList quads;
225 quads.append(quad);
226 apply(window, mask, data, quads);
227
228 offscreenData->maybeRender(window);
229 offscreenData->paint(window, region, data, quads);
230}

291void CrossFadeEffect::drawWindow(EffectWindow *window, int mask, const QRegion &region, WindowPaintData &data)
292{
293 const auto it = d->windows.find(window);
294
295 // paint the new window (if applicable) underneath
296 if (data.crossFadeProgress() > 0 || it == d->windows.end()) {
297 Effect::drawWindow(window, mask, region, data);
298 }
299
300 if (it == d->windows.end()) {
301 return;
302 }
303 CrossFadeWindowData *offscreenData = it->second.get();
304
305 // paint old snapshot on top
306 WindowPaintData previousWindowData = data;
307 previousWindowData.setOpacity((1.0 - data.crossFadeProgress()) * data.opacity());
308
309 const QRectF expandedGeometry = window->expandedGeometry();
310 const QRectF frameGeometry = window->frameGeometry();
311
312 // This is for the case of *non* live effect, when the window buffer we saved has a different size
313 // compared to the size the window has now. The "old" window will be rendered scaled to the current
314 // window geometry, but everything will be scaled, also the shadow if there is any, making the window
315 // frame not line up anymore with window->frameGeometry()
316 // to fix that, we consider how much the shadow will have scaled, and use that as margins to the
317 // current frame geometry. this causes the scaled window to visually line up perfectly with frameGeometry,
318 // having the scaled shadow all outside of it.
319 const qreal widthRatio = offscreenData->frameGeometryAtCapture.width() / frameGeometry.width();
320 const qreal heightRatio = offscreenData->frameGeometryAtCapture.height() / frameGeometry.height();
321
322 const QMarginsF margins(
323 (expandedGeometry.x() - frameGeometry.x()) / widthRatio,
324 (expandedGeometry.y() - frameGeometry.y()) / heightRatio,
325 (frameGeometry.right() - expandedGeometry.right()) / widthRatio,
326 (frameGeometry.bottom() - expandedGeometry.bottom()) / heightRatio);
327
328 QRectF visibleRect = QRectF(QPointF(0, 0), frameGeometry.size()) - margins;
329
330 WindowQuad quad;
331 quad[0] = WindowVertex(visibleRect.topLeft(), QPointF(0, 0));
332 quad[1] = WindowVertex(visibleRect.topRight(), QPointF(1, 0));
333 quad[2] = WindowVertex(visibleRect.bottomRight(), QPointF(1, 1));
334 quad[3] = WindowVertex(visibleRect.bottomLeft(), QPointF(0, 1));
335
336 WindowQuadList quads;
337 quads.append(quad);
338 offscreenData->paint(window, region, previousWindowData, quads);
339}

341void CrossFadeEffect::redirect(EffectWindow *window)
342{
343 if (d->windows.empty()) {
344 connect(effects, &EffectsHandler::windowDeleted, this, &CrossFadeEffect::handleWindowDeleted);
345 }
346
347 std::unique_ptr<CrossFadeWindowData> &offscreenData = d->windows[window];
348 if (offscreenData) {
349 return;
350 }
351 offscreenData = std::make_unique<CrossFadeWindowData>();
352
353 // Avoid including blur and contrast effects. During a normal painting cycle they
354 // won't be included, but since we call effects->drawWindow() outside usual compositing
355 // cycle, we have to prevent backdrop effects kicking in.
356 const QVariant blurRole = window->data(WindowForceBlurRole);
357 window->setData(WindowForceBlurRole, QVariant());
358 const QVariant contrastRole = window->data(WindowForceBackgroundContrastRole);
359 window->setData(WindowForceBackgroundContrastRole, QVariant());
360
361 effects->makeOpenGLContextCurrent();
362 offscreenData->maybeRender(window);
363 offscreenData->frameGeometryAtCapture = window->frameGeometry();
364
365 window->setData(WindowForceBlurRole, blurRole);
366 window->setData(WindowForceBackgroundContrastRole, contrastRole);
367}
