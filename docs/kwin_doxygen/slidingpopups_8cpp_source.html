33SlidingPopupsEffect::SlidingPopupsEffect()
34{
35 initConfig<SlidingPopupsConfig>();
36
37 KWaylandServer::Display *display = effects->waylandDisplay();
38 if (display) {
39 if (!s_slideManagerRemoveTimer) {
40 s_slideManagerRemoveTimer = new QTimer(QCoreApplication::instance());
41 s_slideManagerRemoveTimer->setSingleShot(true);
42 s_slideManagerRemoveTimer->callOnTimeout([]() {
43 s_slideManager->remove();
44 s_slideManager = nullptr;
45 });
46 }
47 s_slideManagerRemoveTimer->stop();
48 if (!s_slideManager) {
49 s_slideManager = new KWaylandServer::SlideManagerInterface(display, s_slideManagerRemoveTimer);
50 }
51 }
52
53 m_slideLength = QFontMetrics(QGuiApplication::font()).height() * 8;
54
55 m_atom = effects->announceSupportProperty("_KDE_SLIDE", this);
56 connect(effects, &EffectsHandler::windowAdded, this, &SlidingPopupsEffect::slotWindowAdded);
57 connect(effects, &EffectsHandler::windowClosed, this, &SlidingPopupsEffect::slideOut);
58 connect(effects, &EffectsHandler::windowDeleted, this, &SlidingPopupsEffect::slotWindowDeleted);
59 connect(effects, &EffectsHandler::propertyNotify, this, &SlidingPopupsEffect::slotPropertyNotify);
60 connect(effects, &EffectsHandler::windowShown, this, &SlidingPopupsEffect::slideIn);
61 connect(effects, &EffectsHandler::windowHidden, this, &SlidingPopupsEffect::slideOut);
62 connect(effects, &EffectsHandler::xcbConnectionChanged, this, [this]() {
63 m_atom = effects->announceSupportProperty(QByteArrayLiteral("_KDE_SLIDE"), this);
64 });
65 connect(effects, &EffectsHandler::desktopChanged,
66 this, &SlidingPopupsEffect::stopAnimations);
67 connect(effects, &EffectsHandler::activeFullScreenEffectChanged,
68 this, &SlidingPopupsEffect::stopAnimations);
69 connect(effects, &EffectsHandler::windowFrameGeometryChanged, this, &SlidingPopupsEffect::slotWindowFrameGeometryChanged);
70
71 reconfigure(ReconfigureAll);
72
73 const EffectWindowList windows = effects->stackingOrder();
74 for (EffectWindow *window : windows) {
75 setupSlideData(window);
76 }
77}

92void SlidingPopupsEffect::reconfigure(ReconfigureFlags flags)
93{
94 SlidingPopupsConfig::self()->read();
95 m_slideInDuration = std::chrono::milliseconds(
96 static_cast<int>(animationTime(SlidingPopupsConfig::slideInTime() != 0 ? SlidingPopupsConfig::slideInTime() : 150)));
97 m_slideOutDuration = std::chrono::milliseconds(
98 static_cast<int>(animationTime(SlidingPopupsConfig::slideOutTime() != 0 ? SlidingPopupsConfig::slideOutTime() : 250)));
99
100 auto animationIt = m_animations.begin();
101 while (animationIt != m_animations.end()) {
102 const auto duration = ((*animationIt).kind == AnimationKind::In)
103 ? m_slideInDuration
104 : m_slideOutDuration;
105 (*animationIt).timeLine.setDuration(duration);
106 ++animationIt;
107 }
108
109 auto dataIt = m_animationsData.begin();
110 while (dataIt != m_animationsData.end()) {
111 (*dataIt).slideInDuration = m_slideInDuration;
112 (*dataIt).slideOutDuration = m_slideOutDuration;
113 ++dataIt;
114 }
115}

131void SlidingPopupsEffect::paintWindow(EffectWindow *w, int mask, QRegion region, WindowPaintData &data)
132{
133 auto animationIt = m_animations.constFind(w);
134 if (animationIt == m_animations.constEnd()) {
135 effects->paintWindow(w, mask, region, data);
136 return;
137 }
138
139 const AnimationData &animData = m_animationsData[w];
140 const qreal slideLength = (animData.slideLength > 0) ? animData.slideLength : m_slideLength;
141
142 const QRectF screenRect = effects->clientArea(FullScreenArea, w->screen(), effects->currentDesktop());
143 int splitPoint = 0;
144 const QRectF geo = w->expandedGeometry();
145 const qreal t = (*animationIt).timeLine.value();
146
147 switch (animData.location) {
148 case Location::Left:
149 if (slideLength < geo.width()) {
150 data.multiplyOpacity(t);
151 }
152 data.translate(-interpolate(std::min(geo.width(), slideLength), 0.0, t));
153 splitPoint = geo.width() - (geo.x() + geo.width() - screenRect.x() - animData.offset);
154 region &= QRegion(geo.x() + splitPoint, geo.y(), geo.width() - splitPoint, geo.height());
155 break;
156 case Location::Top:
157 if (slideLength < geo.height()) {
158 data.multiplyOpacity(t);
159 }
160 data.translate(0.0, -interpolate(std::min(geo.height(), slideLength), 0.0, t));
161 splitPoint = geo.height() - (geo.y() + geo.height() - screenRect.y() - animData.offset);
162 region &= QRegion(geo.x(), geo.y() + splitPoint, geo.width(), geo.height() - splitPoint);
163 break;
164 case Location::Right:
165 if (slideLength < geo.width()) {
166 data.multiplyOpacity(t);
167 }
168 data.translate(interpolate(std::min(geo.width(), slideLength), 0.0, t));
169 splitPoint = screenRect.x() + screenRect.width() - geo.x() - animData.offset;
170 region &= QRegion(geo.x(), geo.y(), splitPoint, geo.height());
171 break;
172 case Location::Bottom:
173 default:
174 if (slideLength < geo.height()) {
175 data.multiplyOpacity(t);
176 }
177 data.translate(0.0, interpolate(std::min(geo.height(), slideLength), 0.0, t));
178 splitPoint = screenRect.y() + screenRect.height() - geo.y() - animData.offset;
179 region &= QRegion(geo.x(), geo.y(), geo.width(), splitPoint);
180 }
181
182 effects->paintWindow(w, mask, region, data);
183}

235void SlidingPopupsEffect::slotPropertyNotify(EffectWindow *w, long atom)
236{
237 if (!w || atom != m_atom || m_atom == XCB_ATOM_NONE) {
238 return;
239 }
240
241 // _KDE_SLIDE atom format(each field is an uint32_t):
242 // <offset> <location> [<slide in duration>] [<slide out duration>] [<slide length>]
243 //
244 // If offset is equal to -1, this effect will decide what offset to use
245 // given edge of the screen, from which the window has to slide.
246 //
247 // If slide in duration is equal to 0 milliseconds, the default slide in
248 // duration will be used. Same with the slide out duration.
249 //
250 // NOTE: If only slide in duration has been provided, then it will be
251 // also used as slide out duration. I.e. if you provided only slide in
252 // duration, then slide in duration == slide out duration.
253
254 const QByteArray rawAtomData = w->readProperty(m_atom, m_atom, 32);
255
256 if (rawAtomData.isEmpty()) {
257 // Property was removed, thus also remove the effect for window
258 if (w->data(WindowClosedGrabRole).value<void *>() == this) {
259 w->setData(WindowClosedGrabRole, QVariant());
260 }
261 m_animations.remove(w);
262 m_animationsData.remove(w);
263 return;
264 }
265
266 // Offset and location are required.
267 if (static_cast<size_t>(rawAtomData.size()) < sizeof(uint32_t) * 2) {
268 return;
269 }
270
271 const auto *atomData = reinterpret_cast<const uint32_t *>(rawAtomData.data());
272 AnimationData &animData = m_animationsData[w];
273 animData.offset = atomData[0];
274
275 switch (atomData[1]) {
276 case 0: // West
277 animData.location = Location::Left;
278 break;
279 case 1: // North
280 animData.location = Location::Top;
281 break;
282 case 2: // East
283 animData.location = Location::Right;
284 break;
285 case 3: // South
286 default:
287 animData.location = Location::Bottom;
288 break;
289 }
290
291 if (static_cast<size_t>(rawAtomData.size()) >= sizeof(uint32_t) * 3) {
292 animData.slideInDuration = std::chrono::milliseconds(atomData[2]);
293 if (static_cast<size_t>(rawAtomData.size()) >= sizeof(uint32_t) * 4) {
294 animData.slideOutDuration = std::chrono::milliseconds(atomData[3]);
295 } else {
296 animData.slideOutDuration = animData.slideInDuration;
297 }
298 } else {
299 animData.slideInDuration = m_slideInDuration;
300 animData.slideOutDuration = m_slideOutDuration;
301 }
302
303 if (static_cast<size_t>(rawAtomData.size()) >= sizeof(uint32_t) * 5) {
304 animData.slideLength = atomData[4];
305 } else {
306 animData.slideLength = 0;
307 }
308
309 setupAnimData(w);
310}

319void SlidingPopupsEffect::setupAnimData(EffectWindow *w)
320{
321 const QRectF screenRect = effects->clientArea(FullScreenArea, w->screen(), effects->currentDesktop());
322 const QRectF windowGeo = w->frameGeometry();
323 AnimationData &animData = m_animationsData[w];
324
325 if (animData.offset == -1) {
326 switch (animData.location) {
327 case Location::Left:
328 animData.offset = std::max<qreal>(windowGeo.left() - screenRect.left(), 0);
329 break;
330 case Location::Top:
331 animData.offset = std::max<qreal>(windowGeo.top() - screenRect.top(), 0);
332 break;
333 case Location::Right:
334 animData.offset = std::max<qreal>(screenRect.right() - windowGeo.right(), 0);
335 break;
336 case Location::Bottom:
337 default:
338 animData.offset = std::max<qreal>(screenRect.bottom() - windowGeo.bottom(), 0);
339 break;
340 }
341 }
342 // sanitize
343 switch (animData.location) {
344 case Location::Left:
345 animData.offset = std::max<qreal>(windowGeo.left() - screenRect.left(), animData.offset);
346 break;
347 case Location::Top:
348 animData.offset = std::max<qreal>(windowGeo.top() - screenRect.top(), animData.offset);
349 break;
350 case Location::Right:
351 animData.offset = std::max<qreal>(screenRect.right() - windowGeo.right(), animData.offset);
352 break;
353 case Location::Bottom:
354 default:
355 animData.offset = std::max<qreal>(screenRect.bottom() - windowGeo.bottom(), animData.offset);
356 break;
357 }
358
359 animData.slideInDuration = (animData.slideInDuration.count() != 0)
360 ? animData.slideInDuration
361 : m_slideInDuration;
362
363 animData.slideOutDuration = (animData.slideOutDuration.count() != 0)
364 ? animData.slideOutDuration
365 : m_slideOutDuration;
366
367 // Grab the window, so other windowClosed effects will ignore it
368 w->setData(WindowClosedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
369}

371void SlidingPopupsEffect::slotWaylandSlideOnShowChanged(EffectWindow *w)
372{
373 if (!w) {
374 return;
375 }
376
377 KWaylandServer::SurfaceInterface *surf = w->surface();
378 if (!surf) {
379 return;
380 }
381
382 if (surf->slideOnShowHide()) {
383 AnimationData &animData = m_animationsData[w];
384
385 animData.offset = surf->slideOnShowHide()->offset();
386
387 switch (surf->slideOnShowHide()->location()) {
388 case KWaylandServer::SlideInterface::Location::Top:
389 animData.location = Location::Top;
390 break;
391 case KWaylandServer::SlideInterface::Location::Left:
392 animData.location = Location::Left;
393 break;
394 case KWaylandServer::SlideInterface::Location::Right:
395 animData.location = Location::Right;
396 break;
397 case KWaylandServer::SlideInterface::Location::Bottom:
398 default:
399 animData.location = Location::Bottom;
400 break;
401 }
402 animData.slideLength = 0;
403 animData.slideInDuration = m_slideInDuration;
404 animData.slideOutDuration = m_slideOutDuration;
405
406 setupAnimData(w);
407 }

410void SlidingPopupsEffect::setupInternalWindowSlide(EffectWindow *w)
411{
412 if (!w) {
413 return;
414 }
415 auto internal = w->internalWindow();
416 if (!internal) {
417 return;
418 }
419 const QVariant slideProperty = internal->property("kwin_slide");
420 if (!slideProperty.isValid()) {
421 return;
422 }
423 Location location;
424 switch (slideProperty.value<KWindowEffects::SlideFromLocation>()) {
425 case KWindowEffects::BottomEdge:
426 location = Location::Bottom;
427 break;
428 case KWindowEffects::TopEdge:
429 location = Location::Top;
430 break;
431 case KWindowEffects::RightEdge:
432 location = Location::Right;
433 break;
434 case KWindowEffects::LeftEdge:
435 location = Location::Left;
436 break;
437 default:
438 return;
439 }
440 AnimationData &animData = m_animationsData[w];
441 animData.location = location;
442 bool intOk = false;
443 animData.offset = internal->property("kwin_slide_offset").toInt(&intOk);
444 if (!intOk) {
445 animData.offset = -1;
446 }
447 animData.slideLength = 0;
448 animData.slideInDuration = m_slideInDuration;
449 animData.slideOutDuration = m_slideOutDuration;
450
451 setupAnimData(w);
452}

474bool SlidingPopupsEffect::eventFilter(QObject *watched, QEvent *event)
475{
476 auto internal = qobject_cast<QWindow *>(watched);
477 if (internal && event->type() == QEvent::DynamicPropertyChange) {
478 QDynamicPropertyChangeEvent *pe = static_cast<QDynamicPropertyChangeEvent *>(event);
479 if (pe->propertyName() == "kwin_slide" || pe->propertyName() == "kwin_slide_offset") {
480 if (auto w = effects->findWindow(internal)) {
481 setupInternalWindowSlide(w);
482 }
483 }
484 }
485 return false;

488void SlidingPopupsEffect::slideIn(EffectWindow *w)
489{
490 if (effects->activeFullScreenEffect()) {
491 return;
492 }
493
494 if (!w->isVisible()) {
495 return;
496 }
497
498 auto dataIt = m_animationsData.constFind(w);
499 if (dataIt == m_animationsData.constEnd()) {
500 return;
501 }
502
503 Animation &animation = m_animations[w];
504 animation.kind = AnimationKind::In;
505 animation.timeLine.setDirection(TimeLine::Forward);
506 animation.timeLine.setDuration((*dataIt).slideInDuration);
507 animation.timeLine.setEasingCurve(QEasingCurve::OutCubic);
508
509 // If the opposite animation (Out) was active and it had shorter duration,
510 // at this point, the timeline can end up in the "done" state. Thus, we have
511 // to reset it.
512 if (animation.timeLine.done()) {
513 animation.timeLine.reset();
514 }
515
516 w->setData(WindowAddedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
517 w->setData(WindowForceBackgroundContrastRole, QVariant(true));
518 w->setData(WindowForceBlurRole, QVariant(true));
519
520 w->addRepaintFull();
521}

523void SlidingPopupsEffect::slideOut(EffectWindow *w)
524{
525 if (effects->activeFullScreenEffect()) {
526 return;
527 }
528
529 if (!w->isVisible()) {
530 return;
531 }
532
533 auto dataIt = m_animationsData.constFind(w);
534 if (dataIt == m_animationsData.constEnd()) {
535 return;
536 }
537
538 Animation &animation = m_animations[w];
539 if (w->isDeleted()) {
540 animation.deletedRef = EffectWindowDeletedRef(w);
541 }
542 animation.visibleRef = EffectWindowVisibleRef(w, EffectWindow::PAINT_DISABLED | EffectWindow::PAINT_DISABLED_BY_DELETE);
543 animation.kind = AnimationKind::Out;
544 animation.timeLine.setDirection(TimeLine::Backward);
545 animation.timeLine.setDuration((*dataIt).slideOutDuration);
546 // this is effectively InCubic because the direction is reversed
547 animation.timeLine.setEasingCurve(QEasingCurve::OutCubic);
548
549 // If the opposite animation (In) was active and it had shorter duration,
550 // at this point, the timeline can end up in the "done" state. Thus, we have
551 // to reset it.
552 if (animation.timeLine.done()) {
553 animation.timeLine.reset();
554 }
555
556 w->setData(WindowClosedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
557 w->setData(WindowForceBackgroundContrastRole, QVariant(true));
558 w->setData(WindowForceBlurRole, QVariant(true));
559
560 w->addRepaintFull();
561}
