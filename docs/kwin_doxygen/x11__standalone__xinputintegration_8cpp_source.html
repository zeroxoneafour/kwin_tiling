40class XInputEventFilter : public X11EventFilter
41{
42public:
43 XInputEventFilter(int xi_opcode)
44 : X11EventFilter(XCB_GE_GENERIC, xi_opcode, QVector<int>{XI_RawMotion, XI_RawButtonPress, XI_RawButtonRelease, XI_RawKeyPress, XI_RawKeyRelease, XI_TouchBegin, XI_TouchUpdate, XI_TouchOwnership, XI_TouchEnd})
45 {
46 }
47 ~XInputEventFilter() override = default;
48
49 bool event(xcb_generic_event_t *event) override
50 {
51 GeEventMemMover ge(event);
52 switch (ge->event_type) {
53 case XI_RawKeyPress: {
54 auto re = reinterpret_cast<xXIRawEvent *>(event);
55 input()->keyboard()->processKey(re->detail - 8, InputRedirection::KeyboardKeyPressed, std::chrono::milliseconds(re->time));
56 break;
57 }
58 case XI_RawKeyRelease: {
59 auto re = reinterpret_cast<xXIRawEvent *>(event);
60 input()->keyboard()->processKey(re->detail - 8, InputRedirection::KeyboardKeyReleased, std::chrono::milliseconds(re->time));
61 break;
62 }
63 case XI_RawButtonPress: {
64 auto e = reinterpret_cast<xXIRawEvent *>(event);
65 switch (e->detail) {
66 // TODO: this currently ignores left handed settings, for current usage not needed
67 // if we want to use also for global mouse shortcuts, this needs to reflect state correctly
68 case XCB_BUTTON_INDEX_1:
69 input()->pointer()->processButton(BTN_LEFT, InputRedirection::PointerButtonPressed, std::chrono::milliseconds(e->time));
70 break;
71 case XCB_BUTTON_INDEX_2:
72 input()->pointer()->processButton(BTN_MIDDLE, InputRedirection::PointerButtonPressed, std::chrono::milliseconds(e->time));
73 break;
74 case XCB_BUTTON_INDEX_3:
75 input()->pointer()->processButton(BTN_RIGHT, InputRedirection::PointerButtonPressed, std::chrono::milliseconds(e->time));
76 break;
77 case XCB_BUTTON_INDEX_4:
78 case XCB_BUTTON_INDEX_5:
79 // vertical axis, ignore on press
80 break;
81 // TODO: further buttons, horizontal scrolling?
82 }
83 }
84 if (m_x11Cursor) {
85 m_x11Cursor->schedulePoll();
86 }
87 break;
88 case XI_RawButtonRelease: {
89 auto e = reinterpret_cast<xXIRawEvent *>(event);
90 switch (e->detail) {
91 // TODO: this currently ignores left handed settings, for current usage not needed
92 // if we want to use also for global mouse shortcuts, this needs to reflect state correctly
93 case XCB_BUTTON_INDEX_1:
94 input()->pointer()->processButton(BTN_LEFT, InputRedirection::PointerButtonReleased, std::chrono::milliseconds(e->time));
95 break;
96 case XCB_BUTTON_INDEX_2:
97 input()->pointer()->processButton(BTN_MIDDLE, InputRedirection::PointerButtonReleased, std::chrono::milliseconds(e->time));
98 break;
99 case XCB_BUTTON_INDEX_3:
100 input()->pointer()->processButton(BTN_RIGHT, InputRedirection::PointerButtonReleased, std::chrono::milliseconds(e->time));
101 break;
102 case XCB_BUTTON_INDEX_4:
103 input()->pointer()->processAxis(InputRedirection::PointerAxisVertical, 120, 1, InputRedirection::PointerAxisSourceWheel, std::chrono::milliseconds(e->time));
104 break;
105 case XCB_BUTTON_INDEX_5:
106 input()->pointer()->processAxis(InputRedirection::PointerAxisVertical, -120, 1, InputRedirection::PointerAxisSourceWheel, std::chrono::milliseconds(e->time));
107 break;
108 // TODO: further buttons, horizontal scrolling?
109 }
110 }
111 if (m_x11Cursor) {
112 m_x11Cursor->schedulePoll();
113 }
114 break;
115 case XI_TouchBegin: {
116 auto e = reinterpret_cast<xXIDeviceEvent *>(event);
117 m_lastTouchPositions.insert(e->detail, QPointF(fixed1616ToReal(e->event_x), fixed1616ToReal(e->event_y)));
118 break;
119 }
120 case XI_TouchUpdate: {
121 auto e = reinterpret_cast<xXIDeviceEvent *>(event);
122 const QPointF touchPosition = QPointF(fixed1616ToReal(e->event_x), fixed1616ToReal(e->event_y));
123 if (e->detail == m_trackingTouchId) {
124 const auto last = m_lastTouchPositions.value(e->detail);
125 workspace()->screenEdges()->gestureRecognizer()->updateSwipeGesture(touchPosition - last);
126 }
127 m_lastTouchPositions.insert(e->detail, touchPosition);
128 break;
129 }
130 case XI_TouchEnd: {
131 auto e = reinterpret_cast<xXIDeviceEvent *>(event);
132 if (e->detail == m_trackingTouchId) {
133 workspace()->screenEdges()->gestureRecognizer()->endSwipeGesture();
134 }
135 m_lastTouchPositions.remove(e->detail);
136 m_trackingTouchId = 0;
137 break;
138 }
139 case XI_TouchOwnership: {
140 auto e = reinterpret_cast<xXITouchOwnershipEvent *>(event);
141 auto it = m_lastTouchPositions.constFind(e->touchid);
142 if (it == m_lastTouchPositions.constEnd()) {
143 XIAllowTouchEvents(display(), e->deviceid, e->sourceid, e->touchid, XIRejectTouch);
144 } else {
145 if (workspace()->screenEdges()->gestureRecognizer()->startSwipeGesture(it.value()) > 0) {
146 m_trackingTouchId = e->touchid;
147 }
148 XIAllowTouchEvents(display(), e->deviceid, e->sourceid, e->touchid, m_trackingTouchId == e->touchid ? XIAcceptTouch : XIRejectTouch);
149 }
150 break;
151 }
152 default:
153 if (m_x11Cursor) {
154 m_x11Cursor->schedulePoll();
155 }
156 break;
157 }
158 return false;
159 }

182class XKeyPressReleaseEventFilter : public X11EventFilter
183{
184public:
185 XKeyPressReleaseEventFilter(uint32_t type)
186 : X11EventFilter(type)
187 {
188 }
189 ~XKeyPressReleaseEventFilter() override = default;
190
191 bool event(xcb_generic_event_t *event) override
192 {
193 xcb_key_press_event_t *ke = reinterpret_cast<xcb_key_press_event_t *>(event);
194 if (ke->event == ke->root) {
195 const uint8_t eventType = event->response_type & ~0x80;
196 if (eventType == XCB_KEY_PRESS) {
197 input()->keyboard()->processKey(ke->detail - 8, InputRedirection::KeyboardKeyPressed, std::chrono::milliseconds(ke->time));
198 } else {
199 input()->keyboard()->processKey(ke->detail - 8, InputRedirection::KeyboardKeyReleased, std::chrono::milliseconds(ke->time));
200 }
201 }
202 return false;
203 }
204};

214void XInputIntegration::init()
215{
216 Display *dpy = display();
217 int xi_opcode, event, error;
218 // init XInput extension
219 if (!XQueryExtension(dpy, "XInputExtension", &xi_opcode, &event, &error)) {
220 qCDebug(KWIN_X11STANDALONE) << "XInputExtension not present";
221 return;
222 }
223
224 // verify that the XInput extension is at at least version 2.0
225 int major = 2, minor = 2;
226 int result = XIQueryVersion(dpy, &major, &minor);
227 if (result != Success) {
228 qCDebug(KWIN_X11STANDALONE) << "Failed to init XInput 2.2, trying 2.0";
229 minor = 0;
230 if (XIQueryVersion(dpy, &major, &minor) != Success) {
231 qCDebug(KWIN_X11STANDALONE) << "Failed to init XInput";
232 return;
233 }
234 }
235 m_hasXInput = true;
236 m_xiOpcode = xi_opcode;
237 m_majorVersion = major;
238 m_minorVersion = minor;
239 qCDebug(KWIN_X11STANDALONE) << "Has XInput support" << m_majorVersion << "." << m_minorVersion;
240}

247void XInputIntegration::startListening()
248{
249 // this assumes KWin is the only one setting events on the root window
250 // given Qt's source code this seems to be true. If it breaks, we need to change
251 XIEventMask evmasks[1];
252 unsigned char mask1[XIMaskLen(XI_LASTEVENT)];
253
254 memset(mask1, 0, sizeof(mask1));
255
256 XISetMask(mask1, XI_RawMotion);
257 XISetMask(mask1, XI_RawButtonPress);
258 XISetMask(mask1, XI_RawButtonRelease);
259 if (m_majorVersion >= 2 && m_minorVersion >= 1) {
260 // we need to listen to all events, which is only available with XInput 2.1
261 XISetMask(mask1, XI_RawKeyPress);
262 XISetMask(mask1, XI_RawKeyRelease);
263 }
264 if (m_majorVersion >= 2 && m_minorVersion >= 2) {
265 // touch events since 2.2
266 XISetMask(mask1, XI_TouchBegin);
267 XISetMask(mask1, XI_TouchUpdate);
268 XISetMask(mask1, XI_TouchOwnership);
269 XISetMask(mask1, XI_TouchEnd);
270 }
271
272 evmasks[0].deviceid = XIAllMasterDevices;
273 evmasks[0].mask_len = sizeof(mask1);
274 evmasks[0].mask = mask1;
275 XISelectEvents(display(), rootWindow(), evmasks, 1);
276
277 m_xiEventFilter = std::make_unique<XInputEventFilter>(m_xiOpcode);
278 m_xiEventFilter->setCursor(m_x11Cursor);
279 m_xiEventFilter->setDisplay(display());
280 m_keyPressFilter = std::make_unique<XKeyPressReleaseEventFilter>(XCB_KEY_PRESS);
281 m_keyReleaseFilter = std::make_unique<XKeyPressReleaseEventFilter>(XCB_KEY_RELEASE);
282
283 // install the input event spies also relevant for X11 platform
284 input()->installInputEventSpy(new ModifierOnlyShortcuts);
285}
