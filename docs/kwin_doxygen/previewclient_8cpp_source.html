362void PreviewClient::requestToggleMaximization(Qt::MouseButtons buttons)
363{
364 if (buttons.testFlag(Qt::LeftButton)) {
365 const bool set = !isMaximized();
366 setMaximizedHorizontally(set);
367 setMaximizedVertically(set);
368 } else if (buttons.testFlag(Qt::RightButton)) {
369 setMaximizedHorizontally(!isMaximizedHorizontally());
370 } else if (buttons.testFlag(Qt::MiddleButton)) {
371 setMaximizedVertically(!isMaximizedVertically());
372 }
373}
