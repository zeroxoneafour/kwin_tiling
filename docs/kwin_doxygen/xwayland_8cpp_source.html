86class XwaylandInputSpy : public QObject, public KWin::InputEventSpy
87{
88public:
89 XwaylandInputSpy()
90 {
91 connect(waylandServer()->seat(), &KWaylandServer::SeatInterface::focusedKeyboardSurfaceAboutToChange,
92 this, [this](KWaylandServer::SurfaceInterface *newSurface) {
93 auto keyboard = waylandServer()->seat()->keyboard();
94 if (!newSurface) {
95 return;
96 }
97
98 if (waylandServer()->xWaylandConnection() == newSurface->client()) {
99 // Since this is a spy but the keyboard interface gets its normal sendKey calls through filters,
100 // there can be a mismatch in both states.
101 // This loop makes sure all key press events are reset before we switch back to the
102 // Xwayland client and the state is correctly restored.
103 for (auto it = m_states.constBegin(); it != m_states.constEnd(); ++it) {
104 if (it.value() == KWaylandServer::KeyboardKeyState::Pressed) {
105 keyboard->sendKey(it.key(), KWaylandServer::KeyboardKeyState::Released, waylandServer()->xWaylandConnection());
106 }
107 }
108 m_states.clear();
109 }
110 });
111 }

113 void setMode(XwaylandEavesdropsMode mode)
114 {
115 static const QSet<quint32> modifierKeys = {
116 Qt::Key_Control,
117 Qt::Key_Shift,
118 Qt::Key_Alt,
119 Qt::Key_Meta,
120 };
121
122 switch (mode) {
123 case None:
124 m_filter = {};
125 break;
126 case Modifiers:
127 m_filter = [](int key, Qt::KeyboardModifiers) {
128 return modifierKeys.contains(key);
129 };
130 break;
131 case Combinations:
132 m_filter = [](int key, Qt::KeyboardModifiers m) {
133 return m != Qt::NoModifier || modifierKeys.contains(key);
134 };
135 break;
136 case All:
137 m_filter = [](int, Qt::KeyboardModifiers) {
138 return true;
139 };
140 break;
141 }
142 }
143
144 void keyEvent(KWin::KeyEvent *event) override
145 {
146 if (event->isAutoRepeat()) {
147 return;
148 }
149
150 Window *window = workspace()->activeWindow();
151 if (!m_filter || !m_filter(event->key(), event->modifiers()) || (window && window->isLockScreen())) {
152 return;
153 }
154
155 auto keyboard = waylandServer()->seat()->keyboard();
156 auto surface = keyboard->focusedSurface();
157 if (!surface) {
158 return;
159 }
160
161 auto client = surface->client();
162 if (waylandServer()->xWaylandConnection() != client) {
163 KWaylandServer::KeyboardKeyState state{event->type() == QEvent::KeyPress};
164 if (!updateKey(event->nativeScanCode(), state)) {
165 return;
166 }
167
168 auto xkb = input()->keyboard()->xkb();
169 keyboard->sendModifiers(xkb->modifierState().depressed,
170 xkb->modifierState().latched,
171 xkb->modifierState().locked,
172 xkb->currentLayout());
173
174 waylandServer()->seat()->keyboard()->sendKey(event->nativeScanCode(), state, waylandServer()->xWaylandConnection());
175 }
176 }
177
178 bool updateKey(quint32 key, KWaylandServer::KeyboardKeyState state)
179 {
180 auto it = m_states.find(key);
181 if (it == m_states.end()) {
182 m_states.insert(key, state);
183 return true;
184 }
185 if (it.value() == state) {
186 return false;
187 }
188 it.value() = state;
189 return true;
190 }
191
192 QHash<quint32, KWaylandServer::KeyboardKeyState> m_states;
193 std::function<bool(int key, Qt::KeyboardModifiers)> m_filter;
194};

227void Xwayland::dispatchEvents(DispatchEventsMode mode)
228{
229 xcb_connection_t *connection = kwinApp()->x11Connection();
230 if (!connection) {
231 qCWarning(KWIN_XWL, "Attempting to dispatch X11 events with no connection");
232 return;
233 }
234
235 const int connectionError = xcb_connection_has_error(connection);
236 if (connectionError) {
237 qCWarning(KWIN_XWL, "The X11 connection broke (error %d)", connectionError);
238 m_launcher->stop();
239 return;
240 }
241
242 auto pollEventFunc = mode == DispatchEventsMode::Poll ? xcb_poll_for_event : xcb_poll_for_queued_event;
243
244 while (xcb_generic_event_t *event = pollEventFunc(connection)) {
245#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
246 long result = 0;
247#else
248 qintptr result = 0;
249#endif
250
251 QAbstractEventDispatcher *dispatcher = QCoreApplication::eventDispatcher();
252 dispatcher->filterNativeEvent(QByteArrayLiteral("xcb_generic_event_t"), event, &result);
253 free(event);
254 }
255
256 xcb_flush(connection);
257}

259void Xwayland::installSocketNotifier()
260{
261 const int fileDescriptor = xcb_get_file_descriptor(kwinApp()->x11Connection());
262
263 m_socketNotifier = new QSocketNotifier(fileDescriptor, QSocketNotifier::Read, this);
264 connect(m_socketNotifier, &QSocketNotifier::activated, this, [this]() {
265 dispatchEvents(DispatchEventsMode::Poll);
266 });
267
268 QAbstractEventDispatcher *dispatcher = QCoreApplication::eventDispatcher();
269 connect(dispatcher, &QAbstractEventDispatcher::aboutToBlock, this, [this]() {
270 dispatchEvents(DispatchEventsMode::EventQueue);
271 });
272 connect(dispatcher, &QAbstractEventDispatcher::awake, this, [this]() {
273 dispatchEvents(DispatchEventsMode::EventQueue);
274 });

286void Xwayland::handleXwaylandFinished()
287{
288 disconnect(workspace(), &Workspace::outputOrderChanged, this, &Xwayland::updatePrimary);
289
290 delete m_xrandrEventsFilter;
291 m_xrandrEventsFilter = nullptr;
292
293 // If Xwayland has crashed, we must deactivate the socket notifier and ensure that no X11
294 // events will be dispatched before blocking; otherwise we will simply hang...
295 uninstallSocketNotifier();
296
297 m_dataBridge.reset();
298 m_selectionOwner.reset();
299
300 m_inputSpy.reset();
301 disconnect(options, &Options::xwaylandEavesdropsChanged, this, &Xwayland::refreshEavesdropping);
302
303 destroyX11Connection();
304}

306void Xwayland::handleXwaylandReady()
307{
308 if (!createX11Connection()) {
309 Q_EMIT errorOccurred();
310 return;
311 }
312
313 qCInfo(KWIN_XWL) << "Xwayland server started on display" << m_launcher->displayName();
314
315 // create selection owner for WM_S0 - magic X display number expected by XWayland
316 m_selectionOwner.reset(new KSelectionOwner("WM_S0", kwinApp()->x11Connection(), kwinApp()->x11RootWindow()));
317 connect(m_selectionOwner.get(), &KSelectionOwner::lostOwnership,
318 this, &Xwayland::handleSelectionLostOwnership);
319 connect(m_selectionOwner.get(), &KSelectionOwner::claimedOwnership,
320 this, &Xwayland::handleSelectionClaimedOwnership);
321 connect(m_selectionOwner.get(), &KSelectionOwner::failedToClaimOwnership,
322 this, &Xwayland::handleSelectionFailedToClaimOwnership);
323 m_selectionOwner->claim(true);
324
325 Cursor *mouseCursor = Cursors::self()->mouse();
326 if (mouseCursor) {
327 Xcb::defineCursor(kwinApp()->x11RootWindow(), mouseCursor->x11Cursor(Qt::ArrowCursor));
328 }
329
330 m_dataBridge = std::make_unique<DataBridge>();
331
332 connect(workspace(), &Workspace::outputOrderChanged, this, &Xwayland::updatePrimary);
333 updatePrimary();
334
335 Xcb::sync(); // Trigger possible errors, there's still a chance to abort
336
337 delete m_xrandrEventsFilter;
338 m_xrandrEventsFilter = new XrandrEventFilter(this);
339
340 refreshEavesdropping();
341 connect(options, &Options::xwaylandEavesdropsChanged, this, &Xwayland::refreshEavesdropping);
342}

344void Xwayland::refreshEavesdropping()
345{
346 if (!waylandServer()->seat()->keyboard()) {
347 return;
348 }
349
350 const bool enabled = options->xwaylandEavesdrops() != None;
351 if (enabled == bool(m_inputSpy)) {
352 if (m_inputSpy) {
353 m_inputSpy->setMode(options->xwaylandEavesdrops());
354 }
355 return;
356 }
357
358 if (enabled) {
359 m_inputSpy.reset(new XwaylandInputSpy);
360 input()->installInputEventSpy(m_inputSpy.get());
361 m_inputSpy->setMode(options->xwaylandEavesdrops());
362 } else {
363 input()->uninstallInputEventSpy(m_inputSpy.get());
364 m_inputSpy.reset();
365 }
366}

368void Xwayland::updatePrimary()
369{
370 if (workspace()->outputOrder().empty()) {
371 return;
372 }
373 Xcb::RandR::ScreenResources resources(kwinApp()->x11RootWindow());
374 xcb_randr_crtc_t *crtcs = resources.crtcs();
375 if (!crtcs) {
376 return;
377 }
378
379 Output *const primaryOutput = workspace()->outputOrder().front();
380 for (int i = 0; i < resources->num_crtcs; ++i) {
381 Xcb::RandR::CrtcInfo crtcInfo(crtcs[i], resources->config_timestamp);
382 const QRect geometry = crtcInfo.rect();
383 if (geometry.topLeft() == primaryOutput->geometry().topLeft()) {
384 auto outputs = crtcInfo.outputs();
385 if (outputs && crtcInfo->num_outputs > 0) {
386 qCDebug(KWIN_XWL) << "Setting primary" << primaryOutput << outputs[0];
387 xcb_randr_set_output_primary(kwinApp()->x11Connection(), kwinApp()->x11RootWindow(), outputs[0]);
388 break;
389 }
390 }
391 }

411bool Xwayland::createX11Connection()
412{
413 xcb_connection_t *connection = xcb_connect_to_fd(m_launcher->xcbConnectionFd(), nullptr);
414
415 const int errorCode = xcb_connection_has_error(connection);
416 if (errorCode) {
417 qCDebug(KWIN_XWL, "Failed to establish the XCB connection (error %d)", errorCode);
418 return false;
419 }
420
421 xcb_screen_t *screen = xcb_setup_roots_iterator(xcb_get_setup(connection)).data;
422 Q_ASSERT(screen);
423
424 m_app->setX11Connection(connection);
425 m_app->setX11RootWindow(screen->root);
426
427 m_app->createAtoms();
428 m_app->installNativeX11EventFilter();
429
430 installSocketNotifier();
431
432 // Note that it's very important to have valid x11RootWindow(), and atoms when the
433 // rest of kwin is notified about the new X11 connection.
434 Q_EMIT m_app->x11ConnectionChanged();
435
436 return true;
437}

439void Xwayland::destroyX11Connection()
440{
441 if (!m_app->x11Connection()) {
442 return;
443 }
444
445 Q_EMIT m_app->x11ConnectionAboutToBeDestroyed();
446
447 Xcb::setInputFocus(XCB_INPUT_FOCUS_POINTER_ROOT);
448 m_app->destroyAtoms();
449 m_app->removeNativeX11EventFilter();
450
451 xcb_disconnect(m_app->x11Connection());
452
453 m_app->setX11Connection(nullptr);
454 m_app->setX11RootWindow(XCB_WINDOW_NONE);
455
456 Q_EMIT m_app->x11ConnectionChanged();
457}
