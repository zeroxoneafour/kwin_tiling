57static int wl_socket_lock(struct wl_socket *socket)
58{
59 struct stat socket_stat;
60
61 snprintf(socket->lock_addr, sizeof socket->lock_addr, "%s%s", socket->addr.sun_path, LOCK_SUFFIX);
62
63 // differing from wl_display, we don't set O_CLOEXEC so that we can pass to the child
64 socket->fd_lock = open(socket->lock_addr, O_CREAT | O_RDWR, (S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP));
65
66 if (socket->fd_lock < 0) {
67 printf("unable to open lockfile %s check permissions\n", socket->lock_addr);
68 goto err;
69 }
70
71 if (flock(socket->fd_lock, LOCK_EX | LOCK_NB) < 0) {
72 printf("unable to lock lockfile %s, maybe another compositor is running\n", socket->lock_addr);
73 goto err_fd;
74 }
75
76 if (lstat(socket->addr.sun_path, &socket_stat) < 0) {
77 if (errno != ENOENT) {
78 printf("did not manage to stat file %s\n", socket->addr.sun_path);
79 goto err_fd;
80 }
81 } else if (socket_stat.st_mode & S_IWUSR || socket_stat.st_mode & S_IWGRP) {
82 unlink(socket->addr.sun_path);
83 }
84
85 return 0;
86err_fd:
87 close(socket->fd_lock);
88 socket->fd_lock = -1;
89err:
90 *socket->lock_addr = 0;
91 /* we did not set this value here, but without lock the
92 * socket won't be created anyway. This prevents the
93 * wl_socket_destroy from unlinking already existing socket
94 * created by other compositor */
95 *socket->addr.sun_path = 0;
96
97 return -1;
98}

100void wl_socket_destroy(struct wl_socket *s)
101{
102 if (s->addr.sun_path[0])
103 unlink(s->addr.sun_path);
104 if (s->fd >= 0)
105 close(s->fd);
106 if (s->lock_addr[0])
107 unlink(s->lock_addr);
108 if (s->fd_lock >= 0)
109 close(s->fd_lock);
110
111 free(s);
112}

124struct wl_socket *wl_socket_create()
125{
126 struct wl_socket *s;
127 int displayno = 0;
128 int name_size;
129
130 /* A reasonable number of maximum default sockets. If
131 * you need more than this, use the explicit add_socket API. */
132 const int MAX_DISPLAYNO = 32;
133 const char *runtime_dir = getenv("XDG_RUNTIME_DIR");
134 if (!runtime_dir) {
135 printf("XDG_RUNTIME_DIR not set");
136 return NULL;
137 }
138
139 s = wl_socket_alloc();
140 if (s == NULL)
141 return NULL;
142
143 do {
144 snprintf(s->display_name, sizeof s->display_name, "wayland-%d", displayno);
145 s->addr.sun_family = AF_LOCAL;
146 name_size = snprintf(s->addr.sun_path, sizeof s->addr.sun_path, "%s/%s", runtime_dir, s->display_name) + 1;
147 assert(name_size > 0);
148
149 if (name_size > (int)sizeof s->addr.sun_path) {
150 goto fail;
151 }
152
153 if (wl_socket_lock(s) < 0)
154 continue;
155
156 s->fd = socket(PF_LOCAL, SOCK_STREAM, 0);
157
158 int size = SUN_LEN(&s->addr);
159 int ret = bind(s->fd, (struct sockaddr*)&s->addr, size);
160 if (ret < 0) {
161 goto fail;
162 }
163 ret = listen(s->fd, 128);
164 if (ret < 0) {
165 goto fail;
166 }
167 return s;
168 } while (displayno++ < MAX_DISPLAYNO);
169
170fail:
171 wl_socket_destroy(s);
172 return NULL;
173}
