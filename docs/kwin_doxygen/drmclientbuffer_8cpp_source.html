33DrmClientBuffer::DrmClientBuffer(wl_resource *resource, DrmClientBufferIntegration *integration)
34 : ClientBuffer(resource, *new DrmClientBufferPrivate)
35{
36 Q_D(DrmClientBuffer);
37
38 EGLDisplay eglDisplay = integration->display()->eglDisplay();
39 eglQueryWaylandBufferWL(eglDisplay, resource, EGL_TEXTURE_FORMAT, &d->textureFormat);
40 eglQueryWaylandBufferWL(eglDisplay, resource, EGL_WIDTH, &d->width);
41 eglQueryWaylandBufferWL(eglDisplay, resource, EGL_HEIGHT, &d->height);
42
43 if (!eglQueryWaylandBufferWL(eglDisplay, resource, EGL_WAYLAND_Y_INVERTED_WL, &d->yInverted)) {
44 // If EGL_WAYLAND_Y_INVERTED_WL is unsupported, we must assume that the buffer is inverted.
45 d->yInverted = true;
46 }

78ClientBuffer *DrmClientBufferIntegration::createBuffer(::wl_resource *resource)
79{
80 EGLDisplay eglDisplay = display()->eglDisplay();
81 static bool resolved = false;
82 if (!resolved && eglDisplay != EGL_NO_DISPLAY) {
83 eglQueryWaylandBufferWL = (eglQueryWaylandBufferWL_func)eglGetProcAddress("eglQueryWaylandBufferWL");
84 resolved = true;
85 }
86
87 EGLint format;
88 if (eglQueryWaylandBufferWL(eglDisplay, resource, EGL_TEXTURE_FORMAT, &format)) {
89 return new DrmClientBuffer(resource, this);
90 }
91 return nullptr;
92}
