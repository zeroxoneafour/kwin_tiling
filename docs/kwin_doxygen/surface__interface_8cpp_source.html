32static QRegion map_helper(const QMatrix4x4 &matrix, const QRegion &region)
33{
34 QRegion result;
35 for (const QRect &rect : region) {
36 result += matrix.mapRect(QRectF(rect)).toAlignedRect();
37 }
38 return result;
39}

95bool SurfaceInterfacePrivate::raiseChild(SubSurfaceInterface *subsurface, SurfaceInterface *anchor)
96{
97 Q_ASSERT(subsurface->parentSurface() == q);
98
99 QList<SubSurfaceInterface *> *anchorList;
100 int anchorIndex;
101
102 pending.below.removeOne(subsurface);
103 pending.above.removeOne(subsurface);
104
105 if (anchor == q) {
106 // Pretend as if the parent surface were before the first child in the above list.
107 anchorList = &pending.above;
108 anchorIndex = -1;
109 } else if (anchorIndex = pending.above.indexOf(anchor->subSurface()); anchorIndex != -1) {
110 anchorList = &pending.above;
111 } else if (anchorIndex = pending.below.indexOf(anchor->subSurface()); anchorIndex != -1) {
112 anchorList = &pending.below;
113 } else {
114 return false; // The anchor belongs to other sub-surface tree.
115 }
116
117 anchorList->insert(anchorIndex + 1, subsurface);
118 pending.childrenChanged = true;
119 return true;
120}

173void SurfaceInterfacePrivate::installPointerConstraint(LockedPointerV1Interface *lock)
174{
175 Q_ASSERT(!lockedPointer);
176 Q_ASSERT(!confinedPointer);
177
178 lockedPointer = lock;
179
180 auto cleanUp = [this]() {
181 lockedPointer = nullptr;
182 QObject::disconnect(constrainsOneShotConnection);
183 constrainsOneShotConnection = QMetaObject::Connection();
184 QObject::disconnect(constrainsUnboundConnection);
185 constrainsUnboundConnection = QMetaObject::Connection();
186 Q_EMIT q->pointerConstraintsChanged();
187 };
188
189 if (lock->lifeTime() == LockedPointerV1Interface::LifeTime::OneShot) {
190 constrainsOneShotConnection = QObject::connect(lock, &LockedPointerV1Interface::lockedChanged, q, [this, cleanUp] {
191 if (lockedPointer->isLocked()) {
192 return;
193 }
194 cleanUp();
195 });
196 }
197 constrainsUnboundConnection = QObject::connect(lock, &LockedPointerV1Interface::destroyed, q, cleanUp);
198 Q_EMIT q->pointerConstraintsChanged();
199}

271 return;
272 }
273 pending.buffer = compositor->display()->clientBufferForResource(buffer);

415QMatrix4x4 SurfaceInterfacePrivate::buildSurfaceToBufferMatrix()
416{
417 // The order of transforms is reversed, i.e. the viewport transform is the first one.
418
419 QMatrix4x4 surfaceToBufferMatrix;
420
421 if (!current.buffer) {
422 return surfaceToBufferMatrix;
423 }
424
425 surfaceToBufferMatrix.scale(current.bufferScale, current.bufferScale);
426 surfaceToBufferMatrix.scale(scaleOverride, scaleOverride);
427
428 switch (current.bufferTransform) {
429 case KWin::Output::Transform::Normal:
430 case KWin::Output::Transform::Flipped:
431 break;
432 case KWin::Output::Transform::Rotated90:
433 case KWin::Output::Transform::Flipped90:
434 surfaceToBufferMatrix.translate(0, bufferSize.height() / current.bufferScale);
435 surfaceToBufferMatrix.rotate(-90, 0, 0, 1);
436 break;
437 case KWin::Output::Transform::Rotated180:
438 case KWin::Output::Transform::Flipped180:
439 surfaceToBufferMatrix.translate(bufferSize.width() / current.bufferScale, bufferSize.height() / current.bufferScale);
440 surfaceToBufferMatrix.rotate(-180, 0, 0, 1);
441 break;
442 case KWin::Output::Transform::Rotated270:
443 case KWin::Output::Transform::Flipped270:
444 surfaceToBufferMatrix.translate(bufferSize.width() / current.bufferScale, 0);
445 surfaceToBufferMatrix.rotate(-270, 0, 0, 1);
446 break;
447 }
448
449 switch (current.bufferTransform) {
450 case KWin::Output::Transform::Flipped:
451 case KWin::Output::Transform::Flipped180:
452 surfaceToBufferMatrix.translate(bufferSize.width() / current.bufferScale, 0);
453 surfaceToBufferMatrix.scale(-1, 1);
454 break;
455 case KWin::Output::Transform::Flipped90:
456 case KWin::Output::Transform::Flipped270:
457 surfaceToBufferMatrix.translate(bufferSize.height() / current.bufferScale, 0);
458 surfaceToBufferMatrix.scale(-1, 1);
459 break;
460 default:
461 break;
462 }
463
464 if (current.viewport.sourceGeometry.isValid()) {
465 surfaceToBufferMatrix.translate(current.viewport.sourceGeometry.x(), current.viewport.sourceGeometry.y());
466 }
467
468 QSizeF sourceSize;
469 if (current.viewport.sourceGeometry.isValid()) {
470 sourceSize = current.viewport.sourceGeometry.size();
471 } else {
472 sourceSize = implicitSurfaceSize;
473 }
474
475 if (sourceSize != surfaceSize) {
476 surfaceToBufferMatrix.scale(sourceSize.width() / surfaceSize.width(), sourceSize.height() / surfaceSize.height());
477 }
478
479 return surfaceToBufferMatrix;
480}

553void SurfaceInterfacePrivate::applyState(SurfaceState *next)
554{
555 const bool bufferChanged = next->bufferIsSet;
556 const bool opaqueRegionChanged = next->opaqueIsSet;
557 const bool scaleFactorChanged = next->bufferScaleIsSet && (current.bufferScale != next->bufferScale);
558 const bool transformChanged = next->bufferTransformIsSet && (current.bufferTransform != next->bufferTransform);
559 const bool shadowChanged = next->shadowIsSet;
560 const bool blurChanged = next->blurIsSet;
561 const bool contrastChanged = next->contrastIsSet;
562 const bool slideChanged = next->slideIsSet;
563 const bool childrenChanged = next->childrenChanged;
564 const bool visibilityChanged = bufferChanged && bool(current.buffer) != bool(next->buffer);
565
566 const QSizeF oldSurfaceSize = surfaceSize;
567 const QSize oldBufferSize = bufferSize;
568 const QMatrix4x4 oldSurfaceToBufferMatrix = surfaceToBufferMatrix;
569 const QRegion oldInputRegion = inputRegion;
570
571 next->mergeInto(&current);
572 scaleOverride = pendingScaleOverride;
573
574 if (lockedPointer) {
575 auto lockedPointerPrivate = LockedPointerV1InterfacePrivate::get(lockedPointer);
576 lockedPointerPrivate->commit();
577 }
578 if (confinedPointer) {
579 auto confinedPointerPrivate = ConfinedPointerV1InterfacePrivate::get(confinedPointer);
580 confinedPointerPrivate->commit();
581 }
582
583 if (bufferRef != current.buffer) {
584 if (bufferRef) {
585 bufferRef->unref();
586 }
587 bufferRef = current.buffer;
588 if (bufferRef) {
589 bufferRef->ref();
590 }
591 }
592
593 // TODO: Refactor the state management code because it gets more clumsy.
594 if (current.buffer) {
595 bufferSize = current.buffer->size();
596
597 implicitSurfaceSize = current.buffer->size() / current.bufferScale;
598 switch (current.bufferTransform) {
599 case KWin::Output::Transform::Rotated90:
600 case KWin::Output::Transform::Rotated270:
601 case KWin::Output::Transform::Flipped90:
602 case KWin::Output::Transform::Flipped270:
603 implicitSurfaceSize.transpose();
604 break;
605 case KWin::Output::Transform::Normal:
606 case KWin::Output::Transform::Rotated180:
607 case KWin::Output::Transform::Flipped:
608 case KWin::Output::Transform::Flipped180:
609 break;
610 }
611
612 if (current.viewport.destinationSize.isValid()) {
613 surfaceSize = current.viewport.destinationSize;
614 } else if (current.viewport.sourceGeometry.isValid()) {
615 surfaceSize = current.viewport.sourceGeometry.size().toSize();
616 } else {
617 surfaceSize = implicitSurfaceSize;
618 }
619
620 const QRectF surfaceRect(QPoint(0, 0), surfaceSize);
621 inputRegion = current.input & surfaceRect.toAlignedRect();
622
623 if (!current.buffer->hasAlphaChannel()) {
624 opaqueRegion = surfaceRect.toAlignedRect();
625 } else {
626 opaqueRegion = current.opaque & surfaceRect.toAlignedRect();
627 }
628
629 QMatrix4x4 scaleOverrideMatrix;
630 if (scaleOverride != 1.) {
631 scaleOverrideMatrix.scale(1. / scaleOverride, 1. / scaleOverride);
632 }
633
634 opaqueRegion = map_helper(scaleOverrideMatrix, opaqueRegion);
635 inputRegion = map_helper(scaleOverrideMatrix, inputRegion);
636 surfaceSize = surfaceSize / scaleOverride;
637 implicitSurfaceSize = implicitSurfaceSize / scaleOverride;
638 } else {
639 surfaceSize = QSizeF(0, 0);
640 implicitSurfaceSize = QSizeF(0, 0);
641 bufferSize = QSize(0, 0);
642 inputRegion = QRegion();
643 opaqueRegion = QRegion();
644 }
645
646 surfaceToBufferMatrix = buildSurfaceToBufferMatrix();
647 bufferToSurfaceMatrix = surfaceToBufferMatrix.inverted();
648 if (opaqueRegionChanged) {
649 Q_EMIT q->opaqueChanged(opaqueRegion);
650 }
651 if (oldInputRegion != inputRegion) {
652 Q_EMIT q->inputChanged(inputRegion);
653 }
654 if (scaleFactorChanged) {
655 Q_EMIT q->bufferScaleChanged(current.bufferScale);
656 }
657 if (transformChanged) {
658 Q_EMIT q->bufferTransformChanged(current.bufferTransform);
659 }
660 if (visibilityChanged) {
661 updateEffectiveMapped();
662 }
663 if (bufferChanged) {
664 if (current.buffer && (!current.damage.isEmpty() || !current.bufferDamage.isEmpty())) {
665 const QRegion windowRegion = QRegion(0, 0, q->size().width(), q->size().height());
666 const QRegion bufferDamage = q->mapFromBuffer(current.bufferDamage);
667 current.damage = windowRegion.intersected(current.damage.united(bufferDamage));
668 Q_EMIT q->damaged(current.damage);
669 }
670 }
671 if (surfaceToBufferMatrix != oldSurfaceToBufferMatrix) {
672 Q_EMIT q->surfaceToBufferMatrixChanged();
673 }
674 if (bufferSize != oldBufferSize) {
675 Q_EMIT q->bufferSizeChanged();
676 }
677 if (surfaceSize != oldSurfaceSize) {
678 Q_EMIT q->sizeChanged();
679 }
680 if (shadowChanged) {
681 Q_EMIT q->shadowChanged();
682 }
683 if (blurChanged) {
684 Q_EMIT q->blurChanged();
685 }
686 if (contrastChanged) {
687 Q_EMIT q->contrastChanged();
688 }
689 if (slideChanged) {
690 Q_EMIT q->slideOnShowHideChanged();
691 }
692 if (childrenChanged) {
693 Q_EMIT q->childSubSurfacesChanged();
694 }
695 // The position of a sub-surface is applied when its parent is committed.
696 for (SubSurfaceInterface *subsurface : std::as_const(current.below)) {
697 auto subsurfacePrivate = SubSurfaceInterfacePrivate::get(subsurface);
698 subsurfacePrivate->parentCommit();
699 }
700 for (SubSurfaceInterface *subsurface : std::as_const(current.above)) {
701 auto subsurfacePrivate = SubSurfaceInterfacePrivate::get(subsurface);
702 subsurfacePrivate->parentCommit();
703 }
704 if (role) {
705 role->commit();
706 }
707 Q_EMIT q->committed();
708}

747void SurfaceInterfacePrivate::updateEffectiveMapped()
748{
749 const bool effectiveMapped = computeEffectiveMapped();
750 if (mapped == effectiveMapped) {
751 return;
752 }
753
754 mapped = effectiveMapped;
755
756 if (mapped) {
757 Q_EMIT q->mapped();
758 } else {
759 Q_EMIT q->unmapped();
760 }

903void SurfaceInterface::setOutputs(const QVector<OutputInterface *> &outputs)
904{
905 QVector<OutputInterface *> removedOutputs = d->outputs;
906 for (auto it = outputs.constBegin(), end = outputs.constEnd(); it != end; ++it) {
907 const auto o = *it;
908 removedOutputs.removeOne(o);
909 }
910 for (auto it = removedOutputs.constBegin(), end = removedOutputs.constEnd(); it != end; ++it) {
911 const auto resources = (*it)->clientResources(client());
912 for (wl_resource *outputResource : resources) {
913 d->send_leave(outputResource);
914 }
915 disconnect(d->outputDestroyedConnections.take(*it));
916 disconnect(d->outputBoundConnections.take(*it));
917 }
918 QVector<OutputInterface *> addedOutputsOutputs = outputs;
919 for (auto it = d->outputs.constBegin(), end = d->outputs.constEnd(); it != end; ++it) {
920 const auto o = *it;
921 addedOutputsOutputs.removeOne(o);
922 }
923 for (auto it = addedOutputsOutputs.constBegin(), end = addedOutputsOutputs.constEnd(); it != end; ++it) {
924 const auto o = *it;
925 const auto resources = o->clientResources(client());
926 for (wl_resource *outputResource : resources) {
927 d->send_enter(outputResource);
928 }
929 d->outputDestroyedConnections[o] = connect(o, &OutputInterface::removed, this, [this, o] {
930 auto outputs = d->outputs;
931 if (outputs.removeOne(o)) {
932 setOutputs(outputs);
933 }
934 });
935
936 Q_ASSERT(!d->outputBoundConnections.contains(o));
937 d->outputBoundConnections[o] = connect(o, &OutputInterface::bound, this, [this](ClientConnection *c, wl_resource *outputResource) {
938 if (c != client()) {
939 return;
940 }
941 d->send_enter(outputResource);
942 });
943 }
944
945 d->outputs = outputs;
946 for (auto child : std::as_const(d->current.below)) {
947 child->surface()->setOutputs(outputs);
948 }
949 for (auto child : std::as_const(d->current.above)) {
950 child->surface()->setOutputs(outputs);
951 }
952}

954SurfaceInterface *SurfaceInterface::surfaceAt(const QPointF &position)
955{
956 if (!isMapped()) {
957 return nullptr;
958 }
959
960 for (auto it = d->current.above.crbegin(); it != d->current.above.crend(); ++it) {
961 const SubSurfaceInterface *current = *it;
962 SurfaceInterface *surface = current->surface();
963 if (auto s = surface->surfaceAt(position - current->position())) {
964 return s;
965 }
966 }

974 const SubSurfaceInterface *current = *it;
