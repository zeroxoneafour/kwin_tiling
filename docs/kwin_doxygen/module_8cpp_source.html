54void Module::importScript()
55{
56 QString path = QFileDialog::getOpenFileName(nullptr, i18n("Import KWin Script"), QDir::homePath(),
57 i18n("*.kwinscript|KWin scripts (*.kwinscript)"));
58
59 if (path.isNull()) {
60 return;
61 }
62
63 using namespace KPackage;
64 PackageStructure *structure = PackageLoader::self()->loadPackageStructure(QStringLiteral("KWin/Script"));
65 Package package(structure);
66
67 KJob *installJob = package.update(path);
68 installJob->setProperty("packagePath", path); // so we can retrieve it later for showing the script's name
69 connect(installJob, &KJob::result, this, &Module::importScriptInstallFinished);
70}

91void Module::importScriptInstallFinished(KJob *job)
92{
93 // if the applet is already installed, just add it to the containment
94 if (job->error() != KJob::NoError) {
95 setErrorMessage(i18nc("Placeholder is error message returned from the install service", "Cannot import selected script.\n%1", job->errorString()));
96 return;
97 }
98
99 using namespace KPackage;
100
101 // so we can show the name of the package we just imported
102 PackageStructure *structure = PackageLoader::self()->loadPackageStructure(QStringLiteral("KWin/Script"));
103 Package package(structure);
104 package.setPath(job->property("packagePath").toString());
105 Q_ASSERT(package.isValid());
106
107 m_infoMessage = i18nc("Placeholder is name of the script that was imported", "The script \"%1\" was successfully imported.", package.metadata().name());
108 m_errorMessage.clear();
109 Q_EMIT messageChanged();
110
111 m_model->clear();
112 m_model->addPlugins(m_kwinScriptsData->pluginMetaDataList(), QString());
113
114 setNeedsSave(false);
115}

136void Module::save()
137{
138 using namespace KPackage;
139 PackageStructure *structure = PackageLoader::self()->loadPackageStructure(QStringLiteral("KWin/Script"));
140 for (const KPluginMetaData &info : std::as_const(m_pendingDeletions)) {
141 // We can get the package root from the entry path
142 QDir root = QFileInfo(info.metaDataFileName()).dir();
143 root.cdUp();
144 KJob *uninstallJob = Package(structure).uninstall(info.pluginId(), root.absolutePath());
145 connect(uninstallJob, &KJob::result, this, [this, uninstallJob]() {
146 if (!uninstallJob->errorString().isEmpty()) {
147 setErrorMessage(i18n("Error when uninstalling KWin Script: %1", uninstallJob->errorString()));
148 } else {
149 load(); // Make sure to reload the KCM to deleted entries to disappear
150 }
151 });
152 }
153
154 m_infoMessage.clear();
155 Q_EMIT messageChanged();
156 m_pendingDeletions.clear();
157 Q_EMIT pendingDeletionsChanged();
158
159 m_model->save();
160 QDBusMessage message = QDBusMessage::createMethodCall("org.kde.KWin", "/Scripting", "org.kde.kwin.Scripting", "start");
161 QDBusConnection::sessionBus().asyncCall(message);
162
163 setNeedsSave(false);
164}
