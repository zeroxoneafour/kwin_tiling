45DrmOutput::DrmOutput(const std::shared_ptr<DrmConnector> &conn)
46 : DrmAbstractOutput(conn->gpu())
47 , m_pipeline(conn->pipeline())
48 , m_connector(conn)
49{
50 RenderLoopPrivate::get(m_renderLoop.get())->canDoTearing = gpu()->asyncPageflipSupported();
51 m_pipeline->setOutput(this);
52 m_renderLoop->setRefreshRate(m_pipeline->mode()->refreshRate());
53
54 Capabilities capabilities = Capability::Dpms;
55 State initialState;
56
57 if (conn->hasOverscan()) {
58 capabilities |= Capability::Overscan;
59 initialState.overscan = conn->overscan();
60 }
61 if (conn->vrrCapable()) {
62 capabilities |= Capability::Vrr;
63 setVrrPolicy(RenderLoop::VrrPolicy::Automatic);
64 }
65 if (conn->hasRgbRange()) {
66 capabilities |= Capability::RgbRange;
67 initialState.rgbRange = conn->rgbRange();
68 }
69
70 const Edid *edid = conn->edid();
71
72 setInformation(Information{
73 .name = conn->connectorName(),
74 .manufacturer = edid->manufacturerString(),
75 .model = conn->modelName(),
76 .serialNumber = edid->serialNumber(),
77 .eisaId = edid->eisaId(),
78 .physicalSize = conn->physicalSize(),
79 .edid = edid->raw(),
80 .subPixel = conn->subpixel(),
81 .capabilities = capabilities,
82 .panelOrientation = DrmConnector::toKWinTransform(conn->panelOrientation()),
83 .internal = conn->isInternal(),
84 .nonDesktop = conn->isNonDesktop(),
85 });
86
87 initialState.modes = getModes();
88 initialState.currentMode = m_pipeline->mode();
89 if (!initialState.currentMode) {
90 initialState.currentMode = initialState.modes.constFirst();
91 }
92
93 setState(initialState);
94
95 m_turnOffTimer.setSingleShot(true);
96 m_turnOffTimer.setInterval(dimAnimationTime());
97 connect(&m_turnOffTimer, &QTimer::timeout, this, [this] {
98 setDrmDpmsMode(DpmsMode::Off);
99 });

138bool DrmOutput::setCursor(CursorSource *source)
139{
140 static bool valid;
141 static const bool forceSoftwareCursor = qEnvironmentVariableIntValue("KWIN_FORCE_SW_CURSOR", &valid) == 1 && valid;
142 // hardware cursors are broken with the NVidia proprietary driver
143 if (forceSoftwareCursor || (!valid && m_gpu->isNVidia())) {
144 m_setCursorSuccessful = false;
145 return false;
146 }
147 const auto layer = m_pipeline->cursorLayer();
148 if (!m_pipeline->crtc() || !layer) {
149 return false;
150 }
151 m_cursor.source = source;
152 if (!m_cursor.source || m_cursor.source->size().isEmpty()) {
153 if (layer->isVisible()) {
154 layer->setVisible(false);
155 m_pipeline->setCursor();
156 }
157 return true;
158 }
159 bool rendered = false;
160 const QMatrix4x4 monitorMatrix = logicalToNativeMatrix(rect(), scale(), transform());
161 const QSizeF cursorSize = m_cursor.source->size();
162 const QRectF cursorRect = QRectF(m_cursor.position, cursorSize);
163 const QRectF nativeCursorRect = monitorMatrix.mapRect(cursorRect);
164 if (nativeCursorRect.width() <= m_gpu->cursorSize().width() && nativeCursorRect.height() <= m_gpu->cursorSize().height()) {
165 if (auto beginInfo = layer->beginFrame()) {
166 RenderTarget *renderTarget = &beginInfo->renderTarget;
167 renderTarget->setDevicePixelRatio(scale());
168
169 RenderLayer renderLayer(m_renderLoop.get());
170 renderLayer.setDelegate(std::make_unique<SceneDelegate>(Compositor::self()->cursorScene()));
171
172 renderLayer.delegate()->prePaint();
173 renderLayer.delegate()->paint(renderTarget, infiniteRegion());
174 renderLayer.delegate()->postPaint();
175
176 rendered = layer->endFrame(infiniteRegion(), infiniteRegion());
177 }
178 }
179 if (!rendered) {
180 if (layer->isVisible()) {
181 layer->setVisible(false);
182 m_pipeline->setCursor();
183 }
184 m_setCursorSuccessful = false;
185 return false;
186 }
187
188 const QSize layerSize = m_gpu->cursorSize() / scale();
189 const QRectF layerRect = monitorMatrix.mapRect(QRectF(m_cursor.position, layerSize));
190 layer->setVisible(cursorRect.intersects(rect()));
191 if (layer->isVisible()) {
192 m_setCursorSuccessful = m_pipeline->setCursor(logicalToNativeMatrix(QRectF(QPoint(), layerRect.size()), scale(), transform()).map(m_cursor.source->hotspot()).toPoint());
193 layer->setVisible(m_setCursorSuccessful);
194 }
195 return m_setCursorSuccessful;
196}

198bool DrmOutput::moveCursor(const QPointF &position)
199{
200 if (!m_setCursorSuccessful || !m_pipeline->crtc()) {
201 return false;
202 }
203 m_cursor.position = position;
204
205 const QSizeF cursorSize = m_cursor.source ? m_cursor.source->size() : QSize(0, 0);
206 const QRectF cursorRect = QRectF(m_cursor.position, cursorSize);
207
208 if (!cursorRect.intersects(rect())) {
209 const auto layer = m_pipeline->cursorLayer();
210 if (layer->isVisible()) {
211 layer->setVisible(false);
212 m_pipeline->setCursor();
213 }
214 return true;
215 }
216 const QMatrix4x4 monitorMatrix = logicalToNativeMatrix(rect(), scale(), transform());
217 const QSizeF layerSize = m_gpu->cursorSize() / scale();
218 const QRectF layerRect = monitorMatrix.mapRect(QRectF(m_cursor.position, layerSize));
219 const auto layer = m_pipeline->cursorLayer();
220 const bool wasVisible = layer->isVisible();
221 layer->setVisible(true);
222 layer->setPosition(layerRect.topLeft().toPoint());
223 m_moveCursorSuccessful = m_pipeline->moveCursor();
224 layer->setVisible(m_moveCursorSuccessful);
225 if (!m_moveCursorSuccessful || !wasVisible) {
226 m_pipeline->setCursor();
227 }
228 return m_moveCursorSuccessful;
229}

243void DrmOutput::setDpmsMode(DpmsMode mode)
244{
245 if (mode == DpmsMode::Off) {
246 if (!m_turnOffTimer.isActive()) {
247 Q_EMIT aboutToTurnOff(std::chrono::milliseconds(m_turnOffTimer.interval()));
248 m_turnOffTimer.start();
249 }
250 if (isEnabled()) {
251 m_gpu->platform()->createDpmsFilter();
252 }
253 } else {
254 m_gpu->platform()->checkOutputsAreOn();
255 if (m_turnOffTimer.isActive() || (mode != dpmsMode() && setDrmDpmsMode(mode))) {
256 Q_EMIT wakeUp();
257 }
258 m_turnOffTimer.stop();
259 }

262bool DrmOutput::setDrmDpmsMode(DpmsMode mode)
263{
264 if (!isEnabled()) {
265 return false;
266 }
267 bool active = mode == DpmsMode::On;
268 bool isActive = dpmsMode() == DpmsMode::On;
269 if (active == isActive) {
270 updateDpmsMode(mode);
271 return true;
272 }
273 m_pipeline->setActive(active);
274 if (DrmPipeline::commitPipelines({m_pipeline}, active ? DrmPipeline::CommitMode::TestAllowModeset : DrmPipeline::CommitMode::CommitModeset) == DrmPipeline::Error::None) {
275 m_pipeline->applyPendingChanges();
276 updateDpmsMode(mode);
277 if (active) {
278 m_gpu->platform()->checkOutputsAreOn();
279 m_renderLoop->uninhibit();
280 m_renderLoop->scheduleRepaint();
281 } else {
282 m_renderLoop->inhibit();
283 m_gpu->platform()->createDpmsFilter();
284 }
285 return true;
286 } else {
287 qCWarning(KWIN_DRM) << "Setting dpms mode failed!";
288 m_pipeline->revertPendingChanges();
289 if (isEnabled() && isActive && !active) {
290 m_gpu->platform()->checkOutputsAreOn();
291 }
292 return false;
293 }

296DrmPlane::Transformations outputToPlaneTransform(DrmOutput::Transform transform)
297{
298 using OutTrans = DrmOutput::Transform;
299 using PlaneTrans = DrmPlane::Transformation;
300
301 // TODO: Do we want to support reflections (flips)?
302
303 switch (transform) {
304 case OutTrans::Normal:
305 case OutTrans::Flipped:
306 return PlaneTrans::Rotate0;
307 case OutTrans::Rotated90:
308 case OutTrans::Flipped90:
309 return PlaneTrans::Rotate90;
310 case OutTrans::Rotated180:
311 case OutTrans::Flipped180:
312 return PlaneTrans::Rotate180;
313 case OutTrans::Rotated270:
314 case OutTrans::Flipped270:
315 return PlaneTrans::Rotate270;
316 default:
317 Q_UNREACHABLE();
318 }
319}

321void DrmOutput::updateModes()
322{
323 State next = m_state;
324 next.modes = getModes();
325
326 if (m_pipeline->crtc()) {
327 const auto currentMode = m_pipeline->connector()->findMode(m_pipeline->crtc()->queryCurrentMode());
328 if (currentMode != m_pipeline->mode()) {
329 // DrmConnector::findCurrentMode might fail
330 m_pipeline->setMode(currentMode ? currentMode : m_pipeline->connector()->modes().constFirst());
331 if (m_gpu->testPendingConfiguration() == DrmPipeline::Error::None) {
332 m_pipeline->applyPendingChanges();
333 m_renderLoop->setRefreshRate(m_pipeline->mode()->refreshRate());
334 } else {
335 qCWarning(KWIN_DRM) << "Setting changed mode failed!";
336 m_pipeline->revertPendingChanges();
337 }
338 }
339 }
340
341 next.currentMode = m_pipeline->mode();
342 if (!next.currentMode) {
343 next.currentMode = next.modes.constFirst();
344 }
345
346 setState(next);
347}

356bool DrmOutput::present()
357{
358 RenderLoopPrivate *renderLoopPrivate = RenderLoopPrivate::get(m_renderLoop.get());
359 const auto type = DrmConnector::kwinToDrmContentType(contentType());
360 if (m_pipeline->syncMode() != renderLoopPrivate->presentMode || type != m_pipeline->contentType()) {
361 m_pipeline->setSyncMode(renderLoopPrivate->presentMode);
362 m_pipeline->setContentType(type);
363 if (DrmPipeline::commitPipelines({m_pipeline}, DrmPipeline::CommitMode::Test) == DrmPipeline::Error::None) {
364 m_pipeline->applyPendingChanges();
365 } else {
366 m_pipeline->revertPendingChanges();
367 }
368 }
369 const bool needsModeset = gpu()->needsModeset();
370 bool success;
371 if (needsModeset) {
372 success = m_pipeline->maybeModeset();
373 } else {
374 DrmPipeline::Error err = m_pipeline->present();
375 success = err == DrmPipeline::Error::None;
376 if (err == DrmPipeline::Error::InvalidArguments) {
377 QTimer::singleShot(0, m_gpu->platform(), &DrmBackend::updateOutputs);
378 }
379 }
380 if (success) {
381 Q_EMIT outputChange(m_pipeline->primaryLayer()->currentDamage());
382 return true;
383 } else if (!needsModeset) {
384 qCWarning(KWIN_DRM) << "Presentation failed!" << strerror(errno);
385 frameFailed();
386 }
387 return false;
388}

400bool DrmOutput::queueChanges(const OutputConfiguration &config)
401{
402 static bool valid;
403 static int envOnlySoftwareRotations = qEnvironmentVariableIntValue("KWIN_DRM_SW_ROTATIONS_ONLY", &valid) == 1 || !valid;
404
405 const auto props = config.constChangeSet(this);
406 const auto mode = props->mode.lock();
407 if (!mode) {
408 return false;
409 }
410 m_pipeline->setMode(std::static_pointer_cast<DrmConnectorMode>(mode));
411 m_pipeline->setOverscan(props->overscan);
412 m_pipeline->setRgbRange(props->rgbRange);
413 m_pipeline->setRenderOrientation(outputToPlaneTransform(props->transform));
414 if (!envOnlySoftwareRotations && m_gpu->atomicModeSetting()) {
415 m_pipeline->setBufferOrientation(m_pipeline->renderOrientation());
416 }
417 m_pipeline->setEnable(props->enabled);
418 return true;
419}

421void DrmOutput::applyQueuedChanges(const OutputConfiguration &config)
422{
423 if (!m_connector->isConnected()) {
424 return;
425 }
426 Q_EMIT aboutToChange();
427 m_pipeline->applyPendingChanges();
428
429 auto props = config.constChangeSet(this);
430
431 State next = m_state;
432 next.enabled = props->enabled && m_pipeline->crtc();
433 next.position = props->pos;
434 next.scale = props->scale;
435 next.transform = props->transform;
436 next.currentMode = m_pipeline->mode();
437 next.overscan = m_pipeline->overscan();
438 next.rgbRange = m_pipeline->rgbRange();
439
440 setState(next);
441 setVrrPolicy(props->vrrPolicy);
442
443 if (!isEnabled() && m_pipeline->needsModeset()) {
444 m_gpu->maybeModeset();
445 }
446
447 m_renderLoop->setRefreshRate(refreshRate());
448 m_renderLoop->scheduleRepaint();
449
450 Q_EMIT changed();
451
452 if (isEnabled() && dpmsMode() == DpmsMode::On) {
453 m_gpu->platform()->turnOutputsOn();
454 }
455}
