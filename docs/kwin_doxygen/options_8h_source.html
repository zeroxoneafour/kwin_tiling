21// Whether to keep all windows mapped when compositing (i.e. whether to have
22// actively updated window pixmaps).
23enum HiddenPreviews {
24 // The normal mode with regard to mapped windows. Hidden (minimized, etc.)
25 // and windows on inactive virtual desktops are not mapped, their pixmaps
26 // are only their icons.
27 HiddenPreviewsNever,
28 // Like normal mode, but shown windows (i.e. on inactive virtual desktops)
29 // are kept mapped, only hidden windows are unmapped.
30 HiddenPreviewsShown,
31 // All windows are kept mapped regardless of their state.
32 HiddenPreviewsAlways
33};
