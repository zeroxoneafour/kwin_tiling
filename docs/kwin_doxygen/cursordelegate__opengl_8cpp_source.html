25void CursorDelegateOpenGL::paint(RenderTarget *renderTarget, const QRegion &region)
26{
27 if (!region.intersects(layer()->mapToGlobal(layer()->rect()).toAlignedRect())) {
28 return;
29 }
30
31 // Render the cursor scene in an offscreen render target.
32 const QSize bufferSize = (Cursors::self()->currentCursor()->rect().size() * renderTarget->devicePixelRatio()).toSize();
33 if (!m_texture || m_texture->size() != bufferSize) {
34 m_texture = std::make_unique<GLTexture>(GL_RGBA8, bufferSize);
35 m_framebuffer = std::make_unique<GLFramebuffer>(m_texture.get());
36 }
37
38 RenderTarget offscreenRenderTarget(m_framebuffer.get());
39 offscreenRenderTarget.setDevicePixelRatio(renderTarget->devicePixelRatio());
40
41 RenderLayer renderLayer(layer()->loop());
42 renderLayer.setDelegate(std::make_unique<SceneDelegate>(Compositor::self()->cursorScene()));
43 renderLayer.delegate()->prePaint();
44 renderLayer.delegate()->paint(&offscreenRenderTarget, infiniteRegion());
45 renderLayer.delegate()->postPaint();
46
47 // Show the rendered cursor scene on the screen.
48 const QRectF cursorRect = layer()->mapToGlobal(layer()->rect());
49 const qreal scale = renderTarget->devicePixelRatio();
50
51 QMatrix4x4 mvp;
52 mvp.ortho(QRect(QPoint(0, 0), renderTarget->size()));
53 mvp.translate(std::round(cursorRect.x() * scale), std::round(cursorRect.y() * scale));
54
55 GLFramebuffer *fbo = std::get<GLFramebuffer *>(renderTarget->nativeHandle());
56 GLFramebuffer::pushFramebuffer(fbo);
57
58 // Don't need to call GLVertexBuffer::beginFrame() and GLVertexBuffer::endOfFrame() because
59 // the GLVertexBuffer::streamingBuffer() is not being used when painting cursor.
60 glEnable(GL_BLEND);
61 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
62
63 m_texture->bind();
64 ShaderBinder binder(ShaderTrait::MapTexture);
65 binder.shader()->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
66 m_texture->render(region, cursorRect.size(), scale);
67 m_texture->unbind();
68 glDisable(GL_BLEND);
69
70 GLFramebuffer::popFramebuffer();
71}
