93void TestScreencastV1Interface::initTestCase()
94{
95 delete m_display;
96 m_display = new KWaylandServer::Display(this);
97 m_display->addSocketName(s_socketName);
98 m_display->start();
99 QVERIFY(m_display->isRunning());
100
101 // setup connection
102 m_connection = new KWayland::Client::ConnectionThread;
103 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
104 m_connection->setSocketName(s_socketName);
105
106 m_thread = new QThread(this);
107 m_connection->moveToThread(m_thread);
108 m_thread->start();
109
110 m_connection->initConnection();
111 QVERIFY(connectedSpy.wait());
112
113 m_queue = new KWayland::Client::EventQueue(this);
114 QVERIFY(!m_queue->isValid());
115 m_queue->setup(m_connection);
116 QVERIFY(m_queue->isValid());
117
118 KWayland::Client::Registry registry;
119
120 QSignalSpy screencastSpy(&registry, &KWayland::Client::Registry::interfacesAnnounced);
121 m_screencastInterface = new KWaylandServer::ScreencastV1Interface(m_display, this);
122 connect(m_screencastInterface,
123 &KWaylandServer::ScreencastV1Interface::windowScreencastRequested,
124 this,
125 [this](KWaylandServer::ScreencastStreamV1Interface *stream, const QString &winid) {
126 stream->sendCreated(123);
127 m_triggered = stream;
128 });
129
130 connect(&registry,
131 &KWayland::Client::Registry::interfaceAnnounced,
132 this,
133 [this, &registry](const QByteArray &interfaceName, quint32 name, quint32 version) {
134 if (interfaceName != "zkde_screencast_unstable_v1")
135 return;
136 m_screencast = new ScreencastV1(this);
137 m_screencast->init(&*registry, name, version);
138 });
139 registry.setEventQueue(m_queue);
140 registry.create(m_connection->display());
141 QVERIFY(registry.isValid());
142 registry.setup();
143 wl_display_flush(m_connection->display());
144
145 QVERIFY(m_screencastInterface);
146 QVERIFY(m_screencast || screencastSpy.wait());
147 QVERIFY(m_screencast);
148}
