22OffscreenSurface::OffscreenSurface(QOffscreenSurface *surface)
23 : QPlatformOffscreenSurface(surface)
24 , m_eglDisplay(kwinApp()->outputBackend()->sceneEglDisplay())
25{
26 const QSize size = surface->size();
27
28 EGLConfig config = configFromFormat(m_eglDisplay, surface->requestedFormat(), EGL_PBUFFER_BIT);
29 if (config == EGL_NO_CONFIG_KHR) {
30 return;
31 }
32
33 const EGLint attributes[] = {
34 EGL_WIDTH, size.width(),
35 EGL_HEIGHT, size.height(),
36 EGL_NONE};
37
38 m_surface = eglCreatePbufferSurface(m_eglDisplay, config, attributes);
39 if (m_surface == EGL_NO_SURFACE) {
40 return;
41 }
42
43 // Requested and actual surface format might be different.
44 m_format = formatFromConfig(m_eglDisplay, config);
45}
