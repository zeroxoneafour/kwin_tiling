271static inline bool exclusiveContains(const QRectF &rect, const QPointF &point)
272{
273 return point.x() >= rect.x() && point.y() >= rect.y() && point.x() < (rect.x() + rect.width()) && point.y() < (rect.y() + rect.height());
274}
