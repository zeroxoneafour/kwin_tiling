82void TestWaylandSurface::init()
83{
84 using namespace KWaylandServer;
85 delete m_display;
86 m_display = new KWaylandServer::Display(this);
87 m_display->addSocketName(s_socketName);
88 m_display->start();
89 QVERIFY(m_display->isRunning());
90 m_display->createShm();
91
92 m_compositorInterface = new CompositorInterface(m_display, m_display);
93 QVERIFY(m_compositorInterface);
94
95 m_idleInhibitInterface = new IdleInhibitManagerV1Interface(m_display, m_display);
96 QVERIFY(m_idleInhibitInterface);
97
98 // setup connection
99 m_connection = new KWayland::Client::ConnectionThread;
100 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
101 m_connection->setSocketName(s_socketName);
102
103 m_thread = new QThread(this);
104 m_connection->moveToThread(m_thread);
105 m_thread->start();
106
107 /*connect(QCoreApplication::eventDispatcher(), &QAbstractEventDispatcher::aboutToBlock, m_connection,
108 [this]() {
109 if (m_connection->display()) {
110 wl_display_flush(m_connection->display());
111 }
112 }
113 );*/
114
115 m_connection->initConnection();
116 QVERIFY(connectedSpy.wait());
117
118 m_queue = new KWayland::Client::EventQueue(this);
119 QVERIFY(!m_queue->isValid());
120 m_queue->setup(m_connection);
121 QVERIFY(m_queue->isValid());
122
123 KWayland::Client::Registry registry;
124 registry.setEventQueue(m_queue);
125 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
126 QSignalSpy shmSpy(&registry, &KWayland::Client::Registry::shmAnnounced);
127 QSignalSpy allAnnounced(&registry, &KWayland::Client::Registry::interfacesAnnounced);
128 registry.create(m_connection->display());
129 QVERIFY(registry.isValid());
130 registry.setup();
131 QVERIFY(allAnnounced.wait());
132 QVERIFY(!compositorSpy.isEmpty());
133 QVERIFY(!shmSpy.isEmpty());
134
135 m_compositor = registry.createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
136 QVERIFY(m_compositor->isValid());
137 m_shm = registry.createShmPool(shmSpy.first().first().value<quint32>(), shmSpy.first().last().value<quint32>(), this);
138 QVERIFY(m_shm->isValid());
139
140 m_idleInhibitManager = registry.createIdleInhibitManager(registry.interface(KWayland::Client::Registry::Interface::IdleInhibitManagerUnstableV1).name,
141 registry.interface(KWayland::Client::Registry::Interface::IdleInhibitManagerUnstableV1).version,
142 this);
143 QVERIFY(m_idleInhibitManager->isValid());
144}

355void TestWaylandSurface::testAttachBuffer()
356{
357 // create the surface
358 QSignalSpy serverSurfaceCreated(m_compositorInterface, &KWaylandServer::CompositorInterface::surfaceCreated);
359 std::unique_ptr<KWayland::Client::Surface> s(m_compositor->createSurface());
360 QVERIFY(serverSurfaceCreated.wait());
361 KWaylandServer::SurfaceInterface *serverSurface = serverSurfaceCreated.first().first().value<KWaylandServer::SurfaceInterface *>();
362 QVERIFY(serverSurface);
363
364 // create three images
365 QImage black(24, 24, QImage::Format_RGB32);
366 black.fill(Qt::black);
367 QImage red(24, 24, QImage::Format_ARGB32); // Note - deliberately not premultiplied
368 red.fill(QColor(255, 0, 0, 128));
369 QImage blue(24, 24, QImage::Format_ARGB32_Premultiplied);
370 blue.fill(QColor(0, 0, 255, 128));
371
372 QSharedPointer<KWayland::Client::Buffer> blackBufferPtr = m_shm->createBuffer(black).toStrongRef();
373 QVERIFY(blackBufferPtr);
374 wl_buffer *blackBuffer = *(blackBufferPtr.data());
375 QSharedPointer<KWayland::Client::Buffer> redBuffer = m_shm->createBuffer(red).toStrongRef();
376 QVERIFY(redBuffer);
377 QSharedPointer<KWayland::Client::Buffer> blueBuffer = m_shm->createBuffer(blue).toStrongRef();
378 QVERIFY(blueBuffer);
379
380 QCOMPARE(blueBuffer->format(), KWayland::Client::Buffer::Format::ARGB32);
381 QCOMPARE(blueBuffer->size(), blue.size());
382 QVERIFY(!blueBuffer->isReleased());
383 QVERIFY(!blueBuffer->isUsed());
384 QCOMPARE(blueBuffer->stride(), blue.bytesPerLine());
385
386 s->attachBuffer(redBuffer.data());
387 s->attachBuffer(blackBuffer);
388 s->damage(QRect(0, 0, 24, 24));
389 s->commit(KWayland::Client::Surface::CommitFlag::None);
390 QSignalSpy damageSpy(serverSurface, &KWaylandServer::SurfaceInterface::damaged);
391 QSignalSpy mappedSpy(serverSurface, &KWaylandServer::SurfaceInterface::mapped);
392 QSignalSpy unmappedSpy(serverSurface, &KWaylandServer::SurfaceInterface::unmapped);
393 QVERIFY(damageSpy.wait());
394 QCOMPARE(mappedSpy.count(), 1);
395 QVERIFY(unmappedSpy.isEmpty());
396
397 // now the ServerSurface should have the black image attached as a buffer
398 KWaylandServer::ClientBuffer *buffer = serverSurface->buffer();
399 buffer->ref();
400 auto shmBuffer = qobject_cast<KWaylandServer::ShmClientBuffer *>(buffer);
401 QVERIFY(shmBuffer);
402 QCOMPARE(shmBuffer->data(), black);
403 QCOMPARE(shmBuffer->data().format(), QImage::Format_RGB32);
404 QCOMPARE(shmBuffer->size(), QSize(24, 24));
405
406 // render another frame
407 s->attachBuffer(redBuffer);
408 s->damage(QRect(0, 0, 24, 24));
409 s->commit(KWayland::Client::Surface::CommitFlag::None);
410 damageSpy.clear();
411 QVERIFY(damageSpy.wait());
412 QCOMPARE(mappedSpy.count(), 1);
413 QVERIFY(unmappedSpy.isEmpty());
414 KWaylandServer::ClientBuffer *buffer2 = serverSurface->buffer();
415 buffer2->ref();
416 auto shmBuffer2 = qobject_cast<KWaylandServer::ShmClientBuffer *>(buffer2);
417 QVERIFY(shmBuffer2);
418 QCOMPARE(shmBuffer2->data().format(), QImage::Format_ARGB32_Premultiplied);
419 QCOMPARE(shmBuffer2->size(), QSize(24, 24));
420 for (int i = 0; i < 24; ++i) {
421 for (int j = 0; j < 24; ++j) {
422 // it's premultiplied in the format
423 QCOMPARE(shmBuffer2->data().pixel(i, j), qRgba(128, 0, 0, 128));
424 }
425 }
426 buffer2->unref();
427 QVERIFY(buffer2->isReferenced());
428 QVERIFY(!redBuffer.data()->isReleased());
429
430 // render another frame
431 blueBuffer->setUsed(true);
432 QVERIFY(blueBuffer->isUsed());
433 s->attachBuffer(blueBuffer.data());
434 s->damage(QRect(0, 0, 24, 24));
435 QSignalSpy frameRenderedSpy(s.get(), &KWayland::Client::Surface::frameRendered);
436 s->commit();
437 damageSpy.clear();
438 QVERIFY(damageSpy.wait());
439 QCOMPARE(mappedSpy.count(), 1);
440 QVERIFY(unmappedSpy.isEmpty());
441 QVERIFY(!buffer2->isReferenced());
442 // TODO: we should have a signal on when the Buffer gets released
443 QCoreApplication::processEvents(QEventLoop::WaitForMoreEvents);
444 if (!redBuffer.data()->isReleased()) {
445 QCoreApplication::processEvents(QEventLoop::WaitForMoreEvents);
446 }
447 QVERIFY(redBuffer.data()->isReleased());
448
449 KWaylandServer::ClientBuffer *buffer3 = serverSurface->buffer();
450 buffer3->ref();
451 auto shmBuffer3 = qobject_cast<KWaylandServer::ShmClientBuffer *>(buffer3);
452 QVERIFY(shmBuffer3);
453 QCOMPARE(shmBuffer3->data().format(), QImage::Format_ARGB32_Premultiplied);
454 QCOMPARE(shmBuffer3->size(), QSize(24, 24));
455 for (int i = 0; i < 24; ++i) {
456 for (int j = 0; j < 24; ++j) {
457 // it's premultiplied in the format
458 QCOMPARE(shmBuffer3->data().pixel(i, j), qRgba(0, 0, 128, 128));
459 }
460 }
461 buffer3->unref();
462 QVERIFY(buffer3->isReferenced());
463
464 serverSurface->frameRendered(1);
465 QVERIFY(frameRenderedSpy.wait());
466
467 // commit a different value shouldn't change our buffer
468 QCOMPARE(serverSurface->buffer(), buffer3);
469 damageSpy.clear();
470 s->setInputRegion(m_compositor->createRegion(QRegion(0, 0, 24, 24)).get());
471 s->commit(KWayland::Client::Surface::CommitFlag::None);
472 wl_display_flush(m_connection->display());
473 QCoreApplication::processEvents();
474 QCoreApplication::processEvents();
475 QCOMPARE(serverSurface->buffer(), buffer3);
476 QVERIFY(damageSpy.isEmpty());
477 QCOMPARE(mappedSpy.count(), 1);
478 QVERIFY(unmappedSpy.isEmpty());
479 QVERIFY(serverSurface->isMapped());
480
481 // clear the surface
482 s->attachBuffer(blackBuffer);
483 s->damage(QRect(0, 0, 1, 1));
484 // TODO: better method
485 s->attachBuffer((wl_buffer *)nullptr);
486 s->damage(QRect(0, 0, 10, 10));
487 s->commit(KWayland::Client::Surface::CommitFlag::None);
488 QVERIFY(unmappedSpy.wait());
489 QCOMPARE(mappedSpy.count(), 1);
490 QCOMPARE(unmappedSpy.count(), 1);
491 QVERIFY(damageSpy.isEmpty());
492 QVERIFY(!serverSurface->isMapped());
493
494 // TODO: add signal test on release
495 buffer->unref();
496}

498void TestWaylandSurface::testMultipleSurfaces()
499{
500 using namespace KWaylandServer;
501 KWayland::Client::Registry registry;
502 registry.setEventQueue(m_queue);
503 QSignalSpy shmSpy(&registry, &KWayland::Client::Registry::shmAnnounced);
504 registry.create(m_connection->display());
505 QVERIFY(registry.isValid());
506 registry.setup();
507 QVERIFY(shmSpy.wait());
508
509 KWayland::Client::ShmPool pool1;
510 KWayland::Client::ShmPool pool2;
511 pool1.setup(registry.bindShm(shmSpy.first().first().value<quint32>(), shmSpy.first().last().value<quint32>()));
512 pool2.setup(registry.bindShm(shmSpy.first().first().value<quint32>(), shmSpy.first().last().value<quint32>()));
513 QVERIFY(pool1.isValid());
514 QVERIFY(pool2.isValid());
515
516 // create the surfaces
517 QSignalSpy serverSurfaceCreated(m_compositorInterface, &KWaylandServer::CompositorInterface::surfaceCreated);
518 std::unique_ptr<KWayland::Client::Surface> s1(m_compositor->createSurface());
519 QVERIFY(serverSurfaceCreated.wait());
520 SurfaceInterface *serverSurface1 = serverSurfaceCreated.first().first().value<KWaylandServer::SurfaceInterface *>();
521 QVERIFY(serverSurface1);
522 // second surface
523 std::unique_ptr<KWayland::Client::Surface> s2(m_compositor->createSurface());
524 QVERIFY(serverSurfaceCreated.wait());
525 SurfaceInterface *serverSurface2 = serverSurfaceCreated.last().first().value<KWaylandServer::SurfaceInterface *>();
526 QVERIFY(serverSurface2);
527 QVERIFY(serverSurface1->resource() != serverSurface2->resource());
528
529 // create two images
530 QImage black(24, 24, QImage::Format_RGB32);
531 black.fill(Qt::black);
532 QImage red(24, 24, QImage::Format_ARGB32_Premultiplied);
533 red.fill(QColor(255, 0, 0, 128));
534
535 auto blackBuffer = pool1.createBuffer(black);
536 auto redBuffer = pool2.createBuffer(red);
537
538 s1->attachBuffer(blackBuffer);
539 s1->damage(QRect(0, 0, 24, 24));
540 s1->commit(KWayland::Client::Surface::CommitFlag::None);
541 QSignalSpy damageSpy1(serverSurface1, &KWaylandServer::SurfaceInterface::damaged);
542 QVERIFY(damageSpy1.wait());
543
544 // now the ServerSurface should have the black image attached as a buffer
545 ClientBuffer *buffer1 = serverSurface1->buffer();
546 QVERIFY(buffer1);
547 QImage buffer1Data = qobject_cast<ShmClientBuffer *>(buffer1)->data();
548 QCOMPARE(buffer1Data, black);
549 // accessing the same buffer is OK
550 QImage buffer1Data2 = qobject_cast<ShmClientBuffer *>(buffer1)->data();
551 QCOMPARE(buffer1Data2, buffer1Data);
552 buffer1Data = QImage();
553 QVERIFY(buffer1Data.isNull());
554 buffer1Data2 = QImage();
555 QVERIFY(buffer1Data2.isNull());
556
557 // attach a buffer for the other surface
558 s2->attachBuffer(redBuffer);
559 s2->damage(QRect(0, 0, 24, 24));
560 s2->commit(KWayland::Client::Surface::CommitFlag::None);
561 QSignalSpy damageSpy2(serverSurface2, &KWaylandServer::SurfaceInterface::damaged);
562 QVERIFY(damageSpy2.wait());
