71void TextInputTest::init()
72{
73 delete m_display;
74 m_display = new KWaylandServer::Display(this);
75 m_display->addSocketName(s_socketName);
76 m_display->start();
77 QVERIFY(m_display->isRunning());
78 m_display->createShm();
79 m_seatInterface = new SeatInterface(m_display, m_display);
80 m_seatInterface->setHasKeyboard(true);
81 m_seatInterface->setHasTouch(true);
82 m_compositorInterface = new CompositorInterface(m_display, m_display);
83 m_textInputManagerV2Interface = new TextInputManagerV2Interface(m_display, m_display);
84
85 // setup connection
86 m_connection = new KWayland::Client::ConnectionThread;
87 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
88 m_connection->setSocketName(s_socketName);
89
90 m_thread = new QThread(this);
91 m_connection->moveToThread(m_thread);
92 m_thread->start();
93
94 m_connection->initConnection();
95 QVERIFY(connectedSpy.wait());
96
97 m_queue = new KWayland::Client::EventQueue(this);
98 m_queue->setup(m_connection);
99
100 KWayland::Client::Registry registry;
101 QSignalSpy interfacesAnnouncedSpy(&registry, &KWayland::Client::Registry::interfacesAnnounced);
102 registry.setEventQueue(m_queue);
103 registry.create(m_connection);
104 QVERIFY(registry.isValid());
105 registry.setup();
106 QVERIFY(interfacesAnnouncedSpy.wait());
107
108 m_seat = registry.createSeat(registry.interface(KWayland::Client::Registry::Interface::Seat).name, registry.interface(KWayland::Client::Registry::Interface::Seat).version, this);
109 QVERIFY(m_seat->isValid());
110 QSignalSpy hasKeyboardSpy(m_seat, &KWayland::Client::Seat::hasKeyboardChanged);
111 QVERIFY(hasKeyboardSpy.wait());
112 m_keyboard = m_seat->createKeyboard(this);
113 QVERIFY(m_keyboard->isValid());
114
115 m_compositor =
116 registry.createCompositor(registry.interface(KWayland::Client::Registry::Interface::Compositor).name, registry.interface(KWayland::Client::Registry::Interface::Compositor).version, this);
117 QVERIFY(m_compositor->isValid());
118
119 m_textInputManagerV2 = registry.createTextInputManager(registry.interface(KWayland::Client::Registry::Interface::TextInputManagerUnstableV2).name,
120 registry.interface(KWayland::Client::Registry::Interface::TextInputManagerUnstableV2).version,
121 this);
122 QVERIFY(m_textInputManagerV2->isValid());

183void TextInputTest::testEnterLeave()
184{
185 // this test verifies that enter leave are sent correctly
186 std::unique_ptr<KWayland::Client::Surface> surface(m_compositor->createSurface());
187
188 std::unique_ptr<KWayland::Client::TextInput> textInput(createTextInput());
189 auto serverSurface = waitForSurface();
190 QVERIFY(serverSurface);
191 QVERIFY(textInput != nullptr);
192 QSignalSpy enteredSpy(textInput.get(), &KWayland::Client::TextInput::entered);
193 QSignalSpy leftSpy(textInput.get(), &KWayland::Client::TextInput::left);
194 QSignalSpy textInputChangedSpy(m_seatInterface, &SeatInterface::focusedTextInputSurfaceChanged);
195
196 // now let's try to enter it
197 QVERIFY(!m_seatInterface->focusedTextInputSurface());
198 m_seatInterface->setFocusedKeyboardSurface(serverSurface);
199 QCOMPARE(m_seatInterface->focusedTextInputSurface(), serverSurface);
200 // text input not yet set for the surface
201 QFETCH(bool, updatesDirectly);
202 QCOMPARE(bool(m_seatInterface->textInputV2()), updatesDirectly);
203 QCOMPARE(textInputChangedSpy.isEmpty(), !updatesDirectly);
204 textInput->enable(surface.get());
205 // this should trigger on server side
206 if (!updatesDirectly) {
207 QVERIFY(textInputChangedSpy.wait());
208 }
209 QCOMPARE(textInputChangedSpy.count(), 1);
210 auto serverTextInput = m_seatInterface->textInputV2();
211 QVERIFY(serverTextInput);
212 QSignalSpy enabledChangedSpy(serverTextInput, &TextInputV2Interface::enabledChanged);
213 if (updatesDirectly) {
214 QVERIFY(enabledChangedSpy.wait());
215 enabledChangedSpy.clear();
216 }
217 QCOMPARE(serverTextInput->surface().data(), serverSurface);
218 QVERIFY(serverTextInput->isEnabled());
219
220 // and trigger an enter
221 if (enteredSpy.isEmpty()) {
222 QVERIFY(enteredSpy.wait());
223 }
224 QCOMPARE(enteredSpy.count(), 1);
225 QCOMPARE(textInput->enteredSurface(), surface.get());
226
227 // now trigger a leave
228 m_seatInterface->setFocusedKeyboardSurface(nullptr);
229 QCOMPARE(textInputChangedSpy.count(), 2);
230 QVERIFY(leftSpy.wait());
231 QVERIFY(!textInput->enteredSurface());
232 QVERIFY(!serverTextInput->isEnabled());
233
234 // if we enter again we should directly get the text input as it's still activated
235 m_seatInterface->setFocusedKeyboardSurface(serverSurface);
236 QCOMPARE(textInputChangedSpy.count(), 3);
237 QVERIFY(m_seatInterface->textInputV2());
238 QVERIFY(enteredSpy.wait());
239 QCOMPARE(enteredSpy.count(), 2);
240 QCOMPARE(textInput->enteredSurface(), surface.get());
241 QVERIFY(serverTextInput->isEnabled());
242
243 // let's deactivate on client side
244 textInput->disable(surface.get());
245 QVERIFY(enabledChangedSpy.wait());
246 QCOMPARE(enabledChangedSpy.count(), 3);
247 QVERIFY(!serverTextInput->isEnabled());
248 // does not trigger a leave
249 QCOMPARE(textInputChangedSpy.count(), 3);
250 // should still be the same text input
251 QCOMPARE(m_seatInterface->textInputV2(), serverTextInput);
252 // reset
253 textInput->enable(surface.get());
254 QVERIFY(enabledChangedSpy.wait());
255
256 // delete the client and wait for the server to catch up
257 QSignalSpy unboundSpy(serverSurface, &QObject::destroyed);
258 surface.reset();
259 QVERIFY(unboundSpy.wait());
260 QVERIFY(leftSpy.wait());
261 QVERIFY(!textInput->enteredSurface());
262}
