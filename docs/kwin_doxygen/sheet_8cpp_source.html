25static QMatrix4x4 createPerspectiveMatrix(const QRectF &rect, const qreal scale)
26{
27 QMatrix4x4 ret;
28
29 const float fovY = std::tan(qDegreesToRadians(60.0f) / 2);
30 const float aspect = 1.0f;
31 const float zNear = 0.1f;
32 const float zFar = 100.0f;
33
34 const float yMax = zNear * fovY;
35 const float yMin = -yMax;
36 const float xMin = yMin * aspect;
37 const float xMax = yMax * aspect;
38
39 ret.frustum(xMin, xMax, yMin, yMax, zNear, zFar);
40
41 const auto deviceRect = scaledRect(rect, scale);
42
43 const float scaleFactor = 1.1 * fovY / yMax;
44 ret.translate(xMin * scaleFactor, yMax * scaleFactor, -1.1);
45 ret.scale((xMax - xMin) * scaleFactor / deviceRect.width(),
46 -(yMax - yMin) * scaleFactor / deviceRect.height(),
47 0.001);
48 ret.translate(-deviceRect.x(), -deviceRect.y());
49
50 return ret;
51}

96void SheetEffect::paintWindow(EffectWindow *w, int mask, QRegion region, WindowPaintData &data)
97{
98 auto animationIt = m_animations.constFind(w);
99 if (animationIt == m_animations.constEnd()) {
100 effects->paintWindow(w, mask, region, data);
101 return;
102 }
103
104 // Perspective projection distorts objects near edges of the viewport
105 // in undesired way. To fix this, the center of the window will be
106 // moved to the origin, after applying perspective projection, the
107 // center is moved back to its "original" projected position. Overall,
108 // this is how the window will be transformed:
109 // [move to the origin] -> [scale] -> [rotate] -> [translate] ->
110 // -> [perspective projection] -> [reverse "move to the origin"]
111 const QMatrix4x4 oldProjMatrix = createPerspectiveMatrix(effects->renderTargetRect(), effects->renderTargetScale());
112 const QRectF windowGeo = w->frameGeometry();
113 const QVector3D invOffset = oldProjMatrix.map(QVector3D(windowGeo.center()));
114 QMatrix4x4 invOffsetMatrix;
115 invOffsetMatrix.translate(invOffset.x(), invOffset.y());
116 data.setProjectionMatrix(invOffsetMatrix * oldProjMatrix);
117
118 // Move the center of the window to the origin.
119 const QRectF screenGeo = effects->virtualScreenGeometry();
120 const QPointF offset = screenGeo.center() - windowGeo.center();
121 data.translate(offset.x(), offset.y());
122
123 const qreal t = (*animationIt).timeLine.value();
124 data.setRotationAxis(Qt::XAxis);
125 data.setRotationAngle(interpolate(60.0, 0.0, t));
126 data *= QVector3D(1.0, t, t);
127 data.translate(0.0, -interpolate(w->y() - (*animationIt).parentY, 0.0, t));
128
129 data.multiplyOpacity(t);
130
131 effects->paintWindow(w, mask, region, data);
132}

134void SheetEffect::postPaintWindow(EffectWindow *w)
135{
136 auto animationIt = m_animations.begin();
137 while (animationIt != m_animations.end()) {
138 EffectWindow *w = animationIt.key();
139 w->addRepaintFull();
140 if ((*animationIt).timeLine.done()) {
141 animationIt = m_animations.erase(animationIt);
142 } else {
143 ++animationIt;
144 }
145 }
146
147 if (m_animations.isEmpty()) {
148 effects->addRepaintFull();
149 }
150
151 effects->postPaintWindow(w);
152}

165void SheetEffect::slotWindowAdded(EffectWindow *w)
166{
167 if (effects->activeFullScreenEffect()) {
168 return;
169 }
170
171 if (!isSheetWindow(w)) {
172 return;
173 }
174
175 Animation &animation = m_animations[w];
176 animation.parentY = 0;
177 animation.timeLine.reset();
178 animation.timeLine.setDuration(m_duration);
179 animation.timeLine.setDirection(TimeLine::Forward);
180 animation.timeLine.setEasingCurve(QEasingCurve::Linear);
181
182 const auto windows = effects->stackingOrder();
183 auto parentIt = std::find_if(windows.constBegin(), windows.constEnd(),
184 [w](EffectWindow *p) {
185 return p->findModal() == w;
186 });
187 if (parentIt != windows.constEnd()) {
188 animation.parentY = (*parentIt)->y();
189 }
190
191 w->setData(WindowAddedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
192
193 w->addRepaintFull();
194}

196void SheetEffect::slotWindowClosed(EffectWindow *w)
197{
198 if (effects->activeFullScreenEffect()) {
199 return;
200 }
201
202 if (!isSheetWindow(w) || w->skipsCloseAnimation()) {
203 return;
204 }
205
206 Animation &animation = m_animations[w];
207 animation.deletedRef = EffectWindowDeletedRef(w);
208 animation.visibleRef = EffectWindowVisibleRef(w, EffectWindow::PAINT_DISABLED_BY_DELETE);
209 animation.timeLine.reset();
210 animation.parentY = 0;
211 animation.timeLine.setDuration(m_duration);
212 animation.timeLine.setDirection(TimeLine::Backward);
213 animation.timeLine.setEasingCurve(QEasingCurve::Linear);
214
215 const auto windows = effects->stackingOrder();
216 auto parentIt = std::find_if(windows.constBegin(), windows.constEnd(),
217 [w](EffectWindow *p) {
218 return p->findModal() == w;
219 });
220 if (parentIt != windows.constEnd()) {
221 animation.parentY = (*parentIt)->y();
222 }
223
224 w->setData(WindowClosedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
225
226 w->addRepaintFull();
227}
