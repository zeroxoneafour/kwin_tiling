40void XdgShellInterfacePrivate::registerPing(quint32 serial)
41{
42 QTimer *timer = new QTimer(q);
43 timer->setInterval(1000);
44 QObject::connect(timer, &QTimer::timeout, q, [this, serial, attempt = 0]() mutable {
45 ++attempt;
46 if (attempt == 1) {
47 Q_EMIT q->pingDelayed(serial);
48 return;
49 }
50 Q_EMIT q->pingTimeout(serial);
51 delete pings.take(serial);
52 });
53 pings.insert(serial, timer);
54 timer->start();
55}

528quint32 XdgToplevelInterface::sendConfigure(const QSize &size, const States &states)
529{
530 // Note that the states listed in the configure event must be an array of uint32_t.
531
532 uint32_t statesData[8] = {0};
533 int i = 0;
534
535 if (states & State::MaximizedHorizontal && states & State::MaximizedVertical) {
536 statesData[i++] = QtWaylandServer::xdg_toplevel::state_maximized;
537 }
538 if (states & State::FullScreen) {
539 statesData[i++] = QtWaylandServer::xdg_toplevel::state_fullscreen;
540 }
541 if (states & State::Resizing) {
542 statesData[i++] = QtWaylandServer::xdg_toplevel::state_resizing;
543 }
544 if (states & State::Activated) {
545 statesData[i++] = QtWaylandServer::xdg_toplevel::state_activated;
546 }
547
548 if (d->resource()->version() >= XDG_TOPLEVEL_STATE_TILED_LEFT_SINCE_VERSION) {
549 if (states & State::TiledLeft) {
550 statesData[i++] = QtWaylandServer::xdg_toplevel::state_tiled_left;
551 }
552 if (states & State::TiledTop) {
553 statesData[i++] = QtWaylandServer::xdg_toplevel::state_tiled_top;
554 }
555 if (states & State::TiledRight) {
556 statesData[i++] = QtWaylandServer::xdg_toplevel::state_tiled_right;
557 }
558 if (states & State::TiledBottom) {
559 statesData[i++] = QtWaylandServer::xdg_toplevel::state_tiled_bottom;
560 }
561 }
562
563 const QByteArray xdgStates = QByteArray::fromRawData(reinterpret_cast<char *>(statesData), sizeof(uint32_t) * i);
564 const quint32 serial = xdgSurface()->shell()->display()->nextSerial();
565
566 d->send_configure(size.width(), size.height(), xdgStates);
567
568 auto xdgSurfacePrivate = XdgSurfaceInterfacePrivate::get(xdgSurface());
569 xdgSurfacePrivate->send_configure(serial);
570 xdgSurfacePrivate->isConfigured = true;
571
572 return serial;
573}
