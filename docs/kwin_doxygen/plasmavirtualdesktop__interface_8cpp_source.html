102void PlasmaVirtualDesktopManagementInterfacePrivate::org_kde_plasma_virtual_desktop_management_bind_resource(Resource *resource)
103{
104 quint32 i = 0;
105 for (auto it = desktops.constBegin(); it != desktops.constEnd(); ++it) {
106 send_desktop_created(resource->handle, (*it)->id(), i++);
107 }
108
109 if (resource->version() >= ORG_KDE_PLASMA_VIRTUAL_DESKTOP_MANAGEMENT_ROWS_SINCE_VERSION) {
110 send_rows(resource->handle, rows);
111 }
112
113 send_done(resource->handle);

153PlasmaVirtualDesktopInterface *PlasmaVirtualDesktopManagementInterface::createDesktop(const QString &id, quint32 position)
154{
155 auto i = d->constFindDesktop(id);
156 if (i != d->desktops.constEnd()) {
157 return *i;
158 }
159
160 const quint32 actualPosition = std::min(position, (quint32)d->desktops.count());
161
162 auto desktop = new PlasmaVirtualDesktopInterface(this);
163 desktop->d->id = id;
164
165 const auto desktopClientResources = desktop->d->resourceMap();
166 for (auto resource : desktopClientResources) {
167 desktop->d->send_desktop_id(resource->handle, id);
168 }
169
170 // activate the first desktop TODO: to be done here?
171 if (d->desktops.isEmpty()) {
172 desktop->d->active = true;
173 }
174
175 d->desktops.insert(actualPosition, desktop);
176
177 const auto clientResources = d->resourceMap();
178 for (auto resource : clientResources) {
179 d->send_desktop_created(resource->handle, id, actualPosition);
180 }
181
182 return desktop;
183}

185void PlasmaVirtualDesktopManagementInterface::removeDesktop(const QString &id)
186{
187 auto deskIt = d->findDesktop(id);
188 if (deskIt == d->desktops.end()) {
189 return;
190 }
191
192 const auto desktopClientResources = (*deskIt)->d->resourceMap();
193 for (auto resource : desktopClientResources) {
194 (*deskIt)->d->send_removed(resource->handle);
195 }
196
197 const auto clientResources = d->resourceMap();
198 for (auto resource : clientResources) {
199 d->send_desktop_removed(resource->handle, id);
200 }
201
202 (*deskIt)->deleteLater();
203 d->desktops.erase(deskIt);
204}
