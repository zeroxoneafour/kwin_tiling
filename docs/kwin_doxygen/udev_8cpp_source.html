95std::vector<UdevDevice::Ptr> UdevEnumerate::find()
96{
97 if (!m_enumerate) {
98 return {};
99 }
100 std::vector<UdevDevice::Ptr> vect;
101 udev_list_entry *it = udev_enumerate_get_list_entry(m_enumerate.get());
102 while (it) {
103 auto current = it;
104 it = udev_list_entry_get_next(it);
105 auto device = m_udev->deviceFromSyspath(udev_list_entry_get_name(current));
106 if (device) {
107 vect.push_back(std::move(device));
108 }
109 }
110 return vect;
111}

113std::vector<UdevDevice::Ptr> Udev::listGPUs()
114{
115 if (!m_udev) {
116 return {};
117 }
118#if defined(Q_OS_FREEBSD)
119 std::vector<UdevDevice::Ptr> r;
120 r.push_back(deviceFromSyspath("/dev/dri/card0"));
121 return r;
122#else
123 UdevEnumerate enumerate(this);
124 enumerate.addMatch(UdevEnumerate::Match::SubSystem, "drm");
125 enumerate.addMatch(UdevEnumerate::Match::SysName, "card[0-9]");
126 enumerate.scan();
127 auto vect = enumerate.find();
128 std::sort(vect.begin(), vect.end(), [](const UdevDevice::Ptr &device1, const UdevDevice::Ptr &device2) {
129 // if set as boot GPU, prefer 1
130 if (device1->isBootVga()) {
131 return true;
132 }
133 // if set as boot GPU, prefer 2
134 if (device2->isBootVga()) {
135 return false;
136 }
137 return true;
138 });
139 return vect;
140#endif
141}

210bool UdevDevice::isBootVga() const
211{
212 auto pci = udev_device_get_parent_with_subsystem_devtype(m_device, "pci", nullptr);
213 if (!pci) {
214 return false;
215 }
216 const char *systAttrValue = udev_device_get_sysattr_value(pci, "boot_vga");
217 return systAttrValue && qstrcmp(systAttrValue, "1") == 0;
218}
