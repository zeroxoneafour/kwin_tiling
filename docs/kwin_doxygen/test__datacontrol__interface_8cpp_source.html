165void DataControlInterfaceTest::init()
166{
167 m_display = new KWaylandServer::Display();
168 m_display->addSocketName(s_socketName);
169 m_display->start();
170 QVERIFY(m_display->isRunning());
171
172 m_seat = new SeatInterface(m_display, this);
173 m_serverCompositor = new CompositorInterface(m_display, this);
174 m_dataControlDeviceManagerInterface = new DataControlDeviceManagerV1Interface(m_display, this);
175
176 // setup connection
177 m_connection = new KWayland::Client::ConnectionThread;
178 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
179 m_connection->setSocketName(s_socketName);
180
181 m_thread = new QThread(this);
182 m_connection->moveToThread(m_thread);
183 m_thread->start();
184
185 m_connection->initConnection();
186 QVERIFY(connectedSpy.wait());
187 QVERIFY(!m_connection->connections().isEmpty());
188
189 m_queue = new KWayland::Client::EventQueue(this);
190 QVERIFY(!m_queue->isValid());
191 m_queue->setup(m_connection);
192 QVERIFY(m_queue->isValid());
193
194 KWayland::Client::Registry registry;
195 connect(&registry, &KWayland::Client::Registry::interfaceAnnounced, this, [this, &registry](const QByteArray &interface, quint32 name, quint32 version) {
196 if (interface == "zwlr_data_control_manager_v1") {
197 m_dataControlDeviceManager = new DataControlDeviceManager;
198 m_dataControlDeviceManager->init(registry.registry(), name, version);
199 }
200 });
201 connect(&registry, &KWayland::Client::Registry::seatAnnounced, this, [this, &registry](quint32 name, quint32 version) {
202 m_clientSeat = registry.createSeat(name, version);
203 });
204 registry.setEventQueue(m_queue);
205 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
206 registry.create(m_connection->display());
207 QVERIFY(registry.isValid());
208 registry.setup();
209 wl_display_flush(m_connection->display());
210
211 QVERIFY(compositorSpy.wait());
212 m_clientCompositor = registry.createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
213 QVERIFY(m_clientCompositor->isValid());
214
215 QVERIFY(m_dataControlDeviceManager);
216}

341void DataControlInterfaceTest::testKlipperCase()
342{
343 // This tests the setup of klipper's real world operation and a race with a common pattern seen between clients and klipper
344 // The client's behaviour is faked with direct access to the seat
345
346 std::unique_ptr<DataControlDevice> dataControlDevice(new DataControlDevice);
347 dataControlDevice->init(m_dataControlDeviceManager->get_data_device(*m_clientSeat));
348
349 QSignalSpy newOfferSpy(dataControlDevice.get(), &DataControlDevice::dataControlOffer);
350 QSignalSpy selectionSpy(dataControlDevice.get(), &DataControlDevice::selection);
351 QSignalSpy serverSelectionChangedSpy(m_seat, &SeatInterface::selectionChanged);
352
353 // Client A has a data source
354 std::unique_ptr<TestDataSource> testSelection(new TestDataSource);
355 m_seat->setSelection(testSelection.get());
356
357 // klipper gets it
358 selectionSpy.wait();
359
360 // Client A deletes it
361 testSelection.reset();
362
363 // klipper gets told
364 selectionSpy.wait();
365
366 // Client A sets something else
367 std::unique_ptr<TestDataSource> testSelection2(new TestDataSource);
368 m_seat->setSelection(testSelection2.get());
369
370 // Meanwhile klipper updates with the old content
371 std::unique_ptr<DataControlSource> source(new DataControlSource);
372 source->init(m_dataControlDeviceManager->create_data_source());
373 source->offer("fromKlipper/test1");
374 source->offer("application/x-kde-onlyReplaceEmpty");
375
376 dataControlDevice->set_selection(source->object());
377
378 QVERIFY(!serverSelectionChangedSpy.wait(10));
379 QCOMPARE(m_seat->selection(), testSelection2.get());
380}
