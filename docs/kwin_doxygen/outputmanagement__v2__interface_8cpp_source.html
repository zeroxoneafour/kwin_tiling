122void OutputConfigurationV2Interface::kde_output_configuration_v2_transform(Resource *resource, wl_resource *outputdevice, int32_t transform)
123{
124 if (invalid) {
125 return;
126 }
127 auto toTransform = [transform]() {
128 switch (transform) {
129 case WL_OUTPUT_TRANSFORM_90:
130 return Output::Transform::Rotated90;
131 case WL_OUTPUT_TRANSFORM_180:
132 return Output::Transform::Rotated180;
133 case WL_OUTPUT_TRANSFORM_270:
134 return Output::Transform::Rotated270;
135 case WL_OUTPUT_TRANSFORM_FLIPPED:
136 return Output::Transform::Flipped;
137 case WL_OUTPUT_TRANSFORM_FLIPPED_90:
138 return Output::Transform::Flipped90;
139 case WL_OUTPUT_TRANSFORM_FLIPPED_180:
140 return Output::Transform::Flipped180;
141 case WL_OUTPUT_TRANSFORM_FLIPPED_270:
142 return Output::Transform::Flipped270;
143 case WL_OUTPUT_TRANSFORM_NORMAL:
144 default:
145 return Output::Transform::Normal;
146 }
147 };
148 auto _transform = toTransform();
149 if (OutputDeviceV2Interface *output = OutputDeviceV2Interface::get(outputdevice)) {
150 config.changeSet(output->handle())->transform = _transform;
151 }
152}

164void OutputConfigurationV2Interface::kde_output_configuration_v2_scale(Resource *resource, wl_resource *outputdevice, wl_fixed_t scale)
165{
166 if (invalid) {
167 return;
168 }
169 qreal doubleScale = wl_fixed_to_double(scale);
170
171 // the fractional scaling protocol only speaks in unit of 120ths
172 // using the same scale throughout makes that simpler
173 // this also eliminates most loss from wl_fixed
174 doubleScale = std::round(doubleScale * 120) / 120;
175
176 if (doubleScale <= 0) {
177 qCWarning(KWIN_CORE) << "Requested to scale output device to" << doubleScale << ", but I can't do that.";
178 return;
179 }
180
181 if (OutputDeviceV2Interface *output = OutputDeviceV2Interface::get(outputdevice)) {
182 config.changeSet(output->handle())->scale = doubleScale;
183 }
184}

253void OutputConfigurationV2Interface::kde_output_configuration_v2_apply(Resource *resource)
254{
255 if (applied) {
256 wl_resource_post_error(resource->handle, error_already_applied, "an output configuration can be applied only once");
257 return;
258 }
259
260 applied = true;
261 if (invalid) {
262 qCWarning(KWIN_CORE) << "Rejecting configuration change because a request output is no longer available";
263 send_failed();
264 return;
265 }
266
267 const auto allOutputs = kwinApp()->outputBackend()->outputs();
268 const bool allDisabled = !std::any_of(allOutputs.begin(), allOutputs.end(), [this](const auto &output) {
269 return config.constChangeSet(output)->enabled;
270 });
271 if (allDisabled) {
272 qCWarning(KWIN_CORE) << "Disabling all outputs through configuration changes is not allowed";
273 send_failed();
274 return;
275 }
276
277 QVector<Output *> sortedOrder;
278 if (!outputOrder.empty()) {
279 const int desktopOutputs = std::count_if(allOutputs.begin(), allOutputs.end(), [](Output *output) {
280 return !output->isNonDesktop();
281 });
282 if (outputOrder.size() != desktopOutputs) {
283 qWarning(KWIN_CORE) << "Provided output order doesn't contain all outputs!";
284 send_failed();
285 return;
286 }
287 outputOrder.erase(std::remove_if(outputOrder.begin(), outputOrder.end(), [this](const auto &pair) {
288 return !config.constChangeSet(pair.second->handle())->enabled;
289 }),
290 outputOrder.end());
291 std::sort(outputOrder.begin(), outputOrder.end(), [](const auto &pair1, const auto &pair2) {
292 return pair1.first < pair2.first;
293 });
294 uint32_t i = 1;
295 for (const auto &[index, name] : std::as_const(outputOrder)) {
296 if (index != i) {
297 qCWarning(KWIN_CORE) << "Provided output order is invalid!";
298 send_failed();
299 return;
300 }
301 i++;
302 }
303 sortedOrder.reserve(outputOrder.size());
304 std::transform(outputOrder.begin(), outputOrder.end(), std::back_inserter(sortedOrder), [](const auto &pair) {
305 return pair.second->handle();
306 });
307 }
308 if (workspace()->applyOutputConfiguration(config, sortedOrder)) {
309 send_applied();
310 } else {
311 qCDebug(KWIN_CORE) << "Applying config failed";
312 send_failed();
313 }
314}
