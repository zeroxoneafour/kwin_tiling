108class KWinSelectionOwner : public KSelectionOwner
109{
110 Q_OBJECT
111public:
112 explicit KWinSelectionOwner()
113 : KSelectionOwner(make_selection_atom())
114 {
115 }
116
117private:
118 bool genericReply(xcb_atom_t target_P, xcb_atom_t property_P, xcb_window_t requestor_P) override
119 {
120 if (target_P == xa_version) {
121 int32_t version[] = {2, 0};
122 xcb_change_property(kwinApp()->x11Connection(), XCB_PROP_MODE_REPLACE, requestor_P,
123 property_P, XCB_ATOM_INTEGER, 32, 2, version);
124 } else {
125 return KSelectionOwner::genericReply(target_P, property_P, requestor_P);
126 }
127 return true;
128 }
129
130 void replyTargets(xcb_atom_t property_P, xcb_window_t requestor_P) override
131 {
132 KSelectionOwner::replyTargets(property_P, requestor_P);
133 xcb_atom_t atoms[1] = {xa_version};
134 // PropModeAppend !
135 xcb_change_property(kwinApp()->x11Connection(), XCB_PROP_MODE_APPEND, requestor_P,
136 property_P, XCB_ATOM_ATOM, 32, 1, atoms);
137 }
138
139 void getAtoms() override
140 {
141 KSelectionOwner::getAtoms();
142 if (xa_version == XCB_ATOM_NONE) {
143 const QByteArray name(QByteArrayLiteral("VERSION"));
144 UniqueCPtr<xcb_intern_atom_reply_t> atom(xcb_intern_atom_reply(
145 kwinApp()->x11Connection(),
146 xcb_intern_atom_unchecked(kwinApp()->x11Connection(), false, name.length(), name.constData()),
147 nullptr));
148 if (atom) {
149 xa_version = atom->atom;
150 }
151 }
152 }
153
154 xcb_atom_t make_selection_atom()
155 {
156 QByteArray screen(QByteArrayLiteral("WM_S0"));
157 UniqueCPtr<xcb_intern_atom_reply_t> atom(xcb_intern_atom_reply(
158 kwinApp()->x11Connection(),
159 xcb_intern_atom_unchecked(kwinApp()->x11Connection(), false, screen.length(), screen.constData()),
160 nullptr));
161 if (!atom) {
162 return XCB_ATOM_NONE;
163 }
164 return atom->atom;
165 }
166 static xcb_atom_t xa_version;
167};

253void ApplicationX11::performStartup()
254{
255 crashChecking();
256
257 owner.reset(new KWinSelectionOwner());
258 connect(owner.get(), &KSelectionOwner::failedToClaimOwnership, [] {
259 fputs(i18n("kwin: unable to claim manager selection, another wm running? (try using --replace)\n").toLocal8Bit().constData(), stderr);
260 ::exit(1);
261 });
262 connect(owner.get(), &KSelectionOwner::lostOwnership, this, &ApplicationX11::lostSelection);
263 connect(owner.get(), &KSelectionOwner::claimedOwnership, this, [this] {
264 installNativeX11EventFilter();
265 // first load options - done internally by a different thread
266 createOptions();
267
268 if (!outputBackend()->initialize()) {
269 std::exit(1);
270 }
271
272 // Check whether another windowmanager is running
273 const uint32_t maskValues[] = {XCB_EVENT_MASK_SUBSTRUCTURE_REDIRECT};
274 UniqueCPtr<xcb_generic_error_t> redirectCheck(xcb_request_check(kwinApp()->x11Connection(),
275 xcb_change_window_attributes_checked(kwinApp()->x11Connection(),
276 kwinApp()->x11RootWindow(),
277 XCB_CW_EVENT_MASK,
278 maskValues)));
279 if (redirectCheck) {
280 fputs(i18n("kwin: another window manager is running (try using --replace)\n").toLocal8Bit().constData(), stderr);
281 if (!wasCrash()) { // if this is a crash-restart, DrKonqi may have stopped the process w/o killing the connection
282 ::exit(1);
283 }
284 }
285
286 // Update the timestamp if a global shortcut is pressed or released. Needed
287 // to ensure that kwin can grab the keyboard.
288 connect(KGlobalAccel::self(), &KGlobalAccel::globalShortcutActiveChanged, this, [this](QAction *triggeredAction) {
289 QVariant timestamp = triggeredAction->property("org.kde.kglobalaccel.activationTimestamp");
290 bool ok = false;
291 const quint32 t = timestamp.toULongLong(&ok);
292 if (ok) {
293 kwinApp()->setX11Time(t);
294 }
295 });
296
297 createInput();
298 createWorkspace();
299 createColorManager();
300 createPlugins();
301
302 Xcb::sync(); // Trigger possible errors, there's still a chance to abort
303
304 notifyKSplash();
305 notifyStarted();
306 });
307 // we need to do an XSync here, otherwise the QPA might crash us later on
308 Xcb::sync();
309 owner->claim(m_replace || wasCrash(), true);
310
311 createAtoms();
312
313 createTabletModeManager();
314}

321void ApplicationX11::crashChecking()
322{
323 setupCrashHandler();
324 if (crashes >= 4) {
325 // Something has gone seriously wrong
326 AlternativeWMDialog dialog;
327 QString cmd = QStringLiteral("kwin_x11");
328 if (dialog.exec() == QDialog::Accepted) {
329 cmd = dialog.selectedWM();
330 } else {
331 ::exit(1);
332 }
333 if (cmd.length() > 500) {
334 qCDebug(KWIN_CORE) << "Command is too long, truncating";
335 cmd = cmd.left(500);
336 }
337 qCDebug(KWIN_CORE) << "Starting" << cmd << "and exiting";
338 char buf[1024];
339 sprintf(buf, "%s &", cmd.toLatin1().data());
340 system(buf);
341 ::exit(1);
342 }
343 // Reset crashes count if we stay up for more that 15 seconds
344 QTimer::singleShot(15 * 1000, this, &Application::resetCrashesCount);
345}

358void ApplicationX11::crashHandler(int signal)
359{
360 crashes++;
361

363 char cmd[1024];

367 sleep(1);

373int main(int argc, char *argv[])
374{
375 KWin::Application::setupMalloc();
376 KWin::Application::setupLocalizedString();
377
378 signal(SIGPIPE, SIG_IGN);
379
380 // enforce xcb plugin, unfortunately command line switch has precedence
381 setenv("QT_QPA_PLATFORM", "xcb", true);
382
383 qunsetenv("QT_DEVICE_PIXEL_RATIO");
384 qunsetenv("QT_SCALE_FACTOR");
385 QCoreApplication::setAttribute(Qt::AA_DisableHighDpiScaling);
386 // KSMServer talks to us directly on DBus.
387 QCoreApplication::setAttribute(Qt::AA_DisableSessionManager);
388 // For sharing thumbnails between our scene graph and qtquick.
389 QCoreApplication::setAttribute(Qt::AA_ShareOpenGLContexts);
390
391 QSurfaceFormat format = QSurfaceFormat::defaultFormat();
392 // shared opengl contexts must have the same reset notification policy
393 format.setOptions(QSurfaceFormat::ResetNotification);
394 // disables vsync for any QtQuick windows we create (BUG 406180)
395 format.setSwapInterval(0);
396 QSurfaceFormat::setDefaultFormat(format);
397
398 KWin::ApplicationX11 a(argc, argv);
399 a.setupTranslator();
400 // reset QT_QPA_PLATFORM so we don't propagate it to our children (e.g. apps launched from the overview effect)
401 qunsetenv("QT_QPA_PLATFORM");
402
403 KSignalHandler::self()->watchSignal(SIGTERM);
404 KSignalHandler::self()->watchSignal(SIGINT);
405 KSignalHandler::self()->watchSignal(SIGHUP);
406 QObject::connect(KSignalHandler::self(), &KSignalHandler::signalReceived,
407 &a, &QCoreApplication::exit);
408
409 KWin::Application::createAboutData();
410
411 QCommandLineOption replaceOption(QStringLiteral("replace"), i18n("Replace already-running ICCCM2.0-compliant window manager"));
412
413 QCommandLineParser parser;
414 a.setupCommandLine(&parser);
415 parser.addOption(replaceOption);
416#if KWIN_BUILD_ACTIVITIES
417 QCommandLineOption noActivitiesOption(QStringLiteral("no-kactivities"),
418 i18n("Disable KActivities integration."));
419 parser.addOption(noActivitiesOption);
420#endif
421
422 parser.process(a);
423 a.processCommandLine(&parser);
424 a.setReplace(parser.isSet(replaceOption));
425#if KWIN_BUILD_ACTIVITIES
426 if (parser.isSet(noActivitiesOption)) {
427 a.setUseKActivities(false);
428 }
429#endif
430
431 // perform sanity checks
432 if (a.platformName().toLower() != QStringLiteral("xcb")) {
433 fprintf(stderr, "%s: FATAL ERROR expecting platform xcb but got platform %s\n",
434 argv[0], qPrintable(a.platformName()));
435 exit(1);
436 }
437 if (!QX11Info::display()) {
438 fprintf(stderr, "%s: FATAL ERROR KWin requires Xlib support in the xcb plugin. Do not configure Qt with -no-xcb-xlib\n",
439 argv[0]);
440 exit(1);
441 }
442
443 a.setSession(KWin::Session::create(KWin::Session::Type::Noop));
444 a.setOutputBackend(std::make_unique<KWin::X11StandaloneBackend>());
445 a.start();
446
447 return a.exec();
448}
