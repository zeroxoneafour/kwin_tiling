51void ThumbnailAsideEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
52{
53 painted = QRegion();
54 effects->paintScreen(mask, region, data);
55
56 const QMatrix4x4 projectionMatrix = data.projectionMatrix();
57 for (const Data &d : std::as_const(windows)) {
58 if (painted.intersects(d.rect)) {
59 WindowPaintData data(projectionMatrix);
60 data.multiplyOpacity(opacity);
61 QRect region;
62 setPositionTransformations(data, region, d.window, d.rect, Qt::KeepAspectRatio);
63 effects->drawWindow(d.window, PAINT_WINDOW_OPAQUE | PAINT_WINDOW_TRANSLUCENT | PAINT_WINDOW_TRANSFORMED, region, data);
64 }
65 }

125void ThumbnailAsideEffect::removeThumbnail(EffectWindow *w)
126{
127 if (!windows.contains(w)) {
128 return;
129 }
130 repaintAll(); // repaint old areas
131 int index = windows[w].index;
132 windows.remove(w);
133 for (QHash<EffectWindow *, Data>::Iterator it = windows.begin();
134 it != windows.end();
135 ++it) {
136 Data &d = *it;
137 if (d.index > index) {
138 --d.index;
139 }
140 }
141 arrange();
142}

144void ThumbnailAsideEffect::arrange()
145{
146 if (windows.size() == 0) {
147 return;
148 }
149 int height = 0;
150 QVector<int> pos(windows.size());
151 qreal mwidth = 0;
152 for (const Data &d : std::as_const(windows)) {
153 height += d.window->height();
154 mwidth = std::max(mwidth, d.window->width());
155 pos[d.index] = d.window->height();
156 }
157 EffectScreen *effectiveScreen = effects->findScreen(screen);
158 if (!effectiveScreen) {
159 effectiveScreen = effects->activeScreen();
160 }
161 QRectF area = effects->clientArea(MaximizeArea, effectiveScreen, effects->currentDesktop());
162 double scale = area.height() / double(height);
163 scale = std::min(scale, maxwidth / double(mwidth)); // don't be wider than maxwidth pixels
164 int add = 0;
165 for (int i = 0;
166 i < windows.size();
167 ++i) {
168 pos[i] = int(pos[i] * scale);
169 pos[i] += spacing + add; // compute offset of each item
170 add = pos[i];
171 }
172 for (QHash<EffectWindow *, Data>::Iterator it = windows.begin();
173 it != windows.end();
174 ++it) {
175 Data &d = *it;
176 int width = int(d.window->width() * scale);
177 d.rect = QRect(area.right() - width, area.bottom() - pos[d.index], width, int(d.window->height() * scale));
178 }
179 repaintAll();
180}
