50static QStringList splitPathList(const QString &input, const QChar delimiter)
51{
52 QStringList ret;
53 QString tmp;
54 for (int i = 0; i < input.size(); i++) {
55 if (input[i] == delimiter) {
56 if (i > 0 && input[i - 1] == '\\') {
57 tmp[tmp.size() - 1] = delimiter;
58 } else if (!tmp.isEmpty()) {
59 ret.append(tmp);
60 tmp = QString();
61 }
62 } else {
63 tmp.append(input[i]);
64 }
65 }
66 if (!tmp.isEmpty()) {
67 ret.append(tmp);
68 }
69 return ret;
70}

119void DrmBackend::checkOutputsAreOn()
120{
121 if (!m_dpmsFilter) {
122 // already disabled, all outputs are on
123 return;
124 }
125 for (Output *output : std::as_const(m_outputs)) {
126 if (output->isEnabled() && output->dpmsMode() != Output::DpmsMode::On) {

146void DrmBackend::reactivate()
147{
148 if (m_active) {
149 return;
150 }
151 m_active = true;
152
153 for (const auto &output : std::as_const(m_outputs)) {
154 output->renderLoop()->uninhibit();
155 output->renderLoop()->scheduleRepaint();
156 }
157
158 // While the session had been inactive, an output could have been added or
159 // removed, we need to re-scan outputs.
160 updateOutputs();
161 Q_EMIT activeChanged();
162}

164void DrmBackend::deactivate()
165{
166 if (!m_active) {
167 return;
168 }
169
170 for (const auto &output : std::as_const(m_outputs)) {
171 output->renderLoop()->inhibit();
172 }
173
174 m_active = false;
175 Q_EMIT activeChanged();

178bool DrmBackend::initialize()
179{
180 // TODO: Pause/Resume individual GPU devices instead.
181 connect(m_session, &Session::devicePaused, this, [this](dev_t deviceId) {
182 if (primaryGpu()->deviceId() == deviceId) {
183 deactivate();
184 }
185 });
186 connect(m_session, &Session::deviceResumed, this, [this](dev_t deviceId) {
187 if (primaryGpu()->deviceId() == deviceId) {
188 reactivate();
189 }
190 });
191 connect(m_session, &Session::awoke, this, &DrmBackend::turnOutputsOn);
192
193 if (!m_explicitGpus.isEmpty()) {
194 for (const QString &fileName : m_explicitGpus) {
195 addGpu(fileName);
196 }
197 } else {
198 const auto devices = m_udev->listGPUs();
199 for (const UdevDevice::Ptr &device : devices) {
200 if (device->seat() == m_session->seat()) {
201 addGpu(device->devNode());
202 }
203 }
204 }
205
206 if (m_gpus.empty()) {
207 qCWarning(KWIN_DRM) << "No suitable DRM devices have been found";
208 return false;
209 }
210
211 // setup udevMonitor
212 if (m_udevMonitor) {
213 m_udevMonitor->filterSubsystemDevType("drm");
214 const int fd = m_udevMonitor->fd();
215 if (fd != -1) {
216 m_socketNotifier = std::make_unique<QSocketNotifier>(fd, QSocketNotifier::Read);
217 connect(m_socketNotifier.get(), &QSocketNotifier::activated, this, &DrmBackend::handleUdevEvent);
218 m_udevMonitor->enable();
219 }
220 }
221 return true;
222}

224void DrmBackend::handleUdevEvent()
225{
226 while (auto device = m_udevMonitor->getDevice()) {
227 if (!m_active) {
228 continue;
229 }
230
231 // Ignore the device seat if the KWIN_DRM_DEVICES envvar is set.
232 if (!m_explicitGpus.isEmpty()) {
233 if (!m_explicitGpus.contains(device->devNode())) {
234 continue;
235 }
236 } else {
237 if (device->seat() != m_session->seat()) {
238 continue;
239 }
240 }
241
242 if (device->action() == QStringLiteral("add")) {
243 if (addGpu(device->devNode())) {
244 updateOutputs();
245 }
246 } else if (device->action() == QStringLiteral("remove")) {
247 DrmGpu *gpu = findGpu(device->devNum());
248 if (gpu) {
249 if (primaryGpu() == gpu) {
250 qCCritical(KWIN_DRM) << "Primary gpu has been removed! Quitting...";
251 QCoreApplication::exit(1);
252 return;
253 } else {
254 gpu->setRemoved();
255 updateOutputs();
256 }
257 }
258 } else if (device->action() == QStringLiteral("change")) {
259 DrmGpu *gpu = findGpu(device->devNum());
260 if (!gpu) {
261 gpu = addGpu(device->devNode());
262 }
263 if (gpu) {
264 qCDebug(KWIN_DRM) << "Received change event for monitored drm device" << gpu->devNode();
265 updateOutputs();
266 }
267 }
268 }

271DrmGpu *DrmBackend::addGpu(const QString &fileName)
272{
273 int fd = m_session->openRestricted(fileName);
274 if (fd < 0) {
275 qCWarning(KWIN_DRM) << "failed to open drm device at" << fileName;
276 return nullptr;
277 }
278
279 // try to make a simple drm get resource call, if it fails it is not useful for us
280 drmModeRes *resources = drmModeGetResources(fd);
281 if (!resources) {
282 qCDebug(KWIN_DRM) << "Skipping KMS incapable drm device node at" << fileName;
283 m_session->closeRestricted(fd);
284 return nullptr;
285 }
286 drmModeFreeResources(resources);
287
288 struct stat buf;
289 if (fstat(fd, &buf) == -1) {
290 qCDebug(KWIN_DRM, "Failed to fstat %s: %s", qPrintable(fileName), strerror(errno));
291 m_session->closeRestricted(fd);
292 return nullptr;
293 }
294
295 qCDebug(KWIN_DRM, "adding GPU %s", qPrintable(fileName));
296 m_gpus.push_back(std::make_unique<DrmGpu>(this, fileName, fd, buf.st_rdev));
297 auto gpu = m_gpus.back().get();
298 m_active = true;
299 connect(gpu, &DrmGpu::outputAdded, this, &DrmBackend::addOutput);
300 connect(gpu, &DrmGpu::outputRemoved, this, &DrmBackend::removeOutput);
301 Q_EMIT gpuAdded(gpu);
302 return gpu;
303}

319void DrmBackend::updateOutputs()
320{
321 for (auto it = m_gpus.begin(); it != m_gpus.end(); ++it) {
322 if ((*it)->isRemoved()) {
323 (*it)->removeOutputs();
324 } else {
325 (*it)->updateOutputs();
326 }
327 }
328
329 Q_EMIT outputsQueried();
330
331 for (auto it = m_gpus.begin(); it != m_gpus.end();) {
332 DrmGpu *gpu = it->get();
333 if (gpu->isRemoved() || (gpu != primaryGpu() && gpu->drmOutputs().isEmpty())) {
334 qCDebug(KWIN_DRM) << "Removing GPU" << (*it)->devNode();
335 const std::unique_ptr<DrmGpu> keepAlive = std::move(*it);
336 it = m_gpus.erase(it);
337 Q_EMIT gpuRemoved(keepAlive.get());
338 } else {
339 it++;
340 }
341 }

466bool DrmBackend::applyOutputChanges(const OutputConfiguration &config)
467{
468 QVector<DrmOutput *> toBeEnabled;
469 QVector<DrmOutput *> toBeDisabled;
470 for (const auto &gpu : std::as_const(m_gpus)) {
471 const auto &outputs = gpu->drmOutputs();
472 for (const auto &output : outputs) {
473 if (output->isNonDesktop()) {
474 continue;
475 }
476 output->queueChanges(config);
477 if (config.constChangeSet(output)->enabled) {
478 toBeEnabled << output;
479 } else {
480 toBeDisabled << output;
481 }
482 }
483 if (gpu->testPendingConfiguration() != DrmPipeline::Error::None) {
484 for (const auto &output : std::as_const(toBeEnabled)) {
485 output->revertQueuedChanges();
486 }
487 for (const auto &output : std::as_const(toBeDisabled)) {
488 output->revertQueuedChanges();
489 }
490 return false;
491 }
492 }
493 // first, apply changes to drm outputs.
494 // This may remove the placeholder output and thus change m_outputs!
495 for (const auto &output : std::as_const(toBeEnabled)) {
496 output->applyQueuedChanges(config);
497 }
498 for (const auto &output : std::as_const(toBeDisabled)) {
499 output->applyQueuedChanges(config);
500 }
501 // only then apply changes to the virtual outputs
502 for (const auto &gpu : std::as_const(m_gpus)) {
503 const auto &outputs = gpu->virtualOutputs();
504 for (const auto &output : outputs) {
505 output->applyChanges(config);
506 }
507 }
508 return true;
509}
