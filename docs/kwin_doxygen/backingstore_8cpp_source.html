34void BackingStore::resize(const QSize &size, const QRegion &staticContents)
35{
36 if (m_backBuffer.size() == size) {
37 return;
38 }
39
40 const QPlatformWindow *platformWindow = static_cast<QPlatformWindow *>(window()->handle());
41 const qreal devicePixelRatio = platformWindow->devicePixelRatio();
42
43 m_backBuffer = QImage(size * devicePixelRatio, QImage::Format_ARGB32_Premultiplied);
44 m_backBuffer.setDevicePixelRatio(devicePixelRatio);
45
46 m_frontBuffer = QImage(size * devicePixelRatio, QImage::Format_ARGB32_Premultiplied);
47 m_frontBuffer.setDevicePixelRatio(devicePixelRatio);

50static QRect scaledRect(const QRect &rect, qreal devicePixelRatio)

55static void blitImage(const QImage &source, QImage &target, const QRegion &region)
56{
57 QPainter painter(&target);
58 for (const QRect &rect : region) {
59 painter.drawImage(rect, source, scaledRect(rect, source.devicePixelRatio()));

63void BackingStore::flush(QWindow *window, const QRegion &region, const QPoint &offset)
64{
65 Window *platformWindow = static_cast<Window *>(window->handle());
66 InternalWindow *internalWindow = platformWindow->internalWindow();
67 if (!internalWindow) {
68 return;
69 }
70
71 blitImage(m_backBuffer, m_frontBuffer, region);
72
73 internalWindow->present(m_frontBuffer, region);
