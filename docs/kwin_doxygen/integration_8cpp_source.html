72bool Integration::hasCapability(Capability cap) const
73{
74 switch (cap) {
75 case ThreadedPixmaps:
76 return true;
77 case OpenGL:
78 return true;
79 case ThreadedOpenGL:
80 return false;
81 case BufferQueueingOpenGL:
82 return false;
83 case MultipleWindows:
84 case NonFullScreenWindows:
85 return true;
86 case RasterGLSurface:
87 return false;
88 default:
89 return QPlatformIntegration::hasCapability(cap);
90 }

93void Integration::initialize()
94{
95 // This method is called from QGuiApplication's constructor, before kwinApp is built
96 QTimer::singleShot(0, this, [this] {
97 // The QPA is initialized before the workspace is created.
98 if (workspace()) {
99 handleWorkspaceCreated();
100 } else {
101 connect(kwinApp(), &Application::workspaceCreated, this, &Integration::handleWorkspaceCreated);
102 }
103 });
104
105 QPlatformIntegration::initialize();
106
107 m_dummyScreen = new PlaceholderScreen();
108 QWindowSystemInterface::handleScreenAdded(m_dummyScreen);
109}

149QPlatformOpenGLContext *Integration::createPlatformOpenGLContext(QOpenGLContext *context) const
150{
151 if (kwinApp()->outputBackend()->sceneEglGlobalShareContext() == EGL_NO_CONTEXT) {
152 qCWarning(KWIN_QPA) << "Attempting to create a QOpenGLContext before the scene is initialized";
153 return nullptr;
154 }
155 const EGLDisplay eglDisplay = kwinApp()->outputBackend()->sceneEglDisplay();
156 if (eglDisplay != EGL_NO_DISPLAY) {
157 EGLPlatformContext *platformContext = new EGLPlatformContext(context, eglDisplay);
158 return platformContext;
159 }
160 return nullptr;
161}

163void Integration::handleWorkspaceCreated()
164{
165 connect(workspace(), &Workspace::outputAdded,
166 this, &Integration::handleOutputEnabled);
167 connect(workspace(), &Workspace::outputRemoved,
168 this, &Integration::handleOutputDisabled);
169
170 const QList<Output *> outputs = workspace()->outputs();
171 for (Output *output : outputs) {
172 handleOutputEnabled(output);
173 }
174}

176void Integration::handleOutputEnabled(Output *output)
177{
178 Screen *platformScreen = new Screen(output, this);
179 QWindowSystemInterface::handleScreenAdded(platformScreen);
180 m_screens.insert(output, platformScreen);
181
182 if (m_dummyScreen) {
183 QWindowSystemInterface::handleScreenRemoved(m_dummyScreen);
184 m_dummyScreen = nullptr;
185 }
186}
187
188void Integration::handleOutputDisabled(Output *output)
189{
190 Screen *platformScreen = m_screens.take(output);
191 if (!platformScreen) {
192 qCWarning(KWIN_QPA) << "Unknown output" << output;
193 return;
194 }
195
196 if (m_screens.isEmpty()) {
197 m_dummyScreen = new PlaceholderScreen();
198 QWindowSystemInterface::handleScreenAdded(m_dummyScreen);
199 }
200
201 QWindowSystemInterface::handleScreenRemoved(platformScreen);
202}
