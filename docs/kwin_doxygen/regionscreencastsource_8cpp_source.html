41void RegionScreenCastSource::updateOutput(Output *output)
42{
43 m_last = output->renderLoop()->lastPresentationTimestamp();
44
45 if (m_renderedTexture) {
46 const std::shared_ptr<GLTexture> outputTexture = Compositor::self()->scene()->textureForOutput(output);
47 const auto outputGeometry = output->geometry();
48 if (!outputTexture || !m_region.intersects(output->geometry())) {
49 return;
50 }
51
52 GLFramebuffer::pushFramebuffer(m_target.get());
53
54 ShaderBinder shaderBinder(ShaderTrait::MapTexture);
55 QMatrix4x4 projectionMatrix;
56 projectionMatrix.ortho(m_region);
57 projectionMatrix.translate(outputGeometry.left() / m_scale, outputGeometry.top() / m_scale);
58
59 shaderBinder.shader()->setUniform(GLShader::ModelViewProjectionMatrix, projectionMatrix);
60
61 outputTexture->bind();
62 outputTexture->render(output->geometry().size(), 1 / m_scale);
63 outputTexture->unbind();
64 GLFramebuffer::popFramebuffer();
65 }
66}

73void RegionScreenCastSource::ensureTexture()
74{
75 if (!m_renderedTexture) {
76 m_renderedTexture.reset(new GLTexture(hasAlphaChannel() ? GL_RGBA8 : GL_RGB8, textureSize()));
77 m_target.reset(new GLFramebuffer(m_renderedTexture.get()));
78 const auto allOutputs = workspace()->outputs();
79 for (auto output : allOutputs) {
80 if (output->geometry().intersects(m_region)) {
81 updateOutput(output);
82 }
83 }
84 }
85}

87void RegionScreenCastSource::render(GLFramebuffer *target)
88{
89 ensureTexture();
90
91 GLFramebuffer::pushFramebuffer(target);
92 auto shader = ShaderManager::instance()->pushShader(ShaderTrait::MapTexture);
93
94 QMatrix4x4 projectionMatrix;
95 projectionMatrix.scale(1, -1);
96 projectionMatrix.ortho(QRect(QPoint(), target->size()));
97 shader->setUniform(GLShader::ModelViewProjectionMatrix, projectionMatrix);
98
99 m_renderedTexture->bind();
100 m_renderedTexture->render(target->size(), m_scale);
101 m_renderedTexture->unbind();
102
103 ShaderManager::instance()->popShader();
104 GLFramebuffer::popFramebuffer();
105}
