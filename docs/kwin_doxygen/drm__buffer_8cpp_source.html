119std::shared_ptr<DrmFramebuffer> DrmFramebuffer::createFramebuffer(const std::shared_ptr<DrmGpuBuffer> &buffer)
120{
121 const auto size = buffer->size();
122 const auto handles = buffer->handles();
123 const auto strides = buffer->strides();
124 const auto offsets = buffer->offsets();
125
126 uint32_t framebufferId = 0;
127 int ret;
128 if (buffer->gpu()->addFB2ModifiersSupported() && buffer->modifier() != DRM_FORMAT_MOD_INVALID) {
129 uint64_t modifier[4];
130 for (uint32_t i = 0; i < 4; i++) {
131 modifier[i] = i < buffer->planeCount() ? buffer->modifier() : 0;
132 }
133 ret = drmModeAddFB2WithModifiers(buffer->gpu()->fd(), size.width(), size.height(), buffer->format(), handles.data(), strides.data(), offsets.data(), modifier, &framebufferId, DRM_MODE_FB_MODIFIERS);
134 } else {
135 ret = drmModeAddFB2(buffer->gpu()->fd(), size.width(), size.height(), buffer->format(), handles.data(), strides.data(), offsets.data(), &framebufferId, 0);
136 if (ret == EOPNOTSUPP && handles.size() == 1) {
137 ret = drmModeAddFB(buffer->gpu()->fd(), size.width(), size.height(), 24, 32, strides[0], handles[0], &framebufferId);
138 }
139 }
140 if (ret == 0) {
141 return std::make_shared<DrmFramebuffer>(buffer, framebufferId);
142 } else {
143 return nullptr;
144 }
145}
