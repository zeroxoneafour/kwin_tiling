100void DrmLeaseDeviceV1Interface::removeOutput(KWin::DrmAbstractOutput *output)
101{
102 const auto it = m_connectors.find(output);
103 if (it != m_connectors.end()) {
104 DrmLeaseConnectorV1Interface *connector = it->second.get();
105 connector->withdraw();
106 for (const auto &lease : std::as_const(m_leases)) {
107 if (lease->connectors().contains(connector)) {
108 lease->connectors().removeOne(connector);
109 lease->revoke();
110 }
111 }
112 for (const auto &leaseRequest : std::as_const(m_leaseRequests)) {
113 if (leaseRequest->connectors().contains(connector)) {
114 leaseRequest->invalidate();
115 }
116 }
117 m_connectors.erase(it);
118 }
119}

121void DrmLeaseDeviceV1Interface::setDrmMaster(bool hasDrmMaster)
122{
123 if (hasDrmMaster == m_hasDrmMaster) {
124 return;
125 }
126 if (hasDrmMaster) {
127 // send pending drm fds
128 while (!m_pendingFds.isEmpty()) {
129 KWin::FileDescriptor fd = m_gpu->createNonMasterFd();
130 send_drm_fd(m_pendingFds.dequeue(), fd.get());
131 }
132 // offer all connectors again
133 for (const auto &[output, connector] : m_connectors) {
134 offerConnector(connector.get());
135 }
136 } else {
137 // withdraw all connectors
138 for (const auto &[output, connector] : m_connectors) {
139 connector->withdraw();
140 }
141 // and revoke all leases
142 for (const auto &lease : std::as_const(m_leases)) {
143 lease->revoke();
144 }
145 }
146 m_hasDrmMaster = hasDrmMaster;
147 done();
148}

158void DrmLeaseDeviceV1Interface::remove()
159{

229void DrmLeaseDeviceV1Interface::wp_drm_lease_device_v1_bind_resource(Resource *resource)
230{
231 if (isGlobalRemoved()) {
232 return;
233 }
234 if (!m_hasDrmMaster) {
235 m_pendingFds << resource->handle;
236 return;
237 }
238 KWin::FileDescriptor fd = m_gpu->createNonMasterFd();
239 send_drm_fd(resource->handle, fd.get());
240 for (const auto &[output, connector] : m_connectors) {
241 if (!connector->withdrawn()) {
242 auto connectorResource = connector->add(resource->client(), 0, s_version);
243 send_connector(resource->handle, connectorResource->handle);
244 connector->send(connectorResource->handle);
245 }
246 }
247 send_done(resource->handle);
248}

328void DrmLeaseRequestV1Interface::wp_drm_lease_request_v1_request_connector(Resource *resource, struct ::wl_resource *connector_handle)
329{
330 if (auto connector = resource_cast<DrmLeaseConnectorV1Interface *>(connector_handle)) {
331 if (connector->device() != m_device) {
332 wl_resource_post_error(resource->handle, WP_DRM_LEASE_REQUEST_V1_ERROR_WRONG_DEVICE, "Requested connector from invalid lease device");
333 } else if (connector->withdrawn()) {
334 qCWarning(KWIN_CORE) << "DrmLease: withdrawn connector requested";
335 invalidate();
336 } else if (m_connectors.contains(connector)) {
337 wl_resource_post_error(resource->handle, WP_DRM_LEASE_REQUEST_V1_ERROR_DUPLICATE_CONNECTOR, "Requested connector twice");
338 } else if (!m_invalid) {
339 m_connectors << connector;
340 }
341 } else {
342 qCWarning(KWIN_CORE, "DrmLease: Invalid connector requested");
343 invalidate();
344 }
345}

347void DrmLeaseRequestV1Interface::wp_drm_lease_request_v1_submit(Resource *resource, uint32_t id)
348{
349 wl_resource *leaseResource = wl_resource_create(resource->client(), &wp_drm_lease_v1_interface, s_version, id);
350 if (!leaseResource) {
351 wl_resource_post_no_memory(resource->handle);
352 return;
353 }
354 DrmLeaseV1Interface *lease = new DrmLeaseV1Interface(m_device, m_connectors, leaseResource);
355 m_device->addLease(lease);
356 if (!m_device->hasDrmMaster()) {
357 qCWarning(KWIN_CORE) << "DrmLease: rejecting lease request without drm master";
358 lease->deny();
359 } else if (m_invalid) {
360 qCWarning(KWIN_CORE) << "DrmLease: rejecting lease request with a withdrawn connector";
361 lease->deny();
362 } else if (m_connectors.isEmpty()) {
363 wl_resource_post_error(resource->handle, WP_DRM_LEASE_REQUEST_V1_ERROR_EMPTY_LEASE, "Requested lease without connectors");
364 } else {
365 QVector<KWin::DrmOutput *> outputs;
366 for (const auto &connector : m_connectors) {
367 outputs.push_back(connector->output());
368 }
369 auto drmLease = m_device->gpu()->leaseOutputs(outputs);
370 if (drmLease) {
371 lease->grant(std::move(drmLease));
372 } else {
373 lease->deny();
374 }
375 }
376 wl_resource_destroy(resource->handle);
377}

401void DrmLeaseV1Interface::grant(std::unique_ptr<KWin::DrmLease> &&lease)
402{
403 KWin::FileDescriptor tmp = std::move(lease->fd());
404 send_lease_fd(tmp.get());
405 m_lease = std::move(lease);
406 connect(m_lease.get(), &KWin::DrmLease::revokeRequested, this, &DrmLeaseV1Interface::revoke);
407 for (const auto &connector : std::as_const(m_connectors)) {
408 connector->withdraw();

422void DrmLeaseV1Interface::revoke()
423{
424 Q_ASSERT(m_lease);
425 if (!m_finished) {
426 m_finished = true;
427 send_finished();
428 }
429 m_lease.reset();
430 // check if we should offer connectors again
431 if (m_device->hasDrmMaster()) {
432 for (const auto &connector : std::as_const(m_connectors)) {
433 m_device->offerConnector(connector);
434 }
435 m_device->done();
436 }
