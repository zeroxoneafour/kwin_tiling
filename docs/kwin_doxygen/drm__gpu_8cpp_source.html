43DrmGpu::DrmGpu(DrmBackend *backend, const QString &devNode, int fd, dev_t deviceId)
44 : m_fd(fd)
45 , m_deviceId(deviceId)
46 , m_devNode(devNode)
47 , m_atomicModeSetting(false)
48 , m_gbmDevice(nullptr)
49 , m_platform(backend)
50{
51 uint64_t capability = 0;
52
53 if (drmGetCap(fd, DRM_CAP_CURSOR_WIDTH, &capability) == 0) {
54 m_cursorSize.setWidth(capability);
55 } else {
56 m_cursorSize.setWidth(64);
57 }
58
59 if (drmGetCap(fd, DRM_CAP_CURSOR_HEIGHT, &capability) == 0) {
60 m_cursorSize.setHeight(capability);
61 } else {
62 m_cursorSize.setHeight(64);
63 }
64
65 int ret = drmGetCap(fd, DRM_CAP_TIMESTAMP_MONOTONIC, &capability);
66 if (ret == 0 && capability == 1) {
67 m_presentationClock = CLOCK_MONOTONIC;
68 } else {
69 m_presentationClock = CLOCK_REALTIME;
70 }
71
72 m_addFB2ModifiersSupported = drmGetCap(fd, DRM_CAP_ADDFB2_MODIFIERS, &capability) == 0 && capability == 1;
73 qCDebug(KWIN_DRM) << "drmModeAddFB2WithModifiers is" << (m_addFB2ModifiersSupported ? "supported" : "not supported") << "on GPU" << m_devNode;
74
75 // find out what driver this kms device is using
76 DrmUniquePtr<drmVersion> version(drmGetVersion(fd));
77 m_isNVidia = strstr(version->name, "nvidia-drm");
78 m_isVirtualMachine = strstr(version->name, "virtio") || strstr(version->name, "qxl")
79 || strstr(version->name, "vmwgfx") || strstr(version->name, "vboxvideo");
80 m_gbmDevice = gbm_create_device(m_fd);
81
82 m_socketNotifier = std::make_unique<QSocketNotifier>(fd, QSocketNotifier::Read);
83 connect(m_socketNotifier.get(), &QSocketNotifier::activated, this, &DrmGpu::dispatchEvents);
84
85 initDrmResources();
86
87 if (m_atomicModeSetting == false) {
88 // only supported with legacy
89 m_asyncPageflipSupported = drmGetCap(fd, DRM_CAP_ASYNC_PAGE_FLIP, &capability) == 0 && capability == 1;
90 }
91}

131void DrmGpu::initDrmResources()
132{
133 // try atomic mode setting
134 bool isEnvVarSet = false;
135 bool noAMS = qEnvironmentVariableIntValue("KWIN_DRM_NO_AMS", &isEnvVarSet) != 0 && isEnvVarSet;
136 if (m_isVirtualMachine && !isEnvVarSet) {
137 qCWarning(KWIN_DRM, "Atomic Mode Setting disabled on GPU %s because of cursor offset issues in virtual machines", qPrintable(m_devNode));
138 } else if (noAMS) {
139 qCWarning(KWIN_DRM) << "Atomic Mode Setting requested off via environment variable. Using legacy mode on GPU" << m_devNode;
140 } else if (drmSetClientCap(m_fd, DRM_CLIENT_CAP_ATOMIC, 1) != 0) {
141 qCWarning(KWIN_DRM) << "drmSetClientCap for Atomic Mode Setting failed. Using legacy mode on GPU" << m_devNode;
142 } else {
143 DrmUniquePtr<drmModePlaneRes> planeResources(drmModeGetPlaneResources(m_fd));
144 if (planeResources) {
145 qCDebug(KWIN_DRM) << "Using Atomic Mode Setting on gpu" << m_devNode;
146 qCDebug(KWIN_DRM) << "Number of planes on GPU" << m_devNode << ":" << planeResources->count_planes;
147 // create the plane objects
148 for (unsigned int i = 0; i < planeResources->count_planes; ++i) {
149 DrmUniquePtr<drmModePlane> kplane(drmModeGetPlane(m_fd, planeResources->planes[i]));
150 auto plane = std::make_unique<DrmPlane>(this, kplane->plane_id);
151 if (plane->init()) {
152 m_allObjects << plane.get();
153 m_planes.push_back(std::move(plane));
154 }
155 }
156 if (m_planes.empty()) {
157 qCWarning(KWIN_DRM) << "Failed to create any plane. Falling back to legacy mode on GPU " << m_devNode;
158 }
159 } else {
160 qCWarning(KWIN_DRM) << "Failed to get plane resources. Falling back to legacy mode on GPU " << m_devNode;
161 }
162 }
163 m_atomicModeSetting = !m_planes.empty();
164
165 DrmUniquePtr<drmModeRes> resources(drmModeGetResources(m_fd));
166 if (!resources) {
167 qCCritical(KWIN_DRM) << "drmModeGetResources for getting CRTCs failed on GPU" << m_devNode;
168 return;
169 }
170 QVector<DrmPlane *> assignedPlanes;
171 for (int i = 0; i < resources->count_crtcs; ++i) {
172 uint32_t crtcId = resources->crtcs[i];
173 QVector<DrmPlane *> primaryCandidates;
174 QVector<DrmPlane *> cursorCandidates;
175 for (const auto &plane : m_planes) {
176 if (plane->isCrtcSupported(i) && !assignedPlanes.contains(plane.get())) {
177 if (plane->type() == DrmPlane::TypeIndex::Primary) {
178 primaryCandidates.push_back(plane.get());
179 } else if (plane->type() == DrmPlane::TypeIndex::Cursor) {
180 cursorCandidates.push_back(plane.get());
181 }
182 }
183 }
184 if (m_atomicModeSetting && primaryCandidates.empty()) {
185 qCWarning(KWIN_DRM) << "Could not find a suitable primary plane for crtc" << resources->crtcs[i];
186 continue;
187 }
188 const auto findBestPlane = [crtcId](const QVector<DrmPlane *> &list) {
189 // if the plane is already used with this crtc, prefer it
190 const auto connected = std::find_if(list.begin(), list.end(), [crtcId](DrmPlane *plane) {
191 return plane->getProp(DrmPlane::PropertyIndex::CrtcId)->pending() == crtcId;
192 });
193 if (connected != list.end()) {
194 return *connected;
195 }
196 // don't take away planes from other crtcs. The kernel currently rejects such commits
197 const auto notconnected = std::find_if(list.begin(), list.end(), [](DrmPlane *plane) {
198 return plane->getProp(DrmPlane::PropertyIndex::CrtcId)->pending() == 0;
199 });
200 if (notconnected != list.end()) {
201 return *notconnected;
202 }
203 return list.empty() ? nullptr : list.front();
204 };
205 DrmPlane *primary = findBestPlane(primaryCandidates);
206 DrmPlane *cursor = findBestPlane(cursorCandidates);
207 assignedPlanes.push_back(primary);
208 if (cursor) {
209 assignedPlanes.push_back(cursor);
210 }
211 auto crtc = std::make_unique<DrmCrtc>(this, crtcId, i, primary, cursor);
212 if (!crtc->init()) {
213 continue;
214 }
215 m_allObjects << crtc.get();
216 m_crtcs.push_back(std::move(crtc));
217 }
218}

220bool DrmGpu::updateOutputs()
221{
222 waitIdle();
223 DrmUniquePtr<drmModeRes> resources(drmModeGetResources(m_fd));
224 if (!resources) {
225 qCWarning(KWIN_DRM) << "drmModeGetResources failed";
226 return false;
227 }
228
229 // In principle these things are supposed to be detected through the wayland protocol.
230 // In practice SteamVR doesn't always behave correctly
231 DrmUniquePtr<drmModeLesseeListRes> lessees{drmModeListLessees(m_fd)};
232 for (const auto &output : std::as_const(m_drmOutputs)) {
233 if (output->lease()) {
234 bool leaseActive = false;
235 for (uint i = 0; i < lessees->count; i++) {
236 if (lessees->lessees[i] == output->lease()->lesseeId()) {
237 leaseActive = true;
238 break;
239 }
240 }
241 if (!leaseActive) {
242 Q_EMIT output->lease()->revokeRequested();
243 }
244 }
245 }
246
247 // check for added and removed connectors
248 QVector<DrmConnector *> existing;
249 QVector<DrmOutput *> addedOutputs;
250 for (int i = 0; i < resources->count_connectors; ++i) {
251 const uint32_t currentConnector = resources->connectors[i];
252 const auto it = std::find_if(m_connectors.begin(), m_connectors.end(), [currentConnector](const auto &connector) {
253 return connector->id() == currentConnector;
254 });
255 if (it == m_connectors.end()) {
256 auto conn = std::make_shared<DrmConnector>(this, currentConnector);
257 if (!conn->init()) {
258 continue;
259 }
260 existing.push_back(conn.get());
261 m_allObjects.push_back(conn.get());
262 m_connectors.push_back(std::move(conn));
263 } else {
264 (*it)->updateProperties();
265 existing.push_back(it->get());
266 }
267 }
268 for (auto it = m_connectors.begin(); it != m_connectors.end();) {
269 DrmConnector *conn = it->get();
270 const auto output = findOutput(conn->id());
271 const bool stillExists = existing.contains(conn);
272 if (!stillExists || !conn->isConnected()) {
273 if (output) {
274 removeOutput(output);
275 }
276 conn->disable();
277 } else if (!output) {
278 qCDebug(KWIN_DRM, "New %soutput on GPU %s: %s", conn->isNonDesktop() ? "non-desktop " : "", qPrintable(m_devNode), qPrintable(conn->modelName()));
279 const auto pipeline = conn->pipeline();
280 m_pipelines << pipeline;
281 auto output = new DrmOutput(*it);
282 m_drmOutputs << output;
283 addedOutputs << output;
284 Q_EMIT outputAdded(output);
285 pipeline->setLayers(m_platform->renderBackend()->createPrimaryLayer(pipeline), m_platform->renderBackend()->createCursorLayer(pipeline));
286 pipeline->setActive(!conn->isNonDesktop());
287 pipeline->applyPendingChanges();
288 }
289 if (stillExists) {
290 it++;
291 } else {
292 m_allObjects.removeOne(it->get());
293 it = m_connectors.erase(it);
294 }
295 }
296
297 // update crtc properties
298 for (const auto &crtc : std::as_const(m_crtcs)) {
299 crtc->updateProperties();
300 }
301 // update plane properties
302 for (const auto &plane : std::as_const(m_planes)) {
303 plane->updateProperties();
304 }
305 DrmPipeline::Error err = testPendingConfiguration();
306 if (err == DrmPipeline::Error::None) {
307 for (const auto &pipeline : std::as_const(m_pipelines)) {
308 pipeline->applyPendingChanges();
309 if (pipeline->output() && !pipeline->crtc()) {
310 pipeline->setEnable(false);
311 pipeline->output()->updateEnabled(false);
312 }
313 }
314 } else if (err == DrmPipeline::Error::NoPermission) {
315 for (const auto &pipeline : std::as_const(m_pipelines)) {
316 pipeline->revertPendingChanges();
317 }
318 for (const auto &output : std::as_const(addedOutputs)) {
319 removeOutput(output);
320 const auto it = std::find_if(m_connectors.begin(), m_connectors.end(), [output](const auto &conn) {
321 return conn.get() == output->connector();
322 });
323 Q_ASSERT(it != m_connectors.end());
324 m_allObjects.removeOne(it->get());
325 m_connectors.erase(it);
326 }
327 QTimer::singleShot(50, m_platform, &DrmBackend::updateOutputs);
328 } else {
329 qCWarning(KWIN_DRM, "Failed to find a working setup for new outputs!");
330 for (const auto &pipeline : std::as_const(m_pipelines)) {
331 pipeline->revertPendingChanges();
332 }
333 for (const auto &output : std::as_const(addedOutputs)) {
334 output->updateEnabled(false);
335 output->pipeline()->setEnable(false);
336 output->pipeline()->applyPendingChanges();
337 }
338 }
339 return true;
340}

356DrmPipeline::Error DrmGpu::checkCrtcAssignment(QVector<DrmConnector *> connectors, const QVector<DrmCrtc *> &crtcs)
357{
358 if (connectors.isEmpty() || crtcs.isEmpty()) {
359 if (m_pipelines.isEmpty()) {
360 // nothing to do
361 return DrmPipeline::Error::None;
362 }
363 // remaining connectors can't be powered
364 for (const auto &conn : std::as_const(connectors)) {
365 qCWarning(KWIN_DRM) << "disabling connector" << conn->modelName() << "without a crtc";
366 conn->pipeline()->setCrtc(nullptr);
367 }
368 return testPipelines();
369 }
370 auto connector = connectors.takeFirst();
371 auto pipeline = connector->pipeline();
372 if (!pipeline->enabled() || !connector->isConnected()) {
373 // disabled pipelines don't need CRTCs
374 pipeline->setCrtc(nullptr);
375 return checkCrtcAssignment(connectors, crtcs);
376 }
377 DrmCrtc *currentCrtc = nullptr;
378 if (m_atomicModeSetting) {
379 // try the crtc that this connector is already connected to first
380 uint32_t id = connector->getProp(DrmConnector::PropertyIndex::CrtcId)->pending();
381 auto it = std::find_if(crtcs.begin(), crtcs.end(), [id](const auto &crtc) {
382 return id == crtc->id();
383 });
384 if (it != crtcs.end()) {
385 currentCrtc = *it;
386 auto crtcsLeft = crtcs;
387 crtcsLeft.removeOne(currentCrtc);
388 pipeline->setCrtc(currentCrtc);
389 do {
390 DrmPipeline::Error err = checkCrtcAssignment(connectors, crtcsLeft);
391 if (err == DrmPipeline::Error::None || err == DrmPipeline::Error::NoPermission || err == DrmPipeline::Error::FramePending) {
392 return err;
393 }
394 } while (pipeline->pruneModifier());
395 }
396 }
397 for (const auto &crtc : std::as_const(crtcs)) {
398 if (connector->isCrtcSupported(crtc) && crtc != currentCrtc) {
399 auto crtcsLeft = crtcs;
400 crtcsLeft.removeOne(crtc);
401 pipeline->setCrtc(crtc);
402 do {
403 DrmPipeline::Error err = checkCrtcAssignment(connectors, crtcsLeft);
404 if (err == DrmPipeline::Error::None || err == DrmPipeline::Error::NoPermission || err == DrmPipeline::Error::FramePending) {
405 return err;
406 }
407 } while (pipeline->pruneModifier());
408 }
409 }
410 return DrmPipeline::Error::InvalidArguments;
411}

413DrmPipeline::Error DrmGpu::testPendingConfiguration()
414{
415 QVector<DrmConnector *> connectors;
416 QVector<DrmCrtc *> crtcs;
417 // only change resources that aren't currently leased away
418 for (const auto &conn : m_connectors) {
419 bool isLeased = std::any_of(m_drmOutputs.cbegin(), m_drmOutputs.cend(), [&conn](const auto output) {
420 return output->lease() && output->pipeline()->connector() == conn.get();
421 });
422 if (!isLeased) {
423 connectors.push_back(conn.get());
424 }
425 }
426 for (const auto &crtc : m_crtcs) {
427 bool isLeased = std::any_of(m_drmOutputs.cbegin(), m_drmOutputs.cend(), [&crtc](const auto output) {
428 return output->lease() && output->pipeline()->crtc() == crtc.get();
429 });
430 if (!isLeased) {
431 crtcs.push_back(crtc.get());
432 }
433 }
434 if (m_atomicModeSetting) {
435 // sort outputs by being already connected (to any CRTC) so that already working outputs get preferred
436 std::sort(connectors.begin(), connectors.end(), [](auto c1, auto c2) {
437 return c1->getProp(DrmConnector::PropertyIndex::CrtcId)->current() > c2->getProp(DrmConnector::PropertyIndex::CrtcId)->current();
438 });
439 }
440 DrmPipeline::Error err = checkCrtcAssignment(connectors, crtcs);
441 if (err == DrmPipeline::Error::None || err == DrmPipeline::Error::NoPermission || err == DrmPipeline::Error::FramePending) {
442 return err;
443 } else {
444 // try again without hw rotation
445 bool hwRotationUsed = false;
446 for (const auto &pipeline : std::as_const(m_pipelines)) {
447 hwRotationUsed |= (pipeline->bufferOrientation() != DrmPlane::Transformations(DrmPlane::Transformation::Rotate0));
448 pipeline->setBufferOrientation(DrmPlane::Transformation::Rotate0);
449 }
450 if (hwRotationUsed) {
451 err = checkCrtcAssignment(connectors, crtcs);
452 }
453 return err;
454 }
455}

457DrmPipeline::Error DrmGpu::testPipelines()
458{
459 QVector<DrmPipeline *> inactivePipelines;
460 std::copy_if(m_pipelines.constBegin(), m_pipelines.constEnd(), std::back_inserter(inactivePipelines), [](const auto pipeline) {
461 return pipeline->enabled() && !pipeline->active();
462 });
463 DrmPipeline::Error test = DrmPipeline::commitPipelines(m_pipelines, DrmPipeline::CommitMode::TestAllowModeset, unusedObjects());
464 if (!inactivePipelines.isEmpty() && test == DrmPipeline::Error::None) {
465 // ensure that pipelines that are set as enabled but currently inactive
466 // still work when they need to be set active again
467 for (const auto pipeline : std::as_const(inactivePipelines)) {
468 pipeline->setActive(true);
469 }
470 test = DrmPipeline::commitPipelines(m_pipelines, DrmPipeline::CommitMode::TestAllowModeset, unusedObjects());
471 for (const auto pipeline : std::as_const(inactivePipelines)) {
472 pipeline->setActive(false);
473 }
474 }
475 return test;
476}

489void DrmGpu::waitIdle()
490{
491 m_socketNotifier->setEnabled(false);
492 while (true) {
493 const bool idle = std::all_of(m_drmOutputs.constBegin(), m_drmOutputs.constEnd(), [](DrmOutput *output) {
494 return !output->pipeline()->pageflipPending();
495 });
496 if (idle) {
497 break;
498 }
499 pollfd pfds[1];
500 pfds[0].fd = m_fd;
501 pfds[0].events = POLLIN;
502
503 const int ready = poll(pfds, 1, 30000);
504 if (ready < 0) {
505 if (errno != EINTR) {
506 qCWarning(KWIN_DRM) << Q_FUNC_INFO << "poll() failed:" << strerror(errno);
507 break;
508 }
509 } else if (ready == 0) {
510 qCWarning(KWIN_DRM) << "No drm events for gpu" << m_devNode << "within last 30 seconds";
511 break;
512 } else {
513 dispatchEvents();
514 }
515 };
516 m_socketNotifier->setEnabled(true);
517}

519static std::chrono::nanoseconds convertTimestamp(const timespec &timestamp)
520{
521 return std::chrono::seconds(timestamp.tv_sec) + std::chrono::nanoseconds(timestamp.tv_nsec);
522}
523
524static std::chrono::nanoseconds convertTimestamp(clockid_t sourceClock, clockid_t targetClock,
525 const timespec &timestamp)
526{
527 if (sourceClock == targetClock) {
528 return convertTimestamp(timestamp);
529 }
530
531 timespec sourceCurrentTime = {};
532 timespec targetCurrentTime = {};
533
534 clock_gettime(sourceClock, &sourceCurrentTime);
535 clock_gettime(targetClock, &targetCurrentTime);
536
537 const auto delta = convertTimestamp(sourceCurrentTime) - convertTimestamp(timestamp);
538 return convertTimestamp(targetCurrentTime) - delta;
539}

541void DrmGpu::pageFlipHandler(int fd, unsigned int sequence, unsigned int sec, unsigned int usec, unsigned int crtc_id, void *user_data)
542{
543 DrmGpu *gpu = static_cast<DrmGpu *>(user_data);
544
545 // The static_cast<> here are for a 32-bit environment where
546 // sizeof(time_t) == sizeof(unsigned int) == 4 . Putting @p sec
547 // into a time_t cuts off the most-significant bit (after the
548 // year 2038), similarly long can't hold all the bits of an
549 // unsigned multiplication.
550 std::chrono::nanoseconds timestamp = convertTimestamp(gpu->presentationClock(), CLOCK_MONOTONIC,
551 {static_cast<time_t>(sec), static_cast<long>(usec * 1000)});
552 if (timestamp == std::chrono::nanoseconds::zero()) {
553 qCDebug(KWIN_DRM, "Got invalid timestamp (sec: %u, usec: %u) on gpu %s",
554 sec, usec, qPrintable(gpu->devNode()));
555 timestamp = std::chrono::steady_clock::now().time_since_epoch();
556 }
557 const auto pipelines = gpu->pipelines();
558 auto it = std::find_if(pipelines.begin(), pipelines.end(), [crtc_id](const auto &pipeline) {
559 return pipeline->currentCrtc() && pipeline->currentCrtc()->id() == crtc_id;
560 });
561 if (it == pipelines.end()) {
562 qCWarning(KWIN_DRM, "received invalid page flip event for crtc %u", crtc_id);
563 } else {
564 (*it)->pageFlipped(timestamp);
565 }
566}

612std::unique_ptr<DrmLease> DrmGpu::leaseOutputs(const QVector<DrmOutput *> &outputs)
613{
614 QVector<uint32_t> objects;
615 for (DrmOutput *output : outputs) {
616 if (output->lease() || !output->addLeaseObjects(objects)) {
617 return nullptr;
618 }
619 }
620
621 uint32_t lesseeId;
622 FileDescriptor fd{drmModeCreateLease(m_fd, objects.constData(), objects.count(), 0, &lesseeId)};
623 if (!fd.isValid()) {
624 qCWarning(KWIN_DRM) << "Could not create DRM lease!" << strerror(errno);

716bool DrmGpu::maybeModeset()
717{
718 auto pipelines = m_pipelines;
719 for (const auto &output : std::as_const(m_drmOutputs)) {
720 if (output->lease()) {
721 pipelines.removeOne(output->pipeline());
722 }
723 }
724 bool presentPendingForAll = std::all_of(pipelines.constBegin(), pipelines.constEnd(), [](const auto &pipeline) {
725 return pipeline->modesetPresentPending() || !pipeline->activePending();
726 });
727 if (!presentPendingForAll) {
728 // commit only once all pipelines are ready for presentation
729 return true;
730 }
731 // make sure there's no pending pageflips
732 waitIdle();
733 const DrmPipeline::Error err = DrmPipeline::commitPipelines(pipelines, DrmPipeline::CommitMode::CommitModeset, unusedObjects());
734 for (DrmPipeline *pipeline : std::as_const(pipelines)) {
735 if (pipeline->modesetPresentPending()) {
736 pipeline->resetModesetPresentPending();
737 if (err != DrmPipeline::Error::None) {
738 pipeline->output()->frameFailed();
739 }
740 }
741 }
742 if (err == DrmPipeline::Error::None) {
743 return true;
744 } else {
745 if (err != DrmPipeline::Error::FramePending) {
746 QTimer::singleShot(0, m_platform, &DrmBackend::updateOutputs);
747 }
748 return false;
749 }
750}

764 }
