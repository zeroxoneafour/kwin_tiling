50bool Tile::supportsResizeGravity(Gravity gravity)
51{
52 if (!m_parentTile) {
53 return false;
54 }
55
56 switch (gravity) {
57 case Gravity::Left:
58 return m_relativeGeometry.left() > 0.0;
59 case Gravity::Top:
60 return m_relativeGeometry.top() > 0.0;
61 case Gravity::Right:
62 return m_relativeGeometry.right() < 1.0;
63 case Gravity::Bottom:
64 return m_relativeGeometry.bottom() < 1.0;
65 case Gravity::TopLeft:
66 return m_relativeGeometry.top() > 0.0 && m_relativeGeometry.left() > 0.0;
67 case Gravity::TopRight:
68 return m_relativeGeometry.top() > 0.0 && m_relativeGeometry.right() < 1.0;
69 case Gravity::BottomLeft:
70 return m_relativeGeometry.bottom() < 1.0 && m_relativeGeometry.left() > 0.0;
71 case Gravity::BottomRight:
72 return m_relativeGeometry.bottom() < 1.0 && m_relativeGeometry.right() < 1.0;
73 case Gravity::None:
74 default:
75 return false;
76 }
77}

95void Tile::setRelativeGeometry(const QRectF &geom)
96{
97 QRectF constrainedGeom = geom;
98 constrainedGeom.setWidth(std::max(constrainedGeom.width(), s_minimumSize.width()));
99 constrainedGeom.setHeight(std::max(constrainedGeom.height(), s_minimumSize.height()));
100
101 if (m_relativeGeometry == constrainedGeom) {
102 return;
103 }
104
105 m_relativeGeometry = constrainedGeom;
106
107 Q_EMIT relativeGeometryChanged();
108 Q_EMIT absoluteGeometryChanged();
109 Q_EMIT windowGeometryChanged();
110
111 for (auto *w : std::as_const(m_windows)) {
112 w->moveResize(windowGeometry());
113 }
114}

198void Tile::resizeFromGravity(Gravity gravity, int x_root, int y_root)
199{
200 if (!m_parentTile) {
201 return;
202 }
203
204 const QRectF outGeom = m_tiling->output()->fractionalGeometry();
205 const QPointF relativePos = QPointF((x_root - outGeom.x()) / outGeom.width(), (y_root - outGeom.y()) / outGeom.height());
206 QRectF newGeom = m_relativeGeometry;
207
208 switch (gravity) {
209 case Gravity::TopLeft:
210 newGeom.setTopLeft(relativePos - QPointF(m_padding / outGeom.width(), m_padding / outGeom.height()));
211 break;
212 case Gravity::BottomRight:
213 newGeom.setBottomRight(relativePos + QPointF(m_padding / outGeom.width(), m_padding / outGeom.height()));
214 break;
215 case Gravity::BottomLeft:
216 newGeom.setBottomLeft(relativePos + QPointF(-m_padding / outGeom.width(), m_padding / outGeom.height()));
217 break;
218 case Gravity::TopRight:
219 newGeom.setTopRight(relativePos + QPointF(m_padding / outGeom.width(), -m_padding / outGeom.height()));
220 break;
221 case Gravity::Top:
222 newGeom.setTop(relativePos.y() - m_padding / outGeom.height());
223 break;
224 case Gravity::Bottom:
225 newGeom.setBottom(relativePos.y() + m_padding / outGeom.height());
226 break;
227 case Gravity::Left:
228 newGeom.setLeft(relativePos.x() - m_padding / outGeom.width());
229 break;
230 case Gravity::Right:
231 newGeom.setRight(relativePos.x() + m_padding / outGeom.width());
232 break;
233 case Gravity::None:
234 Q_UNREACHABLE();
235 break;
236 }
237
238 setRelativeGeometry(newGeom);
239}

241void Tile::resizeByPixels(qreal delta, Qt::Edge edge)
242{
243 if (!m_parentTile) {
244 return;
245 }
246
247 const auto outGeom = m_tiling->output()->fractionalGeometry();
248 auto newGeom = m_relativeGeometry;
249
250 switch (edge) {
251 case Qt::LeftEdge: {
252 qreal relativeDelta = delta / outGeom.width();
253 newGeom.setLeft(newGeom.left() + relativeDelta);
254 break;
255 }
256 case Qt::TopEdge: {
257 qreal relativeDelta = delta / outGeom.height();
258 newGeom.setTop(newGeom.top() + relativeDelta);
259 break;
260 }
261 case Qt::RightEdge: {
262 qreal relativeDelta = delta / outGeom.width();
263 newGeom.setRight(newGeom.right() + relativeDelta);
264 break;
265 }
266 case Qt::BottomEdge: {
267 qreal relativeDelta = delta / outGeom.height();
268 newGeom.setBottom(newGeom.bottom() + relativeDelta);
269 break;
270 }
271 }
272 setRelativeGeometry(newGeom);
273}

301void Tile::insertChild(int position, Tile *item)
302{
303 Q_ASSERT(position >= 0);
304 const bool wasEmpty = m_children.isEmpty();
305 item->setParent(this);
306
307#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
308 m_children.insert(std::clamp(position, 0, m_children.length()), item);
309#else
310 m_children.insert(std::clamp<qsizetype>(position, 0, m_children.length()), item);
311#endif
312
313 if (wasEmpty) {
314 Q_EMIT isLayoutChanged(true);
315 for (auto *w : std::as_const(m_windows)) {
316 Tile *tile = m_tiling->bestTileForPosition(w->moveResizeGeometry().center());
317 w->setTile(tile);
318 }
319 }
320
321 Q_EMIT childTilesChanged();
322}

330void Tile::removeChild(Tile *child)
331{
332 const bool wasEmpty = m_children.isEmpty();
333 const int idx = m_children.indexOf(child);
334 m_children.removeAll(child);
335 if (m_children.isEmpty() && !wasEmpty) {
336 Q_EMIT isLayoutChanged(false);
337 }
338 if (idx > -1) {
339 for (int i = idx; i < m_children.count(); ++i) {
340 Q_EMIT m_children[i]->rowChanged(i);
341 }
342 }
343 Q_EMIT childTilesChanged();
344}
