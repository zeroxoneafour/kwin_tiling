137void RenderLayer::updateBoundingRect()
138{
139 QRectF boundingRect = rect();
140 for (const RenderLayer *sublayer : std::as_const(m_sublayers)) {
141 boundingRect |= sublayer->boundingRect().translated(sublayer->geometry().topLeft());
142 }
143
144 if (m_boundingRect != boundingRect) {
145 m_boundingRect = boundingRect;
146 if (m_superlayer) {
147 m_superlayer->updateBoundingRect();
148 }

206void RenderLayer::updateEffectiveVisibility()
207{
208 const bool effectiveVisible = computeEffectiveVisibility();
209 if (m_effectiveVisible == effectiveVisible) {
210 return;
211 }
212
213 m_effectiveVisible = effectiveVisible;
214
215 if (effectiveVisible) {
216 addRepaintFull();
217 } else {
218 if (m_outputLayer) {
219 m_outputLayer->addRepaint(mapToGlobal(boundingRect()).toAlignedRect());
220 }
221 }
222
223 for (RenderLayer *sublayer : std::as_const(m_sublayers)) {
224 sublayer->updateEffectiveVisibility();
225 }
226}
