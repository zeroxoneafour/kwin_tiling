39QVariant DecorationsModel::data(const QModelIndex &index, int role) const
40{
41 if (!index.isValid() || index.column() != 0 || index.row() < 0 || index.row() >= int(m_plugins.size())) {
42 return QVariant();
43 }
44 const KDecoration2::DecorationThemeMetaData &d = m_plugins.at(index.row());
45 switch (role) {
46 case Qt::DisplayRole:
47 return d.visibleName();
48 case PluginNameRole:
49 return d.pluginId();
50 case ThemeNameRole:
51 return d.themeName();
52 case ConfigurationRole:
53 return d.hasConfiguration();
54 case RecommendedBorderSizeRole:
55 return Utils::borderSizeToString(d.borderSize());
56 }
57 return QVariant();
58}

115void DecorationsModel::init()
116{
117 beginResetModel();
118 m_plugins.clear();
119 const auto plugins = KPluginMetaData::findPlugins(s_pluginName);
120 for (const auto &info : plugins) {
121 std::unique_ptr<KDecoration2::DecorationThemeProvider> themeFinder(
122 KPluginFactory::instantiatePlugin<KDecoration2::DecorationThemeProvider>(info).plugin);
123 KDecoration2::DecorationThemeMetaData data;
124 const auto decoSettingsMap = info.rawData().value("org.kde.kdecoration2").toObject().toVariantMap();
125 if (themeFinder) {
126 const QString &kns = findKNewStuff(decoSettingsMap);
127 if (!kns.isEmpty() && !m_knsProviders.contains(kns)) {
128 m_knsProviders.append(kns);
129 }
130 if (isThemeEngine(decoSettingsMap)) {
131 const QString keyword = themeListKeyword(decoSettingsMap);
132 if (keyword.isNull()) {
133 // We cannot list the themes
134 continue;
135 }
136 const auto themesList = themeFinder->themes();
137 for (const KDecoration2::DecorationThemeMetaData &data : themesList) {
138 m_plugins.emplace_back(data);
139 }
140
141 // it's a theme engine, we don't want to show this entry
142 continue;
143 }
144 }
145 data.setHasConfiguration(isConfigureable(decoSettingsMap));
146 data.setBorderSize(recommendedBorderSize(decoSettingsMap));
147 data.setVisibleName(info.name().isEmpty() ? info.pluginId() : info.name());
148 data.setPluginId(info.pluginId());
149 data.setThemeName(data.visibleName());
150
151 m_plugins.emplace_back(std::move(data));
152 }
153 endResetModel();
154}

156QModelIndex DecorationsModel::findDecoration(const QString &pluginName, const QString &themeName) const
157{
158 auto it = std::find_if(m_plugins.cbegin(), m_plugins.cend(), [pluginName, themeName](const KDecoration2::DecorationThemeMetaData &d) {
159 return d.pluginId() == pluginName && d.themeName() == themeName;
160 });
161 if (it == m_plugins.cend()) {
162 return QModelIndex();
163 }
164 const auto distance = std::distance(m_plugins.cbegin(), it);
165 return createIndex(distance, 0);
