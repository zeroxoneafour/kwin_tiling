131OffscreenQuickView::OffscreenQuickView(QObject *parent, QWindow *renderWindow, ExportMode exportMode)
132 : QObject(parent)
133 , d(new OffscreenQuickView::Private)
134{
135 d->m_renderControl = std::make_unique<EffectQuickRenderControl>(this, renderWindow);
136
137 d->m_view = std::make_unique<QQuickWindow>(d->m_renderControl.get());
138 d->m_view->setFlags(Qt::FramelessWindowHint);
139 d->m_view->setColor(Qt::transparent);
140
141 if (exportMode == ExportMode::Image) {
142 d->m_useBlit = true;
143 }
144
145 const bool usingGl = d->m_view->rendererInterface()->graphicsApi() == QSGRendererInterface::OpenGL;
146
147 if (!usingGl) {
148 qCDebug(LIBKWINEFFECTS) << "QtQuick Software rendering mode detected";
149 d->m_useBlit = true;
150#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
151 d->m_renderControl->initialize(nullptr);
152#else
153 d->m_renderControl->initialize();
154#endif
155 } else {
156 QSurfaceFormat format;
157 format.setOption(QSurfaceFormat::ResetNotification);
158 format.setDepthBufferSize(16);
159 format.setStencilBufferSize(8);
160
161 auto shareContext = QOpenGLContext::globalShareContext();
162 d->m_glcontext.reset(new QOpenGLContext);
163 d->m_glcontext->setShareContext(shareContext);
164 d->m_glcontext->setFormat(format);
165 d->m_glcontext->create();
166
167 // and the offscreen surface
168 d->m_offscreenSurface.reset(new QOffscreenSurface);
169 d->m_offscreenSurface->setFormat(d->m_glcontext->format());
170 d->m_offscreenSurface->create();
171
172 d->m_glcontext->makeCurrent(d->m_offscreenSurface.get());
173#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
174 d->m_renderControl->initialize(d->m_glcontext.get());
175#else
176 d->m_view->setGraphicsDevice(QQuickGraphicsDevice::fromOpenGLContext(d->m_glcontext.get()));
177 d->m_renderControl->initialize();
178#endif
179 d->m_glcontext->doneCurrent();
180
181 // On Wayland, contexts are implicitly shared and QOpenGLContext::globalShareContext() is null.
182 if (shareContext && !d->m_glcontext->shareContext()) {
183 qCDebug(LIBKWINEFFECTS) << "Failed to create a shared context, falling back to raster rendering";
184 // still render via GL, but blit for presentation
185 d->m_useBlit = true;
186 }
187 }
188
189 auto updateSize = [this]() {
190 contentItem()->setSize(d->m_view->size());
191 };
192 updateSize();
193 connect(d->m_view.get(), &QWindow::widthChanged, this, updateSize);
194 connect(d->m_view.get(), &QWindow::heightChanged, this, updateSize);
195
196 d->m_repaintTimer = std::make_unique<QTimer>();
197 d->m_repaintTimer->setSingleShot(true);
198 d->m_repaintTimer->setInterval(10);
199
200 connect(d->m_repaintTimer.get(), &QTimer::timeout, this, &OffscreenQuickView::update);
201 connect(d->m_renderControl.get(), &QQuickRenderControl::renderRequested, this, &OffscreenQuickView::handleRenderRequested);
202 connect(d->m_renderControl.get(), &QQuickRenderControl::sceneChanged, this, &OffscreenQuickView::handleSceneChanged);
203
204#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
205 d->touchDevice = new QTouchDevice{};
206 d->touchDevice->setCapabilities(QTouchDevice::Position);
207 d->touchDevice->setType(QTouchDevice::TouchScreen);
208 d->touchDevice->setMaximumTouchPoints(10);
209#else
210 d->touchDevice = new QPointingDevice({}, {}, QInputDevice::DeviceType::TouchScreen, {}, QInputDevice::Capability::Position, 10, {});
211#endif
212}

259void OffscreenQuickView::update()
260{
261 if (!d->m_visible) {
262 return;
263 }
264 if (d->m_view->size().isEmpty()) {
265 return;
266 }
267
268 bool usingGl = d->m_glcontext != nullptr;
269
270 if (usingGl) {
271 if (!d->m_glcontext->makeCurrent(d->m_offscreenSurface.get())) {
272 // probably a context loss event, kwin is about to reset all the effects anyway
273 return;
274 }
275
276 const QSize nativeSize = d->m_view->size() * d->m_view->effectiveDevicePixelRatio();
277 if (!d->m_fbo || d->m_fbo->size() != nativeSize) {
278 d->m_textureExport.reset(nullptr);
279 d->m_fbo.reset(new QOpenGLFramebufferObject(nativeSize, QOpenGLFramebufferObject::CombinedDepthStencil));
280 if (!d->m_fbo->isValid()) {
281 d->m_fbo.reset();
282 d->m_glcontext->doneCurrent();
283 return;
284 }
285 }
286#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
287 d->m_view->setRenderTarget(d->m_fbo.get());
288#else
289 d->m_view->setRenderTarget(QQuickRenderTarget::fromOpenGLTexture(d->m_fbo->texture(), d->m_fbo->size()));
290#endif
291 }
292
293 d->m_renderControl->polishItems();
294 d->m_renderControl->sync();
295
296 d->m_renderControl->render();
297 if (usingGl) {
298#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
299 d->m_view->resetOpenGLState();
300#else
301 QQuickOpenGLUtils::resetOpenGLState();
302#endif
303 }
304
305 if (d->m_useBlit) {
306#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
307 d->m_image = d->m_renderControl->grab();
308#else
309 d->m_image = d->m_view->grabWindow();
310#endif
311 }
312
313 if (usingGl) {
314 QOpenGLFramebufferObject::bindDefault();
315 d->m_glcontext->doneCurrent();
316 }
317 Q_EMIT repaintNeeded();
318}

320void OffscreenQuickView::forwardMouseEvent(QEvent *e)
321{
322 if (!d->m_visible) {
323 return;
324 }
325 switch (e->type()) {
326 case QEvent::MouseMove:
327 case QEvent::MouseButtonPress:
328 case QEvent::MouseButtonRelease: {
329 QMouseEvent *me = static_cast<QMouseEvent *>(e);
330 const QPoint widgetPos = d->m_view->mapFromGlobal(me->pos());
331 QMouseEvent cloneEvent(me->type(), widgetPos, me->pos(), me->button(), me->buttons(), me->modifiers());
332 QCoreApplication::sendEvent(d->m_view.get(), &cloneEvent);
333 e->setAccepted(cloneEvent.isAccepted());
334
335 if (e->type() == QEvent::MouseButtonPress) {
336 const ulong doubleClickInterval = static_cast<ulong>(QGuiApplication::styleHints()->mouseDoubleClickInterval());
337 const bool doubleClick = (me->timestamp() - d->lastMousePressTime < doubleClickInterval) && me->button() == d->lastMousePressButton;
338 d->lastMousePressTime = me->timestamp();
339 d->lastMousePressButton = me->button();
340 if (doubleClick) {
341 d->lastMousePressButton = Qt::NoButton;
342 QMouseEvent doubleClickEvent(QEvent::MouseButtonDblClick, me->localPos(), me->windowPos(), me->screenPos(), me->button(), me->buttons(), me->modifiers());
343 QCoreApplication::sendEvent(d->m_view.get(), &doubleClickEvent);
344 }
345 }
346
347 return;
348 }
349 case QEvent::HoverEnter:
350 case QEvent::HoverLeave:
351 case QEvent::HoverMove: {
352 QHoverEvent *he = static_cast<QHoverEvent *>(e);
353 const QPointF widgetPos = d->m_view->mapFromGlobal(he->pos());
354 const QPointF oldWidgetPos = d->m_view->mapFromGlobal(he->oldPos());
355 QHoverEvent cloneEvent(he->type(), widgetPos, oldWidgetPos, he->modifiers());
356 QCoreApplication::sendEvent(d->m_view.get(), &cloneEvent);
357 e->setAccepted(cloneEvent.isAccepted());
358 return;
359 }
360 case QEvent::Wheel: {
361 QWheelEvent *we = static_cast<QWheelEvent *>(e);
362 const QPointF widgetPos = d->m_view->mapFromGlobal(we->position().toPoint());
363 QWheelEvent cloneEvent(widgetPos, we->globalPosition(), we->pixelDelta(), we->angleDelta(), we->buttons(),
364 we->modifiers(), we->phase(), we->inverted());
365 QCoreApplication::sendEvent(d->m_view.get(), &cloneEvent);
366 e->setAccepted(cloneEvent.isAccepted());
367 return;
368 }
369 default:
370 return;
371 }
372}

382bool OffscreenQuickView::forwardTouchDown(qint32 id, const QPointF &pos, std::chrono::microseconds time)
383{
384 d->updateTouchState(Qt::TouchPointPressed, id, pos);
385
386#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)

530void OffscreenQuickView::Private::updateTouchState(Qt::TouchPointState state, qint32 id, const QPointF &pos)
531{
532 // Remove the points that were previously in a released state, since they
533 // are no longer relevant. Additionally, reset the state of all remaining
534 // points to Stationary so we only have one touch point with a different
535 // state.
536 touchPoints.erase(std::remove_if(touchPoints.begin(), touchPoints.end(), [](QTouchEvent::TouchPoint &point) {
537 if (point.state() == Qt::TouchPointReleased) {
538 return true;
539 }
540#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
541 point.setState(Qt::TouchPointStationary);
542#else
543 QMutableEventPoint::setState(point, QEventPoint::Stationary);
544#endif
545 return false;
546 }),
547 touchPoints.end());
548
549 // QtQuick Pointer Handlers incorrectly consider a touch point with ID 0
550 // to be an invalid touch point. This has been fixed in Qt 6 but could not
551 // be fixed for Qt 5. Instead, we offset kwin's internal IDs with this
552 // offset to trick QtQuick into treating them as valid points.
553 static const qint32 idOffset = 111;
554
555 // Find the touch point that has changed. This is separate from the above
556 // loop because removing the released touch points invalidates iterators.
557 auto changed = std::find_if(touchPoints.begin(), touchPoints.end(), [id](const QTouchEvent::TouchPoint &point) {
558 return point.id() == id + idOffset;
559 });
560
561 switch (state) {
562 case Qt::TouchPointPressed: {
563 if (changed != touchPoints.end()) {
564 return;
565 }
566
567 QTouchEvent::TouchPoint point;
568#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
569 point.setState(Qt::TouchPointPressed);
570 point.setId(id + idOffset);
571 point.setScreenPos(pos);
572 point.setScenePos(m_view->mapFromGlobal(pos.toPoint()));
573 point.setPos(m_view->mapFromGlobal(pos.toPoint()));
574#else
575 QMutableEventPoint::setState(point, QEventPoint::Pressed);
576 QMutableEventPoint::setId(point, id + idOffset);
577 QMutableEventPoint::setGlobalPosition(point, pos);
578 QMutableEventPoint::setScenePosition(point, m_view->mapFromGlobal(pos.toPoint()));
579 QMutableEventPoint::setPosition(point, m_view->mapFromGlobal(pos.toPoint()));
580#endif
581
582 touchPoints.append(point);
583 } break;
584 case Qt::TouchPointMoved: {
585 if (changed == touchPoints.end()) {
586 return;
587 }
588
589 auto &point = *changed;
590#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
591 point.setLastPos(point.pos());
592 point.setLastScenePos(point.scenePos());
593 point.setLastScreenPos(point.screenPos());
594 point.setState(Qt::TouchPointMoved);
595 point.setScenePos(m_view->mapFromGlobal(pos.toPoint()));
596 point.setPos(m_view->mapFromGlobal(pos.toPoint()));
597 point.setScreenPos(pos);
598#else
599 QMutableEventPoint::setGlobalLastPosition(point, point.globalPosition());
600 QMutableEventPoint::setState(point, QEventPoint::Updated);
601 QMutableEventPoint::setScenePosition(point, m_view->mapFromGlobal(pos.toPoint()));
602 QMutableEventPoint::setPosition(point, m_view->mapFromGlobal(pos.toPoint()));
603 QMutableEventPoint::setGlobalPosition(point, pos);
604#endif
605 } break;
606 case Qt::TouchPointReleased: {
607 if (changed == touchPoints.end()) {
608 return;
609 }
610
611 auto &point = *changed;
612#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
613 point.setLastPos(point.pos());
614 point.setLastScreenPos(point.screenPos());
615 point.setState(Qt::TouchPointReleased);
616#else
617 QMutableEventPoint::setGlobalLastPosition(point, point.globalPosition());
618 QMutableEventPoint::setState(point, QEventPoint::Released);
619#endif
620 } break;
621 default:
622 break;
623 }
624
625 // The touch state value is used in QTouchEvent and includes all the states
626 // that the current touch points are in.
627#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
628 touchState = std::accumulate(touchPoints.begin(), touchPoints.end(), Qt::TouchPointStates{}, [](auto init, const auto &point) {
629 return init | point.state();
630 });
631#endif
632}

665void OffscreenQuickScene::setSource(const QUrl &source, const QVariantMap &initialProperties)
666{
667 if (!d->qmlComponent) {
668 d->qmlComponent.reset(new QQmlComponent(d->qmlEngine.get()));
669 }
670
671 d->qmlComponent->loadUrl(source);
672 if (d->qmlComponent->isError()) {
673 qCWarning(LIBKWINEFFECTS).nospace() << "Failed to load effect quick view " << source << ": " << d->qmlComponent->errors();
674 d->qmlComponent.reset();
675 return;
676 }
677
678 d->quickItem.reset();
679
680 std::unique_ptr<QObject> qmlObject(d->qmlComponent->createWithInitialProperties(initialProperties));
681 QQuickItem *item = qobject_cast<QQuickItem *>(qmlObject.get());
682 if (!item) {
683 qCWarning(LIBKWINEFFECTS) << "Root object of effect quick view" << source << "is not a QQuickItem";
684 return;
685 }
686
687 qmlObject.release();
688 d->quickItem.reset(item);
689
690 item->setParentItem(contentItem());
691
692 auto updateSize = [item, this]() {
693 item->setSize(contentItem()->size());
694 };
695 updateSize();
696 connect(contentItem(), &QQuickItem::widthChanged, item, updateSize);
697 connect(contentItem(), &QQuickItem::heightChanged, item, updateSize);
698}
