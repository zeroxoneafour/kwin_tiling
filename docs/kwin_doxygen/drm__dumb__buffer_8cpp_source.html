39bool DrmDumbBuffer::map(QImage::Format format)
40{
41 drm_mode_map_dumb mapArgs;
42 memset(&mapArgs, 0, sizeof mapArgs);
43 mapArgs.handle = m_handles[0];
44 if (drmIoctl(m_gpu->fd(), DRM_IOCTL_MODE_MAP_DUMB, &mapArgs) != 0) {
45 return false;
46 }
47#ifdef KWIN_UNIT_TEST
48 m_memory = reinterpret_cast<void *>(mapArgs.offset);
49#else
50 void *address = mmap(nullptr, m_bufferSize, PROT_WRITE, MAP_SHARED, m_gpu->fd(), mapArgs.offset);
51 if (address == MAP_FAILED) {
52 return false;
53 }
54 m_memory = address;
55#endif
56 m_image = std::make_unique<QImage>((uchar *)m_memory, m_size.width(), m_size.height(), m_strides[0], format);
57 return !m_image->isNull();
58}
