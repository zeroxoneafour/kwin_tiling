43// number of key frames for bouncing animation
44static const int BOUNCE_FRAMES = 20;
45// duration between two key frames in msec
46static const int BOUNCE_FRAME_DURATION = 30;
47// duration of one bounce animation
48static const int BOUNCE_DURATION = BOUNCE_FRAME_DURATION * BOUNCE_FRAMES;
49// number of key frames for blinking animation
50static const int BLINKING_FRAMES = 5;
51// duration between two key frames in msec
52static const int BLINKING_FRAME_DURATION = 100;
53// duration of one blinking animation
54static const int BLINKING_DURATION = BLINKING_FRAME_DURATION * BLINKING_FRAMES;
55// const int color_to_pixmap[] = { 0, 1, 2, 3, 2, 1 };
56static const int FRAME_TO_BOUNCE_YOFFSET[] = {
57 -5, -1, 2, 5, 8, 10, 12, 13, 15, 15, 15, 15, 14, 12, 10, 8, 5, 2, -1, -5};
58static const QSize BOUNCE_SIZES[] = {

61 0, 0, 0, 1, 2, 2, 1, 0, 3, 4, 4, 3, 0, 1, 2, 2, 1, 0, 0, 0};

132void StartupFeedbackEffect::reconfigure(Effect::ReconfigureFlags flags)
133{
134 KConfigGroup c = m_configWatcher->config()->group("FeedbackStyle");
135 const bool busyCursor = c.readEntry("BusyCursor", true);
136
137 c = m_configWatcher->config()->group("BusyCursorSettings");
138 m_timeout = std::chrono::seconds(c.readEntry("Timeout", s_startupDefaultTimeout));
139 m_startupInfo->setTimeout(m_timeout.count());
140 const bool busyBlinking = c.readEntry("Blinking", false);
141 const bool busyBouncing = c.readEntry("Bouncing", true);
142 if (!busyCursor) {
143 m_type = NoFeedback;
144 } else if (busyBouncing) {
145 m_type = BouncingFeedback;
146 } else if (busyBlinking) {
147 m_type = BlinkingFeedback;
148 if (effects->compositingType() == OpenGLCompositing) {
149 ensureResources();
150 m_blinkingShader = ShaderManager::instance()->generateShaderFromFile(ShaderTrait::MapTexture, QString(), QStringLiteral(":/effects/startupfeedback/shaders/blinking-startup.frag"));
151 if (m_blinkingShader->isValid()) {
152 qCDebug(KWIN_STARTUPFEEDBACK) << "Blinking Shader is valid";
153 } else {
154 qCDebug(KWIN_STARTUPFEEDBACK) << "Blinking Shader is not valid";
155 }
156 }
157 } else {
158 m_type = PassiveFeedback;
159 }
160 if (m_active) {
161 stop();
162 start(m_startups[m_currentStartup]);
163 }
164}

166void StartupFeedbackEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
167{
168 int time = 0;
169 if (m_lastPresentTime.count()) {
170 time = (presentTime - m_lastPresentTime).count();
171 }
172 m_lastPresentTime = presentTime;
173
174 if (m_active && effects->isCursorHidden()) {
175 stop();
176 }
177 if (m_active) {
178 // need the unclipped version
179 switch (m_type) {
180 case BouncingFeedback:
181 m_progress = (m_progress + time) % BOUNCE_DURATION;
182 m_frame = qRound((qreal)m_progress / (qreal)BOUNCE_FRAME_DURATION) % BOUNCE_FRAMES;
183 m_currentGeometry = feedbackRect(); // bounce alters geometry with m_frame
184 data.paint = data.paint.united(m_currentGeometry);
185 break;
186 case BlinkingFeedback:
187 m_progress = (m_progress + time) % BLINKING_DURATION;
188 m_frame = qRound((qreal)m_progress / (qreal)BLINKING_FRAME_DURATION) % BLINKING_FRAMES;
189 break;
190 default:
191 break; // nothing
192 }
193 }
194 effects->prePaintScreen(data, presentTime);
195}

197void StartupFeedbackEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
198{
199 effects->paintScreen(mask, region, data);
200 if (m_active) {
201 GLTexture *texture;
202 switch (m_type) {
203 case BouncingFeedback:
204 texture = m_bouncingTextures[FRAME_TO_BOUNCE_TEXTURE[m_frame]].get();
205 break;
206 case BlinkingFeedback: // fall through
207 case PassiveFeedback:
208 texture = m_texture.get();
209 break;
210 default:
211 return; // safety
212 }
213 glEnable(GL_BLEND);
214 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
215 texture->bind();
216 if (m_type == BlinkingFeedback && m_blinkingShader && m_blinkingShader->isValid()) {
217 const QColor &blinkingColor = BLINKING_COLORS[FRAME_TO_BLINKING_COLOR[m_frame]];
218 ShaderManager::instance()->pushShader(m_blinkingShader.get());
219 m_blinkingShader->setUniform(GLShader::Color, blinkingColor);
220 } else {
221 ShaderManager::instance()->pushShader(ShaderTrait::MapTexture);
222 }
223 const auto scale = effects->renderTargetScale();
224 QMatrix4x4 mvp = data.projectionMatrix();
225 mvp.translate(m_currentGeometry.x() * scale, m_currentGeometry.y() * scale);
226 ShaderManager::instance()->getBoundShader()->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
227 texture->render(m_currentGeometry.size(), scale);
228 ShaderManager::instance()->popShader();
229 texture->unbind();
230 glDisable(GL_BLEND);
231 }
232}

256void StartupFeedbackEffect::gotNewStartup(const QString &id, const QIcon &icon)
257{
258 Startup &startup = m_startups[id];
259 startup.icon = icon;
260
261 startup.expiredTimer.reset(new QTimer());
262 // Stop the animation if the startup doesn't finish within reasonable interval.
263 connect(startup.expiredTimer.get(), &QTimer::timeout, this, [this, id]() {
264 gotRemoveStartup(id);
265 });
266 startup.expiredTimer->setSingleShot(true);
267 startup.expiredTimer->start(m_timeout);
268
269 m_currentStartup = id;
270 start(startup);
271}

296void StartupFeedbackEffect::start(const Startup &startup)
297{
298 if (m_type == NoFeedback || m_splashVisible || effects->isCursorHidden()) {
299 return;
300 }
301 if (!m_active) {
302 effects->startMousePolling();
303 }
304 m_active = true;
305
306 // read details about the mouse-cursor theme define per default
307 KConfigGroup mousecfg(effects->inputConfig(), "Mouse");
308 m_cursorSize = mousecfg.readEntry("cursorSize", 24);
309
310 int iconSize = m_cursorSize / 1.5;
311 if (!iconSize) {
312 iconSize = QApplication::style()->pixelMetric(QStyle::PM_SmallIconSize);
313 }
314 // get ratio for bouncing cursor so we don't need to manually calculate the sizes for each icon size
315 if (m_type == BouncingFeedback) {
316 m_bounceSizesRatio = iconSize / 16.0;
317 }
318 const QPixmap iconPixmap = startup.icon.pixmap(iconSize);
319 prepareTextures(iconPixmap);
320 m_dirtyRect = m_currentGeometry = feedbackRect();
321 effects->addRepaint(m_dirtyRect);
322}

324void StartupFeedbackEffect::stop()
325{
326 if (m_active) {
327 effects->stopMousePolling();
328 }
329 m_active = false;
330 m_lastPresentTime = std::chrono::milliseconds::zero();
331 effects->makeOpenGLContextCurrent();
332 switch (m_type) {
333 case BouncingFeedback:
334 for (int i = 0; i < 5; ++i) {
335 m_bouncingTextures[i].reset();
336 }
337 break;
338 case BlinkingFeedback:
339 case PassiveFeedback:
340 m_texture.reset();
341 break;
342 case NoFeedback:
343 return; // don't want the full repaint
344 default:
345 break; // impossible
346 }
347 effects->addRepaintFull();
348}

350void StartupFeedbackEffect::prepareTextures(const QPixmap &pix)
351{
352 effects->makeOpenGLContextCurrent();
353 switch (m_type) {
354 case BouncingFeedback:
355 for (int i = 0; i < 5; ++i) {
356 m_bouncingTextures[i].reset(new GLTexture(scalePixmap(pix, BOUNCE_SIZES[i])));
357 m_bouncingTextures[i]->setFilter(GL_LINEAR);
358 m_bouncingTextures[i]->setWrapMode(GL_CLAMP_TO_EDGE);
359 }
360 break;
361 case BlinkingFeedback:
362 case PassiveFeedback:
363 m_texture.reset(new GLTexture(pix));
364 m_texture->setFilter(GL_LINEAR);
365 m_texture->setWrapMode(GL_CLAMP_TO_EDGE);
366 break;
367 default:
368 // for safety
369 m_active = false;
370 m_lastPresentTime = std::chrono::milliseconds::zero();
371 break;
372 }
373}

375QImage StartupFeedbackEffect::scalePixmap(const QPixmap &pm, const QSize &size) const
376{
377 const QSize &adjustedSize = size * m_bounceSizesRatio;
378 QImage scaled = pm.toImage().scaled(adjustedSize, Qt::IgnoreAspectRatio, Qt::SmoothTransformation);
379 if (scaled.format() != QImage::Format_ARGB32_Premultiplied && scaled.format() != QImage::Format_ARGB32) {
380 scaled = scaled.convertToFormat(QImage::Format_ARGB32);
381 }
382
383 QImage result(20 * m_bounceSizesRatio, 20 * m_bounceSizesRatio, QImage::Format_ARGB32);
384 QPainter p(&result);
385 p.setCompositionMode(QPainter::CompositionMode_Source);
386 p.fillRect(result.rect(), Qt::transparent);
387 p.drawImage((20 * m_bounceSizesRatio - adjustedSize.width()) / 2, (20 * m_bounceSizesRatio - adjustedSize.height()) / 2, scaled, 0, 0, adjustedSize.width(), adjustedSize.height() * m_bounceSizesRatio);
388 return result;
389}

391QRect StartupFeedbackEffect::feedbackRect() const
392{
393 int xDiff;
394 if (m_cursorSize <= 16) {
395 xDiff = 8 + 7;
396 } else if (m_cursorSize <= 32) {
397 xDiff = 16 + 7;
398 } else if (m_cursorSize <= 48) {
399 xDiff = 24 + 7;
400 } else {
401 xDiff = 32 + 7;
402 }
403 int yDiff = xDiff;
404 GLTexture *texture = nullptr;
405 int yOffset = 0;
406 switch (m_type) {
407 case BouncingFeedback:
408 texture = m_bouncingTextures[FRAME_TO_BOUNCE_TEXTURE[m_frame]].get();
409 yOffset = FRAME_TO_BOUNCE_YOFFSET[m_frame] * m_bounceSizesRatio;
410 break;
411 case BlinkingFeedback: // fall through
412 case PassiveFeedback:
413 texture = m_texture.get();
414 break;
415 default:
416 // nothing
417 break;
418 }
419 const QPoint cursorPos = effects->cursorPos().toPoint() + QPoint(xDiff, yDiff + yOffset);
420 QRect rect;
421 if (texture) {
422 rect = QRect(cursorPos, texture->size());
423 }
424 return rect;
425}
