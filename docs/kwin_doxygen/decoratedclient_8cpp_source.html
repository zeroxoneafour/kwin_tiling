28DecoratedClientImpl::DecoratedClientImpl(Window *window, KDecoration2::DecoratedClient *decoratedClient, KDecoration2::Decoration *decoration)
29 : QObject()
30 , ApplicationMenuEnabledDecoratedClientPrivate(decoratedClient, decoration)
31 , m_window(window)
32 , m_clientSize(window->clientSize().toSize())
33{
34 window->setDecoratedClient(QPointer<DecoratedClientImpl>(this));
35 connect(window, &Window::activeChanged, this, [decoratedClient, window]() {
36 Q_EMIT decoratedClient->activeChanged(window->isActive());
37 });
38 connect(window, &Window::clientGeometryChanged, this, [decoratedClient, this]() {
39 if (m_window->clientSize() == m_clientSize) {
40 return;
41 }
42 const auto oldSize = m_clientSize;
43 m_clientSize = m_window->clientSize().toSize();
44 if (oldSize.width() != m_clientSize.width()) {
45 Q_EMIT decoratedClient->widthChanged(m_clientSize.width());
46 }
47 if (oldSize.height() != m_clientSize.height()) {
48 Q_EMIT decoratedClient->heightChanged(m_clientSize.height());
49 }
50 Q_EMIT decoratedClient->sizeChanged(m_clientSize);
51 });
52 connect(window, &Window::desktopChanged, this, [decoratedClient, window]() {
53 Q_EMIT decoratedClient->onAllDesktopsChanged(window->isOnAllDesktops());
54 });
55 connect(window, &Window::captionChanged, this, [decoratedClient, window]() {
56 Q_EMIT decoratedClient->captionChanged(window->caption());
57 });
58 connect(window, &Window::iconChanged, this, [decoratedClient, window]() {
59 Q_EMIT decoratedClient->iconChanged(window->icon());
60 });
61 connect(window, &Window::shadeChanged, this, &Decoration::DecoratedClientImpl::signalShadeChange);
62 connect(window, &Window::keepAboveChanged, decoratedClient, &KDecoration2::DecoratedClient::keepAboveChanged);
63 connect(window, &Window::keepBelowChanged, decoratedClient, &KDecoration2::DecoratedClient::keepBelowChanged);
64 connect(window, &Window::quickTileModeChanged, decoratedClient, [this, decoratedClient]() {
65 Q_EMIT decoratedClient->adjacentScreenEdgesChanged(adjacentScreenEdges());
66 });
67 connect(window, &Window::closeableChanged, decoratedClient, &KDecoration2::DecoratedClient::closeableChanged);
68 connect(window, &Window::shadeableChanged, decoratedClient, &KDecoration2::DecoratedClient::shadeableChanged);
69 connect(window, &Window::minimizeableChanged, decoratedClient, &KDecoration2::DecoratedClient::minimizeableChanged);
70 connect(window, &Window::maximizeableChanged, decoratedClient, &KDecoration2::DecoratedClient::maximizeableChanged);
71
72 connect(window, &Window::paletteChanged, decoratedClient, &KDecoration2::DecoratedClient::paletteChanged);
73
74 connect(window, &Window::hasApplicationMenuChanged, decoratedClient, &KDecoration2::DecoratedClient::hasApplicationMenuChanged);
75 connect(window, &Window::applicationMenuActiveChanged, decoratedClient, &KDecoration2::DecoratedClient::applicationMenuActiveChanged);
76
77 m_toolTipWakeUp.setSingleShot(true);
78 connect(&m_toolTipWakeUp, &QTimer::timeout, this, [this]() {
79 int fallAsleepDelay = QApplication::style()->styleHint(QStyle::SH_ToolTip_FallAsleepDelay);
80 this->m_toolTipFallAsleep.setRemainingTime(fallAsleepDelay);
81
82 QToolTip::showText(Cursors::self()->mouse()->pos().toPoint(), this->m_toolTipText);
83 m_toolTipShowing = true;
84 });
85}

178void DecoratedClientImpl::requestShowToolTip(const QString &text)
179{
180 if (!workspace()->decorationBridge()->showToolTips()) {
181 return;
182 }
183
184 m_toolTipText = text;
185
186 int wakeUpDelay = QApplication::style()->styleHint(QStyle::SH_ToolTip_WakeUpDelay);
187 m_toolTipWakeUp.start(m_toolTipFallAsleep.hasExpired() ? wakeUpDelay : 20);
188}

257Qt::Edges DecoratedClientImpl::adjacentScreenEdges() const
258{
259 Qt::Edges edges;
260 const QuickTileMode mode = m_window->quickTileMode();
261 if (mode.testFlag(QuickTileFlag::Left)) {
262 edges |= Qt::LeftEdge;
263 if (!mode.testFlag(QuickTileFlag::Top) && !mode.testFlag(QuickTileFlag::Bottom)) {
264 // using complete side
265 edges |= Qt::TopEdge | Qt::BottomEdge;
266 }
267 }
268 if (mode.testFlag(QuickTileFlag::Top)) {
269 edges |= Qt::TopEdge;
270 }
271 if (mode.testFlag(QuickTileFlag::Right)) {
272 edges |= Qt::RightEdge;
273 if (!mode.testFlag(QuickTileFlag::Top) && !mode.testFlag(QuickTileFlag::Bottom)) {
274 // using complete side
275 edges |= Qt::TopEdge | Qt::BottomEdge;
276 }
277 }
278 if (mode.testFlag(QuickTileFlag::Bottom)) {
279 edges |= Qt::BottomEdge;
280 }
281 return edges;
282}
