53void ColorPickerEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
54{
55 effects->paintScreen(mask, region, data);
56
57 const QRectF geo = effects->renderTargetRect();
58 if (m_scheduledPosition != QPoint(-1, -1) && exclusiveContains(geo, m_scheduledPosition)) {
59 uint8_t data[4];
60 constexpr GLsizei PIXEL_SIZE = 1;
61 const QPoint screenPosition(m_scheduledPosition.x() - geo.x(), m_scheduledPosition.y() - geo.y());
62 const QPoint texturePosition(screenPosition.x() * effects->renderTargetScale(), (geo.height() - screenPosition.y() - PIXEL_SIZE) * effects->renderTargetScale());
63
64 glReadnPixels(texturePosition.x(), texturePosition.y(), PIXEL_SIZE, PIXEL_SIZE, GL_RGBA, GL_UNSIGNED_BYTE, 4, data);
65 QDBusConnection::sessionBus().send(m_replyMessage.createReply(QColor(data[0], data[1], data[2])));
66 m_picking = false;
67 m_scheduledPosition = QPoint(-1, -1);
68 }
69}

71QColor ColorPickerEffect::pick()
72{
73 if (!calledFromDBus()) {
74 return QColor();
75 }
76 if (m_picking) {
77 sendErrorReply(QDBusError::Failed, "Color picking is already in progress");
78 return QColor();
79 }
80 m_picking = true;
81 m_replyMessage = message();
82 setDelayedReply(true);
83 showInfoMessage();
84 effects->startInteractivePositionSelection(
85 [this](const QPointF &p) {
86 hideInfoMessage();
87 if (p == QPointF(-1, -1)) {
88 // error condition
89 QDBusConnection::sessionBus().send(m_replyMessage.createErrorReply(QStringLiteral("org.kde.kwin.ColorPicker.Error.Cancelled"), "Color picking got cancelled"));
90 m_picking = false;
91 } else {
92 m_scheduledPosition = p;
93 effects->addRepaintFull();
94 }
95 });
96 return QColor();
97}
