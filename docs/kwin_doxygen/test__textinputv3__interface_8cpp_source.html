36 void commit_string(const QString &text);
37 void delete_surrounding_text(quint32 before_length, quint32 after_length);
38 void preedit_string(const QString &text, quint32 cursor_begin, quint32 cursor_end);

51 {
52 Q_EMIT surface_leave(surface);

60 before = before_length;

62 }

64 {
65 Q_EMIT commit_string(commitText);

129void TestTextInputV3Interface::initTestCase()
130{
131 m_display.addSocketName(s_socketName);
132 m_display.start();
133 QVERIFY(m_display.isRunning());
134
135 m_seat = new SeatInterface(&m_display, this);
136 m_seat->setHasKeyboard(true);
137
138 m_serverCompositor = new CompositorInterface(&m_display, this);
139 new TextInputManagerV3Interface(&m_display);
140
141 m_connection = new KWayland::Client::ConnectionThread;
142 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
143 m_connection->setSocketName(s_socketName);
144
145 m_thread = new QThread(this);
146 m_connection->moveToThread(m_thread);
147 m_thread->start();
148
149 m_connection->initConnection();
150 QVERIFY(connectedSpy.wait());
151 QVERIFY(!m_connection->connections().isEmpty());
152
153 m_queue = new KWayland::Client::EventQueue(this);
154 QVERIFY(!m_queue->isValid());
155 m_queue->setup(m_connection);
156 QVERIFY(m_queue->isValid());
157
158 auto registry = new KWayland::Client::Registry(this);
159 connect(registry, &KWayland::Client::Registry::interfaceAnnounced, this, [this, registry](const QByteArray &interface, quint32 id, quint32 version) {
160 if (interface == QByteArrayLiteral("zwp_text_input_manager_v3")) {
161 m_clientTextInputManagerV3 = new TextInputManagerV3();
162 m_clientTextInputManagerV3->init(*registry, id, version);
163 }
164 });
165
166 connect(registry, &KWayland::Client::Registry::seatAnnounced, this, [this, registry](quint32 name, quint32 version) {
167 m_clientSeat = registry->createSeat(name, version);
168 });
169
170 QSignalSpy allAnnouncedSpy(registry, &KWayland::Client::Registry::interfaceAnnounced);
171 QSignalSpy compositorSpy(registry, &KWayland::Client::Registry::compositorAnnounced);
172 QSignalSpy shmSpy(registry, &KWayland::Client::Registry::shmAnnounced);
173 registry->setEventQueue(m_queue);
174 registry->create(m_connection->display());
175 QVERIFY(registry->isValid());
176 registry->setup();
177 QVERIFY(allAnnouncedSpy.wait());
178
179 m_clientCompositor = registry->createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
180 QVERIFY(m_clientCompositor->isValid());
181 // create a text input v3
182 m_clientTextInputV3 = new TextInputV3();
183 m_clientTextInputV3->init(m_clientTextInputManagerV3->get_text_input(*m_clientSeat));
184 QVERIFY(m_clientTextInputV3);
185}

212void TestTextInputV3Interface::testEnableDisable()
213{
214 // create a surface
215 QSignalSpy serverSurfaceCreatedSpy(m_serverCompositor, &CompositorInterface::surfaceCreated);
216 std::unique_ptr<KWayland::Client::Surface> clientSurface(m_clientCompositor->createSurface(this));
217 QVERIFY(serverSurfaceCreatedSpy.wait());
218 SurfaceInterface *serverSurface = serverSurfaceCreatedSpy.first().first().value<SurfaceInterface *>();
219 QVERIFY(serverSurface);
220
221 m_serverTextInputV3 = m_seat->textInputV3();
222 QVERIFY(m_serverTextInputV3);
223
224 QSignalSpy focusedSurfaceChangedSpy(m_seat, &SeatInterface::focusedTextInputSurfaceChanged);
225 QSignalSpy textInputEnabledSpy(m_serverTextInputV3, &TextInputV3Interface::enabledChanged);
226 QSignalSpy cursorRectangleChangedSpy(m_serverTextInputV3, &TextInputV3Interface::cursorRectangleChanged);
227 QSignalSpy enableRequestedSpy(m_serverTextInputV3, &TextInputV3Interface::enableRequested);
228
229 QSignalSpy surfaceEnterSpy(m_clientTextInputV3, &TextInputV3::surface_enter);
230 QSignalSpy surfaceLeaveSpy(m_clientTextInputV3, &TextInputV3::surface_leave);
231
232 // Enter the textinput
233
234 QCOMPARE(focusedSurfaceChangedSpy.count(), 0);
235
236 // Make sure that entering surface does not trigger the text input
237 m_seat->setFocusedTextInputSurface(serverSurface);
238 QVERIFY(surfaceEnterSpy.wait());
239 QCOMPARE(surfaceEnterSpy.count(), 1);
240 QCOMPARE(focusedSurfaceChangedSpy.count(), 1);
241 QCOMPARE(textInputEnabledSpy.count(), 0);
242
243 // Now enable the textInput, we should not get event just yet
244 m_clientTextInputV3->enable();
245 m_clientTextInputV3->set_cursor_rectangle(0, 0, 20, 20);
246 m_clientTextInputV3->set_surrounding_text("KDE Plasma Desktop", 0, 3);
247 QCOMPARE(textInputEnabledSpy.count(), 0);
248 QCOMPARE(cursorRectangleChangedSpy.count(), 0);
249
250 // after we do commit we should get event
251 m_clientTextInputV3->commit();
252 QVERIFY(textInputEnabledSpy.wait());
253 m_totalCommits++;
254
255 QCOMPARE(enableRequestedSpy.count(), 0);
256 QCOMPARE(textInputEnabledSpy.count(), 1);
257 QCOMPARE(cursorRectangleChangedSpy.count(), 1);
258 QCOMPARE(m_serverTextInputV3->cursorRectangle(), QRect(0, 0, 20, 20));
259 QCOMPARE(m_serverTextInputV3->surroundingText(), QString("KDE Plasma Desktop"));
260 QCOMPARE(m_serverTextInputV3->surroundingTextCursorPosition(), 0);
261 QCOMPARE(m_serverTextInputV3->surroundingTextSelectionAnchor(), 3);
262
263 // Do another enable when it's already enabled.
264 m_clientTextInputV3->enable();
265 m_clientTextInputV3->set_cursor_rectangle(0, 0, 20, 20);
266 m_clientTextInputV3->set_surrounding_text("KDE Plasma Desktop", 0, 3);
267 m_clientTextInputV3->commit();
268 QVERIFY(enableRequestedSpy.wait());
269 QCOMPARE(textInputEnabledSpy.count(), 1);
270 QCOMPARE(cursorRectangleChangedSpy.count(), 1);
271 QCOMPARE(m_serverTextInputV3->cursorRectangle(), QRect(0, 0, 20, 20));
272 QCOMPARE(m_serverTextInputV3->surroundingText(), QString("KDE Plasma Desktop"));
273 QCOMPARE(m_serverTextInputV3->surroundingTextCursorPosition(), 0);
274 QCOMPARE(m_serverTextInputV3->surroundingTextSelectionAnchor(), 3);
275 m_totalCommits++;
276
277 // disabling we should not get the event
278 m_clientTextInputV3->disable();
279 QCOMPARE(textInputEnabledSpy.count(), 1);
280
281 // after we do commit we should get event
282 m_clientTextInputV3->commit();
283 QVERIFY(textInputEnabledSpy.wait());
284 QCOMPARE(textInputEnabledSpy.count(), 2);
285 m_totalCommits++;
286
287 // Lets try leaving the surface and make sure event propogage
288 m_seat->setFocusedTextInputSurface(nullptr);
289 QVERIFY(surfaceLeaveSpy.wait());
290 QCOMPARE(surfaceLeaveSpy.count(), 1);
291}

293void TestTextInputV3Interface::testEvents()
294{
295 // create a surface
296 QSignalSpy serverSurfaceCreatedSpy(m_serverCompositor, &CompositorInterface::surfaceCreated);
297 std::unique_ptr<KWayland::Client::Surface> clientSurface(m_clientCompositor->createSurface(this));
298 QVERIFY(serverSurfaceCreatedSpy.wait());
299 SurfaceInterface *serverSurface = serverSurfaceCreatedSpy.first().first().value<SurfaceInterface *>();
300 QVERIFY(serverSurface);
301
302 m_serverTextInputV3 = m_seat->textInputV3();
303 QVERIFY(m_serverTextInputV3);
304
305 QSignalSpy focusedSurfaceChangedSpy(m_seat, &SeatInterface::focusedTextInputSurfaceChanged);
306 QSignalSpy textInputEnabledSpy(m_serverTextInputV3, &TextInputV3Interface::enabledChanged);
307 QSignalSpy doneSpy(m_clientTextInputV3, &TextInputV3::done);
308
309 // Enter the textinput
310 QCOMPARE(focusedSurfaceChangedSpy.count(), 0);
311
312 // Make sure that entering surface does not trigger the text input
313 m_seat->setFocusedTextInputSurface(serverSurface);
314 // FIXME: somehow this triggers BEFORE setFocusedTextInputSurface returns :(
315 // QVERIFY(focusedSurfaceChangedSpy.wait());
316 QCOMPARE(focusedSurfaceChangedSpy.count(), 1);
317
318 // Now enable the textInput
319 m_clientTextInputV3->enable();
320 m_clientTextInputV3->commit();
321 m_totalCommits++;
322 QVERIFY(textInputEnabledSpy.wait());
323 QVERIFY(doneSpy.wait());
324 QCOMPARE(doneSpy.count(), 1);
325
326 QSignalSpy preEditSpy(m_clientTextInputV3, &TextInputV3::preedit_string);
327 QSignalSpy commitStringSpy(m_clientTextInputV3, &TextInputV3::commit_string);
328 QSignalSpy deleteSurroundingSpy(m_clientTextInputV3, &TextInputV3::delete_surrounding_text);
329
330 m_serverTextInputV3->sendPreEditString("Hello KDE community!", 1, 2);
331 m_serverTextInputV3->deleteSurroundingText(6, 10);
332 m_serverTextInputV3->commitString("Plasma");
333 m_serverTextInputV3->done();
334
335 QVERIFY(doneSpy.wait());
336 QCOMPARE(doneSpy.count(), 2);
337 QCOMPARE(preEditSpy.count(), 1);
338 QCOMPARE(commitStringSpy.count(), 1);
339 QCOMPARE(deleteSurroundingSpy.count(), 1);
340
341 QCOMPARE(preEditSpy.last().at(0).value<QString>(), "Hello KDE community!");
342 QCOMPARE(preEditSpy.last().at(1).value<quint32>(), 1);
343 QCOMPARE(preEditSpy.last().at(2).value<quint32>(), 2);
344 QCOMPARE(commitStringSpy.last().at(0).value<QString>(), "Plasma");
345 QCOMPARE(deleteSurroundingSpy.last().at(0).value<quint32>(), 6);
346 QCOMPARE(deleteSurroundingSpy.last().at(1).value<quint32>(), 10);
347
348 // zwp_text_input_v3.done event have serial of total commits
349 QCOMPARE(doneSpy.last().at(0).value<quint32>(), m_totalCommits);
350
351 // Now disable the textInput
352 m_clientTextInputV3->disable();
353 m_clientTextInputV3->commit();
354 m_totalCommits++;
355 QVERIFY(textInputEnabledSpy.wait());
356}
