96QRectF WaylandWindow::resizeWithChecks(const QRectF &geometry, const QSizeF &size)
97{
98 const QRectF area = workspace()->clientArea(WorkArea, this, geometry.center());
99
100 qreal width = size.width();
101 qreal height = size.height();
102
103 // don't allow growing larger than workarea
104 if (width > area.width()) {
105 width = area.width();
106 }
107 if (height > area.height()) {
108 height = area.height();
109 }
110 return QRectF(geometry.topLeft(), QSizeF(width, height));
111}

113void WaylandWindow::killWindow()
114{
115 if (!surface()) {
116 return;
117 }
118 auto c = surface()->client();
119 if (c->processId() == getpid() || c->processId() == 0) {
120 c->destroy();
121 return;
122 }
123 ::kill(c->processId(), SIGTERM);
124 // give it time to terminate and only if terminate fails, try destroy Wayland connection
125 QTimer::singleShot(5000, c, &ClientConnection::destroy);
126}

133bool WaylandWindow::belongsToSameApplication(const Window *other, SameApplicationChecks checks) const
134{
135 if (checks.testFlag(SameApplicationCheck::AllowCrossProcesses)) {
136 if (other->desktopFileName() == desktopFileName()) {

139 }
140 if (auto s = other->surface()) {

187void WaylandWindow::updateCaption()
188{
189 const QString oldSuffix = m_captionSuffix;
190 const auto shortcut = shortcutCaptionSuffix();
191 m_captionSuffix = shortcut;
192 if ((!isSpecialWindow() || isToolbar()) && findWindowWithSameCaption()) {
193 int i = 2;
194 do {
195 m_captionSuffix = shortcut + QLatin1String(" <") + QString::number(i) + QLatin1Char('>');
196 i++;
197 } while (findWindowWithSameCaption());
198 }
199 if (m_captionSuffix != oldSuffix) {
200 Q_EMIT captionChanged();
201 }
202}

289void WaylandWindow::updateGeometry(const QRectF &rect)
290{
291 const QRectF oldClientGeometry = m_clientGeometry;
292 const QRectF oldFrameGeometry = m_frameGeometry;
293 const QRectF oldBufferGeometry = m_bufferGeometry;
294 const Output *oldOutput = m_output;
295
296 m_clientGeometry = frameRectToClientRect(rect);
297 m_frameGeometry = rect;
298 m_bufferGeometry = frameRectToBufferRect(rect);
299
300 WaylandGeometryTypes changedGeometries;
301
302 if (m_clientGeometry != oldClientGeometry) {
303 changedGeometries |= WaylandGeometryClient;
304 }
305 if (m_frameGeometry != oldFrameGeometry) {
306 changedGeometries |= WaylandGeometryFrame;
307 }
308 if (m_bufferGeometry != oldBufferGeometry) {
309 changedGeometries |= WaylandGeometryBuffer;
310 }
311
312 if (!changedGeometries) {
313 return;
314 }
315
316 m_output = workspace()->outputAt(rect.center());
317 updateWindowRules(Rules::Position | Rules::Size);
318
319 if (changedGeometries & WaylandGeometryBuffer) {
320 Q_EMIT bufferGeometryChanged(this, oldBufferGeometry);
321 }
322 if (changedGeometries & WaylandGeometryClient) {
323 Q_EMIT clientGeometryChanged(this, oldClientGeometry);
324 }
325 if (changedGeometries & WaylandGeometryFrame) {
326 Q_EMIT frameGeometryChanged(this, oldFrameGeometry);
327 }
328 if (oldOutput != m_output) {
329 Q_EMIT outputChanged();
330 }
331 Q_EMIT geometryShapeChanged(this, oldFrameGeometry);
332}
