147void TestServerSideDecoration::testCreate()
148{
149 using namespace KWaylandServer;
150 QFETCH(KWaylandServer::ServerSideDecorationManagerInterface::Mode, serverMode);
151 m_serverSideDecorationManagerInterface->setDefaultMode(serverMode);
152 QCOMPARE(m_serverSideDecorationManagerInterface->defaultMode(), serverMode);
153
154 QSignalSpy serverSurfaceCreated(m_compositorInterface, &CompositorInterface::surfaceCreated);
155 QSignalSpy decorationCreated(m_serverSideDecorationManagerInterface, &ServerSideDecorationManagerInterface::decorationCreated);
156
157 std::unique_ptr<KWayland::Client::Surface> surface(m_compositor->createSurface());
158 QVERIFY(serverSurfaceCreated.wait());
159
160 auto serverSurface = serverSurfaceCreated.first().first().value<SurfaceInterface *>();
161 QVERIFY(!ServerSideDecorationInterface::get(serverSurface));
162
163 // create server side deco
164 std::unique_ptr<KWayland::Client::ServerSideDecoration> serverSideDecoration(m_serverSideDecorationManager->create(surface.get()));
165 QCOMPARE(serverSideDecoration->mode(), KWayland::Client::ServerSideDecoration::Mode::None);
166 QSignalSpy modeChangedSpy(serverSideDecoration.get(), &KWayland::Client::ServerSideDecoration::modeChanged);
167
168 QVERIFY(decorationCreated.wait());
169
170 auto serverDeco = decorationCreated.first().first().value<ServerSideDecorationInterface *>();
171 QVERIFY(serverDeco);
172 QCOMPARE(serverDeco, ServerSideDecorationInterface::get(serverSurface));
173 QCOMPARE(serverDeco->surface(), serverSurface);
174
175 // after binding the client should get the default mode
176 QVERIFY(modeChangedSpy.wait());
177 QCOMPARE(modeChangedSpy.count(), 1);
178 QTEST(serverSideDecoration->mode(), "clientMode");
179
180 // and destroy
181 QSignalSpy destroyedSpy(serverDeco, &QObject::destroyed);
182 serverSideDecoration.reset();
183 QVERIFY(destroyedSpy.wait());

212void TestServerSideDecoration::testRequest()
213{
214 using namespace KWaylandServer;
215 QFETCH(KWaylandServer::ServerSideDecorationManagerInterface::Mode, defaultMode);
216 m_serverSideDecorationManagerInterface->setDefaultMode(defaultMode);
217 QCOMPARE(m_serverSideDecorationManagerInterface->defaultMode(), defaultMode);
218
219 QSignalSpy serverSurfaceCreated(m_compositorInterface, &CompositorInterface::surfaceCreated);
220 QSignalSpy decorationCreated(m_serverSideDecorationManagerInterface, &ServerSideDecorationManagerInterface::decorationCreated);
221
222 // create server side deco
223 std::unique_ptr<KWayland::Client::Surface> surface(m_compositor->createSurface());
224 std::unique_ptr<KWayland::Client::ServerSideDecoration> serverSideDecoration(m_serverSideDecorationManager->create(surface.get()));
225 QCOMPARE(serverSideDecoration->mode(), KWayland::Client::ServerSideDecoration::Mode::None);
226 QSignalSpy modeChangedSpy(serverSideDecoration.get(), &KWayland::Client::ServerSideDecoration::modeChanged);
227 QVERIFY(decorationCreated.wait());
228
229 auto serverDeco = decorationCreated.first().first().value<ServerSideDecorationInterface *>();
230 QVERIFY(serverDeco);
231 QSignalSpy preferredModeChangedSpy(serverDeco, &ServerSideDecorationInterface::preferredModeChanged);
232
233 // after binding the client should get the default mode
234 QVERIFY(modeChangedSpy.wait());
235 QCOMPARE(modeChangedSpy.count(), 1);
236 QTEST(serverSideDecoration->mode(), "clientMode");
237
238 // request a change
239 QFETCH(KWayland::Client::ServerSideDecoration::Mode, clientRequestMode);
240 serverSideDecoration->requestMode(clientRequestMode);
241 // mode not yet changed
242 QTEST(serverSideDecoration->mode(), "clientMode");
243
244 QVERIFY(preferredModeChangedSpy.wait());
245 QCOMPARE(preferredModeChangedSpy.count(), 1);
246 QFETCH(ServerSideDecorationManagerInterface::Mode, serverRequestedMode);
247 QCOMPARE(serverDeco->preferredMode(), serverRequestedMode);
248
249 // mode not yet changed
250 QCOMPARE(serverDeco->mode(), defaultMode);
251 serverDeco->setMode(serverRequestedMode);
252 QCOMPARE(serverDeco->mode(), serverRequestedMode);
253
254 // should be sent to client
255 QVERIFY(modeChangedSpy.wait());
256 QCOMPARE(modeChangedSpy.count(), 2);
257 QCOMPARE(serverSideDecoration->mode(), clientRequestMode);
258}
