33std::optional<OutputLayerBeginFrameInfo> DrmQPainterLayer::beginFrame()
34{
35 if (!doesSwapchainFit()) {
36 m_swapchain = std::make_shared<DumbSwapchain>(m_pipeline->gpu(), m_pipeline->bufferSize(), DRM_FORMAT_XRGB8888);
37 }
38 QRegion needsRepaint;
39 if (!m_swapchain->acquireBuffer(&needsRepaint)) {
40 return std::nullopt;
41 }
42 return OutputLayerBeginFrameInfo{
43 .renderTarget = RenderTarget(m_swapchain->currentBuffer()->image()),
44 .repaint = needsRepaint,
45 };
46}

48bool DrmQPainterLayer::endFrame(const QRegion &renderedRegion, const QRegion &damagedRegion)
49{
50 m_currentDamage = damagedRegion;
51 m_swapchain->releaseBuffer(m_swapchain->currentBuffer(), damagedRegion);
52 m_currentFramebuffer = DrmFramebuffer::createFramebuffer(m_swapchain->currentBuffer());
53 if (!m_currentFramebuffer) {
54 qCWarning(KWIN_DRM, "Failed to create dumb framebuffer: %s", strerror(errno));
55 }
56 return m_currentFramebuffer != nullptr;
57}

59bool DrmQPainterLayer::checkTestBuffer()
60{
61 if (!doesSwapchainFit()) {
62 m_swapchain = std::make_shared<DumbSwapchain>(m_pipeline->gpu(), m_pipeline->bufferSize(), DRM_FORMAT_XRGB8888);
63 if (!m_swapchain->isEmpty()) {
64 m_currentFramebuffer = DrmFramebuffer::createFramebuffer(m_swapchain->currentBuffer());
65 if (!m_currentFramebuffer) {
66 qCWarning(KWIN_DRM, "Failed to create dumb framebuffer: %s", strerror(errno));
67 }
68 } else {
69 m_currentFramebuffer.reset();
70 }
71 }
72 return m_currentFramebuffer != nullptr;
73}
