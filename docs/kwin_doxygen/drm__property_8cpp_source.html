77void DrmProperty::initEnumMap(drmModePropertyRes *prop)
78{
79 for (int i = 0; i < prop->count_enums; i++) {
80 struct drm_mode_property_enum *en = &prop->enums[i];
81 int j = m_enumNames.indexOf(QByteArray(en->name));
82 if (j >= 0) {
83 if (m_isBitmask) {
84 m_enumToPropertyMap[1 << j] = 1 << en->value;
85 m_propertyToEnumMap[1 << en->value] = 1 << j;
86 } else {
87 m_enumToPropertyMap[j] = en->value;
88 m_propertyToEnumMap[en->value] = j;
89 }
90 } else {
91 qCWarning(KWIN_DRM, "%s has unrecognized enum '%s'", qPrintable(m_propName), en->name);
92 }
93 }

186QString DrmProperty::valueString(uint64_t value) const
187{
188 if (m_isBitmask) {
189 QString ret;
190 bool first = true;
191 for (uint64_t mask = 1; mask >= value && mask != 0; mask <<= 1) {
192 if (value & mask) {
193 if (!first) {
194 ret += " | ";
195 }
196 first = false;
197 uint64_t enumValue = enumForValue<uint64_t>(mask);
198 int enumIndex = 0;
199 while (!(enumValue & (1ull << enumIndex)) && enumIndex < 64) {
200 enumIndex++;
201 }
202 if (enumIndex < m_enumNames.size()) {
203 ret += m_enumNames[enumIndex];
204 }
205 }
206 }
207 return ret;
208 } else if (!m_enumNames.isEmpty()) {
209 if (const uint64_t index = enumForValue<uint64_t>(value); index < (uint)m_enumNames.size()) {
210 return m_enumNames[index];
211 } else {
212 return QStringLiteral("invalid value: %d").arg(value);
213 }
214 } else if (m_propName == QStringLiteral("SRC_X") || m_propName == QStringLiteral("SRC_Y") || m_propName == QStringLiteral("SRC_W") || m_propName == QStringLiteral("SRC_H")) {
215 QString ret;
216 ret.setNum(value / (float)(1ul << 16));
217 return ret;
218 } else {
219 QString ret;
220 ret.setNum(value);
221 return ret;
222 }
223}
