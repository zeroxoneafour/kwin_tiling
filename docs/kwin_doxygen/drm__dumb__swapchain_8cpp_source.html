19DumbSwapchain::DumbSwapchain(DrmGpu *gpu, const QSize &size, uint32_t drmFormat)
20 : m_size(size)
21 , m_format(drmFormat)
22{
23 for (int i = 0; i < 2; i++) {
24 auto buffer = DrmDumbBuffer::createDumbBuffer(gpu, size, drmFormat);
25 if (!buffer->map(QImage::Format::Format_ARGB32)) {
26 break;
27 }
28 buffer->image()->fill(Qt::black);
29 m_slots.append(Slot{
30 .buffer = buffer,
31 .age = 0,
32 });
33 }
34 m_damageJournal.setCapacity(2);
35 if (m_slots.count() < 2) {
36 qCWarning(KWIN_DRM) << "Failed to create dumb buffers for swapchain!";
37 m_slots.clear();
38 }
39}

58void DumbSwapchain::releaseBuffer(const std::shared_ptr<DrmDumbBuffer> &buffer, const QRegion &damage)
59{
60 Q_ASSERT(m_slots[index].buffer == buffer);
61
62 for (qsizetype i = 0; i < m_slots.count(); ++i) {
