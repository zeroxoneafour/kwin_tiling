70void DataSourceInterfacePrivate::data_source_set_actions(Resource *resource, uint32_t dnd_actions)
71{
72 // verify that the no other actions are sent
73 if (dnd_actions
74 & ~(QtWaylandServer::wl_data_device_manager::dnd_action_copy | QtWaylandServer::wl_data_device_manager::dnd_action_move
75 | QtWaylandServer::wl_data_device_manager::dnd_action_ask)) {
76 wl_resource_post_error(resource->handle, error_invalid_action_mask, "Invalid action mask");
77 return;
78 }
79 DataDeviceManagerInterface::DnDActions supportedActions;
80 if (dnd_actions & QtWaylandServer::wl_data_device_manager::dnd_action_copy) {
81 supportedActions |= DataDeviceManagerInterface::DnDAction::Copy;
82 }
83 if (dnd_actions & QtWaylandServer::wl_data_device_manager::dnd_action_move) {
84 supportedActions |= DataDeviceManagerInterface::DnDAction::Move;
85 }
86 if (dnd_actions & QtWaylandServer::wl_data_device_manager::dnd_action_ask) {
87 supportedActions |= DataDeviceManagerInterface::DnDAction::Ask;
88 }
89 if (supportedDnDActions != supportedActions) {
90 supportedDnDActions = supportedActions;
91 Q_EMIT q->supportedDragAndDropActionsChanged();
92 }
93}

161void DataSourceInterface::dndAction(DataDeviceManagerInterface::DnDAction action)
162{
163 d->selectedDndAction = action;
164
165 if (d->resource()->version() < WL_DATA_SOURCE_ACTION_SINCE_VERSION) {
166 return;
167 }
168 uint32_t wlAction = QtWaylandServer::wl_data_device_manager::dnd_action_none;
169 if (action == DataDeviceManagerInterface::DnDAction::Copy) {
170 wlAction = QtWaylandServer::wl_data_device_manager::dnd_action_copy;
171 } else if (action == DataDeviceManagerInterface::DnDAction::Move) {
172 wlAction = QtWaylandServer::wl_data_device_manager::dnd_action_move;
173 } else if (action == DataDeviceManagerInterface::DnDAction::Ask) {
174 wlAction = QtWaylandServer::wl_data_device_manager::dnd_action_ask;
175 }
176 d->send_action(wlAction);
177}
