18inline DmaBufAttributes dmaBufAttributesForBo(gbm_bo *bo)
19{
20 DmaBufAttributes attributes;
21 attributes.planeCount = gbm_bo_get_plane_count(bo);
22 attributes.width = gbm_bo_get_width(bo);
23 attributes.height = gbm_bo_get_height(bo);
24 attributes.format = gbm_bo_get_format(bo);
25 attributes.modifier = gbm_bo_get_modifier(bo);
26
27#if HAVE_GBM_BO_GET_FD_FOR_PLANE
28 for (int i = 0; i < attributes.planeCount; ++i) {
29 attributes.fd[i] = FileDescriptor{gbm_bo_get_fd_for_plane(bo, i)};
30 attributes.offset[i] = gbm_bo_get_offset(bo, i);
31 attributes.pitch[i] = gbm_bo_get_stride_for_plane(bo, i);
32 }
33#else
34 if (attributes.planeCount > 1) {
35 return attributes;
36 }
37
38 attributes.fd[0] = FileDescriptor{gbm_bo_get_fd(bo)};
39 attributes.offset[0] = gbm_bo_get_offset(bo, 0);
40 attributes.pitch[0] = gbm_bo_get_stride_for_plane(bo, 0);
41#endif
42
43 return attributes;
44}

57inline gbm_bo *createGbmBo(gbm_device *device, const QSize &size, quint32 format, const QVector<uint64_t> &modifiers)
58{
59 gbm_bo *bo = nullptr;
60 if (modifiers.count() > 0 && !(modifiers.count() == 1 && modifiers[0] == DRM_FORMAT_MOD_INVALID)) {
61#if HAVE_GBM_BO_CREATE_WITH_MODIFIERS2
62 bo = gbm_bo_create_with_modifiers2(device,
63 size.width(),
64 size.height(),
65 format,
66 modifiers.constData(), modifiers.count(),
67 GBM_BO_USE_RENDERING);
68#else
69 bo = gbm_bo_create_with_modifiers(device,
70 size.width(),
71 size.height(),
72 format,
73 modifiers.constData(), modifiers.count());
74#endif
75 }
76
77 if (!bo && (modifiers.isEmpty() || modifiers.contains(DRM_FORMAT_MOD_INVALID))) {
78 bo = gbm_bo_create(device,
79 size.width(),
80 size.height(),
81 format,
82 GBM_BO_USE_LINEAR);
83 }
84 return bo;
85}
