45std::unique_ptr<Policy> Policy::create(Xkb *xkb, KeyboardLayout *layout, const KConfigGroup &config, const QString &policy)
46{
47 if (policy.toLower() == QStringLiteral("desktop")) {
48 return std::make_unique<VirtualDesktopPolicy>(xkb, layout, config);

97VirtualDesktopPolicy::VirtualDesktopPolicy(Xkb *xkb, KeyboardLayout *layout, const KConfigGroup &config)
98 : Policy(xkb, layout, config)
99{
100 connect(VirtualDesktopManager::self(), &VirtualDesktopManager::currentChanged,
101 this, &VirtualDesktopPolicy::desktopChanged);
102
103 connect(workspace()->sessionManager(), &SessionManager::prepareSessionSaveRequested, this, [this](const QString &name) {
104 clearLayouts();
105
106 for (auto i = m_layouts.constBegin(); i != m_layouts.constEnd(); ++i) {
107 if (const uint layout = *i) {
108 m_config.writeEntry(defaultLayoutEntryKey() % QString::number(i.key()->x11DesktopNumber()), layout);
109 }
110 }
111 });
112
113 connect(workspace()->sessionManager(), &SessionManager::loadSessionRequested, this, [this, xkb](const QString &name) {
114 if (xkb->numberOfLayouts() > 1) {
115 const auto &desktops = VirtualDesktopManager::self()->desktops();
116 for (KWin::VirtualDesktop *const desktop : desktops) {
117 const uint layout = m_config.readEntry(defaultLayoutEntryKey() % QString::number(desktop->x11DesktopNumber()), 0u);
118 if (layout) {
119 m_layouts.insert(desktop, layout);
120 connect(desktop, &VirtualDesktop::aboutToBeDestroyed, this, [this, desktop]() {
121 m_layouts.remove(desktop);
122 });
123 }
124 }
125 desktopChanged();
126 }
127 });

229ApplicationPolicy::ApplicationPolicy(KWin::Xkb *xkb, KWin::KeyboardLayout *layout, const KConfigGroup &config)
230 : Policy(xkb, layout, config)
231{
232 connect(workspace(), &Workspace::windowActivated, this, &ApplicationPolicy::windowActivated);
233
234 connect(workspace()->sessionManager(), &SessionManager::prepareSessionSaveRequested, this, [this](const QString &name) {
235 clearLayouts();
236
237 for (auto i = m_layouts.constBegin(); i != m_layouts.constEnd(); ++i) {
238 if (const uint layout = *i) {
239 const QString desktopFileName = i.key()->desktopFileName();
240 if (!desktopFileName.isEmpty()) {
241 m_config.writeEntry(defaultLayoutEntryKey() % desktopFileName, layout);
242 }
243 }
244 }
245 });
246
247 connect(workspace()->sessionManager(), &SessionManager::loadSessionRequested, this, [this, xkb](const QString &name) {
248 if (xkb->numberOfLayouts() > 1) {
249 const QString keyPrefix = defaultLayoutEntryKey();
250 const QStringList keyList = m_config.keyList().filter(keyPrefix);
251 for (const QString &key : keyList) {
252 m_layoutsRestored.insert(
253 QStringView(key).mid(keyPrefix.size()).toLatin1(),
254 m_config.readEntry(key, 0));
255 }
256 }
257 m_layoutsRestored.squeeze();
258 });
259}

265void ApplicationPolicy::windowActivated(Window *window)
266{
267 if (!window) {
268 return;
269 }
270 // ignore some special types
271 if (window->isDesktop() || window->isDock()) {
272 return;
273 }
274 auto it = m_layouts.constFind(window);
275 if (it != m_layouts.constEnd()) {
276 setLayout(it.value());
277 return;
278 };
279 for (it = m_layouts.constBegin(); it != m_layouts.constEnd(); it++) {
280 if (Window::belongToSameApplication(window, it.key())) {
281 const uint layout = it.value();
282 setLayout(layout);
283 layoutChanged(layout);
284 return;
285 }
286 }
287 setLayout(m_layoutsRestored.take(window->desktopFileName()));
288 if (const uint index = m_xkb->currentLayout()) {
289 layoutChanged(index);
290 }
291}

298void ApplicationPolicy::layoutChanged(uint index)
299{
300 auto window = workspace()->activeWindow();
301 if (!window) {
302 return;
303 }
304 // ignore some special types
305 if (window->isDesktop() || window->isDock()) {
306 return;
307 }
308
309 auto it = m_layouts.find(window);
310 if (it == m_layouts.end()) {
311 m_layouts.insert(window, index);
312 connect(window, &Window::windowClosed, this, [this, window]() {
313 m_layouts.remove(window);
314 });
315 } else {
316 if (it.value() == index) {
317 return;
318 }
319 it.value() = index;
320 }
321 // update all layouts for the application
322 for (it = m_layouts.begin(); it != m_layouts.end(); it++) {
323 if (Window::belongToSameApplication(it.key(), window)) {
324 it.value() = index;
325 }
326 }
327}
