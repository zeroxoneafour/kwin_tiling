38void CompositorInterfacePrivate::compositor_create_surface(Resource *resource, uint32_t id)
39{
40 wl_resource *surfaceResource = wl_resource_create(resource->client(), &wl_surface_interface, resource->version(), id);
41 if (!surfaceResource) {
42 wl_resource_post_no_memory(resource->handle);
43 return;
44 }
45 Q_EMIT q->surfaceCreated(new SurfaceInterface(q, surfaceResource));
