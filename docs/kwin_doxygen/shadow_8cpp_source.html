117QVector<uint32_t> Shadow::readX11ShadowProperty(xcb_window_t id)
118{
119 QVector<uint32_t> ret;
120 if (id != XCB_WINDOW_NONE) {
121 Xcb::Property property(false, id, atoms->kde_net_wm_shadow, XCB_ATOM_CARDINAL, 0, 12);
122 uint32_t *shadow = property.value<uint32_t *>();
123 if (shadow) {
124 ret.reserve(12);
125 for (int i = 0; i < 12; ++i) {
126 ret << shadow[i];
127 }
128 }
129 }
130 return ret;
131}

133bool Shadow::init(const QVector<uint32_t> &data)
134{
135 QVector<Xcb::WindowGeometry> pixmapGeometries(ShadowElementsCount);
136 QVector<xcb_get_image_cookie_t> getImageCookies(ShadowElementsCount);
137 auto *c = kwinApp()->x11Connection();
138 for (int i = 0; i < ShadowElementsCount; ++i) {
139 pixmapGeometries[i] = Xcb::WindowGeometry(data[i]);
140 }
141 auto discardReplies = [&getImageCookies](int start) {
142 for (int i = start; i < getImageCookies.size(); ++i) {
143 xcb_discard_reply(kwinApp()->x11Connection(), getImageCookies.at(i).sequence);
144 }
145 };
146 for (int i = 0; i < ShadowElementsCount; ++i) {
147 auto &geo = pixmapGeometries[i];
148 if (geo.isNull()) {
149 discardReplies(0);
150 return false;
151 }
152 getImageCookies[i] = xcb_get_image_unchecked(c, XCB_IMAGE_FORMAT_Z_PIXMAP, data[i],
153 0, 0, geo->width, geo->height, ~0);
154 }
155 for (int i = 0; i < ShadowElementsCount; ++i) {
156 auto *reply = xcb_get_image_reply(c, getImageCookies.at(i), nullptr);
157 if (!reply) {
158 discardReplies(i + 1);
159 return false;
160 }
161 auto &geo = pixmapGeometries[i];
162 QImage image(xcb_get_image_data(reply), geo->width, geo->height, QImage::Format_ARGB32);
163 m_shadowElements[i] = image.copy();
164 free(reply);
165 }
166 m_offset = QMargins(data[ShadowElementsCount + 3],
167 data[ShadowElementsCount],
168 data[ShadowElementsCount + 1],
169 data[ShadowElementsCount + 2]);
170 Q_EMIT offsetChanged();
171 Q_EMIT textureChanged();
172 return true;
173}

175bool Shadow::init(KDecoration2::Decoration *decoration)
176{
177 if (m_decorationShadow) {
178 // disconnect previous connections
179 disconnect(m_decorationShadow.data(), &KDecoration2::DecorationShadow::innerShadowRectChanged, m_window, &Window::updateShadow);
180 disconnect(m_decorationShadow.data(), &KDecoration2::DecorationShadow::shadowChanged, m_window, &Window::updateShadow);
181 disconnect(m_decorationShadow.data(), &KDecoration2::DecorationShadow::paddingChanged, m_window, &Window::updateShadow);
182 }
183 m_decorationShadow = decoration->shadow();
184 if (!m_decorationShadow) {
185 return false;
186 }
187 // setup connections - all just mapped to recreate
188 connect(m_decorationShadow.data(), &KDecoration2::DecorationShadow::innerShadowRectChanged, m_window, &Window::updateShadow);
189 connect(m_decorationShadow.data(), &KDecoration2::DecorationShadow::shadowChanged, m_window, &Window::updateShadow);
190 connect(m_decorationShadow.data(), &KDecoration2::DecorationShadow::paddingChanged, m_window, &Window::updateShadow);
191
192 m_offset = m_decorationShadow->padding();
193 Q_EMIT offsetChanged();
194 Q_EMIT textureChanged();
195 return true;
196}

207bool Shadow::init(const QPointer<KWaylandServer::ShadowInterface> &shadow)
208{
209 if (!shadow) {
210 return false;
211 }
212
213 m_shadowElements[ShadowElementTop] = shadowTileForBuffer(shadow->top());
214 m_shadowElements[ShadowElementTopRight] = shadowTileForBuffer(shadow->topRight());
215 m_shadowElements[ShadowElementRight] = shadowTileForBuffer(shadow->right());
216 m_shadowElements[ShadowElementBottomRight] = shadowTileForBuffer(shadow->bottomRight());
217 m_shadowElements[ShadowElementBottom] = shadowTileForBuffer(shadow->bottom());
218 m_shadowElements[ShadowElementBottomLeft] = shadowTileForBuffer(shadow->bottomLeft());
219 m_shadowElements[ShadowElementLeft] = shadowTileForBuffer(shadow->left());
220 m_shadowElements[ShadowElementTopLeft] = shadowTileForBuffer(shadow->topLeft());
221
222 m_offset = shadow->offset().toMargins();
223 Q_EMIT offsetChanged();
224 Q_EMIT textureChanged();
225 return true;
226}

228bool Shadow::init(const QWindow *window)
229{
230 const bool isEnabled = window->property("kwin_shadow_enabled").toBool();
231 if (!isEnabled) {
232 return false;
233 }
234
235 const QImage leftTile = window->property("kwin_shadow_left_tile").value<QImage>();
236 const QImage topLeftTile = window->property("kwin_shadow_top_left_tile").value<QImage>();
237 const QImage topTile = window->property("kwin_shadow_top_tile").value<QImage>();
238 const QImage topRightTile = window->property("kwin_shadow_top_right_tile").value<QImage>();
239 const QImage rightTile = window->property("kwin_shadow_right_tile").value<QImage>();
240 const QImage bottomRightTile = window->property("kwin_shadow_bottom_right_tile").value<QImage>();
241 const QImage bottomTile = window->property("kwin_shadow_bottom_tile").value<QImage>();
242 const QImage bottomLeftTile = window->property("kwin_shadow_bottom_left_tile").value<QImage>();
243
244 m_shadowElements[ShadowElementLeft] = leftTile;
245 m_shadowElements[ShadowElementTopLeft] = topLeftTile;
246 m_shadowElements[ShadowElementTop] = topTile;
247 m_shadowElements[ShadowElementTopRight] = topRightTile;
248 m_shadowElements[ShadowElementRight] = rightTile;
249 m_shadowElements[ShadowElementBottomRight] = bottomRightTile;
250 m_shadowElements[ShadowElementBottom] = bottomTile;
251 m_shadowElements[ShadowElementBottomLeft] = bottomLeftTile;
252
253 m_offset = window->property("kwin_shadow_padding").value<QMargins>();
254 Q_EMIT offsetChanged();
255 Q_EMIT textureChanged();
256 return true;
257}

259bool Shadow::updateShadow()
260{
261 if (!m_window) {
262 return false;
263 }
264
265 if (m_decorationShadow) {
266 if (m_window) {
267 if (m_window->decoration()) {
268 if (init(m_window->decoration())) {
269 return true;
270 }
271 }
272 }
273 return false;
274 }
275
276 if (waylandServer()) {
277 if (m_window && m_window->surface()) {
278 if (const auto &s = m_window->surface()->shadow()) {
279 if (init(s)) {
280 return true;
281 }
282 }
283 }
284 }
285
286 if (InternalWindow *window = qobject_cast<InternalWindow *>(m_window)) {
287 if (init(window->handle())) {
288 return true;
289 }
290 }
291
292 auto data = Shadow::readX11ShadowProperty(m_window->window());
293 if (data.isEmpty()) {
294 return false;
295 }
296
297 init(data);
298
299 return true;
300}
