67void VirtualBackend::setVirtualOutputs(const QVector<QRect> &geometries, QVector<qreal> scales)
68{
69 Q_ASSERT(scales.size() == 0 || scales.size() == geometries.size());
70
71 const QVector<VirtualOutput *> removed = m_outputs;
72
73 for (int i = 0; i < geometries.size(); i++) {
74 createOutput(geometries[i].topLeft(), geometries[i].size(), scales.value(i, 1.0));
75 }
76
77 for (VirtualOutput *output : removed) {
78 output->updateEnabled(false);
79 m_outputs.removeOne(output);
80 Q_EMIT outputRemoved(output);
81 output->unref();
82 }
83
84 Q_EMIT outputsQueried();
85}
