281QString Workspace::getPlacementTrackerHash()
282{
283 QStringList hashes;
284 for (const auto &output : std::as_const(m_outputs)) {
285 QCryptographicHash hash(QCryptographicHash::Md5);
286 if (!output->edid().isEmpty()) {
287 hash.addData(output->edid());
288 } else {
289 hash.addData(output->name().toLatin1());
290 }
291 const auto geometry = output->geometry();
292 hash.addData(reinterpret_cast<const char *>(&geometry), sizeof(geometry));
293 hashes.push_back(QString::fromLatin1(hash.result().toHex()));
294 }
295 std::sort(hashes.begin(), hashes.end());
296 const auto hash = QCryptographicHash::hash(hashes.join(QString()).toLatin1(), QCryptographicHash::Md5);
297 return QString::fromLatin1(hash.toHex());
298}

300void Workspace::initializeX11()
301{
302 if (!kwinApp()->x11Connection()) {
303 return;
304 }
305
306 atoms->retrieveHelpers();
307
308 // first initialize the extensions
309 Xcb::Extensions::self();
310 m_colorMapper.reset(new ColorMapper(this));
311 connect(this, &Workspace::windowActivated, m_colorMapper.get(), &ColorMapper::update);
312
313 // Call this before XSelectInput() on the root window
314 m_startup.reset(new KStartupInfo(
315 KStartupInfo::DisableKWinModule | KStartupInfo::AnnounceSilenceChanges, this));
316
317 // Select windowmanager privileges
318 selectWmInputEventMask();
319
320 if (kwinApp()->operationMode() == Application::OperationModeX11) {
321 m_wasUserInteractionFilter.reset(new WasUserInteractionX11Filter);
322 m_movingClientFilter.reset(new MovingClientX11Filter);
323 }
324 if (Xcb::Extensions::self()->isSyncAvailable()) {
325 m_syncAlarmFilter.reset(new SyncAlarmX11Filter);
326 }
327 kwinApp()->updateXTime(); // Needed for proper initialization of user_time in Client ctor
328
329 const uint32_t nullFocusValues[] = {true};
330 m_nullFocus.reset(new Xcb::Window(QRect(-1, -1, 1, 1), XCB_WINDOW_CLASS_INPUT_ONLY, XCB_CW_OVERRIDE_REDIRECT, nullFocusValues));
331 m_nullFocus->map();
332
333 RootInfo *rootInfo = RootInfo::create();
334 const auto vds = VirtualDesktopManager::self();
335 vds->setRootInfo(rootInfo);
336 rootInfo->activate();
337
338 // TODO: only in X11 mode
339 // Extra NETRootInfo instance in Client mode is needed to get the values of the properties
340 NETRootInfo client_info(kwinApp()->x11Connection(), NET::ActiveWindow | NET::CurrentDesktop);
341 bool sessionRestored = false;
342#ifndef QT_NO_SESSIONMANAGER
343 sessionRestored = qApp->isSessionRestored();
344#endif
345 if (!sessionRestored) {
346 m_initialDesktop = client_info.currentDesktop();
347 vds->setCurrent(m_initialDesktop);
348 }
349
350 // TODO: better value
351 rootInfo->setActiveWindow(XCB_WINDOW_NONE);
352 focusToNull();
353
354 if (!sessionRestored) {
355 ++block_focus; // Because it will be set below
356 }
357
358 {
359 // Begin updates blocker block
360 StackingUpdatesBlocker blocker(this);
361
362 Xcb::Tree tree(kwinApp()->x11RootWindow());
363 xcb_window_t *wins = xcb_query_tree_children(tree.data());
364
365 QVector<Xcb::WindowAttributes> windowAttributes(tree->children_len);
366 QVector<Xcb::WindowGeometry> windowGeometries(tree->children_len);
367
368 // Request the attributes and geometries of all toplevel windows
369 for (int i = 0; i < tree->children_len; i++) {
370 windowAttributes[i] = Xcb::WindowAttributes(wins[i]);
371 windowGeometries[i] = Xcb::WindowGeometry(wins[i]);
372 }
373
374 // Get the replies
375 for (int i = 0; i < tree->children_len; i++) {
376 Xcb::WindowAttributes attr(windowAttributes.at(i));
377
378 if (attr.isNull()) {
379 continue;
380 }
381
382 if (attr->override_redirect) {
383 if (attr->map_state == XCB_MAP_STATE_VIEWABLE && attr->_class != XCB_WINDOW_CLASS_INPUT_ONLY) {
384 // ### This will request the attributes again
385 createUnmanaged(wins[i]);
386 }
387 } else if (attr->map_state != XCB_MAP_STATE_UNMAPPED) {
388 if (Application::wasCrash()) {
389 fixPositionAfterCrash(wins[i], windowGeometries.at(i).data());
390 }
391
392 // ### This will request the attributes again
393 createX11Window(wins[i], true);
394 }
395 }
396
397 // Propagate windows, will really happen at the end of the updates blocker block
398 updateStackingOrder(true);
399
400 saveOldScreenSizes();
401 updateClientArea();
402
403 // NETWM spec says we have to set it to (0,0) if we don't support it
404 NETPoint *viewports = new NETPoint[VirtualDesktopManager::self()->count()];
405 rootInfo->setDesktopViewport(VirtualDesktopManager::self()->count(), *viewports);
406 delete[] viewports;
407
408 NETSize desktop_geometry;
409 desktop_geometry.width = m_geometry.width();
410 desktop_geometry.height = m_geometry.height();
411 rootInfo->setDesktopGeometry(desktop_geometry);
412 setShowingDesktop(false);
413
414 } // End updates blocker block
415
416 // TODO: only on X11?
417 Window *newActiveWindow = nullptr;
418 if (!sessionRestored) {
419 --block_focus;
420 newActiveWindow = findClient(Predicate::WindowMatch, client_info.activeWindow());
421 }
422 if (newActiveWindow == nullptr && activeWindow() == nullptr && should_get_focus.count() == 0) {
423 // No client activated in manage()
424 if (newActiveWindow == nullptr) {
425 newActiveWindow = topWindowOnDesktop(VirtualDesktopManager::self()->currentDesktop());
426 }
427 if (newActiveWindow == nullptr) {
428 newActiveWindow = findDesktop(true, VirtualDesktopManager::self()->currentDesktop());
429 }
430 }
431 if (newActiveWindow != nullptr) {
432 activateWindow(newActiveWindow);
433 }
434}

437{
438 // We expect that other components will unregister their X11 event filters after the
439 // connection to the X server has been lost.
440
441 StackingUpdatesBlocker blocker(this);
442
443 // Use stacking_order, so that kwin --replace keeps stacking order.
444 const QList<X11Window *> orderedClients = ensureStackingOrder(m_x11Clients);
445 for (X11Window *client : orderedClients) {
446 client->releaseWindow(true);

458
459 VirtualDesktopManager *desktopManager = VirtualDesktopManager::self();
460 desktopManager->setRootInfo(nullptr);
461
462 X11Window::cleanupX11();
463 RootInfo::destroy();

465
466 m_colorMapper.reset();
467 m_movingClientFilter.reset();
468 m_startup.reset();
469 m_nullFocus.reset();
470 m_syncAlarmFilter.reset();
471 m_wasUserInteractionFilter.reset();
472}

546QMap<Output *, QJsonObject> outputsConfig(const QVector<Output *> &outputs, const QString &hash)
547{
548 const QString kscreenJsonPath = QStandardPaths::locate(QStandardPaths::GenericDataLocation, QStringLiteral("kscreen/") % hash);
549 if (kscreenJsonPath.isEmpty()) {
550 return {};
551 }
552
553 QFile f(kscreenJsonPath);
554 if (!f.open(QIODevice::ReadOnly)) {
555 qCWarning(KWIN_CORE) << "Could not open file" << kscreenJsonPath;
556 return {};
557 }
558
559 QJsonParseError error;
560 const auto doc = QJsonDocument::fromJson(f.readAll(), &error);
561 if (error.error != QJsonParseError::NoError) {
562 qCWarning(KWIN_CORE) << "Failed to parse" << kscreenJsonPath << error.errorString();
563 return {};
564 }
565
566 QHash<Output *, bool> duplicate;
567 QHash<Output *, QString> outputHashes;
568 for (Output *output : outputs) {
569 const QString hash = outputHash(output);
570 const auto it = std::find_if(outputHashes.cbegin(), outputHashes.cend(), [hash](const auto &value) {
571 return value == hash;
572 });
573 if (it == outputHashes.cend()) {
574 duplicate[output] = false;
575 } else {
576 duplicate[output] = true;
577 duplicate[it.key()] = true;
578 }
579 outputHashes[output] = hash;
580 }
581
582 QMap<Output *, QJsonObject> ret;
583 const auto outputsJson = doc.array();
584 for (const auto &outputJson : outputsJson) {
585 const auto outputObject = outputJson.toObject();
586 const auto id = outputObject["id"];
587 const auto output = std::find_if(outputs.begin(), outputs.end(), [&duplicate, &id, &outputObject](Output *output) {
588 if (outputHash(output) != id.toString()) {
589 return false;
590 }
591 if (duplicate[output]) {
592 // can't distinguish between outputs by hash alone, need to look at connector names
593 const auto metadata = outputObject[QStringLiteral("metadata")];
594 const auto outputName = metadata[QStringLiteral("name")].toString();
595 return outputName == output->name();
596 } else {
597 return true;
598 }
599 });
600 if (output != outputs.end()) {
601 ret[*output] = outputObject;
602 }
603 }
604 return ret;
605}

654void Workspace::updateOutputConfiguration()
655{
656 // There's conflict between this code and setVirtualOutputs(), need to adjust the tests.
657 if (QStandardPaths::isTestModeEnabled()) {
658 return;
659 }
660
661 const auto outputs = kwinApp()->outputBackend()->outputs();
662 if (outputs.empty()) {
663 // nothing to do
664 return;
665 }
666 const QString hash = KWinKScreenIntegration::connectedOutputsHash(outputs);
667 const auto outputsInfo = KWinKScreenIntegration::outputsConfig(outputs, hash);
668 m_outputsHash = hash;
669
670 // Update the output order to a fallback list, to avoid dangling pointers
671 const auto setFallbackOutputOrder = [this, &outputs]() {
672 auto newOrder = outputs;
673 newOrder.erase(std::remove_if(newOrder.begin(), newOrder.end(), [](Output *o) {
674 return !o->isEnabled();
675 }),
676 newOrder.end());
677 std::sort(newOrder.begin(), newOrder.end(), [](Output *left, Output *right) {
678 return left->name() < right->name();
679 });
680 setOutputOrder(newOrder);
681 };
682
683 std::vector<std::pair<uint32_t, Output *>> outputOrder;
684 OutputConfiguration cfg;
685 // default position goes from left to right
686 QPoint pos(0, 0);
687 for (const auto &output : std::as_const(outputs)) {
688 if (output->isPlaceholder() || output->isNonDesktop()) {
689 continue;
690 }
691 auto props = cfg.changeSet(output);
692 const QJsonObject outputInfo = outputsInfo[output];
693 qCDebug(KWIN_CORE) << "Reading output configuration for " << output;
694 if (!outputInfo.isEmpty()) {
695 props->enabled = outputInfo["enabled"].toBool(true);
696 if (outputInfo["primary"].toBool()) {
697 outputOrder.push_back(std::make_pair(1, output));
698 if (!props->enabled) {
699 qCWarning(KWIN_CORE) << "KScreen config would disable the primary output!";
700 setFallbackOutputOrder();
701 return;
702 }
703 } else if (int prio = outputInfo["priority"].toInt(); prio > 0) {
704 outputOrder.push_back(std::make_pair(prio, output));
705 if (!props->enabled) {
706 qCWarning(KWIN_CORE) << "KScreen config would disable an output with priority!";
707 setFallbackOutputOrder();
708 return;
709 }
710 } else {
711 outputOrder.push_back(std::make_pair(0, output));
712 }
713 const QJsonObject pos = outputInfo["pos"].toObject();
714 props->pos = QPoint(pos["x"].toInt(), pos["y"].toInt());
715 if (const QJsonValue scale = outputInfo["scale"]; !scale.isUndefined()) {
716 props->scale = scale.toDouble(1.);
717 }
718 props->transform = KWinKScreenIntegration::toDrmTransform(outputInfo["rotation"].toInt());
719
720 props->overscan = static_cast<uint32_t>(outputInfo["overscan"].toInt(props->overscan));
721 props->vrrPolicy = static_cast<RenderLoop::VrrPolicy>(outputInfo["vrrpolicy"].toInt(static_cast<uint32_t>(props->vrrPolicy)));
722 props->rgbRange = static_cast<Output::RgbRange>(outputInfo["rgbrange"].toInt(static_cast<uint32_t>(props->rgbRange)));
723
724 if (const QJsonObject modeInfo = outputInfo["mode"].toObject(); !modeInfo.isEmpty()) {
725 if (auto mode = KWinKScreenIntegration::parseMode(output, modeInfo)) {
726 props->mode = mode;
727 }
728 }
729 } else {
730 props->enabled = true;
731 props->pos = pos;
732 props->transform = output->panelOrientation();
733 outputOrder.push_back(std::make_pair(0, output));
734 }
735 pos.setX(pos.x() + output->geometry().width());
736 }
737 bool allDisabled = std::all_of(outputs.begin(), outputs.end(), [&cfg](const auto &output) {
738 return !cfg.changeSet(output)->enabled;
739 });
740 if (allDisabled) {
741 qCWarning(KWIN_CORE) << "KScreen config would disable all outputs!";
742 setFallbackOutputOrder();
743 return;
744 }
745 std::erase_if(outputOrder, [&cfg](const auto &pair) {
746 return !cfg.constChangeSet(pair.second)->enabled;
747 });
748 std::sort(outputOrder.begin(), outputOrder.end(), [](const auto &left, const auto &right) {
749 if (left.first == right.first) {
750 // sort alphabetically as a fallback
751 return left.second->name() < right.second->name();
752 } else if (left.first == 0) {
753 return false;
754 } else {
755 return left.first < right.first;
756 }
757 });
758 if (!kwinApp()->outputBackend()->applyOutputChanges(cfg)) {
759 qCWarning(KWIN_CORE) << "Applying KScreen config failed!";
760 setFallbackOutputOrder();
761 return;
762 }
763 QVector<Output *> order;
764 order.reserve(outputOrder.size());
765 std::transform(outputOrder.begin(), outputOrder.end(), std::back_inserter(order), [](const auto &pair) {
766 return pair.second;
767 });
768 setOutputOrder(order);
769}

778void Workspace::constrain(Window *below, Window *above)
779{
780 if (below == above) {
781 return;
782 }
783
784 QList<Constraint *> parents;
785 QList<Constraint *> children;
786 for (Constraint *constraint : std::as_const(m_constraints)) {
787 if (constraint->below == below && constraint->above == above) {
788 return;
789 }
790 if (constraint->below == above) {
791 children << constraint;
792 } else if (constraint->above == below) {
793 parents << constraint;
794 }
795 }
796
797 Constraint *constraint = new Constraint();
798 constraint->parents = parents;
799 constraint->below = below;
800 constraint->above = above;
801 constraint->children = children;
802 m_constraints << constraint;
803
804 for (Constraint *parent : std::as_const(parents)) {
805 parent->children << constraint;
806 }
807
808 for (Constraint *child : std::as_const(children)) {
809 child->parents << constraint;
810 }
811
812 updateStackingOrder();
813}

815void Workspace::unconstrain(Window *below, Window *above)
816{
817 Constraint *constraint = nullptr;
818 for (int i = 0; i < m_constraints.count(); ++i) {
819 if (m_constraints[i]->below == below && m_constraints[i]->above == above) {
820 constraint = m_constraints.takeAt(i);
821 break;
822 }
823 }
824
825 if (!constraint) {
826 return;
827 }
828
829 const QList<Constraint *> parents = constraint->parents;
830 for (Constraint *parent : parents) {
831 parent->children.removeOne(constraint);
832 }
833
834 const QList<Constraint *> children = constraint->children;
835 for (Constraint *child : children) {
836 child->parents.removeOne(constraint);
837 }
838
839 delete constraint;
840 updateStackingOrder();
841}

855void Workspace::replaceInStack(Window *original, Deleted *deleted)
856{
857 const int unconstraintedIndex = unconstrained_stacking_order.indexOf(original);
858 if (unconstraintedIndex != -1) {
859 unconstrained_stacking_order.replace(unconstraintedIndex, deleted);
860 } else {
861 // This can be the case only if an override-redirect window is unmapped.
862 unconstrained_stacking_order.append(deleted);
863 }
864
865 const int index = stacking_order.indexOf(original);
866 if (index != -1) {
867 stacking_order.replace(index, deleted);
868 } else {
869 // This can be the case only if an override-redirect window is unmapped.
870 stacking_order.append(deleted);
871 }
872
873 for (Constraint *constraint : std::as_const(m_constraints)) {
874 if (constraint->below == original) {
875 constraint->below = deleted;
876 } else if (constraint->above == original) {
877 constraint->above = deleted;
878 }
879 }

882void Workspace::removeFromStack(Window *window)
883{
884 unconstrained_stacking_order.removeAll(window);
885 stacking_order.removeAll(window);
886
887 for (int i = m_constraints.count() - 1; i >= 0; --i) {
888 Constraint *constraint = m_constraints[i];
889 const bool isBelow = (constraint->below == window);
890 const bool isAbove = (constraint->above == window);
891 if (!isBelow && !isAbove) {
892 continue;
893 }
894 if (isBelow) {
895 for (Constraint *child : std::as_const(constraint->children)) {
896 child->parents.removeOne(constraint);
897 }
898 } else {
899 for (Constraint *parent : std::as_const(constraint->parents)) {
900 parent->children.removeOne(constraint);
901 }
902 }
903 delete m_constraints.takeAt(i);
904 }

1106void Workspace::updateToolWindows(bool also_hide)
1107{
1108 // TODO: What if Client's transiency/group changes? should this be called too? (I'm paranoid, am I not?)
1109 if (!options->isHideUtilityWindowsForInactive()) {
1110 for (auto it = m_x11Clients.constBegin(); it != m_x11Clients.constEnd(); ++it) {
1111 (*it)->showClient();
1112 }
1113 return;
1114 }
1115 const Group *group = nullptr;
1116 auto window = m_activeWindow;
1117 // Go up in transiency hiearchy, if the top is found, only tool transients for the top mainwindow
1118 // will be shown; if a group transient is group, all tools in the group will be shown
1119 while (window != nullptr) {
1120 if (!window->isTransient()) {
1121 break;
1122 }
1123 if (window->groupTransient()) {
1124 group = window->group();
1125 break;
1126 }
1127 window = window->transientFor();
1128 }
1129 // Use stacking order only to reduce flicker, it doesn't matter if block_stacking_updates == 0,
1130 // I.e. if it's not up to date
1131
1132 // SELI TODO: But maybe it should - what if a new window has been added that's not in stacking order yet?
1133 QVector<Window *> to_show, to_hide;
1134 for (auto it = stacking_order.constBegin(); it != stacking_order.constEnd(); ++it) {
1135 auto c = *it;
1136 if (!c->isClient()) {
1137 continue;
1138 }
1139 if (c->isUtility() || c->isMenu() || c->isToolbar()) {
1140 bool show = true;
1141 if (!c->isTransient()) {
1142 if (!c->group() || c->group()->members().count() == 1) { // Has its own group, keep always visible
1143 show = true;
1144 } else if (window != nullptr && c->group() == window->group()) {
1145 show = true;
1146 } else {
1147 show = false;
1148 }
1149 } else {
1150 if (group != nullptr && c->group() == group) {
1151 show = true;
1152 } else if (window != nullptr && window->hasTransient(c, true)) {
1153 show = true;
1154 } else {
1155 show = false;
1156 }
1157 }
1158 if (!show && also_hide) {
1159 const auto mainwindows = c->mainWindows();
1160 // Don't hide utility windows which are standalone(?) or
1161 // have e.g. kicker as mainwindow
1162 if (mainwindows.isEmpty()) {
1163 show = true;
1164 }
1165 for (auto it2 = mainwindows.constBegin(); it2 != mainwindows.constEnd(); ++it2) {
1166 if ((*it2)->isSpecialWindow()) {
1167 show = true;
1168 }
1169 }
1170 if (!show) {
1171 to_hide.append(c);
1172 }
1173 }
1174 if (show) {
1175 to_show.append(c);
1176 }
1177 }
1178 } // First show new ones, then hide
1179 for (int i = to_show.size() - 1; i >= 0; --i) { // From topmost
1180 // TODO: Since this is in stacking order, the order of taskbar entries changes :(
1181 to_show.at(i)->showClient();
1182 }
1183 if (also_hide) {
1184 for (auto it = to_hide.constBegin(); it != to_hide.constEnd(); ++it) { // From bottommost
1185 (*it)->hideClient();
1186 }
1187 updateToolWindowsTimer.stop();
1188 } else { // setActiveWindow() is after called with NULL window, quickly followed
1189 // by setting a new window, which would result in flickering
1190 resetUpdateToolWindowsTimer();
1191 }
1192}

1455{
1456 Output *bestOutput = nullptr;

1458
1459 for (Output *output : std::as_const(m_outputs)) {
1460 const QRect &geo = output->geometry();

1465 distance = std::min(distance, QPointF(geo.topRight() - pos).manhattanLength());
1466 distance = std::min(distance, QPointF(geo.bottomRight() - pos).manhattanLength());
1467 distance = std::min(distance, QPointF(geo.bottomLeft() - pos).manhattanLength());

1469 minDistance = distance;
1470 bestOutput = output;
1471 }
1472 }

1476Output *Workspace::findOutput(Output *reference, Direction direction, bool wrapAround) const
1477{
1478 QList<Output *> relevantOutputs;
1479 std::copy_if(m_outputs.begin(), m_outputs.end(), std::back_inserter(relevantOutputs), [reference, direction](Output *output) {
1480 switch (direction) {
1481 case DirectionEast:
1482 case DirectionWest:
1483 // filter for outputs on same horizontal line
1484 return output->geometry().top() <= reference->geometry().bottom() && output->geometry().bottom() >= reference->geometry().top();
1485 case DirectionSouth:
1486 case DirectionNorth:
1487 // filter for outputs on same vertical line
1488 return output->geometry().left() <= reference->geometry().right() && output->geometry().right() >= reference->geometry().left();
1489 default:
1490 // take all outputs
1491 return true;
1492 }
1493 });
1494
1495 std::sort(relevantOutputs.begin(), relevantOutputs.end(), [direction](const Output *o1, const Output *o2) {
1496 switch (direction) {
1497 case DirectionEast:
1498 case DirectionWest:
1499 // order outputs from left to right
1500 return o1->geometry().center().x() < o2->geometry().center().x();
1501 case DirectionSouth:
1502 case DirectionNorth:
1503 // order outputs from top to bottom
1504 return o1->geometry().center().y() < o2->geometry().center().y();
1505 default:
1506 // order outputs from top to bottom, then left to right
1507 return (o1->geometry().center().y() < o2->geometry().center().y() || (o1->geometry().center().y() == o2->geometry().center().y() && o1->geometry().center().x() < o2->geometry().center().x()));
1508 }
1509 });
1510
1511 const int index = relevantOutputs.indexOf(reference);
1512 Q_ASSERT(index != -1);
1513 switch (direction) {
1514 case DirectionEast:
1515 case DirectionSouth:
1516 case DirectionNext:
1517 // go forward in the list
1518 return relevantOutputs[wrapAround ? (index + 1) % relevantOutputs.count() : std::min(index + 1, (int)relevantOutputs.count() - 1)];
1519 case DirectionWest:
1520 case DirectionNorth:
1521 case DirectionPrev:
1522 // go backward in the list
1523 return relevantOutputs[wrapAround ? (index + relevantOutputs.count() - 1) % relevantOutputs.count() : std::max(index - 1, 0)];
1524 default:
1525 Q_UNREACHABLE();
1526 }
1527}

1620 QuickTileFlag::Bottom | QuickTileFlag::Right,
1621 };

1623 for (const QuickTileMode &quickTileMode : quickTileModes) {
1624 Tile *quickTile = tileManager->quickTile(quickTileMode);

1626 if (windows.isEmpty()) {
1627 continue;
1628 }
1629
1630 Output *bestOutput = outputAt(output->geometry().center());

1636 }
1637 }

1642 output->unref();

2558void Workspace::updateClientArea()
2559{
2560 const QVector<VirtualDesktop *> desktops = VirtualDesktopManager::self()->desktops();
2561
2562 QHash<const VirtualDesktop *, QRectF> workAreas;
2563 QHash<const VirtualDesktop *, StrutRects> restrictedAreas;
2564 QHash<const VirtualDesktop *, QHash<const Output *, QRectF>> screenAreas;
2565
2566 for (const VirtualDesktop *desktop : desktops) {
2567 workAreas[desktop] = m_geometry;
2568
2569 for (const Output *output : std::as_const(m_outputs)) {
2570 screenAreas[desktop][output] = output->fractionalGeometry();
2571 }
2572 }
2573
2574 for (Window *window : std::as_const(m_allClients)) {
2575 if (!window->hasStrut()) {
2576 continue;
2577 }
2578 QRectF r = adjustClientArea(window, m_geometry);
2579
2580 // This happens sometimes when the workspace size changes and the
2581 // struted windows haven't repositioned yet
2582 if (!r.isValid()) {
2583 continue;
2584 }
2585 // sanity check that a strut doesn't exclude a complete screen geometry
2586 // this is a violation to EWMH, as KWin just ignores the strut
2587 for (const Output *output : std::as_const(m_outputs)) {
2588 if (!r.intersects(output->geometry())) {
2589 qCDebug(KWIN_CORE) << "Adjusted client area would exclude a complete screen, ignore";
2590 r = m_geometry;
2591 break;
2592 }
2593 }
2594 StrutRects strutRegion = window->strutRects();
2595 const QRect clientsScreenRect = window->output()->geometry();
2596 for (int i = strutRegion.size() - 1; i >= 0; --i) {
2597 const StrutRect clipped = StrutRect(strutRegion[i].intersected(clientsScreenRect), strutRegion[i].area());
2598 if (clipped.isEmpty()) {
2599 strutRegion.removeAt(i);
2600 } else {
2601 strutRegion[i] = clipped;
2602 }
2603 }
2604
2605 // Ignore offscreen xinerama struts. These interfere with the larger monitors on the setup
2606 // and should be ignored so that applications that use the work area to work out where
2607 // windows can go can use the entire visible area of the larger monitors.
2608 // This goes against the EWMH description of the work area but it is a toss up between
2609 // having unusable sections of the screen (Which can be quite large with newer monitors)
2610 // or having some content appear offscreen (Relatively rare compared to other).
2611 bool hasOffscreenStrut = hasOffscreenXineramaStrut(window);
2612
2613 const auto vds = window->isOnAllDesktops() ? desktops : window->desktops();
2614 for (VirtualDesktop *vd : vds) {
2615 if (!hasOffscreenStrut) {
2616 workAreas[vd] &= r;
2617 }
2618 restrictedAreas[vd] += strutRegion;
2619 for (Output *output : std::as_const(m_outputs)) {
2620 const auto geo = screenAreas[vd][output].intersected(adjustClientArea(window, output->fractionalGeometry()));
2621 // ignore the geometry if it results in the screen getting removed completely
2622 if (!geo.isEmpty()) {
2623 screenAreas[vd][output] = geo;
2624 }
2625 }
2626 }
2627 }
2628
2629 if (m_workAreas != workAreas || m_restrictedAreas != restrictedAreas || m_screenAreas != screenAreas) {
2630 m_workAreas = workAreas;
2631 m_screenAreas = screenAreas;
2632
2633 m_inUpdateClientArea = true;
2634 m_oldRestrictedAreas = m_restrictedAreas;
2635 m_restrictedAreas = restrictedAreas;
2636
2637 if (rootInfo()) {
2638 for (VirtualDesktop *desktop : desktops) {
2639 const QRectF &workArea = m_workAreas[desktop];
2640 NETRect r(Xcb::toXNative(workArea));
2641 rootInfo()->setWorkArea(desktop->x11DesktopNumber(), r);
2642 }
2643 }
2644
2645 for (auto it = m_allClients.constBegin(); it != m_allClients.constEnd(); ++it) {
2646 (*it)->checkWorkspacePosition();
2647 }
2648
2649 m_oldRestrictedAreas.clear(); // reset, no longer valid or needed
2650 m_inUpdateClientArea = false;
2651 }
