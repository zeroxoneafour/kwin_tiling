117WindowQuadList SurfaceItem::buildQuads() const
118{
119 if (!pixmap()) {
120 return {};
121 }
122
123 const QVector<QRectF> region = shape();
124 const auto size = pixmap()->size();
125
126 WindowQuadList quads;
127 quads.reserve(region.count());
128
129 for (const QRectF rect : region) {
130 WindowQuad quad;
131
132 // Use toPoint to round the device position to match what we eventually
133 // do for the geometry, otherwise we end up with mismatched UV
134 // coordinates as the texture size is going to be in (rounded) device
135 // coordinates as well.
136 const QPointF bufferTopLeft = m_surfaceToBufferMatrix.map(rect.topLeft()).toPoint();
137 const QPointF bufferTopRight = m_surfaceToBufferMatrix.map(rect.topRight()).toPoint();
138 const QPointF bufferBottomRight = m_surfaceToBufferMatrix.map(rect.bottomRight()).toPoint();
139 const QPointF bufferBottomLeft = m_surfaceToBufferMatrix.map(rect.bottomLeft()).toPoint();
140
141 quad[0] = WindowVertex(rect.topLeft(), QPointF{bufferTopLeft.x() / size.width(), bufferTopLeft.y() / size.height()});
142 quad[1] = WindowVertex(rect.topRight(), QPointF{bufferTopRight.x() / size.width(), bufferTopRight.y() / size.height()});
143 quad[2] = WindowVertex(rect.bottomRight(), QPointF{bufferBottomRight.x() / size.width(), bufferBottomRight.y() / size.height()});
144 quad[3] = WindowVertex(rect.bottomLeft(), QPointF{bufferBottomLeft.x() / size.width(), bufferBottomLeft.y() / size.height()});
145
146 quads << quad;
147 }
148
149 return quads;
150}
