25 : Effect()
26{

28 ensureResources();

55qreal transformAngle(EffectScreen::Transform current, EffectScreen::Transform old)
56{
57 auto ensureShort = [](int angle) {
58 return angle > 180 ? angle - 360 : angle < -180 ? angle + 360
59 : angle;
60 };
61 // % 4 to ignore flipped cases (for now)
62 return ensureShort((int(current) % 4 - int(old) % 4) * 90);
63}

65void ScreenTransformEffect::addScreen(EffectScreen *screen)
66{
67 connect(screen, &EffectScreen::changed, this, [this, screen] {
68 auto &state = m_states[screen];
69 if (screen->transform() == state.m_oldTransform) {
70 effects->makeOpenGLContextCurrent();
71 m_states.remove(screen);
72 return;
73 }
74 state.m_timeLine.setDuration(std::chrono::milliseconds(long(animationTime(250))));
75 state.m_timeLine.setEasingCurve(QEasingCurve::InOutCubic);
76 state.m_angle = transformAngle(screen->transform(), state.m_oldTransform);
77 Q_ASSERT(state.m_angle != 0);
78 effects->addRepaintFull();
79 });
80 connect(screen, &EffectScreen::aboutToChange, this, [this, screen] {
81 effects->makeOpenGLContextCurrent();
82 auto &state = m_states[screen];
83 state.m_oldTransform = screen->transform();
84 state.m_oldGeometry = screen->geometry();
85 state.m_prev.texture.reset(new GLTexture(GL_RGBA8, screen->geometry().size() * screen->devicePixelRatio()));
86 state.m_prev.framebuffer.reset(new GLFramebuffer(state.m_prev.texture.get()));
87
88 // Rendering the current scene into a texture
89 GLFramebuffer::pushFramebuffer(state.m_prev.framebuffer.get());
90 effects->renderScreen(screen);
91 GLFramebuffer::popFramebuffer();
92
93 // Now, the effect can cross-fade between current and previous state.
94 state.m_captured = true;
95 });
96}

105void ScreenTransformEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
106{
107 auto it = m_states.find(data.screen);
108 if (it != m_states.end() && it->m_captured) {
109 it->m_timeLine.advance(presentTime);
110 if (it->m_timeLine.done()) {
111 m_states.remove(data.screen);
112 }
113 }
114
115 effects->prePaintScreen(data, presentTime);
116}

118static GLVertexBuffer *texturedRectVbo(const QRectF &geometry, qreal scale)
119{
120 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
121 vbo->reset();
122
123 const GLVertexAttrib attribs[] = {
124 {VA_Position, 2, GL_FLOAT, offsetof(GLVertex2D, position)},
125 {VA_TexCoord, 2, GL_FLOAT, offsetof(GLVertex2D, texcoord)},
126 };
127 vbo->setAttribLayout(attribs, 2, sizeof(GLVertex2D));
128
129 auto map = static_cast<GLVertex2D *>(vbo->map(6 * sizeof(GLVertex2D)));
130
131 auto deviceGeometry = scaledRect(geometry, scale);
132
133 // first triangle
134 map[0] = GLVertex2D{
135 .position = QVector2D(deviceGeometry.left(), deviceGeometry.top()),
136 .texcoord = QVector2D(0.0, 1.0),
137 };
138 map[1] = GLVertex2D{
139 .position = QVector2D(deviceGeometry.right(), deviceGeometry.bottom()),
140 .texcoord = QVector2D(1.0, 0.0),
141 };
142 map[2] = GLVertex2D{
143 .position = QVector2D(deviceGeometry.left(), deviceGeometry.bottom()),
144 .texcoord = QVector2D(0.0, 0.0),
145 };
146
147 // second triangle
148 map[3] = GLVertex2D{
149 .position = QVector2D(deviceGeometry.left(), deviceGeometry.top()),
150 .texcoord = QVector2D(0.0, 1.0),
151 };
152 map[4] = GLVertex2D{
153 .position = QVector2D(deviceGeometry.right(), deviceGeometry.top()),
154 .texcoord = QVector2D(1.0, 1.0),
155 };
156 map[5] = GLVertex2D{
157 .position = QVector2D(deviceGeometry.right(), deviceGeometry.bottom()),
158 .texcoord = QVector2D(1.0, 0.0),
159 };
160
161 vbo->unmap();
162 return vbo;
163}

165static qreal lerp(qreal a, qreal b, qreal t)
166{
167 return (1 - t) * a + t * b;
168}

170static QRectF lerp(const QRectF &a, const QRectF &b, qreal t)
171{
172 QRectF ret;
173 ret.setWidth(lerp(a.width(), b.width(), t));
174 ret.setHeight(lerp(a.height(), b.height(), t));
175 ret.moveCenter(b.center());
176 return ret;

179void ScreenTransformEffect::paintScreen(int mask, const QRegion &region, KWin::ScreenPaintData &data)
180{
181 EffectScreen *screen = data.screen();
182
183 auto it = m_states.find(screen);
184 if (it == m_states.end() || !it->m_captured) {
185 effects->paintScreen(mask, region, data);
186 return;
187 }
188
189 // Render the screen in an offscreen texture.
190 const QSize nativeSize = screen->geometry().size() * screen->devicePixelRatio();
191 if (!it->m_current.texture || it->m_current.texture->size() != nativeSize) {
192 it->m_current.texture.reset(new GLTexture(GL_RGBA8, nativeSize));
193 it->m_current.framebuffer.reset(new GLFramebuffer(it->m_current.texture.get()));
194 }
195
196 GLFramebuffer::pushFramebuffer(it->m_current.framebuffer.get());
197 effects->paintScreen(mask, region, data);
198 GLFramebuffer::popFramebuffer();
199
200 const qreal blendFactor = it->m_timeLine.value();
201 const QRectF screenRect = screen->geometry();
202 const qreal angle = it->m_angle * (1 - blendFactor);
203
204 const auto scale = effects->renderTargetScale();
205
206 // Projection matrix + rotate transform.
207 const QVector3D transformOrigin(screenRect.center());
208 QMatrix4x4 modelViewProjectionMatrix(data.projectionMatrix());
209 modelViewProjectionMatrix.translate(transformOrigin * scale);
210 modelViewProjectionMatrix.rotate(angle, 0, 0, 1);
211 modelViewProjectionMatrix.translate(-transformOrigin * scale);
212
213 glActiveTexture(GL_TEXTURE1);
214 it->m_prev.texture->bind();
215 glActiveTexture(GL_TEXTURE0);
216 it->m_current.texture->bind();
217
218 // Clear the background.
219 glClearColor(0, 0, 0, 0);
220 glClear(GL_COLOR_BUFFER_BIT);
221
222 ShaderManager *sm = ShaderManager::instance();
223 sm->pushShader(m_shader.get());
224 m_shader->setUniform(m_modelViewProjectioMatrixLocation, modelViewProjectionMatrix);
225 m_shader->setUniform(m_blendFactorLocation, float(blendFactor));
226 m_shader->setUniform(m_currentTextureLocation, 0);
227 m_shader->setUniform(m_previousTextureLocation, 1);
228
229 GLVertexBuffer *vbo = texturedRectVbo(lerp(it->m_oldGeometry, screenRect, blendFactor), scale);
230 vbo->bindArrays();
231 vbo->draw(GL_TRIANGLES, 0, 6);
232 vbo->unbindArrays();
233 sm->popShader();
234
235 glActiveTexture(GL_TEXTURE1);
236 it->m_prev.texture->unbind();
237 glActiveTexture(GL_TEXTURE0);
238 it->m_current.texture->unbind();
239
240 effects->addRepaintFull();
241}
