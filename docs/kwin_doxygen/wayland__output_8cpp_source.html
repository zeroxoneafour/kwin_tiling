95void WaylandCursor::sync()
96{
97 if (!m_enabled) {
98 m_surface->attachBuffer(KWayland::Client::Buffer::Ptr());
99 m_surface->commit(KWayland::Client::Surface::CommitFlag::None);
100 } else {
101 m_surface->attachBuffer(m_buffer);
102 m_surface->setScale(std::ceil(m_scale));
103 m_surface->damageBuffer(QRect(0, 0, INT32_MAX, INT32_MAX));
104 m_surface->commit(KWayland::Client::Surface::CommitFlag::None);
105 }
106
107 if (m_pointer) {
108 m_pointer->setCursor(m_surface.get(), m_hotspot);
109 }
110}

113 : Output(backend)
114 , m_renderLoop(std::make_unique<RenderLoop>())
115 , m_surface(backend->display()->compositor()->createSurface())
116 , m_xdgShellSurface(backend->display()->xdgShell()->createSurface(m_surface.get()))

119{

122 m_xdgDecoration->setMode(KWayland::Client::XdgDecoration::Mode::ServerSide);
123 }
124
125 setInformation(Information{
126 .name = name,
127 .model = name,
128 .capabilities = Capability::Dpms,
129 });
130
131 m_turnOffTimer.setSingleShot(true);
132 m_turnOffTimer.setInterval(dimAnimationTime());
133 connect(&m_turnOffTimer, &QTimer::timeout, this, [this] {
134 updateDpmsMode(DpmsMode::Off);
135 });
136
137 connect(m_surface.get(), &KWayland::Client::Surface::frameRendered, this, [this]() {
138 RenderLoopPrivate *renderLoopPrivate = RenderLoopPrivate::get(renderLoop());
139 renderLoopPrivate->notifyFrameCompleted(std::chrono::steady_clock::now().time_since_epoch());
140 });
141
142 updateWindowTitle();
143
144 connect(m_xdgShellSurface.get(), &XdgShellSurface::configureRequested, this, &WaylandOutput::handleConfigure);
145 connect(m_xdgShellSurface.get(), &XdgShellSurface::closeRequested, qApp, &QCoreApplication::quit);
146 connect(this, &WaylandOutput::enabledChanged, this, &WaylandOutput::updateWindowTitle);
147 connect(this, &WaylandOutput::dpmsModeChanged, this, &WaylandOutput::updateWindowTitle);

182bool WaylandOutput::setCursor(CursorSource *source)
183{
184 if (m_hasPointerLock) {
185 return false;
186 }
187
188 OutputLayer *cursorLayer = Compositor::self()->backend()->cursorLayer(this);
189 if (source) {
190 cursorLayer->setSize(source->size());
191 cursorLayer->setScale(scale());
192 cursorLayer->setHotspot(source->hotspot());
193 } else {
194 cursorLayer->setSize(QSize());
195 cursorLayer->setHotspot(QPoint());
196 }
197
198 std::optional<OutputLayerBeginFrameInfo> beginInfo = cursorLayer->beginFrame();
199 if (!beginInfo) {
200 return false;
201 }
202
203 RenderTarget *renderTarget = &beginInfo->renderTarget;
204 renderTarget->setDevicePixelRatio(scale());
205
206 RenderLayer renderLayer(m_renderLoop.get());
207 renderLayer.setDelegate(std::make_unique<SceneDelegate>(Compositor::self()->cursorScene()));
208
209 renderLayer.delegate()->prePaint();
210 renderLayer.delegate()->paint(renderTarget, infiniteRegion());
211 renderLayer.delegate()->postPaint();
212
213 cursorLayer->endFrame(infiniteRegion(), infiniteRegion());
214 return true;
215}

250void WaylandOutput::setDpmsMode(DpmsMode mode)
251{
252 if (mode == DpmsMode::Off) {
253 if (!m_turnOffTimer.isActive()) {
254 Q_EMIT aboutToTurnOff(std::chrono::milliseconds(m_turnOffTimer.interval()));
255 m_turnOffTimer.start();
256 }
257 m_backend->createDpmsFilter();
258 } else {
259 m_turnOffTimer.stop();
260 m_backend->clearDpmsFilter();
261
262 if (mode != dpmsMode()) {
263 updateDpmsMode(mode);
264 Q_EMIT wakeUp();
265 }
266 }
267}

293void WaylandOutput::updateWindowTitle()
294{
295 QString grab;
296 if (m_hasPointerLock) {
297 grab = i18n("Press right control to ungrab pointer");
298 } else if (m_backend->display()->pointerConstraints()) {
299 grab = i18n("Press right control key to grab pointer");
300 }
301
302 QString title = i18nc("Title of nested KWin Wayland with Wayland socket identifier as argument",
303 "KDE Wayland Compositor %1 (%2)", name(), waylandServer()->socketName());
304
305 if (!isEnabled()) {
306 title += i18n("- Output disabled");
307 } else if (dpmsMode() != DpmsMode::On) {
308 title += i18n("- Output dimmed");
309 } else if (!grab.isEmpty()) {
310 title += QStringLiteral(" ") + grab;
311 }
312 m_xdgShellSurface->setTitle(title);
313}

315void WaylandOutput::lockPointer(Pointer *pointer, bool lock)
316{
317 if (!lock) {
318 const bool surfaceWasLocked = m_pointerLock && m_hasPointerLock;
319 m_pointerLock.reset();
320 m_hasPointerLock = false;
321 if (surfaceWasLocked) {
322 updateWindowTitle();
323 m_cursor->enable();
324 Q_EMIT m_backend->pointerLockChanged(false);
325 }
326 return;
327 }
328
329 Q_ASSERT(!m_pointerLock);
330 m_pointerLock.reset(m_backend->display()->pointerConstraints()->lockPointer(surface(), pointer, nullptr, PointerConstraints::LifeTime::OneShot));
331 if (!m_pointerLock->isValid()) {
332 m_pointerLock.reset();
333 return;
334 }
335 connect(m_pointerLock.get(), &LockedPointer::locked, this, [this]() {
336 m_hasPointerLock = true;
337 updateWindowTitle();
338 m_cursor->disable();
339 Q_EMIT m_backend->pointerLockChanged(true);
340 });
341 connect(m_pointerLock.get(), &LockedPointer::unlocked, this, [this]() {
342 m_pointerLock.reset();
343 m_hasPointerLock = false;
344 updateWindowTitle();
345 m_cursor->enable();
346 Q_EMIT m_backend->pointerLockChanged(false);
347 });
348}
