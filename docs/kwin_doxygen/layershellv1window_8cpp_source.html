22static NET::WindowType scopeToType(const QString &scope)
23{
24 static const QHash<QString, NET::WindowType> scopeToType{
25 {QStringLiteral("desktop"), NET::Desktop},
26 {QStringLiteral("dock"), NET::Dock},
27 {QStringLiteral("crititical-notification"), NET::CriticalNotification},
28 {QStringLiteral("notification"), NET::Notification},
29 {QStringLiteral("tooltip"), NET::Tooltip},
30 {QStringLiteral("on-screen-display"), NET::OnScreenDisplay},
31 {QStringLiteral("dialog"), NET::Dialog},
32 {QStringLiteral("splash"), NET::Splash},
33 {QStringLiteral("utility"), NET::Utility},
34 };
35 return scopeToType.value(scope.toLower(), NET::Normal);
36}

140StrutRect LayerShellV1Window::strutRect(StrutArea area) const
141{
142 switch (area) {
143 case StrutAreaLeft:
144 if (m_shellSurface->exclusiveEdge() == Qt::LeftEdge) {
145 return StrutRect(x(), y(), m_shellSurface->exclusiveZone(), height(), StrutAreaLeft);
146 }
147 return StrutRect();
148 case StrutAreaRight:
149 if (m_shellSurface->exclusiveEdge() == Qt::RightEdge) {
150 return StrutRect(x() + width() - m_shellSurface->exclusiveZone(), y(),
151 m_shellSurface->exclusiveZone(), height(), StrutAreaRight);
152 }
153 return StrutRect();
154 case StrutAreaTop:
155 if (m_shellSurface->exclusiveEdge() == Qt::TopEdge) {
156 return StrutRect(x(), y(), width(), m_shellSurface->exclusiveZone(), StrutAreaTop);
157 }
158 return StrutRect();
159 case StrutAreaBottom:
160 if (m_shellSurface->exclusiveEdge() == Qt::BottomEdge) {
161 return StrutRect(x(), y() + height() - m_shellSurface->exclusiveZone(),
162 width(), m_shellSurface->exclusiveZone(), StrutAreaBottom);
163 }
164 return StrutRect();
165 default:
166 return StrutRect();
167 }
168}

175void LayerShellV1Window::destroyWindow()
176{
177 markAsZombie();
178 cleanTabBox();
179 Deleted *deleted = Deleted::create(this);
180 Q_EMIT windowClosed(this, deleted);
181 StackingUpdatesBlocker blocker(workspace());
182 cleanGrouping();
183 waylandServer()->removeWindow(this);
184 deleted->unrefWindow();
185 scheduleRearrange();
186 delete this;
187}

194Layer LayerShellV1Window::belongsToLayer() const
195{
196 if (!isNormalWindow()) {
197 return WaylandWindow::belongsToLayer();
198 }
199 switch (m_shellSurface->layer()) {
200 case LayerSurfaceV1Interface::BackgroundLayer:
201 return DesktopLayer;
202 case LayerSurfaceV1Interface::BottomLayer:
203 return BelowLayer;
204 case LayerSurfaceV1Interface::TopLayer:
205 return AboveLayer;
206 case LayerSurfaceV1Interface::OverlayLayer:
207 return UnmanagedLayer;
208 default:
209 Q_UNREACHABLE();
210 }

218void LayerShellV1Window::moveResizeInternal(const QRectF &rect, MoveResizeMode mode)
219{
220 if (areGeometryUpdatesBlocked()) {
221 setPendingMoveResizeMode(mode);
222 return;
223 }
224
225 const QSizeF requestedClientSize = frameSizeToClientSize(rect.size());
226 if (requestedClientSize != clientSize()) {
227 m_shellSurface->sendConfigure(rect.size().toSize());
228 } else {
229 updateGeometry(rect);
230 return;
231 }
232
233 // The surface position is updated synchronously.
234 QRectF updateRect = m_frameGeometry;
235 updateRect.moveTopLeft(rect.topLeft());
236 updateGeometry(updateRect);
237}

258void LayerShellV1Window::handleAcceptsFocusChanged()
259{
260 switch (m_shellSurface->layer()) {
261 case LayerSurfaceV1Interface::TopLayer:
262 case LayerSurfaceV1Interface::OverlayLayer:
263 if (wantsInput()) {
264 workspace()->activateWindow(this);
265 }
266 break;
267 case LayerSurfaceV1Interface::BackgroundLayer:
268 case LayerSurfaceV1Interface::BottomLayer:
269 break;
270 }
271}
