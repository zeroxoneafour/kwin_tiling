158void TabBoxHandlerPrivate::updateHighlightWindows()
159{
160 if (!isShown || config.tabBoxMode() != TabBoxConfig::ClientTabBox) {
161 return;
162 }
163
164 TabBoxClient *currentClient = q->client(index);
165 QWindow *w = window();
166
167 if (q->isKWinCompositing()) {
168 if (lastRaisedClient) {
169 q->elevateClient(lastRaisedClient, w, false);
170 }
171 lastRaisedClient = currentClient;
172 // don't elevate desktop
173 const auto desktop = q->desktopClient().toStrongRef();
174 if (currentClient && (!desktop || currentClient->internalId() != desktop->internalId())) {
175 q->elevateClient(currentClient, w, true);
176 }
177 } else {
178 if (lastRaisedClient) {
179 q->shadeClient(lastRaisedClient, true);
180 if (lastRaisedClientSucc) {
181 q->restack(lastRaisedClient, lastRaisedClientSucc);
182 }
183 // TODO lastRaisedClient->setMinimized( lastRaisedClientWasMinimized );
184 }
185
186 lastRaisedClient = currentClient;
187 if (lastRaisedClient) {
188 q->shadeClient(lastRaisedClient, false);
189 // TODO if ( (lastRaisedClientWasMinimized = lastRaisedClient->isMinimized()) )
190 // lastRaisedClient->setMinimized( false );
191 TabBoxClientList order = q->stackingOrder();
192 int succIdx = order.count() + 1;
193 for (int i = 0; i < order.count(); ++i) {
194 if (order.at(i).toStrongRef() == lastRaisedClient) {
195 succIdx = i + 1;
196 break;
197 }
198 }
199 lastRaisedClientSucc = (succIdx < order.count()) ? order.at(succIdx).toStrongRef().data() : nullptr;
200 q->raiseClient(lastRaisedClient);
201 }
202 }
203
204 if (config.isShowTabBox() && w) {
205 q->highlightWindows(currentClient, w);
206 } else {
207 q->highlightWindows(currentClient);
208 }
209}

211void TabBoxHandlerPrivate::endHighlightWindows(bool abort)
212{
213 TabBoxClient *currentClient = q->client(index);
214 if (isHighlightWindows() && q->isKWinCompositing()) {
215 const auto stackingOrder = q->stackingOrder();
216 for (const QWeakPointer<TabBoxClient> &clientPointer : stackingOrder) {
217 if (QSharedPointer<TabBoxClient> client = clientPointer.toStrongRef()) {
218 if (client != currentClient) { // to not mess up with wanted ShadeActive/ShadeHover state
219 q->shadeClient(client.data(), true);
220 }
221 }
222 }
223 }
224 QWindow *w = window();
225 if (currentClient) {
226 q->elevateClient(currentClient, w, false);
227 }
228 if (abort && lastRaisedClient && lastRaisedClientSucc) {
229 q->restack(lastRaisedClient, lastRaisedClientSucc);
230 }
231 lastRaisedClient = nullptr;
232 lastRaisedClientSucc = nullptr;
233 // highlight windows
234 q->highlightWindows();
235}

238QObject *TabBoxHandlerPrivate::createSwitcherItem(bool desktopMode)
239{
240 // first try look'n'feel package
241 QString file = QStandardPaths::locate(
242 QStandardPaths::GenericDataLocation,
243 QStringLiteral("plasma/look-and-feel/%1/contents/%2")
244 .arg(config.layoutName(),
245 desktopMode ? QStringLiteral("desktopswitcher/DesktopSwitcher.qml") : QStringLiteral("windowswitcher/WindowSwitcher.qml")));
246 if (file.isNull()) {
247 const QString folderName = desktopMode ? QLatin1String("kwin/desktoptabbox/") : QLatin1String("kwin/tabbox/");
248 auto findSwitcher = [this, desktopMode, folderName] {
249 const QString type = desktopMode ? QStringLiteral("KWin/DesktopSwitcher") : QStringLiteral("KWin/WindowSwitcher");
250 auto offers = KPackage::PackageLoader::self()->findPackages(type, folderName,
251 [this](const KPluginMetaData &data) {
252 return data.pluginId().compare(config.layoutName(), Qt::CaseInsensitive) == 0;
253 });
254 if (offers.isEmpty()) {
255 // load default
256 offers = KPackage::PackageLoader::self()->findPackages(type, folderName,
257 [](const KPluginMetaData &data) {
258 return data.pluginId().compare(QStringLiteral("informative"), Qt::CaseInsensitive) == 0;
259 });
260 if (offers.isEmpty()) {
261 qCDebug(KWIN_TABBOX) << "could not find default window switcher layout";
262 return KPluginMetaData();
263 }
264 }
265 return offers.first();
266 };
267 auto service = findSwitcher();
268 if (!service.isValid()) {
269 return nullptr;
270 }
271 if (service.value(QStringLiteral("X-Plasma-API")) != QLatin1String("declarativeappletscript")) {
272 qCDebug(KWIN_TABBOX) << "Window Switcher Layout is no declarativeappletscript";
273 return nullptr;
274 }
275 auto findScriptFile = [service, folderName] {
276 const QString pluginName = service.pluginId();
277 const QString scriptName = service.value(QStringLiteral("X-Plasma-MainScript"));
278 return QStandardPaths::locate(QStandardPaths::GenericDataLocation, folderName + pluginName + QLatin1String("/contents/") + scriptName);
279 };
280 file = findScriptFile();
281 }
282 if (file.isNull()) {
283 qCDebug(KWIN_TABBOX) << "Could not find QML file for window switcher";
284 return nullptr;
285 }
286 m_qmlComponent->loadUrl(QUrl::fromLocalFile(file));
287 if (m_qmlComponent->isError()) {
288 qCDebug(KWIN_TABBOX) << "Component failed to load: " << m_qmlComponent->errors();
289 QStringList args;
290 args << QStringLiteral("--passivepopup") << i18n("The Window Switcher installation is broken, resources are missing.\n"
291 "Contact your distribution about this.")
292 << QStringLiteral("20");
293 KProcess::startDetached(QStringLiteral("kdialog"), args);
294 m_qmlComponent.reset(nullptr);
295 } else {
296 QObject *object = m_qmlComponent->create(m_qmlContext.get());
297 if (desktopMode) {
298 m_desktopTabBoxes.insert(config.layoutName(), object);
299 } else {
300 m_clientTabBoxes.insert(config.layoutName(), object);
301 }
302 return object;
303 }
304 return nullptr;
305}

308void TabBoxHandlerPrivate::show()
309{
310#ifndef KWIN_UNIT_TEST
311 if (!m_qmlContext) {
312 qmlRegisterType<SwitcherItem>("org.kde.kwin", 3, 0, "TabBoxSwitcher");
313 m_qmlContext.reset(new QQmlContext(Scripting::self()->qmlEngine()));
314 }
315 if (!m_qmlComponent) {
316 m_qmlComponent.reset(new QQmlComponent(Scripting::self()->qmlEngine()));
317 }
318 const bool desktopMode = (config.tabBoxMode() == TabBoxConfig::DesktopTabBox);
319 auto findMainItem = [this](const QMap<QString, QObject *> &tabBoxes) -> QObject * {
320 auto it = tabBoxes.constFind(config.layoutName());
321 if (it != tabBoxes.constEnd()) {
322 return it.value();
323 }
324 return nullptr;
325 };
326 m_mainItem = nullptr;
327 m_mainItem = desktopMode ? findMainItem(m_desktopTabBoxes) : findMainItem(m_clientTabBoxes);
328 if (!m_mainItem) {
329 m_mainItem = createSwitcherItem(desktopMode);
330 if (!m_mainItem) {
331 return;
332 }
333 }
334 if (SwitcherItem *item = switcherItem()) {
335 // In case the model isn't yet set (see below), index will be reset and therefore we
336 // need to save the current index row (https://bugs.kde.org/show_bug.cgi?id=333511).
337 int indexRow = index.row();
338 if (!item->model()) {
339 QAbstractItemModel *model = nullptr;
340 if (desktopMode) {
341 model = desktopModel();
342 } else {
343 model = clientModel();
344 }
345 item->setModel(model);
346 }
347 item->setAllDesktops(config.clientDesktopMode() == TabBoxConfig::AllDesktopsClients);
348 item->setCurrentIndex(indexRow);
349 item->setNoModifierGrab(q->noModifierGrab());
350 // everything is prepared, so let's make the whole thing visible
351 item->setVisible(true);
352 }
353 if (QWindow *w = window()) {
354 wheelAngleDelta = 0;
355 w->installEventFilter(q);
356 // pretend to activate the window to enable accessibility notifications
357 QWindowSystemInterface::handleWindowActivated(w, Qt::TabFocusReason);
358 }
359#endif
360}

437QModelIndex TabBoxHandler::nextPrev(bool forward) const
438{
439 QModelIndex ret;
440 QAbstractItemModel *model;
441 switch (d->config.tabBoxMode()) {
442 case TabBoxConfig::ClientTabBox:
443 model = d->clientModel();
444 break;
445 case TabBoxConfig::DesktopTabBox:
446 model = d->desktopModel();
447 break;
448 default:
449 Q_UNREACHABLE();
450 }
451 if (forward) {
452 int column = d->index.column() + 1;
453 int row = d->index.row();
454 if (column == model->columnCount()) {
455 column = 0;
456 row++;
457 if (row == model->rowCount()) {
458 row = 0;
459 }
460 }
461 ret = model->index(row, column);
462 if (!ret.isValid()) {
463 ret = model->index(0, 0);
464 }
465 } else {
466 int column = d->index.column() - 1;
467 int row = d->index.row();
468 if (column < 0) {
469 column = model->columnCount() - 1;
470 row--;
471 if (row < 0) {
472 row = model->rowCount() - 1;
473 }
474 }
475 ret = model->index(row, column);
476 if (!ret.isValid()) {
477 row = model->rowCount() - 1;
478 for (int i = model->columnCount() - 1; i >= 0; i--) {
479 ret = model->index(row, i);
480 if (ret.isValid()) {
481 break;
482 }
483 }
484 }
485 }
486 if (ret.isValid()) {
487 return ret;
488 } else {
489 return d->index;
490 }
491}

587void TabBoxHandler::createModel(bool partialReset)
588{
589 switch (d->config.tabBoxMode()) {
590 case TabBoxConfig::ClientTabBox: {
591 d->clientModel()->createClientList(partialReset);
592 // TODO: C++11 use lambda function
593 bool lastRaised = false;
594 bool lastRaisedSucc = false;
595 const auto clients = stackingOrder();
596 for (const auto &clientPointer : clients) {
597 QSharedPointer<TabBoxClient> client = clientPointer.toStrongRef();
598 if (!client) {
599 continue;
600 }
601 if (client.data() == d->lastRaisedClient) {
602 lastRaised = true;
603 }
604 if (client.data() == d->lastRaisedClientSucc) {
605 lastRaisedSucc = true;
606 }
607 }
608 if (d->lastRaisedClient && !lastRaised) {
609 d->lastRaisedClient = nullptr;
610 }
611 if (d->lastRaisedClientSucc && !lastRaisedSucc) {
612 d->lastRaisedClientSucc = nullptr;
613 }
614 break;
615 }
616 case TabBoxConfig::DesktopTabBox:
617 d->desktopModel()->createDesktopList();
618 break;
619 }
620}

638bool TabBoxHandler::eventFilter(QObject *watched, QEvent *e)
639{
640 if (e->type() == QEvent::Wheel && watched == d->window()) {
641 QWheelEvent *event = static_cast<QWheelEvent *>(e);
642 // On x11 the delta for vertical scrolling might also be on X for whatever reason
643 const int delta = std::abs(event->angleDelta().x()) > std::abs(event->angleDelta().y()) ? event->angleDelta().x() : event->angleDelta().y();
644 d->wheelAngleDelta += delta;
645 while (d->wheelAngleDelta <= -120) {
646 d->wheelAngleDelta += 120;
647 const QModelIndex index = nextPrev(true);
648 if (index.isValid()) {
649 setCurrentIndex(index);
650 }
651 }
652 while (d->wheelAngleDelta >= 120) {
653 d->wheelAngleDelta -= 120;
654 const QModelIndex index = nextPrev(false);
655 if (index.isValid()) {
656 setCurrentIndex(index);
657 }
658 }
659 return true;
660 }
661 // pass on
662 return QObject::eventFilter(watched, e);
663}
