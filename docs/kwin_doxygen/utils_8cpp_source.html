51QString buttonsToString(const DecorationButtonsList &buttons)
52{
53 auto buttonToString = [](KDecoration2::DecorationButtonType button) -> QChar {
54 const auto it = s_buttonNames.constFind(button);
55 if (it != s_buttonNames.constEnd()) {
56 return it.value();
57 }
58 return QChar();
59 };
60 QString ret;
61 for (auto button : buttons) {
62 ret.append(buttonToString(button));
63 }
64 return ret;
65}
66
67DecorationButtonsList buttonsFromString(const QString &buttons)
68{
69 DecorationButtonsList ret;
70 for (auto it = buttons.begin(); it != buttons.end(); ++it) {
71 for (auto it2 = s_buttonNames.constBegin(); it2 != s_buttonNames.constEnd(); ++it2) {
72 if (it2.value() == (*it)) {
73 ret << it2.key();
74 }
75 }
76 }
77 return ret;
78}

80DecorationButtonsList readDecorationButtons(const KConfigGroup &config, const QString &key, const DecorationButtonsList &defaultValue)
81{
