24static bool shouldIgnoreDevice(InputDevice *device)
25{
26 if (qobject_cast<FakeInputDevice*>(device)) {
27 return true;
28 }
29
30 auto libinput_device = qobject_cast<LibInput::Device *>(device);
31 if (!libinput_device) {
32 return false;
33 }
34
35 bool ignore = false;
36 if (auto udev = libinput_device_get_udev_device(libinput_device->device()); udev) {
37 ignore = udev_device_has_tag(udev, "kwin-ignore-tablet-mode");
38 udev_device_unref(udev);
39 }
40 return ignore;
41}

43class TabletModeSwitchEventSpy : public QObject, public InputEventSpy
44{
45public:
46 explicit TabletModeSwitchEventSpy(TabletModeManager *parent)
47 : QObject(parent)
48 , m_parent(parent)
49 {
50 }
51
52 void switchEvent(SwitchEvent *event) override
53 {
54 if (!event->device()->isTabletModeSwitch()) {
55 return;
56 }
57
58 switch (event->state()) {
59 case SwitchEvent::State::Off:
60 m_parent->setIsTablet(false);
61 break;
62 case SwitchEvent::State::On:
63 m_parent->setIsTablet(true);
64 break;
65 default:
66 Q_UNREACHABLE();
67 }
68 }
69
70private:
71 TabletModeManager *const m_parent;
72};

74class TabletModeTouchpadRemovedSpy : public QObject
75{
76public:
77 explicit TabletModeTouchpadRemovedSpy(TabletModeManager *parent)
78 : QObject(parent)
79 , m_parent(parent)
80 {
81 connect(input(), &InputRedirection::deviceAdded, this, &TabletModeTouchpadRemovedSpy::refresh);
82 connect(input(), &InputRedirection::deviceRemoved, this, &TabletModeTouchpadRemovedSpy::refresh);
83
84 check();
85 }
86
87 void refresh(InputDevice *inputDevice)
88 {
89 if (inputDevice->isTouch() || inputDevice->isPointer()) {
90 check();
91 }
92 }
93
94 void check()
95 {
96 const auto devices = input()->devices();
97 const bool hasTouch = std::any_of(devices.constBegin(), devices.constEnd(), [](InputDevice *device) {
98 return device->isTouch() && !shouldIgnoreDevice(device);
99 });
100 m_parent->setTabletModeAvailable(hasTouch);
101
102 const bool hasPointer = std::any_of(devices.constBegin(), devices.constEnd(), [](InputDevice *device) {
103 return device->isPointer() && !shouldIgnoreDevice(device);
104 });
105 m_parent->setIsTablet(hasTouch && !hasPointer);
106 }
107
108private:
109 TabletModeManager *const m_parent;
110};

112TabletModeManager::TabletModeManager()
113{
114 if (waylandServer()) {
115 if (input()->hasTabletModeSwitch()) {
116 input()->installInputEventSpy(new TabletModeSwitchEventSpy(this));

118 hasTabletModeInputChanged(false);

125 refreshSettings();

127 QDBusConnection::sessionBus().registerObject(QStringLiteral("/org/kde/KWin"),

130 QDBusConnection::ExportAllProperties | QDBusConnection::ExportAllSignals);
131
132 if (waylandServer()) {
133 connect(input(), &InputRedirection::hasTabletModeSwitchChanged, this, &TabletModeManager::hasTabletModeInputChanged);

137void KWin::TabletModeManager::refreshSettings()
138{
139 KSharedConfig::Ptr kwinSettings = kwinApp()->config();
140 KConfigGroup cg = kwinSettings->group("Input");
141 const QString tabletModeConfig = cg.readPathEntry("TabletMode", QStringLiteral("auto"));
142 const bool oldEffectiveTabletMode = effectiveTabletMode();
143 if (tabletModeConfig == QStringLiteral("on")) {
144 m_configuredMode = ConfiguredMode::On;
145 if (!m_detecting) {
146 Q_EMIT tabletModeAvailableChanged(true);
147 }
148 } else if (tabletModeConfig == QStringLiteral("off")) {
149 m_configuredMode = ConfiguredMode::Off;
150 } else {
151 m_configuredMode = ConfiguredMode::Auto;
152 }
153 if (effectiveTabletMode() != oldEffectiveTabletMode) {
154 Q_EMIT tabletModeChanged(effectiveTabletMode());
155 }
156}
