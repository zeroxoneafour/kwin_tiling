88bool Display::start()
89{
90 if (d->running) {
91 return true;
92 }
93
94 const int fileDescriptor = wl_event_loop_get_fd(d->loop);
95 if (fileDescriptor == -1) {
96 qCWarning(KWIN_CORE) << "Did not get the file descriptor for the event loop";
97 return false;
98 }
99
100 d->socketNotifier = new QSocketNotifier(fileDescriptor, QSocketNotifier::Read, this);
101 connect(d->socketNotifier, &QSocketNotifier::activated, this, &Display::dispatchEvents);
102
103 QAbstractEventDispatcher *dispatcher = QCoreApplication::eventDispatcher();
104 connect(dispatcher, &QAbstractEventDispatcher::aboutToBlock, this, &Display::flush);
105
106 d->running = true;
107 Q_EMIT runningChanged(true);
108
109 return true;
110}

181ClientConnection *Display::getConnection(wl_client *client)
182{
183 Q_ASSERT(client);
184 auto it = std::find_if(d->clients.constBegin(), d->clients.constEnd(), [client](ClientConnection *c) {
185 return c->client() == client;
186 });
187 if (it != d->clients.constEnd()) {
188 return *it;
189 }
190 // no ConnectionData yet, create it
191 auto c = new ClientConnection(client, this);
192 d->clients << c;
193 connect(c, &ClientConnection::disconnected, this, [this](ClientConnection *c) {
194 const int index = d->clients.indexOf(c);
195 Q_ASSERT(index != -1);
196 d->clients.remove(index);
197 Q_ASSERT(d->clients.indexOf(c) == -1);
198 Q_EMIT clientDisconnected(c);
199 });
200 Q_EMIT clientConnected(c);
201 return c;
202}

244{

270ClientBuffer *Display::clientBufferForResource(wl_resource *resource) const
271{
272 ClientBuffer *buffer = d->resourceToBuffer.value(resource);
273 if (buffer) {
274 return buffer;
275 }
276
277 for (ClientBufferIntegration *integration : std::as_const(d->bufferIntegrations)) {
278 ClientBuffer *buffer = integration->createBuffer(resource);
279 if (buffer) {
280 d->registerClientBuffer(buffer);
281 return buffer;
282 }
283 }
284 return nullptr;
285}
