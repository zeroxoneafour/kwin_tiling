15// in-place vertical mirroring
16static void mirrorVertically(uchar *data, int height, int stride)
17{
18 const int halfHeight = height / 2;
19 std::vector<uchar> temp(stride);
20 for (int y = 0; y < halfHeight; ++y) {
21 auto cur = &data[y * stride], dest = &data[(height - y - 1) * stride];
22 memcpy(temp.data(), cur, stride);
23 memcpy(cur, dest, stride);
24 memcpy(dest, temp.data(), stride);
25 }
26}

28static void grabTexture(GLTexture *texture, QImage *image)
29{
30 const bool invert = !texture->isYInverted();
31 Q_ASSERT(texture->size() == image->size());
32 bool isGLES = GLPlatform::instance()->isGLES();
33 bool invertNeeded = isGLES ^ invert;
34 const bool invertNeededAndSupported = invertNeeded && GLPlatform::instance()->supports(PackInvert);
35 GLboolean prev;
36 if (invertNeededAndSupported) {
37 glGetBooleanv(GL_PACK_INVERT_MESA, &prev);
38 glPixelStorei(GL_PACK_INVERT_MESA, GL_TRUE);
39 }
40
41 texture->bind();
42 if (GLPlatform::instance()->isGLES()) {
43 glReadPixels(0, 0, image->width(), image->height(), image->hasAlphaChannel() ? GL_BGRA : GL_BGR, GL_UNSIGNED_BYTE, (GLvoid *)image->bits());
44 } else if (GLPlatform::instance()->glVersion() >= kVersionNumber(4, 5)) {
45 glGetTextureImage(texture->texture(), 0, image->hasAlphaChannel() ? GL_BGRA : GL_BGR, GL_UNSIGNED_BYTE, image->sizeInBytes(), image->bits());
46 } else {
47 glGetTexImage(texture->target(), 0, image->hasAlphaChannel() ? GL_BGRA : GL_BGR, GL_UNSIGNED_BYTE, image->bits());
48 }
49
50 if (invertNeededAndSupported) {
51 if (!prev) {
52 glPixelStorei(GL_PACK_INVERT_MESA, prev);
53 }
54 } else if (invertNeeded) {
55 mirrorVertically(image->bits(), image->height(), image->bytesPerLine());
56 }
57}
