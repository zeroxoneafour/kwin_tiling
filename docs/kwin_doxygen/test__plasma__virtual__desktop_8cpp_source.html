73void TestVirtualDesktop::init()
74{
75 using namespace KWaylandServer;
76 delete m_display;
77 m_display = new KWaylandServer::Display(this);
78 m_display->addSocketName(s_socketName);
79 m_display->start();
80 QVERIFY(m_display->isRunning());
81
82 // setup connection
83 m_connection = new KWayland::Client::ConnectionThread;
84 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
85 m_connection->setSocketName(s_socketName);
86
87 m_thread = new QThread(this);
88 m_connection->moveToThread(m_thread);
89 m_thread->start();
90
91 m_connection->initConnection();
92 QVERIFY(connectedSpy.wait());
93
94 m_queue = new KWayland::Client::EventQueue(this);
95 QVERIFY(!m_queue->isValid());
96 m_queue->setup(m_connection);
97 QVERIFY(m_queue->isValid());
98
99 KWayland::Client::Registry registry;
100 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
101
102 QSignalSpy plasmaVirtualDesktopManagementSpy(&registry, &KWayland::Client::Registry::plasmaVirtualDesktopManagementAnnounced);
103
104 QSignalSpy windowManagementSpy(&registry, &KWayland::Client::Registry::plasmaWindowManagementAnnounced);
105
106 QVERIFY(!registry.eventQueue());
107 registry.setEventQueue(m_queue);
108 QCOMPARE(registry.eventQueue(), m_queue);
109 registry.create(m_connection->display());
110 QVERIFY(registry.isValid());
111 registry.setup();
112
113 m_compositorInterface = new CompositorInterface(m_display, m_display);
114 QVERIFY(compositorSpy.wait());
115 m_compositor = registry.createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
116
117 m_plasmaVirtualDesktopManagementInterface = new PlasmaVirtualDesktopManagementInterface(m_display, m_display);
118
119 QVERIFY(plasmaVirtualDesktopManagementSpy.wait());
120 m_plasmaVirtualDesktopManagement = registry.createPlasmaVirtualDesktopManagement(plasmaVirtualDesktopManagementSpy.first().first().value<quint32>(),
121 plasmaVirtualDesktopManagementSpy.first().last().value<quint32>(),
122 this);
123
124 m_windowManagementInterface = new PlasmaWindowManagementInterface(m_display, m_display);
125 m_windowManagementInterface->setPlasmaVirtualDesktopManagementInterface(m_plasmaVirtualDesktopManagementInterface);
126
127 QVERIFY(windowManagementSpy.wait());
128 m_windowManagement =
129 registry.createPlasmaWindowManagement(windowManagementSpy.first().first().value<quint32>(), windowManagementSpy.first().last().value<quint32>(), this);
130
131 QSignalSpy windowSpy(m_windowManagement, &KWayland::Client::PlasmaWindowManagement::windowCreated);
132 m_windowInterface = m_windowManagementInterface->createWindow(this, QUuid::createUuid());
133 m_windowInterface->setPid(1337);
134
135 QVERIFY(windowSpy.wait());
136 m_window = windowSpy.first().first().value<KWayland::Client::PlasmaWindow *>();
137}

320void TestVirtualDesktop::testActivate()
321{
322 // rebuild some desktops
323 testCreate();
324
325 KWaylandServer::PlasmaVirtualDesktopInterface *desktop1Int = m_plasmaVirtualDesktopManagementInterface->desktops().first();
326 KWayland::Client::PlasmaVirtualDesktop *desktop1 = m_plasmaVirtualDesktopManagement->desktops().first();
327 QVERIFY(desktop1->isActive());
328 QVERIFY(desktop1Int->isActive());
329
330 KWaylandServer::PlasmaVirtualDesktopInterface *desktop2Int = m_plasmaVirtualDesktopManagementInterface->desktops()[1];
331 KWayland::Client::PlasmaVirtualDesktop *desktop2 = m_plasmaVirtualDesktopManagement->desktops()[1];
332 QVERIFY(!desktop2Int->isActive());
333
334 QSignalSpy requestActivateSpy(desktop2Int, &KWaylandServer::PlasmaVirtualDesktopInterface::activateRequested);
335 QSignalSpy activatedSpy(desktop2, &KWayland::Client::PlasmaVirtualDesktop::activated);
336
337 desktop2->requestActivate();
338 requestActivateSpy.wait();
339
340 // This simulates a compositor which supports only one active desktop at a time
341 for (auto *deskInt : m_plasmaVirtualDesktopManagementInterface->desktops()) {
342 if (deskInt->id() == desktop2->id()) {
343 deskInt->setActive(true);
344 } else {
345 deskInt->setActive(false);
346 }
347 }
348 activatedSpy.wait();
349
350 // correct state in the server
351 QVERIFY(desktop2Int->isActive());
352 QVERIFY(!desktop1Int->isActive());
353 // correct state in the client
354 QVERIFY(desktop2Int->isActive());
355 QVERIFY(!desktop1Int->isActive());
356
357 // Test the deactivated signal
358 QSignalSpy deactivatedSpy(desktop2, &KWayland::Client::PlasmaVirtualDesktop::deactivated);
359
360 for (auto *deskInt : m_plasmaVirtualDesktopManagementInterface->desktops()) {
361 if (deskInt->id() == desktop1->id()) {
362 deskInt->setActive(true);
363 } else {
364 deskInt->setActive(false);
365 }
366 }
367 deactivatedSpy.wait();
368}

370void TestVirtualDesktop::testEnterLeaveDesktop()
371{
372 testCreate();
373
374 QSignalSpy enterRequestedSpy(m_windowInterface, &KWaylandServer::PlasmaWindowInterface::enterPlasmaVirtualDesktopRequested);
375 m_window->requestEnterVirtualDesktop(QStringLiteral("0-1"));
376 enterRequestedSpy.wait();
377
378 QCOMPARE(enterRequestedSpy.takeFirst().at(0).toString(), QStringLiteral("0-1"));
379
380 QSignalSpy virtualDesktopEnteredSpy(m_window, &KWayland::Client::PlasmaWindow::plasmaVirtualDesktopEntered);
381
382 // agree to the request
383 m_windowInterface->addPlasmaVirtualDesktop(QStringLiteral("0-1"));
384 QCOMPARE(m_windowInterface->plasmaVirtualDesktops().length(), 1);
385 QCOMPARE(m_windowInterface->plasmaVirtualDesktops().first(), QStringLiteral("0-1"));
386
387 // check if the client received the enter
388 virtualDesktopEnteredSpy.wait();
389 QCOMPARE(virtualDesktopEnteredSpy.takeFirst().at(0).toString(), QStringLiteral("0-1"));
390 QCOMPARE(m_window->plasmaVirtualDesktops().length(), 1);
391 QCOMPARE(m_window->plasmaVirtualDesktops().first(), QStringLiteral("0-1"));
392
393 // add another desktop, server side
394 m_windowInterface->addPlasmaVirtualDesktop(QStringLiteral("0-3"));
395 virtualDesktopEnteredSpy.wait();
396 QCOMPARE(virtualDesktopEnteredSpy.takeFirst().at(0).toString(), QStringLiteral("0-3"));
397 QCOMPARE(m_windowInterface->plasmaVirtualDesktops().length(), 2);
398 QCOMPARE(m_window->plasmaVirtualDesktops().length(), 2);
399 QCOMPARE(m_window->plasmaVirtualDesktops()[1], QStringLiteral("0-3"));
400
401 // try to add an invalid desktop
402 m_windowInterface->addPlasmaVirtualDesktop(QStringLiteral("invalid"));
403 QCOMPARE(m_window->plasmaVirtualDesktops().length(), 2);
404
405 // remove a desktop
406 QSignalSpy leaveRequestedSpy(m_windowInterface, &KWaylandServer::PlasmaWindowInterface::leavePlasmaVirtualDesktopRequested);
407 m_window->requestLeaveVirtualDesktop(QStringLiteral("0-1"));
408 leaveRequestedSpy.wait();
409
410 QCOMPARE(leaveRequestedSpy.takeFirst().at(0).toString(), QStringLiteral("0-1"));
411
412 QSignalSpy virtualDesktopLeftSpy(m_window, &KWayland::Client::PlasmaWindow::plasmaVirtualDesktopLeft);
413
414 // agree to the request
415 m_windowInterface->removePlasmaVirtualDesktop(QStringLiteral("0-1"));
416 QCOMPARE(m_windowInterface->plasmaVirtualDesktops().length(), 1);
417 QCOMPARE(m_windowInterface->plasmaVirtualDesktops().first(), QStringLiteral("0-3"));
418
419 // check if the client received the leave
420 virtualDesktopLeftSpy.wait();
421 QCOMPARE(virtualDesktopLeftSpy.takeFirst().at(0).toString(), QStringLiteral("0-1"));
422 QCOMPARE(m_window->plasmaVirtualDesktops().length(), 1);
423 QCOMPARE(m_window->plasmaVirtualDesktops().first(), QStringLiteral("0-3"));
424
425 // Destroy desktop 2
426 m_plasmaVirtualDesktopManagementInterface->removeDesktop(QStringLiteral("0-3"));
427 // the window should receive a left signal from the destroyed desktop
428 virtualDesktopLeftSpy.wait();
429
430 QCOMPARE(m_window->plasmaVirtualDesktops().length(), 0);
431}

437 QSignalSpy virtualDesktopLeftSpy(m_window, &KWayland::Client::PlasmaWindow::plasmaVirtualDesktopLeft);
