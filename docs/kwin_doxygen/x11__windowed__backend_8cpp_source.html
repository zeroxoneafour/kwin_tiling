178bool X11WindowedBackend::initialize()
179{
180 m_display = XOpenDisplay(m_options.display.toLatin1().constData());
181 if (!m_display) {
182 return false;
183 }
184
185 m_connection = XGetXCBConnection(m_display);
186 m_screenNumber = XDefaultScreen(m_display);
187 XSetEventQueueOwner(m_display, XCBOwnsEventQueue);
188
189 int screen = m_screenNumber;
190 for (xcb_screen_iterator_t it = xcb_setup_roots_iterator(xcb_get_setup(m_connection));
191 it.rem;
192 --screen, xcb_screen_next(&it)) {
193 if (screen == m_screenNumber) {
194 m_screen = it.data;
195 }
196 }
197
198 const xcb_query_extension_reply_t *presentExtension = xcb_get_extension_data(m_connection, &xcb_present_id);
199 if (presentExtension && presentExtension->present) {
200 m_presentOpcode = presentExtension->major_opcode;
201 xcb_present_query_version_cookie_t cookie = xcb_present_query_version(m_connection, 1, 2);
202 xcb_present_query_version_reply_t *reply = xcb_present_query_version_reply(m_connection, cookie, nullptr);
203 if (!reply) {
204 qCWarning(KWIN_X11WINDOWED) << "Requested Present extension version is unsupported";
205 return false;
206 }
207 m_presentMajorVersion = reply->major_version;
208 m_presentMinorVersion = reply->minor_version;
209 free(reply);
210 } else {
211 qCWarning(KWIN_X11WINDOWED) << "Present X11 extension is unavailable";
212 return false;
213 }
214
215 const xcb_query_extension_reply_t *shmExtension = xcb_get_extension_data(m_connection, &xcb_shm_id);
216 if (shmExtension && shmExtension->present) {
217 xcb_shm_query_version_cookie_t cookie = xcb_shm_query_version(m_connection);
218 xcb_shm_query_version_reply_t *reply = xcb_shm_query_version_reply(m_connection, cookie, nullptr);
219 if (!reply) {
220 qCWarning(KWIN_X11WINDOWED) << "Requested SHM extension version is unsupported";
221 } else {
222 if (!reply->shared_pixmaps) {
223 qCWarning(KWIN_X11WINDOWED) << "X server supports SHM extension but not shared pixmaps";
224 } else {
225 m_hasShm = true;
226 }
227 free(reply);
228 }
229 }
230
231 initXInput();
232 XRenderUtils::init(m_connection, m_screen->root);
233 createOutputs();
234
235 m_pointerDevice = std::make_unique<X11WindowedInputDevice>();
236 m_pointerDevice->setPointer(true);
237 m_keyboardDevice = std::make_unique<X11WindowedInputDevice>();
238 m_keyboardDevice->setKeyboard(true);
239 if (m_hasXInput) {
240 m_touchDevice = std::make_unique<X11WindowedInputDevice>();
241 m_touchDevice->setTouch(true);
242 }
243
244 m_eventNotifier = std::make_unique<QSocketNotifier>(xcb_get_file_descriptor(m_connection), QSocketNotifier::Read);
245 auto processXcbEvents = [this] {
246 while (auto event = xcb_poll_for_event(m_connection)) {
247 handleEvent(event);
248 free(event);
249 }
250 xcb_flush(m_connection);
251 };
252 connect(m_eventNotifier.get(), &QSocketNotifier::activated, this, processXcbEvents);
253 connect(QCoreApplication::eventDispatcher(), &QAbstractEventDispatcher::aboutToBlock, this, processXcbEvents);
254 connect(QCoreApplication::eventDispatcher(), &QAbstractEventDispatcher::awake, this, processXcbEvents);
255
256 Q_EMIT outputsQueried();
257 return true;
258}

260void X11WindowedBackend::initXInput()
261{
262#if HAVE_X11_XINPUT
263 int xi_opcode, event, error;
264 // init XInput extension
265 if (!XQueryExtension(m_display, "XInputExtension", &xi_opcode, &event, &error)) {
266 qCDebug(KWIN_X11WINDOWED) << "XInputExtension not present";
267 return;
268 }
269
270 // verify that the XInput extension is at at least version 2.0
271 int major = 2, minor = 2;
272 int result = XIQueryVersion(m_display, &major, &minor);
273 if (result != Success) {
274 qCDebug(KWIN_X11WINDOWED) << "Failed to init XInput 2.2, trying 2.0";
275 minor = 0;
276 if (XIQueryVersion(m_display, &major, &minor) != Success) {
277 qCDebug(KWIN_X11WINDOWED) << "Failed to init XInput";
278 return;
279 }
280 }
281 m_xiOpcode = xi_opcode;
282 m_majorVersion = major;
283 m_minorVersion = minor;
284 m_hasXInput = m_majorVersion >= 2 && m_minorVersion >= 2;
285#endif
286}

300void X11WindowedBackend::createOutputs()
301{
302 Xcb::Atom protocolsAtom(QByteArrayLiteral("WM_PROTOCOLS"), false, m_connection);
303 Xcb::Atom deleteWindowAtom(QByteArrayLiteral("WM_DELETE_WINDOW"), false, m_connection);
304
305 // we need to multiply the initial window size with the scale in order to
306 // create an output window of this size in the end
307 const QSize pixelSize = m_options.outputSize * m_options.outputScale;
308 for (int i = 0; i < m_options.outputCount; ++i) {
309 auto *output = new X11WindowedOutput(this);
310 output->init(pixelSize, m_options.outputScale);
311
312 m_protocols = protocolsAtom;
313 m_deleteWindowProtocol = deleteWindowAtom;
314
315 xcb_change_property(m_connection,
316 XCB_PROP_MODE_REPLACE,
317 output->window(),
318 m_protocols,
319 XCB_ATOM_ATOM,
320 32, 1,
321 &m_deleteWindowProtocol);
322
323 m_outputs << output;
324 Q_EMIT outputAdded(output);
325 output->updateEnabled(true);
326 }
327
328 updateWindowTitle();
329
330 xcb_flush(m_connection);
331}

341void X11WindowedBackend::handleEvent(xcb_generic_event_t *e)
342{
343 const uint8_t eventType = e->response_type & ~0x80;
344 switch (eventType) {
345 case XCB_BUTTON_PRESS:
346 case XCB_BUTTON_RELEASE:
347 handleButtonPress(reinterpret_cast<xcb_button_press_event_t *>(e));
348 break;
349 case XCB_MOTION_NOTIFY: {
350 auto event = reinterpret_cast<xcb_motion_notify_event_t *>(e);
351 const X11WindowedOutput *output = findOutput(event->event);
352 if (!output) {
353 break;
354 }
355 const QPointF position = output->mapFromGlobal(QPointF(event->root_x, event->root_y));
356 Q_EMIT m_pointerDevice->pointerMotionAbsolute(position, std::chrono::milliseconds(event->time), m_pointerDevice.get());
357 } break;
358 case XCB_KEY_PRESS:
359 case XCB_KEY_RELEASE: {
360 auto event = reinterpret_cast<xcb_key_press_event_t *>(e);
361 if (eventType == XCB_KEY_PRESS) {
362 if (!m_keySymbols) {
363 m_keySymbols = xcb_key_symbols_alloc(m_connection);
364 }
365 const xcb_keysym_t kc = xcb_key_symbols_get_keysym(m_keySymbols, event->detail, 0);
366 if (kc == XK_Control_R) {
367 grabKeyboard(event->time);
368 }
369 Q_EMIT m_keyboardDevice->keyChanged(event->detail - 8,
370 InputRedirection::KeyboardKeyPressed,
371 std::chrono::milliseconds(event->time),
372 m_keyboardDevice.get());
373 } else {
374 Q_EMIT m_keyboardDevice->keyChanged(event->detail - 8,
375 InputRedirection::KeyboardKeyReleased,
376 std::chrono::milliseconds(event->time),
377 m_keyboardDevice.get());
378 }
379 } break;
380 case XCB_CONFIGURE_NOTIFY:
381 updateSize(reinterpret_cast<xcb_configure_notify_event_t *>(e));
382 break;
383 case XCB_ENTER_NOTIFY: {
384 auto event = reinterpret_cast<xcb_enter_notify_event_t *>(e);
385 const X11WindowedOutput *output = findOutput(event->event);
386 if (!output) {
387 break;
388 }
389 const QPointF position = output->mapFromGlobal(QPointF(event->root_x, event->root_y));
390 Q_EMIT m_pointerDevice->pointerMotionAbsolute(position, std::chrono::milliseconds(event->time), m_pointerDevice.get());
391 } break;
392 case XCB_CLIENT_MESSAGE:
393 handleClientMessage(reinterpret_cast<xcb_client_message_event_t *>(e));
394 break;
395 case XCB_EXPOSE:
396 handleExpose(reinterpret_cast<xcb_expose_event_t *>(e));
397 break;
398 case XCB_MAPPING_NOTIFY:
399 if (m_keySymbols) {
400 xcb_refresh_keyboard_mapping(m_keySymbols, reinterpret_cast<xcb_mapping_notify_event_t *>(e));
401 }
402 break;
403 case XCB_GE_GENERIC: {
404 xcb_ge_generic_event_t *ev = reinterpret_cast<xcb_ge_generic_event_t *>(e);
405 if (ev->extension == m_presentOpcode) {
406 handlePresentEvent(ev);
407 } else if (ev->extension == m_xiOpcode) {
408 handleXinputEvent(ev);
409 }
410 break;
411 }
412 default:
413 break;
414 }
415}

417void X11WindowedBackend::grabKeyboard(xcb_timestamp_t time)
418{
419 const bool oldState = m_keyboardGrabbed;
420 if (m_keyboardGrabbed) {
421 xcb_ungrab_keyboard(m_connection, time);
422 xcb_ungrab_pointer(m_connection, time);
423 m_keyboardGrabbed = false;
424 } else {
425 const X11WindowedOutput *output = static_cast<X11WindowedOutput *>(m_outputs[0]);
426 const auto c = xcb_grab_keyboard_unchecked(m_connection, false, output->window(), time,
427 XCB_GRAB_MODE_ASYNC, XCB_GRAB_MODE_ASYNC);
428 UniqueCPtr<xcb_grab_keyboard_reply_t> grab(xcb_grab_keyboard_reply(m_connection, c, nullptr));
429 if (!grab) {
430 return;
431 }
432 if (grab->status == XCB_GRAB_STATUS_SUCCESS) {
433 const auto c = xcb_grab_pointer_unchecked(m_connection, false, output->window(),
434 XCB_EVENT_MASK_BUTTON_PRESS | XCB_EVENT_MASK_BUTTON_RELEASE | XCB_EVENT_MASK_POINTER_MOTION | XCB_EVENT_MASK_ENTER_WINDOW | XCB_EVENT_MASK_LEAVE_WINDOW,
435 XCB_GRAB_MODE_ASYNC, XCB_GRAB_MODE_ASYNC,
436 output->window(), XCB_CURSOR_NONE, time);
437 UniqueCPtr<xcb_grab_pointer_reply_t> grab(xcb_grab_pointer_reply(m_connection, c, nullptr));
438 if (!grab || grab->status != XCB_GRAB_STATUS_SUCCESS) {
439 xcb_ungrab_keyboard(m_connection, time);
440 return;
441 }
442 m_keyboardGrabbed = true;
443 }
444 }
445 if (oldState != m_keyboardGrabbed) {
446 updateWindowTitle();
447 xcb_flush(m_connection);
448 }
449}

460void X11WindowedBackend::handleClientMessage(xcb_client_message_event_t *event)
461{
462 auto it = std::find_if(m_outputs.begin(), m_outputs.end(),
463 [event](X11WindowedOutput *output) {
464 return output->window() == event->window;
465 });
466 if (it == m_outputs.end()) {
467 return;
468 }
469 if (event->type == m_protocols && m_protocols != XCB_ATOM_NONE) {
470 if (event->data.data32[0] == m_deleteWindowProtocol && m_deleteWindowProtocol != XCB_ATOM_NONE) {
471 if (m_outputs.count() == 1) {
472 qCDebug(KWIN_X11WINDOWED) << "Backend window is going to be closed, shutting down.";
473 QCoreApplication::quit();
474 } else {
475 // remove the window
476 qCDebug(KWIN_X11WINDOWED) << "Removing one output window.";
477
478 auto removedOutput = *it;
479 it = m_outputs.erase(it);
480
481 removedOutput->updateEnabled(false);
482 Q_EMIT outputRemoved(removedOutput);
483 removedOutput->unref();
484 Q_EMIT outputsQueried();
485 }
486 }
487 }

490void X11WindowedBackend::handleButtonPress(xcb_button_press_event_t *event)
491{
492 const X11WindowedOutput *output = findOutput(event->event);
493 if (!output) {
494 return;
495 }
496 bool const pressed = (event->response_type & ~0x80) == XCB_BUTTON_PRESS;
497 if (event->detail >= XCB_BUTTON_INDEX_4 && event->detail <= 7) {
498 // wheel
499 if (!pressed) {
500 return;
501 }
502 const int delta = (event->detail == XCB_BUTTON_INDEX_4 || event->detail == 6) ? -120 : 120;
503 static const qreal s_defaultAxisStepDistance = 10.0;
504 InputRedirection::PointerAxis axis;
505 if (event->detail > 5) {
506 axis = InputRedirection::PointerAxisHorizontal;
507 } else {
508 axis = InputRedirection::PointerAxisVertical;
509 }
510 Q_EMIT m_pointerDevice->pointerAxisChanged(axis,
511 delta * s_defaultAxisStepDistance,
512 delta,
513 InputRedirection::PointerAxisSourceUnknown,
514 std::chrono::milliseconds(event->time),
515 m_pointerDevice.get());
516 return;
517 }
518 uint32_t button = 0;
519 switch (event->detail) {
520 case XCB_BUTTON_INDEX_1:
521 button = BTN_LEFT;
522 break;
523 case XCB_BUTTON_INDEX_2:
524 button = BTN_MIDDLE;
525 break;
526 case XCB_BUTTON_INDEX_3:
527 button = BTN_RIGHT;
528 break;
529 default:
530 button = event->detail + BTN_LEFT - 1;
531 return;
532 }
533
534 const QPointF position = output->mapFromGlobal(QPointF(event->root_x, event->root_y));
535 Q_EMIT m_pointerDevice->pointerMotionAbsolute(position, std::chrono::milliseconds(event->time), m_pointerDevice.get());
536
537 if (pressed) {
538 Q_EMIT m_pointerDevice->pointerButtonChanged(button, InputRedirection::PointerButtonPressed, std::chrono::milliseconds(event->time), m_pointerDevice.get());
539 } else {
540 Q_EMIT m_pointerDevice->pointerButtonChanged(button, InputRedirection::PointerButtonReleased, std::chrono::milliseconds(event->time), m_pointerDevice.get());
541 }
542}

568void X11WindowedBackend::handleXinputEvent(xcb_ge_generic_event_t *ge)
569{
570#if HAVE_X11_XINPUT
571 auto te = reinterpret_cast<xXIDeviceEvent *>(ge);
572 const X11WindowedOutput *output = findOutput(te->event);
573 if (!output) {
574 return;
575 }
576
577 const QPointF position = output->mapFromGlobal(QPointF(fixed1616ToReal(te->root_x), fixed1616ToReal(te->root_y)));
578
579 switch (ge->event_type) {
580 case XI_TouchBegin: {
581 Q_EMIT m_touchDevice->touchDown(te->detail, position, std::chrono::milliseconds(te->time), m_touchDevice.get());
582 Q_EMIT m_touchDevice->touchFrame(m_touchDevice.get());
583 break;
584 }
585 case XI_TouchUpdate: {
586 Q_EMIT m_touchDevice->touchMotion(te->detail, position, std::chrono::milliseconds(te->time), m_touchDevice.get());
587 Q_EMIT m_touchDevice->touchFrame(m_touchDevice.get());
588 break;
589 }
590 case XI_TouchEnd: {
591 Q_EMIT m_touchDevice->touchUp(te->detail, std::chrono::milliseconds(te->time), m_touchDevice.get());
592 Q_EMIT m_touchDevice->touchFrame(m_touchDevice.get());
593 break;
594 }
595 case XI_TouchOwnership: {
596 auto te = reinterpret_cast<xXITouchOwnershipEvent *>(ge);
597 XIAllowTouchEvents(m_display, te->deviceid, te->sourceid, te->touchid, XIAcceptTouch);
598 break;
599 }
600 }
601#endif
602}
