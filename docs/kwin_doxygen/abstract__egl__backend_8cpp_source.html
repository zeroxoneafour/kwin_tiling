68{
69 const EGLDisplay eglDisplay = kwinApp()->outputBackend()->sceneEglDisplay();
70 if (eglDisplay == EGL_NO_DISPLAY || s_globalShareContext == EGL_NO_CONTEXT) {

102bool AbstractEglBackend::initEglAPI()
103{
104 EGLint major, minor;
105 if (eglInitialize(m_display, &major, &minor) == EGL_FALSE) {
106 qCWarning(KWIN_OPENGL) << "eglInitialize failed";
107 EGLint error = eglGetError();
108 if (error != EGL_SUCCESS) {
109 qCWarning(KWIN_OPENGL) << "Error during eglInitialize " << error;
110 }
111 return false;
112 }
113 EGLint error = eglGetError();
114 if (error != EGL_SUCCESS) {
115 qCWarning(KWIN_OPENGL) << "Error during eglInitialize " << error;
116 return false;
117 }
118 qCDebug(KWIN_OPENGL) << "Egl Initialize succeeded";
119
120 if (eglBindAPI(isOpenGLES() ? EGL_OPENGL_ES_API : EGL_OPENGL_API) == EGL_FALSE) {
121 qCCritical(KWIN_OPENGL) << "bind OpenGL API failed";
122 return false;
123 }
124 qCDebug(KWIN_OPENGL) << "EGL version: " << major << "." << minor;
125 const QByteArray eglExtensions = eglQueryString(m_display, EGL_EXTENSIONS);
126 setExtensions(eglExtensions.split(' '));
127
128 const QByteArray requiredExtensions[] = {
129 QByteArrayLiteral("EGL_KHR_no_config_context"),
130 QByteArrayLiteral("EGL_KHR_surfaceless_context"),
131 };
132 for (const QByteArray &extensionName : requiredExtensions) {
133 if (!hasExtension(extensionName)) {
134 qCWarning(KWIN_OPENGL) << extensionName << "extension is unsupported";
135 return false;
136 }
137 }
138
139 setSupportsNativeFence(hasExtension(QByteArrayLiteral("EGL_ANDROID_native_fence_sync")));
140 return true;
141}

182void AbstractEglBackend::initWayland()
183{
184 if (!WaylandServer::self()) {
185 return;
186 }
187 if (hasExtension(QByteArrayLiteral("EGL_WL_bind_wayland_display"))) {
188 m_functions.eglBindWaylandDisplayWL = (eglBindWaylandDisplayWL_func)eglGetProcAddress("eglBindWaylandDisplayWL");
189 m_functions.eglUnbindWaylandDisplayWL = (eglUnbindWaylandDisplayWL_func)eglGetProcAddress("eglUnbindWaylandDisplayWL");
190 m_functions.eglQueryWaylandBufferWL = (eglQueryWaylandBufferWL_func)eglGetProcAddress("eglQueryWaylandBufferWL");
191 // only bind if not already done
192 if (waylandServer()->display()->eglDisplay() != eglDisplay()) {
193 if (!m_functions.eglBindWaylandDisplayWL(eglDisplay(), *(WaylandServer::self()->display()))) {
194 m_functions.eglUnbindWaylandDisplayWL = nullptr;
195 m_functions.eglQueryWaylandBufferWL = nullptr;
196 } else {
197 waylandServer()->display()->setEglDisplay(eglDisplay());
198 }
199 }
200 }
201
202 Q_ASSERT(!m_dmaBuf);
203 m_dmaBuf = EglDmabuf::factory(this);
204}

258EGLContext AbstractEglBackend::createContextInternal(EGLContext sharedContext)
259{
260 const bool haveRobustness = hasExtension(QByteArrayLiteral("EGL_EXT_create_context_robustness"));
261 const bool haveCreateContext = hasExtension(QByteArrayLiteral("EGL_KHR_create_context"));
262 const bool haveContextPriority = hasExtension(QByteArrayLiteral("EGL_IMG_context_priority"));
263 const bool haveResetOnVideoMemoryPurge = hasExtension(QByteArrayLiteral("EGL_NV_robustness_video_memory_purge"));
264
265 std::vector<std::unique_ptr<AbstractOpenGLContextAttributeBuilder>> candidates;
266 if (isOpenGLES()) {
267 if (haveCreateContext && haveRobustness && haveContextPriority && haveResetOnVideoMemoryPurge) {
268 auto glesRobustPriority = std::make_unique<EglOpenGLESContextAttributeBuilder>();
269 glesRobustPriority->setResetOnVideoMemoryPurge(true);
270 glesRobustPriority->setVersion(2);
271 glesRobustPriority->setRobust(true);
272 glesRobustPriority->setHighPriority(true);
273 candidates.push_back(std::move(glesRobustPriority));
274 }
275
276 if (haveCreateContext && haveRobustness && haveContextPriority) {
277 auto glesRobustPriority = std::make_unique<EglOpenGLESContextAttributeBuilder>();
278 glesRobustPriority->setVersion(2);
279 glesRobustPriority->setRobust(true);
280 glesRobustPriority->setHighPriority(true);
281 candidates.push_back(std::move(glesRobustPriority));
282 }
283 if (haveCreateContext && haveRobustness) {
284 auto glesRobust = std::make_unique<EglOpenGLESContextAttributeBuilder>();
285 glesRobust->setVersion(2);
286 glesRobust->setRobust(true);
287 candidates.push_back(std::move(glesRobust));
288 }
289 if (haveContextPriority) {
290 auto glesPriority = std::make_unique<EglOpenGLESContextAttributeBuilder>();
291 glesPriority->setVersion(2);
292 glesPriority->setHighPriority(true);
293 candidates.push_back(std::move(glesPriority));
294 }
295 auto gles = std::make_unique<EglOpenGLESContextAttributeBuilder>();
296 gles->setVersion(2);
297 candidates.push_back(std::move(gles));
298 } else {
299 if (haveCreateContext) {
300 if (haveRobustness && haveContextPriority && haveResetOnVideoMemoryPurge) {
301 auto robustCorePriority = std::make_unique<EglContextAttributeBuilder>();
302 robustCorePriority->setResetOnVideoMemoryPurge(true);
303 robustCorePriority->setVersion(3, 1);
304 robustCorePriority->setRobust(true);
305 robustCorePriority->setHighPriority(true);
306 candidates.push_back(std::move(robustCorePriority));
307 }
308 if (haveRobustness && haveContextPriority) {
309 auto robustCorePriority = std::make_unique<EglContextAttributeBuilder>();
310 robustCorePriority->setVersion(3, 1);
311 robustCorePriority->setRobust(true);
312 robustCorePriority->setHighPriority(true);
313 candidates.push_back(std::move(robustCorePriority));
314 }
315 if (haveRobustness) {
316 auto robustCore = std::make_unique<EglContextAttributeBuilder>();
317 robustCore->setVersion(3, 1);
318 robustCore->setRobust(true);
319 candidates.push_back(std::move(robustCore));
320 }
321 if (haveContextPriority) {
322 auto corePriority = std::make_unique<EglContextAttributeBuilder>();
323 corePriority->setVersion(3, 1);
324 corePriority->setHighPriority(true);
325 candidates.push_back(std::move(corePriority));
326 }
327 auto core = std::make_unique<EglContextAttributeBuilder>();
328 core->setVersion(3, 1);
329 candidates.push_back(std::move(core));
330 }
331 if (haveRobustness && haveCreateContext && haveContextPriority) {
332 auto robustPriority = std::make_unique<EglContextAttributeBuilder>();
333 robustPriority->setRobust(true);
334 robustPriority->setHighPriority(true);
335 candidates.push_back(std::move(robustPriority));
336 }
337 if (haveRobustness && haveCreateContext) {
338 auto robust = std::make_unique<EglContextAttributeBuilder>();
339 robust->setRobust(true);
340 candidates.push_back(std::move(robust));
341 }
342 candidates.emplace_back(new EglContextAttributeBuilder);
343 }
344
345 EGLContext ctx = EGL_NO_CONTEXT;
346 for (auto it = candidates.begin(); it != candidates.end(); it++) {
347 const auto attribs = (*it)->build();
348 ctx = eglCreateContext(m_display, config(), sharedContext, attribs.data());
349 if (ctx != EGL_NO_CONTEXT) {
350 qCDebug(KWIN_OPENGL) << "Created EGL context with attributes:" << (*it).get();
351 break;
352 }
353 }
354
355 if (ctx == EGL_NO_CONTEXT) {
356 qCCritical(KWIN_OPENGL) << "Create Context failed";
357 }
358 return ctx;
359}

400EGLImageKHR AbstractEglBackend::importDmaBufAsImage(const DmaBufAttributes &dmabuf) const
401{
402 QVector<EGLint> attribs;
403 attribs.reserve(6 + dmabuf.planeCount * 10 + 1);
404
405 attribs
406 << EGL_WIDTH << dmabuf.width
407 << EGL_HEIGHT << dmabuf.height
408 << EGL_LINUX_DRM_FOURCC_EXT << dmabuf.format;
409
410 attribs
411 << EGL_DMA_BUF_PLANE0_FD_EXT << dmabuf.fd[0].get()
412 << EGL_DMA_BUF_PLANE0_OFFSET_EXT << dmabuf.offset[0]
413 << EGL_DMA_BUF_PLANE0_PITCH_EXT << dmabuf.pitch[0];
414 if (dmabuf.modifier != DRM_FORMAT_MOD_INVALID) {
415 attribs
416 << EGL_DMA_BUF_PLANE0_MODIFIER_LO_EXT << EGLint(dmabuf.modifier & 0xffffffff)
417 << EGL_DMA_BUF_PLANE0_MODIFIER_HI_EXT << EGLint(dmabuf.modifier >> 32);
418 }
419
420 if (dmabuf.planeCount > 1) {
421 attribs
422 << EGL_DMA_BUF_PLANE1_FD_EXT << dmabuf.fd[1].get()
423 << EGL_DMA_BUF_PLANE1_OFFSET_EXT << dmabuf.offset[1]
424 << EGL_DMA_BUF_PLANE1_PITCH_EXT << dmabuf.pitch[1];
425 if (dmabuf.modifier != DRM_FORMAT_MOD_INVALID) {
426 attribs
427 << EGL_DMA_BUF_PLANE1_MODIFIER_LO_EXT << EGLint(dmabuf.modifier & 0xffffffff)
428 << EGL_DMA_BUF_PLANE1_MODIFIER_HI_EXT << EGLint(dmabuf.modifier >> 32);
429 }
430 }
431
432 if (dmabuf.planeCount > 2) {
433 attribs
434 << EGL_DMA_BUF_PLANE2_FD_EXT << dmabuf.fd[2].get()
435 << EGL_DMA_BUF_PLANE2_OFFSET_EXT << dmabuf.offset[2]
436 << EGL_DMA_BUF_PLANE2_PITCH_EXT << dmabuf.pitch[2];
437 if (dmabuf.modifier != DRM_FORMAT_MOD_INVALID) {
438 attribs
439 << EGL_DMA_BUF_PLANE2_MODIFIER_LO_EXT << EGLint(dmabuf.modifier & 0xffffffff)
440 << EGL_DMA_BUF_PLANE2_MODIFIER_HI_EXT << EGLint(dmabuf.modifier >> 32);
441 }
442 }
443
444 if (dmabuf.planeCount > 3) {
445 attribs
446 << EGL_DMA_BUF_PLANE3_FD_EXT << dmabuf.fd[3].get()
447 << EGL_DMA_BUF_PLANE3_OFFSET_EXT << dmabuf.offset[3]
448 << EGL_DMA_BUF_PLANE3_PITCH_EXT << dmabuf.pitch[3];
449 if (dmabuf.modifier != DRM_FORMAT_MOD_INVALID) {
450 attribs
451 << EGL_DMA_BUF_PLANE3_MODIFIER_LO_EXT << EGLint(dmabuf.modifier & 0xffffffff)
452 << EGL_DMA_BUF_PLANE3_MODIFIER_HI_EXT << EGLint(dmabuf.modifier >> 32);
453 }
454 }
455
456 attribs << EGL_NONE;
457
458 return eglCreateImageKHR(m_display, EGL_NO_CONTEXT, EGL_LINUX_DMA_BUF_EXT, nullptr, attribs.data());
459}

481bool AbstractEglBackend::initBufferConfigs()
482{
483 const EGLint config_attribs[] = {
484 EGL_SURFACE_TYPE,
485 EGL_WINDOW_BIT,
486 EGL_RED_SIZE,
487 1,
488 EGL_GREEN_SIZE,
489 1,
490 EGL_BLUE_SIZE,
491 1,
492 EGL_ALPHA_SIZE,
493 0,
494 EGL_RENDERABLE_TYPE,
495 isOpenGLES() ? EGL_OPENGL_ES2_BIT : EGL_OPENGL_BIT,
496 EGL_CONFIG_CAVEAT,
497 EGL_NONE,
498 EGL_NONE,
499 };
500
501 EGLint count;
502 EGLConfig configs[1024];
503 if (eglChooseConfig(eglDisplay(), config_attribs, configs, 1, &count) == EGL_FALSE) {
504 qCCritical(KWIN_OPENGL) << "choose config failed";
505 return false;
506 }
507 if (count != 1) {
508 qCCritical(KWIN_OPENGL) << "choose config did not return a config" << count;
509 return false;
510 }
511 setConfig(configs[0]);
512
513 return true;
514}
