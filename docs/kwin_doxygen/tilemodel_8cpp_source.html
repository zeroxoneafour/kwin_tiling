67QModelIndex TileModel::index(int row, int column, const QModelIndex &parent) const
68{
69 if (column > 0 || !hasIndex(row, column, parent)) {
70 return QModelIndex();
71 }
72
73 CustomTile *parentItem;
74
75 if (!parent.isValid()) {
76 parentItem = m_tileManager->rootTile();
77 } else {
78 parentItem = static_cast<CustomTile *>(parent.internalPointer());
79 }
80
81 CustomTile *childItem = static_cast<CustomTile *>(parentItem->childTile(row));
82 if (childItem) {
83 return createIndex(row, column, childItem);
84 }
85 return QModelIndex();

88QModelIndex TileModel::parent(const QModelIndex &index) const
89{
90 if (!index.isValid()) {
91 return QModelIndex();
92 }
93
94 CustomTile *childItem = static_cast<CustomTile *>(index.internalPointer());
95 CustomTile *parentItem = static_cast<CustomTile *>(childItem->parentTile());
96
97 if (!parentItem || parentItem == m_tileManager->rootTile()) {
98 return QModelIndex();

101 return createIndex(parentItem->row(), 0, parentItem);

125void TileModel::beginInsertTile(CustomTile *tile, int position)
126{
127 Q_ASSERT(position >= 0);
128 CustomTile *parentTile = static_cast<CustomTile *>(tile->parentTile());
129 Q_ASSERT(parentTile);
130
131 auto index = parentTile == m_tileManager->rootTile() ? QModelIndex() : createIndex(parentTile->row(), 0, parentTile);
132 const int pos = std::clamp(position, 0, parentTile->childCount());
133
134 beginInsertRows(index, pos, pos);
135}
