622void Options::setGlPreferBufferSwap(char glPreferBufferSwap)
623{
624 if (glPreferBufferSwap == 'a') {
625 // buffer copying is very fast with the nvidia blob
626 // but due to restrictions in DRI2 *incredibly* slow for all MESA drivers
627 // see https://www.x.org/releases/X11R7.7/doc/dri2proto/dri2proto.txt, item 2.5
628 if (GLPlatform::instance()->driver() == Driver_NVidia) {
629 glPreferBufferSwap = CopyFrontBuffer;
630 } else if (GLPlatform::instance()->driver() != Driver_Unknown) { // undetected, finally resolved when context is initialized
631 glPreferBufferSwap = ExtendDamage;
632 }
633 }
634 if (m_glPreferBufferSwap == (GlSwapStrategy)glPreferBufferSwap) {
635 return;
636 }
637 m_glPreferBufferSwap = (GlSwapStrategy)glPreferBufferSwap;
638 Q_EMIT glPreferBufferSwapChanged();
639}

682void Options::setGlPlatformInterface(OpenGLPlatformInterface interface)
683{
684 // check environment variable
685 const QByteArray envOpenGLInterface(qgetenv("KWIN_OPENGL_INTERFACE"));
686 if (!envOpenGLInterface.isEmpty()) {
687 if (qstrcmp(envOpenGLInterface, "egl") == 0) {
688 qCDebug(KWIN_CORE) << "Forcing EGL native interface through environment variable";
689 interface = EglPlatformInterface;
690 } else if (qstrcmp(envOpenGLInterface, "glx") == 0) {
691 qCDebug(KWIN_CORE) << "Forcing GLX native interface through environment variable";
692 interface = GlxPlatformInterface;
693 }
694 }
695 if (kwinApp()->shouldUseWaylandForCompositing() && interface == GlxPlatformInterface) {
696 // Glx is impossible on Wayland, enforce egl
697 qCDebug(KWIN_CORE) << "Forcing EGL native interface for Wayland mode";
698 interface = EglPlatformInterface;
699 }
700#if !HAVE_EPOXY_GLX
701 qCDebug(KWIN_CORE) << "Forcing EGL native interface as compiled without GLX support";
702 interface = EglPlatformInterface;
703#endif
704 if (QOpenGLContext::openGLModuleType() == QOpenGLContext::LibGLES) {
705 qCDebug(KWIN_CORE) << "Forcing EGL native interface as Qt uses OpenGL ES";
706 interface = EglPlatformInterface;
707 } else if (qstrcmp(qgetenv("KWIN_COMPOSE"), "O2ES") == 0) {
708 qCDebug(KWIN_CORE) << "Forcing EGL native interface as OpenGL ES requested through KWIN_COMPOSE environment variable.";
709 interface = EglPlatformInterface;
710 }
711
712 if (m_glPlatformInterface == interface) {
713 return;
714 }
715 m_glPlatformInterface = interface;
716 Q_EMIT glPlatformInterfaceChanged();
717}

741void Options::loadConfig()
742{
743 m_settings->load();
744
745 syncFromKcfgc();
746
747 // Electric borders
748 KConfigGroup config(m_settings->config(), "Windows");
749 OpTitlebarDblClick = windowOperation(config.readEntry("TitlebarDoubleClickCommand", "Maximize"), true);
750 setOperationMaxButtonLeftClick(windowOperation(config.readEntry("MaximizeButtonLeftClickCommand", "Maximize"), true));
751 setOperationMaxButtonMiddleClick(windowOperation(config.readEntry("MaximizeButtonMiddleClickCommand", "Maximize (vertical only)"), true));
752 setOperationMaxButtonRightClick(windowOperation(config.readEntry("MaximizeButtonRightClickCommand", "Maximize (horizontal only)"), true));
753
754 // Mouse bindings
755 config = KConfigGroup(m_settings->config(), "MouseBindings");
756 // TODO: add properties for missing options
757 CmdTitlebarWheel = mouseWheelCommand(config.readEntry("CommandTitlebarWheel", "Nothing"));
758 CmdAllModKey = (config.readEntry("CommandAllKey", "Meta") == QStringLiteral("Meta")) ? Qt::Key_Meta : Qt::Key_Alt;
759 CmdAllWheel = mouseWheelCommand(config.readEntry("CommandAllWheel", "Nothing"));

826bool Options::loadCompositingConfig(bool force)
827{
828 KConfigGroup config(m_settings->config(), "Compositing");
829
830 bool useCompositing = false;
831 CompositingType compositingMode = NoCompositing;
832 QString compositingBackend = config.readEntry("Backend", "OpenGL");
833 if (compositingBackend == "QPainter") {
834 compositingMode = QPainterCompositing;
835 } else {
836 compositingMode = OpenGLCompositing;
837 }
838
839 if (const char *c = getenv("KWIN_COMPOSE")) {
840 switch (c[0]) {
841 case 'O':
842 qCDebug(KWIN_CORE) << "Compositing forced to OpenGL mode by environment variable";
843 compositingMode = OpenGLCompositing;
844 useCompositing = true;
845 break;
846 case 'Q':
847 qCDebug(KWIN_CORE) << "Compositing forced to QPainter mode by environment variable";
848 compositingMode = QPainterCompositing;
849 useCompositing = true;
850 break;
851 case 'N':
852 if (getenv("KDE_FAILSAFE")) {
853 qCDebug(KWIN_CORE) << "Compositing disabled forcefully by KDE failsafe mode";
854 } else {
855 qCDebug(KWIN_CORE) << "Compositing disabled forcefully by environment variable";
856 }
857 compositingMode = NoCompositing;
858 break;
859 default:
860 qCDebug(KWIN_CORE) << "Unknown KWIN_COMPOSE mode set, ignoring";
861 break;
862 }
863 }
864 setCompositingMode(compositingMode);
865
866 const bool platformSupportsNoCompositing = kwinApp()->outputBackend()->supportedCompositors().contains(NoCompositing);
867 if (m_compositingMode == NoCompositing && platformSupportsNoCompositing) {
868 setUseCompositing(false);
869 return false; // do not even detect compositing preferences if explicitly disabled
870 }
871
872 // it's either enforced by env or by initial resume from "suspend" or we check the settings
873 setUseCompositing(useCompositing || force || config.readEntry("Enabled", Options::defaultUseCompositing() || !platformSupportsNoCompositing));
874
875 if (!m_useCompositing) {
876 return false; // not enforced or necessary and not "enabled" by settings
877 }
878 return true;
879}
