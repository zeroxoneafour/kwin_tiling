43class WaylandEventThread : public QThread
44{
45 Q_OBJECT
46
47public:
48 WaylandEventThread(wl_display *display)
49 : m_display(display)
50 , m_fd(wl_display_get_fd(display))
51 , m_quitPipe{-1, -1}
52 , m_reading(true)
53 , m_quitting(false)
54 {
55 if (pipe(m_quitPipe) == -1) {
56 qCWarning(KWIN_WAYLAND_BACKEND) << "Failed to create quite pipe in WaylandEventThread";
57 }
58 }
59
60 ~WaylandEventThread() override
61 {
62 if (m_quitPipe[0] != -1) {
63 close(m_quitPipe[0]);
64 close(m_quitPipe[1]);
65 }
66 }
67
68 void dispatch()
69 {
70 while (true) {
71 if (wl_display_dispatch_pending(m_display) < 0) {
72 qFatal("Wayland connection broke");
73 }
74
75 wl_display_flush(m_display);
76
77 if (m_reading.loadAcquire()) {
78 break;
79 }
80
81 if (wl_display_prepare_read(m_display) == 0) {
82 QMutexLocker lock(&m_mutex);
83 m_reading.storeRelease(true);
84 m_cond.wakeOne();
85 break;
86 }
87 }
88 }
89
90 void stop()
91 {
92 if (m_quitPipe[1] != -1) {
93 write(m_quitPipe[1], "\0", 1);
94 }
95
96 m_mutex.lock();
97 m_quitting = true;
98 m_cond.wakeOne();
99 m_mutex.unlock();
100
101 wait();
102 }
103
104Q_SIGNALS:
105 void available();
106
107protected:
108 void run() override
109 {
110 while (true) {
111 m_reading.storeRelease(false);
112
113 Q_EMIT available();
114
115 m_mutex.lock();
116 while (!m_reading.loadRelaxed() && !m_quitting) {
117 m_cond.wait(&m_mutex);
118 }
119 m_mutex.unlock();
120
121 if (m_quitting) {
122 break;
123 }
124
125 pollfd fds[2] = { { m_fd, POLLIN, 0 }, { m_quitPipe[0], POLLIN, 0 } };
126 poll(fds, 2, -1);
127
128 if (fds[1].revents & POLLIN) {
129 wl_display_cancel_read(m_display);
130 break;
131 }
132
133 if (fds[0].revents & POLLIN) {
134 wl_display_read_events(m_display);
135 } else {
136 wl_display_cancel_read(m_display);
137 }
138 }
139 }
140
141private:
142 wl_display *const m_display;
143 int m_fd;
144 int m_quitPipe[2];
145 QAtomicInteger<bool> m_reading;
146 QMutex m_mutex;
147 QWaitCondition m_cond;
148 bool m_quitting;
149};

221bool WaylandDisplay::initialize(const QString &socketName)
222{
223 m_display = wl_display_connect(socketName.toUtf8());
224 if (!m_display) {
225 return false;
226 }
227
228 m_eventThread = std::make_unique<WaylandEventThread>(m_display);
229 connect(m_eventThread.get(), &WaylandEventThread::available, this, &WaylandDisplay::flush, Qt::QueuedConnection);
230 m_eventThread->start();
231
232 static wl_registry_listener registryListener {
233 .global = registry_global,
234 .global_remove = registry_global_remove,
235 };
236 m_registry = wl_display_get_registry(m_display);
237 wl_registry_add_listener(m_registry, &registryListener, this);
238 wl_display_roundtrip(m_display);
239 wl_display_roundtrip(m_display); // get dmabuf formats
240
241 return true;
242}
