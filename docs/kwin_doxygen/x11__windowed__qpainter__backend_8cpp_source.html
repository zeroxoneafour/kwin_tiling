24X11WindowedQPainterLayerBuffer::X11WindowedQPainterLayerBuffer(const QSize &size, X11WindowedOutput *output)
25 : m_connection(output->backend()->connection())
26 , m_size(size)
27{
28 QImage::Format format;
29 int bytesPerPixel;
30 switch (output->depth()) {
31 case 24:
32 case 32:
33 format = QImage::Format_ARGB32_Premultiplied;
34 bytesPerPixel = 4;
35 break;
36 case 30:
37 format = QImage::Format_A2RGB30_Premultiplied;
38 bytesPerPixel = 4;
39 break;
40 default:
41 qCWarning(KWIN_X11WINDOWED) << "Unsupported output depth:" << output->depth() << ". Falling back to ARGB32";
42 format = QImage::Format_ARGB32_Premultiplied;
43 bytesPerPixel = 4;
44 break;
45 }
46
47 int shmId = shmget(IPC_PRIVATE, size.width() * size.height() * bytesPerPixel, IPC_CREAT | 0600);
48 if (shmId < 0) {
49 qCWarning(KWIN_X11WINDOWED) << "shmget() failed:" << strerror(errno);
50 return;
51 }
52
53 m_buffer = shmat(shmId, nullptr, 0 /*read/write*/);
54 shmctl(shmId, IPC_RMID, nullptr);
55 if (reinterpret_cast<long>(m_buffer) == -1) {
56 qCWarning(KWIN_X11WINDOWED) << "shmat() failed:" << strerror(errno);
57 return;
58 }
59
60 xcb_shm_seg_t segment = xcb_generate_id(m_connection);
61 xcb_shm_attach(m_connection, segment, shmId, false);
62
63 m_pixmap = xcb_generate_id(m_connection);
64 xcb_shm_create_pixmap(m_connection, m_pixmap, output->window(), size.width(), size.height(), output->depth(), segment, 0);
65 xcb_shm_detach(m_connection, segment);
66
67 m_view = std::make_unique<QImage>(static_cast<uchar *>(m_buffer), size.width(), size.height(), format);
68}

70X11WindowedQPainterLayerBuffer::~X11WindowedQPainterLayerBuffer()
71{
72 if (m_pixmap != XCB_PIXMAP_NONE) {
73 xcb_free_pixmap(m_connection, m_pixmap);
74 }
75 m_view.reset();
76 if (reinterpret_cast<long>(m_buffer) != -1) {
77 shmdt(m_buffer);
78 }
79}

125std::optional<OutputLayerBeginFrameInfo> X11WindowedQPainterPrimaryLayer::beginFrame()
126{
127 const QSize bufferSize = m_output->pixelSize();
128 if (!m_swapchain || m_swapchain->size() != bufferSize) {
129 m_swapchain = std::make_unique<X11WindowedQPainterLayerSwapchain>(bufferSize, m_output);
130 }
131
132 QRegion repaint = m_output->exposedArea() + m_output->rect();
133 m_output->clearExposedArea();
134
135 m_buffer = m_swapchain->acquire();
136 return OutputLayerBeginFrameInfo{
137 .renderTarget = RenderTarget(m_buffer->view()),
138 .repaint = repaint,
139 };
140}
