21static const QColor s_lineColor = QColor(128, 128, 128, 128);

23static QRegion computeDirtyRegion(const QRectF &windowRect)
24{
25 const QMargins outlineMargins(
26 s_lineWidth / 2,
27 s_lineWidth / 2,
28 s_lineWidth / 2,
29 s_lineWidth / 2);
30
31 QRegion dirtyRegion;
32
33 const QList<EffectScreen *> screens = effects->screens();
34 for (EffectScreen *screen : screens) {
35 const QRectF screenRect = effects->clientArea(ScreenArea, screen, effects->currentDesktop());
36
37 QRectF screenWindowRect = windowRect;
38 screenWindowRect.moveCenter(screenRect.center());
39
40 QRectF verticalBarRect(0, 0, s_lineWidth, screenRect.height());
41 verticalBarRect.moveCenter(screenRect.center());
42 verticalBarRect.adjust(-1, -1, 1, 1);
43 dirtyRegion += verticalBarRect.toAlignedRect();
44
45 QRectF horizontalBarRect(0, 0, screenRect.width(), s_lineWidth);
46 horizontalBarRect.moveCenter(screenRect.center());
47 horizontalBarRect.adjust(-1, -1, 1, 1);
48 dirtyRegion += horizontalBarRect.toAlignedRect();
49
50 const QRectF outlineOuterRect = screenWindowRect
51 .marginsAdded(outlineMargins)
52 .adjusted(-1, -1, 1, 1);
53 const QRectF outlineInnerRect = screenWindowRect
54 .marginsRemoved(outlineMargins)
55 .adjusted(1, 1, -1, -1);
56 dirtyRegion += QRegion(outlineOuterRect.toRect()) - QRegion(outlineInnerRect.toRect());
57 }
58
59 return dirtyRegion;
60}

91void SnapHelperEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
92{
93 effects->paintScreen(mask, region, data);
94
95 const qreal opacityFactor = m_animation.active
96 ? m_animation.timeLine.value()
97 : 1.0;
98 const QList<EffectScreen *> screens = effects->screens();
99
100 const auto scale = effects->renderTargetScale();
101
102 // Display the guide
103 if (effects->isOpenGLCompositing()) {
104 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
105 vbo->reset();
106 vbo->setUseColor(true);
107 ShaderBinder binder(ShaderTrait::UniformColor);
108 binder.shader()->setUniform(GLShader::ModelViewProjectionMatrix, data.projectionMatrix());
109 glEnable(GL_BLEND);
110 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
111
112 QColor color = s_lineColor;
113 color.setAlphaF(color.alphaF() * opacityFactor);
114 vbo->setColor(color);
115
116 glLineWidth(s_lineWidth);
117 QVector<float> verts;
118 verts.reserve(screens.count() * 24);
119 for (EffectScreen *screen : screens) {
120 const QRectF rect = effects->clientArea(ScreenArea, screen, effects->currentDesktop());
121 const int midX = rect.x() + rect.width() / 2;
122 const int midY = rect.y() + rect.height() / 2;
123 const int halfWidth = m_geometry.width() / 2;
124 const int halfHeight = m_geometry.height() / 2;
125
126 // Center vertical line.
127 verts << (rect.x() + rect.width() / 2) * scale << rect.y() * scale;
128 verts << (rect.x() + rect.width() / 2) * scale << (rect.y() + rect.height()) * scale;
129
130 // Center horizontal line.
131 verts << rect.x() * scale << (rect.y() + rect.height() / 2) * scale;
132 verts << (rect.x() + rect.width()) * scale << (rect.y() + rect.height() / 2) * scale;
133
134 // Top edge of the window outline.
135 verts << (midX - halfWidth - s_lineWidth / 2) * scale << (midY - halfHeight) * scale;
136 verts << (midX + halfWidth + s_lineWidth / 2) * scale << (midY - halfHeight) * scale;
137
138 // Right edge of the window outline.
139 verts << (midX + halfWidth) * scale << (midY - halfHeight + s_lineWidth / 2) * scale;
140 verts << (midX + halfWidth) * scale << (midY + halfHeight - s_lineWidth / 2) * scale;
141
142 // Bottom edge of the window outline.
143 verts << (midX + halfWidth + s_lineWidth / 2) * scale << (midY + halfHeight) * scale;
144 verts << (midX - halfWidth - s_lineWidth / 2) * scale << (midY + halfHeight) * scale;
145
146 // Left edge of the window outline.
147 verts << (midX - halfWidth) * scale << (midY + halfHeight - s_lineWidth / 2) * scale;
148 verts << (midX - halfWidth) * scale << (midY - halfHeight + s_lineWidth / 2) * scale;
149 }
150 vbo->setData(verts.count() / 2, 2, verts.data(), nullptr);
151 vbo->render(GL_LINES);
152
153 glDisable(GL_BLEND);
154 glLineWidth(1.0);
155 } else if (effects->compositingType() == QPainterCompositing) {
156 QPainter *painter = effects->scenePainter();
157 painter->save();
158 QColor color = s_lineColor;
159 color.setAlphaF(color.alphaF() * opacityFactor);
160 QPen pen(color);
161 pen.setWidth(s_lineWidth);
162 painter->setPen(pen);
163 painter->setBrush(Qt::NoBrush);
164
165 for (EffectScreen *screen : screens) {
166 const QRectF rect = effects->clientArea(ScreenArea, screen, effects->currentDesktop());
167 // Center lines.
168 painter->drawLine(rect.center().x(), rect.y(), rect.center().x(), rect.y() + rect.height());
169 painter->drawLine(rect.x(), rect.center().y(), rect.x() + rect.width(), rect.center().y());
170
171 // Window outline.
172 QRectF outlineRect(0, 0, m_geometry.width(), m_geometry.height());
173 outlineRect.moveCenter(rect.center());
174 painter->drawRect(outlineRect);
175 }
176 painter->restore();
177 }
178}
