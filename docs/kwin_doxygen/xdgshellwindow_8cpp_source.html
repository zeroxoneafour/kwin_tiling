43XdgSurfaceWindow::XdgSurfaceWindow(XdgSurfaceInterface *shellSurface)
44 : WaylandWindow(shellSurface->surface())
45 , m_shellSurface(shellSurface)
46 , m_configureTimer(new QTimer(this))
47{
48 setupPlasmaShellIntegration();
49 connect(shellSurface, &XdgSurfaceInterface::configureAcknowledged,
50 this, &XdgSurfaceWindow::handleConfigureAcknowledged);
51 connect(shellSurface, &XdgSurfaceInterface::resetOccurred,
52 this, &XdgSurfaceWindow::destroyWindow);
53 connect(shellSurface->surface(), &SurfaceInterface::committed,
54 this, &XdgSurfaceWindow::handleCommit);
55#if 0 // TODO: Refactor kwin core in order to uncomment this code.
56 connect(shellSurface->surface(), &SurfaceInterface::mapped,
57 this, &XdgSurfaceWindow::setReadyForPainting);
58#endif
59 connect(shellSurface, &XdgSurfaceInterface::aboutToBeDestroyed,
60 this, &XdgSurfaceWindow::destroyWindow);
61 connect(shellSurface->surface(), &SurfaceInterface::aboutToBeDestroyed,
62 this, &XdgSurfaceWindow::destroyWindow);

64 // The effective window geometry is determined by two things: (a) the rectangle that bounds
65 // the main surface and all of its sub-surfaces, (b) the client-specified window geometry, if
66 // any. If the client hasn't provided the window geometry, we fallback to the bounding sub-
67 // surface rectangle. If the client has provided the window geometry, we intersect it with
68 // the bounding rectangle and that will be the effective window geometry. It's worth to point
69 // out that geometry updates do not occur that frequently, so we don't need to recompute the
70 // bounding geometry every time the client commits the surface.
71
72 SubSurfaceMonitor *treeMonitor = new SubSurfaceMonitor(surface(), this);
73
74 connect(treeMonitor, &SubSurfaceMonitor::subSurfaceAdded,
75 this, &XdgSurfaceWindow::setHaveNextWindowGeometry);
76 connect(treeMonitor, &SubSurfaceMonitor::subSurfaceRemoved,
77 this, &XdgSurfaceWindow::setHaveNextWindowGeometry);
78 connect(treeMonitor, &SubSurfaceMonitor::subSurfaceMoved,
79 this, &XdgSurfaceWindow::setHaveNextWindowGeometry);
80 connect(treeMonitor, &SubSurfaceMonitor::subSurfaceResized,
81 this, &XdgSurfaceWindow::setHaveNextWindowGeometry);
82 connect(shellSurface, &XdgSurfaceInterface::windowGeometryChanged,
83 this, &XdgSurfaceWindow::setHaveNextWindowGeometry);
84 connect(surface(), &SurfaceInterface::sizeChanged,
85 this, &XdgSurfaceWindow::setHaveNextWindowGeometry);

87 // Configure events are not sent immediately, but rather scheduled to be sent when the event
88 // loop is about to be idle. By doing this, we can avoid sending configure events that do
89 // nothing, and implementation-wise, it's simpler.
90
91 m_configureTimer->setSingleShot(true);
92 connect(m_configureTimer, &QTimer::timeout, this, &XdgSurfaceWindow::sendConfigure);

129void XdgSurfaceWindow::sendConfigure()
130{
131 XdgSurfaceConfigure *configureEvent = sendRoleConfigure();
132
133 // The configure event inherits configure flags from the previous event.
134 if (!m_configureEvents.isEmpty()) {
135 const XdgSurfaceConfigure *previousEvent = m_configureEvents.constLast();
136 configureEvent->flags = previousEvent->flags;
137 }
138
139 configureEvent->gravity = m_nextGravity;
140 configureEvent->flags |= m_configureFlags;
141 m_configureFlags = {};
142
143 m_configureEvents.append(configureEvent);

151void XdgSurfaceWindow::handleCommit()
152{
153 if (!surface()->buffer()) {
154 return;
155 }
156
157 if (m_lastAcknowledgedConfigureSerial.has_value()) {
158 const quint32 serial = m_lastAcknowledgedConfigureSerial.value();
159 while (!m_configureEvents.isEmpty()) {
160 if (serial < m_configureEvents.constFirst()->serial) {
161 break;
162 }
163 m_lastAcknowledgedConfigure.reset(m_configureEvents.takeFirst());
164 }
165 }
166
167 handleRolePrecommit();
168 if (haveNextWindowGeometry()) {
169 handleNextWindowGeometry();
170 resetHaveNextWindowGeometry();
171 }
172
173 handleRoleCommit();
174 m_lastAcknowledgedConfigure.reset();
175 m_lastAcknowledgedConfigureSerial.reset();
176
177 setReadyForPainting();
178 updateDepth();
179}

189void XdgSurfaceWindow::maybeUpdateMoveResizeGeometry(const QRectF &rect)
190{
191 // We are about to send a configure event, ignore the committed window geometry.
192 if (m_configureTimer->isActive()) {
193 return;
194 }
195
196 // If there are unacknowledged configure events that change the geometry, don't sync
197 // the move resize geometry in order to avoid rolling back to old state. When the last
198 // configure event is acknowledged, the move resize geometry will be synchronized.
199 for (int i = m_configureEvents.count() - 1; i >= 0; --i) {
200 if (m_configureEvents[i]->flags & XdgSurfaceConfigure::ConfigurePosition) {
201 return;
202 }
203 }
204
205 setMoveResizeGeometry(rect);
206}

208static QRectF gravitateGeometry(const QRectF &rect, const QRectF &bounds, Gravity gravity)
209{
210 QRectF geometry = rect;
211
212 switch (gravity) {
213 case Gravity::TopLeft:
214 geometry.moveRight(bounds.right());
215 geometry.moveBottom(bounds.bottom());
216 break;
217 case Gravity::Top:
218 case Gravity::TopRight:
219 geometry.moveLeft(bounds.left());
220 geometry.moveBottom(bounds.bottom());
221 break;
222 case Gravity::Right:
223 case Gravity::BottomRight:
224 case Gravity::Bottom:
225 case Gravity::None:
226 geometry.moveLeft(bounds.left());
227 geometry.moveTop(bounds.top());
228 break;
229 case Gravity::BottomLeft:
230 case Gravity::Left:
231 geometry.moveRight(bounds.right());
232 geometry.moveTop(bounds.top());
233 break;
234 }
235
236 return geometry;
237}

239void XdgSurfaceWindow::handleNextWindowGeometry()
240{
241 const QRectF boundingGeometry = surface()->boundingRect();
242
243 // The effective window geometry is defined as the intersection of the window geometry
244 // and the rectangle that bounds the main surface and all of its sub-surfaces. If the
245 // client hasn't specified the window geometry, we must fallback to the bounding geometry.
246 // Note that the xdg-shell spec is not clear about when exactly we have to clamp the
247 // window geometry.
248
249 m_windowGeometry = m_shellSurface->windowGeometry();
250 if (m_windowGeometry.isValid()) {
251 m_windowGeometry &= boundingGeometry;
252 } else {
253 m_windowGeometry = boundingGeometry;
254 }
255
256 if (m_windowGeometry.isEmpty()) {
257 qCWarning(KWIN_CORE) << "Committed empty window geometry, dealing with a buggy client!";
258 }
259
260 QRectF frameGeometry(pos(), clientSizeToFrameSize(m_windowGeometry.size()));
261 if (const XdgSurfaceConfigure *configureEvent = lastAcknowledgedConfigure()) {
262 if (configureEvent->flags & XdgSurfaceConfigure::ConfigurePosition) {
263 frameGeometry = gravitateGeometry(frameGeometry, configureEvent->bounds, configureEvent->gravity);
264 }
265 }
266
267 if (!isInteractiveMoveResize()) {
268 // Both the compositor and the client can change the window geometry. If the client
269 // sets a new window geometry, the compositor's move-resize geometry will be invalid.
270 maybeUpdateMoveResizeGeometry(frameGeometry);
271 }
272
273 updateGeometry(frameGeometry);
274}

291void XdgSurfaceWindow::moveResizeInternal(const QRectF &rect, MoveResizeMode mode)
292{
293 if (areGeometryUpdatesBlocked()) {
294 setPendingMoveResizeMode(mode);
295 return;
296 }
297
298 Q_EMIT frameGeometryAboutToChange(this);
299
300 if (mode != MoveResizeMode::Move) {
301 const QSizeF requestedClientSize = frameSizeToClientSize(rect.size());
302 if (requestedClientSize == clientSize()) {
303 updateGeometry(rect);
304 } else {
305 m_configureFlags |= XdgSurfaceConfigure::ConfigurePosition;
306 scheduleConfigure();
307 }
308 } else {
309 // If the window is moved, cancel any queued window position updates.
310 for (XdgSurfaceConfigure *configureEvent : std::as_const(m_configureEvents)) {
311 configureEvent->flags.setFlag(XdgSurfaceConfigure::ConfigurePosition, false);
312 }
313 m_configureFlags.setFlag(XdgSurfaceConfigure::ConfigurePosition, false);
314 updateGeometry(QRectF(rect.topLeft(), size()));
315 }
316}

352void XdgSurfaceWindow::updateShowOnScreenEdge()
353{
354 if (!workspace()->screenEdges()) {
355 return;
356 }
357 if (!readyForPainting() || !m_plasmaShellSurface || m_plasmaShellSurface->role() != PlasmaShellSurfaceInterface::Role::Panel) {
358 workspace()->screenEdges()->reserve(this, ElectricNone);
359 return;
360 }
361 const PlasmaShellSurfaceInterface::PanelBehavior panelBehavior = m_plasmaShellSurface->panelBehavior();
362 if ((panelBehavior == PlasmaShellSurfaceInterface::PanelBehavior::AutoHide && isHidden()) || panelBehavior == PlasmaShellSurfaceInterface::PanelBehavior::WindowsCanCover) {
363 // Screen edge API requires an edge, thus we need to figure out which edge the window borders.
364 const QRect clientGeometry = frameGeometry().toRect(); // converted here to match output checks
365 Qt::Edges edges;
366
367 const auto outputs = workspace()->outputs();
368 for (const Output *output : outputs) {
369 const QRect screenGeometry = output->geometry();
370 if (screenGeometry.left() == clientGeometry.left()) {
371 edges |= Qt::LeftEdge;
372 }
373 if (screenGeometry.right() == clientGeometry.right()) {
374 edges |= Qt::RightEdge;
375 }
376 if (screenGeometry.top() == clientGeometry.top()) {
377 edges |= Qt::TopEdge;
378 }
379 if (screenGeometry.bottom() == clientGeometry.bottom()) {
380 edges |= Qt::BottomEdge;
381 }
382 }
383
384 // A panel might border multiple screen edges. E.g. a horizontal panel at the bottom will
385 // also border the left and right edge. Let's remove such cases.
386 if (edges & Qt::LeftEdge && edges & Qt::RightEdge) {
387 edges = edges & (~(Qt::LeftEdge | Qt::RightEdge));
388 }
389 if (edges & Qt::TopEdge && edges & Qt::BottomEdge) {
390 edges = edges & (~(Qt::TopEdge | Qt::BottomEdge));
391 }
392
393 // It's still possible that a panel borders two edges, e.g. bottom and left
394 // in that case the one which is sharing more with the edge wins.
395 auto check = [clientGeometry](Qt::Edges edges, Qt::Edge horizontal, Qt::Edge vertical) {
396 if (edges & horizontal && edges & vertical) {
397 if (clientGeometry.width() >= clientGeometry.height()) {
398 return edges & ~horizontal;
399 } else {
400 return edges & ~vertical;
401 }
402 }
403 return edges;
404 };
405 edges = check(edges, Qt::LeftEdge, Qt::TopEdge);
406 edges = check(edges, Qt::LeftEdge, Qt::BottomEdge);
407 edges = check(edges, Qt::RightEdge, Qt::TopEdge);
408 edges = check(edges, Qt::RightEdge, Qt::BottomEdge);
409
410 ElectricBorder border = ElectricNone;
411 if (edges & Qt::LeftEdge) {
412 border = ElectricLeft;
413 }
414 if (edges & Qt::RightEdge) {
415 border = ElectricRight;
416 }
417 if (edges & Qt::TopEdge) {
418 border = ElectricTop;
419 }
420 if (edges & Qt::BottomEdge) {
421 border = ElectricBottom;
422 }
423 workspace()->screenEdges()->reserve(this, border);
424 } else {
425 workspace()->screenEdges()->reserve(this, ElectricNone);
426 }
427}

435void XdgSurfaceWindow::installPlasmaShellSurface(PlasmaShellSurfaceInterface *shellSurface)
436{
437 m_plasmaShellSurface = shellSurface;
438
439 auto updatePosition = [this, shellSurface] {
440 move(shellSurface->position());
441 };
442 auto moveUnderCursor = [this] {
443 // Wait for the first commit
444 auto connection = new QMetaObject::Connection;
445 *connection = connect(this, &Window::windowShown, [this, connection] () {
446 disconnect(*connection);
447 if (input()->hasPointer()) {
448 move(input()->globalPointer());
449 keepInArea(workspace()->clientArea(PlacementArea, this));
450 }
451 });
452 };
453 auto updateRole = [this, shellSurface] {
454 NET::WindowType type = NET::Unknown;
455 switch (shellSurface->role()) {
456 case PlasmaShellSurfaceInterface::Role::Desktop:
457 type = NET::Desktop;
458 break;
459 case PlasmaShellSurfaceInterface::Role::Panel:
460 type = NET::Dock;
461 break;
462 case PlasmaShellSurfaceInterface::Role::OnScreenDisplay:
463 type = NET::OnScreenDisplay;
464 break;
465 case PlasmaShellSurfaceInterface::Role::Notification:
466 type = NET::Notification;
467 break;
468 case PlasmaShellSurfaceInterface::Role::ToolTip:
469 type = NET::Tooltip;
470 break;
471 case PlasmaShellSurfaceInterface::Role::CriticalNotification:
472 type = NET::CriticalNotification;
473 break;
474 case PlasmaShellSurfaceInterface::Role::AppletPopup:
475 type = NET::AppletPopup;
476 break;
477 case PlasmaShellSurfaceInterface::Role::Normal:
478 default:
479 type = NET::Normal;
480 break;
481 }
482 if (m_windowType == type) {
483 return;
484 }
485 m_windowType = type;
486 switch (m_windowType) {
487 case NET::Desktop:
488 case NET::Dock:
489 case NET::OnScreenDisplay:
490 case NET::Notification:
491 case NET::CriticalNotification:
492 case NET::Tooltip:
493 case NET::AppletPopup:
494 setOnAllDesktops(true);
495#if KWIN_BUILD_ACTIVITIES
496 setOnAllActivities(true);
497#endif
498 break;
499 default:
500 break;
501 }
502 workspace()->updateClientArea();
503 };
504 connect(shellSurface, &PlasmaShellSurfaceInterface::positionChanged, this, updatePosition);
505 connect(shellSurface, &PlasmaShellSurfaceInterface::openUnderCursorRequested, this, moveUnderCursor);
506 connect(shellSurface, &PlasmaShellSurfaceInterface::roleChanged, this, updateRole);
507 connect(shellSurface, &PlasmaShellSurfaceInterface::panelBehaviorChanged, this, [this] {
508 updateShowOnScreenEdge();
509 workspace()->updateClientArea();
510 });
511 connect(shellSurface, &PlasmaShellSurfaceInterface::panelAutoHideHideRequested, this, [this] {
512 if (m_plasmaShellSurface->panelBehavior() == PlasmaShellSurfaceInterface::PanelBehavior::AutoHide) {
513 hideClient();
514 m_plasmaShellSurface->hideAutoHidingPanel();
515 }
516 updateShowOnScreenEdge();
517 });
518 connect(shellSurface, &PlasmaShellSurfaceInterface::panelAutoHideShowRequested, this, [this] {
519 showClient();
520 workspace()->screenEdges()->reserve(this, ElectricNone);
521 m_plasmaShellSurface->showAutoHidingPanel();
522 });
523 connect(shellSurface, &PlasmaShellSurfaceInterface::panelTakesFocusChanged, this, [this] {
524 if (m_plasmaShellSurface->panelTakesFocus()) {
525 workspace()->activateWindow(this);
526 }
527 });
528 if (shellSurface->isPositionSet()) {
529 updatePosition();
530 }
531 if (shellSurface->wantsOpenUnderCursor()) {
532 moveUnderCursor();
533 }
534 updateRole();
535 updateShowOnScreenEdge();
536 connect(this, &XdgSurfaceWindow::frameGeometryChanged,
537 this, &XdgSurfaceWindow::updateShowOnScreenEdge);
538 connect(this, &XdgSurfaceWindow::windowShown,
539 this, &XdgSurfaceWindow::updateShowOnScreenEdge);
540
541 setSkipTaskbar(shellSurface->skipTaskbar());
542 connect(shellSurface, &PlasmaShellSurfaceInterface::skipTaskbarChanged, this, [this] {
543 setSkipTaskbar(m_plasmaShellSurface->skipTaskbar());
544 });
545
546 setSkipSwitcher(shellSurface->skipSwitcher());
547 connect(shellSurface, &PlasmaShellSurfaceInterface::skipSwitcherChanged, this, [this] {
548 setSkipSwitcher(m_plasmaShellSurface->skipSwitcher());
549 });
550}

779{

788 const bool right = windowRect.right() == outputRect.right();
789 const bool top = windowRect.top() == outputRect.top();
790 const bool bottom = windowRect.bottom() == outputRect.bottom();
791 const bool horizontal = width() >= height();

793 switch (area) {
794 case StrutAreaTop:
795 if (top && horizontal) {

799 case StrutAreaRight:

808 return StrutRect();
809 case StrutAreaLeft:
810 if (left && !horizontal) {
811 return StrutRect(windowRect, StrutAreaLeft);

816 }

1360void XdgToplevelWindow::initialize()
1361{
1362 bool needsPlacement = isPlaceable();
1363 setupWindowRules();
1364
1365 // Move or resize the window only if enforced by a window rule.
1366 const QPointF forcedPosition = rules()->checkPositionSafe(invalidPoint, true);
1367 if (forcedPosition != invalidPoint) {
1368 move(forcedPosition);
1369 }
1370 const QSizeF forcedSize = rules()->checkSize(QSize(), true);
1371 if (forcedSize.isValid()) {
1372 resize(forcedSize);
1373 }
1374
1375 maximize(rules()->checkMaximize(initialMaximizeMode(), true));
1376 setFullScreen(rules()->checkFullScreen(initialFullScreenMode(), true), false);
1377 setOnActivities(rules()->checkActivity(activities(), true));
1378 setDesktops(rules()->checkDesktops(desktops(), true));
1379 setDesktopFileName(rules()->checkDesktopFile(desktopFileName(), true));
1380 if (rules()->checkMinimize(isMinimized(), true)) {
1381 minimize(true); // No animation.
1382 }
1383 setSkipTaskbar(rules()->checkSkipTaskbar(skipTaskbar(), true));
1384 setSkipPager(rules()->checkSkipPager(skipPager(), true));
1385 setSkipSwitcher(rules()->checkSkipSwitcher(skipSwitcher(), true));
1386 setKeepAbove(rules()->checkKeepAbove(keepAbove(), true));
1387 setKeepBelow(rules()->checkKeepBelow(keepBelow(), true));
1388 setShortcut(rules()->checkShortcut(shortcut().toString(), true));
1389 setNoBorder(rules()->checkNoBorder(noBorder(), true));
1390
1391 // Don't place the client if its position is set by a rule.
1392 if (rules()->checkPosition(invalidPoint, true) != invalidPoint) {
1393 needsPlacement = false;
1394 }
1395
1396 // Don't place the client if the maximize state is set by a rule.
1397 if (requestedMaximizeMode() != MaximizeRestore) {
1398 needsPlacement = false;
1399 }
1400
1401 workspace()->rulebook()->discardUsed(this, false); // Remove Apply Now rules.
1402 updateWindowRules(Rules::All);
1403
1404 if (isRequestedFullScreen()) {
1405 needsPlacement = false;
1406 }
1407 if (needsPlacement) {
1408 const QRectF area = workspace()->clientArea(PlacementArea, this, workspace()->activeOutput());
1409 workspace()->placement()->place(this, area);
1410 }
1411
1412 configureDecoration();
1413 scheduleConfigure();
1414 updateColorScheme();
1415 setupWindowManagementInterface();
1416
1417 m_isInitialized = true;
1418}

1640 if (!isResizable() || isAppletPopup()) {

1647 const QRectF oldGeometry = moveResizeGeometry();

1649 mode = rules()->checkMaximize(mode);
1650 if (m_requestedMaximizeMode == mode) {
1651 return;
1652 }
1653
1654 Q_EMIT clientMaximizedStateAboutToChange(this, mode);
1655 m_requestedMaximizeMode = mode;
1656
1657 // call into decoration update borders
1658 if (m_nextDecoration && !(options->borderlessMaximizedWindows() && m_requestedMaximizeMode == KWin::MaximizeFull)) {
1659 changeMaximizeRecursion = true;
1660 const auto c = m_nextDecoration->client().toStrongRef();
1661 if ((m_requestedMaximizeMode & MaximizeVertical) != (oldMode & MaximizeVertical)) {
1662 Q_EMIT c->maximizedVerticallyChanged(m_requestedMaximizeMode & MaximizeVertical);
1663 }
1664 if ((m_requestedMaximizeMode & MaximizeHorizontal) != (oldMode & MaximizeHorizontal)) {
1665 Q_EMIT c->maximizedHorizontallyChanged(m_requestedMaximizeMode & MaximizeHorizontal);
1666 }
1667 if ((m_requestedMaximizeMode == MaximizeFull) != (oldMode == MaximizeFull)) {
1668 Q_EMIT c->maximizedChanged(m_requestedMaximizeMode == MaximizeFull);
1669 }
1670 changeMaximizeRecursion = false;
1671 }
1672
1673 if (options->borderlessMaximizedWindows()) {
1674 setNoBorder(m_requestedMaximizeMode == MaximizeFull);
1675 }
1676
1677 if (quickTileMode() == QuickTileMode(QuickTileFlag::None)) {
1678 QRectF savedGeometry = geometryRestore();
1679 if (!(oldMode & MaximizeVertical)) {
1680 savedGeometry.setTop(oldGeometry.top());
1681 savedGeometry.setBottom(oldGeometry.bottom());
1682 }
1683 if (!(oldMode & MaximizeHorizontal)) {
1684 savedGeometry.setLeft(oldGeometry.left());
1685 savedGeometry.setRight(oldGeometry.right());
1686 }
1687 setGeometryRestore(savedGeometry);
1688 }
1689
1690 const MaximizeMode delta = m_requestedMaximizeMode ^ oldMode;
1691 QRectF geometry = oldGeometry;
1692
1693 if (delta & MaximizeHorizontal) {
1694 if (m_requestedMaximizeMode & MaximizeHorizontal) {
1695 // Stretch the window vertically to fit the size of the maximize area.
1696 geometry.setX(clientArea.x());
1697 geometry.setWidth(clientArea.width());
1698 } else if (geometryRestore().isValid()) {
1699 // The window is no longer maximized horizontally and the saved geometry is valid.
1700 geometry.setX(geometryRestore().x());
1701 geometry.setWidth(geometryRestore().width());
1702 } else {
1703 // The window is no longer maximized horizontally and the saved geometry is
1704 // invalid. This would happen if the window had been mapped in the maximized state.
1705 // We ask the client to resize the window horizontally to its preferred size.
1706 geometry.setX(clientArea.x());
1707 geometry.setWidth(0);
1708 }
1709 }
1710
1711 if (delta & MaximizeVertical) {
1712 if (m_requestedMaximizeMode & MaximizeVertical) {
1713 // Stretch the window horizontally to fit the size of the maximize area.
1714 geometry.setY(clientArea.y());
1715 geometry.setHeight(clientArea.height());
1716 } else if (geometryRestore().isValid()) {
1717 // The window is no longer maximized vertically and the saved geometry is valid.
1718 geometry.setY(geometryRestore().y());
1719 geometry.setHeight(geometryRestore().height());
1720 } else {
1721 // The window is no longer maximized vertically and the saved geometry is
1722 // invalid. This would happen if the window had been mapped in the maximized state.
1723 // We ask the client to resize the window vertically to its preferred size.
1724 geometry.setY(clientArea.y());
1725 geometry.setHeight(0);

1734 updateQuickTileMode(QuickTileFlag::None);

1736 } else {
1737 updateQuickTileMode(QuickTileFlag::None);

1840QRectF XdgPopupWindow::transientPlacement(const QRectF &bounds) const
1841{
1842 const XdgPositioner positioner = m_shellSurface->positioner();
1843
1844 const QSize desiredSize = positioner.size();
1845
1846 const QPointF parentPosition = transientFor()->framePosToClientPos(transientFor()->pos());
1847
1848 // returns if a target is within the supplied bounds, optional edges argument states which side to check
1849 auto inBounds = [bounds](const QRectF &target, Qt::Edges edges = Qt::LeftEdge | Qt::RightEdge | Qt::TopEdge | Qt::BottomEdge) -> bool {
1850 if (edges & Qt::LeftEdge && target.left() < bounds.left()) {
1851 return false;
1852 }
1853 if (edges & Qt::TopEdge && target.top() < bounds.top()) {
1854 return false;
1855 }
1856 if (edges & Qt::RightEdge && target.right() > bounds.right()) {
1857 // normal QRect::right issue cancels out
1858 return false;
1859 }
1860 if (edges & Qt::BottomEdge && target.bottom() > bounds.bottom()) {
1861 return false;
1862 }
1863 return true;
1864 };
1865
1866 QRectF popupRect(popupOffset(positioner.anchorRect(), positioner.anchorEdges(), positioner.gravityEdges(), desiredSize) + positioner.offset() + parentPosition, desiredSize);
1867
1868 // if that fits, we don't need to do anything
1869 if (inBounds(popupRect)) {
1870 return popupRect;
1871 }
1872 // otherwise apply constraint adjustment per axis in order XDG Shell Popup states
1873
1874 if (positioner.flipConstraintAdjustments() & Qt::Horizontal) {
1875 if (!inBounds(popupRect, Qt::LeftEdge | Qt::RightEdge)) {
1876 // flip both edges (if either bit is set, XOR both)
1877 auto flippedAnchorEdge = positioner.anchorEdges();
1878 if (flippedAnchorEdge & (Qt::LeftEdge | Qt::RightEdge)) {
1879 flippedAnchorEdge ^= (Qt::LeftEdge | Qt::RightEdge);
1880 }
1881 auto flippedGravity = positioner.gravityEdges();
1882 if (flippedGravity & (Qt::LeftEdge | Qt::RightEdge)) {
1883 flippedGravity ^= (Qt::LeftEdge | Qt::RightEdge);
1884 }
1885 auto flippedPopupRect = QRectF(popupOffset(positioner.anchorRect(), flippedAnchorEdge, flippedGravity, desiredSize) + positioner.offset() + parentPosition, desiredSize);
1886
1887 // if it still doesn't fit we should continue with the unflipped version
1888 if (inBounds(flippedPopupRect, Qt::LeftEdge | Qt::RightEdge)) {
1889 popupRect.moveLeft(flippedPopupRect.left());
1890 }
1891 }
1892 }
1893 if (positioner.slideConstraintAdjustments() & Qt::Horizontal) {
1894 if (!inBounds(popupRect, Qt::LeftEdge)) {
1895 popupRect.moveLeft(bounds.left());
1896 }
1897 if (!inBounds(popupRect, Qt::RightEdge)) {
1898 popupRect.moveRight(bounds.right());
1899 }
1900 }
1901 if (positioner.resizeConstraintAdjustments() & Qt::Horizontal) {
1902 QRectF unconstrainedRect = popupRect;
1903
1904 if (!inBounds(unconstrainedRect, Qt::LeftEdge)) {
1905 unconstrainedRect.setLeft(bounds.left());
1906 }
1907 if (!inBounds(unconstrainedRect, Qt::RightEdge)) {
1908 unconstrainedRect.setRight(bounds.right());
1909 }
1910
1911 if (unconstrainedRect.isValid()) {
1912 popupRect = unconstrainedRect;
1913 }
1914 }
1915
1916 if (positioner.flipConstraintAdjustments() & Qt::Vertical) {
1917 if (!inBounds(popupRect, Qt::TopEdge | Qt::BottomEdge)) {
1918 // flip both edges (if either bit is set, XOR both)
1919 auto flippedAnchorEdge = positioner.anchorEdges();
1920 if (flippedAnchorEdge & (Qt::TopEdge | Qt::BottomEdge)) {
1921 flippedAnchorEdge ^= (Qt::TopEdge | Qt::BottomEdge);
1922 }
1923 auto flippedGravity = positioner.gravityEdges();
1924 if (flippedGravity & (Qt::TopEdge | Qt::BottomEdge)) {
1925 flippedGravity ^= (Qt::TopEdge | Qt::BottomEdge);
1926 }
1927 auto flippedPopupRect = QRectF(popupOffset(positioner.anchorRect(), flippedAnchorEdge, flippedGravity, desiredSize) + positioner.offset() + parentPosition, desiredSize);
1928
1929 // if it still doesn't fit we should continue with the unflipped version
1930 if (inBounds(flippedPopupRect, Qt::TopEdge | Qt::BottomEdge)) {
1931 popupRect.moveTop(flippedPopupRect.top());
1932 }
1933 }
1934 }
1935 if (positioner.slideConstraintAdjustments() & Qt::Vertical) {
1936 if (!inBounds(popupRect, Qt::TopEdge)) {
1937 popupRect.moveTop(bounds.top());
1938 }
1939 if (!inBounds(popupRect, Qt::BottomEdge)) {
1940 popupRect.moveBottom(bounds.bottom());
1941 }
1942 }
1943 if (positioner.resizeConstraintAdjustments() & Qt::Vertical) {
1944 QRectF unconstrainedRect = popupRect;
1945
1946 if (!inBounds(unconstrainedRect, Qt::TopEdge)) {
1947 unconstrainedRect.setTop(bounds.top());
1948 }
1949 if (!inBounds(unconstrainedRect, Qt::BottomEdge)) {
1950 unconstrainedRect.setBottom(bounds.bottom());
1951 }
1952
1953 if (unconstrainedRect.isValid()) {
1954 popupRect = unconstrainedRect;
1955 }
1956 }
1957
1958 return popupRect;
1959}
