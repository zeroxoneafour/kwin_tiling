92void TestWindowManagement::init()
93{
94 using namespace KWaylandServer;
95 qRegisterMetaType<KWaylandServer::PlasmaWindowManagementInterface::ShowingDesktopState>("ShowingDesktopState");
96 delete m_display;
97 m_display = new KWaylandServer::Display(this);
98 m_display->addSocketName(s_socketName);
99 m_display->start();
100 QVERIFY(m_display->isRunning());
101
102 // setup connection
103 m_connection = new KWayland::Client::ConnectionThread;
104 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
105 m_connection->setSocketName(s_socketName);
106
107 m_thread = new QThread(this);
108 m_connection->moveToThread(m_thread);
109 m_thread->start();
110
111 m_connection->initConnection();
112 QVERIFY(connectedSpy.wait());
113
114 m_queue = new KWayland::Client::EventQueue(this);
115 QVERIFY(!m_queue->isValid());
116 m_queue->setup(m_connection);
117 QVERIFY(m_queue->isValid());
118
119 m_registry = new KWayland::Client::Registry(this);
120 QSignalSpy compositorSpy(m_registry, &KWayland::Client::Registry::compositorAnnounced);
121
122 QSignalSpy windowManagementSpy(m_registry, &KWayland::Client::Registry::plasmaWindowManagementAnnounced);
123
124 QVERIFY(!m_registry->eventQueue());
125 m_registry->setEventQueue(m_queue);
126 QCOMPARE(m_registry->eventQueue(), m_queue);
127 m_registry->create(m_connection->display());
128 QVERIFY(m_registry->isValid());
129 m_registry->setup();
130
131 m_compositorInterface = new CompositorInterface(m_display, m_display);
132 QVERIFY(compositorSpy.wait());
133 m_compositor = m_registry->createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
134
135 m_windowManagementInterface = new PlasmaWindowManagementInterface(m_display, m_display);
136
137 QVERIFY(windowManagementSpy.wait());
138 m_windowManagement = m_registry->createPlasmaWindowManagement(windowManagementSpy.first().first().value<quint32>(),
139 windowManagementSpy.first().last().value<quint32>(),
140 this);
141
142 QSignalSpy windowSpy(m_windowManagement, &KWayland::Client::PlasmaWindowManagement::windowCreated);
143 m_windowInterface = m_windowManagementInterface->createWindow(this, QUuid::createUuid());
144 m_windowInterface->setPid(1337);
145
146 QVERIFY(windowSpy.wait());
147 m_window = windowSpy.first().first().value<KWayland::Client::PlasmaWindow *>();
148
149 QSignalSpy serverSurfaceCreated(m_compositorInterface, &KWaylandServer::CompositorInterface::surfaceCreated);
150 m_surface = m_compositor->createSurface(this);
151 QVERIFY(m_surface);
152
153 QVERIFY(serverSurfaceCreated.wait());
154 m_surfaceInterface = serverSurfaceCreated.first().first().value<KWaylandServer::SurfaceInterface *>();
155 QVERIFY(m_surfaceInterface);
156}

183void TestWindowManagement::cleanup()
184{
185 if (m_surface) {
186 delete m_surface;
187 m_surface = nullptr;
188 }
189 if (m_compositor) {
190 delete m_compositor;
191 m_compositor = nullptr;
192 }
193 if (m_queue) {
194 delete m_queue;
195 m_queue = nullptr;
196 }
197 if (m_windowManagement) {
198 delete m_windowManagement;
199 m_windowManagement = nullptr;
200 }
201 if (m_registry) {
202 delete m_registry;
203 m_registry = nullptr;
204 }
205 if (m_thread) {
206 if (m_connection) {
207 m_connection->flush();
208 m_connection->deleteLater();
209 }
210 m_thread->quit();
211 m_thread->wait();
212 delete m_thread;
213 m_thread = nullptr;
214 }
215 m_connection = nullptr;
216
217 m_display->dispatchEvents();
218
219 QVERIFY(m_surfaceInterface.isNull());
220
221 delete m_display;
222 m_display = nullptr;
223
224 // these are the children of the display
225 m_windowManagementInterface = nullptr;
226 m_windowInterface = nullptr;
227}

294void TestWindowManagement::testCreateAfterUnmap()
295{
296 // this test verifies that we don't get a protocol error on client side when creating an already unmapped window.
297 QCOMPARE(m_windowManagement->children().count(), 1);
298
299 QSignalSpy windowAddedSpy(m_windowManagement, &KWayland::Client::PlasmaWindowManagement::windowCreated);
300
301 // create and unmap in one go
302 // client will first handle the create, the unmap will be sent once the server side is bound
303 auto serverWindow = m_windowManagementInterface->createWindow(m_windowManagementInterface, QUuid::createUuid());
304 delete serverWindow;
305 QCOMPARE(m_windowManagementInterface->children().count(), 0);
306
307 windowAddedSpy.wait();
308 auto window = dynamic_cast<KWayland::Client::PlasmaWindow *>(m_windowManagement->children().last());
309 QVERIFY(window);
310 // now this is not yet on the server, on the server it will be after next roundtrip
311 // which we can trigger by waiting for destroy of the newly created window.
312 // why destroy? Because we will get the unmap which triggers a destroy
313 QSignalSpy clientDestroyedSpy(window, &QObject::destroyed);
314 QVERIFY(clientDestroyedSpy.wait());
315 // Verify that any wrappers created for our temporary window are now gone
316 QCOMPARE(m_windowManagement->children().count(), 1);
317}

475 auto parentWindow = m_windowManagement->windows().first();
476 QVERIFY(parentWindow);
477 QVERIFY(parentWindow->parentWindow().isNull());
478
479 // now let's create a second window
480 QSignalSpy windowAddedSpy(m_windowManagement, &KWayland::Client::PlasmaWindowManagement::windowCreated);
481 std::unique_ptr<KWaylandServer::PlasmaWindowInterface> serverTransient(m_windowManagementInterface->createWindow(this, QUuid::createUuid()));
482 serverTransient->setParentWindow(m_windowInterface);
483 QVERIFY(windowAddedSpy.wait());
484 auto transient = windowAddedSpy.first().first().value<KWayland::Client::PlasmaWindow *>();
485 QCOMPARE(transient->parentWindow().data(), parentWindow);
486
487 // let's unset the parent
488 QSignalSpy parentWindowChangedSpy(transient, &KWayland::Client::PlasmaWindow::parentWindowChanged);
489 serverTransient->setParentWindow(nullptr);
490 QVERIFY(parentWindowChangedSpy.wait());
491 QVERIFY(transient->parentWindow().isNull());
492
493 // and set it again
494 serverTransient->setParentWindow(m_windowInterface);
495 QVERIFY(parentWindowChangedSpy.wait());
496 QCOMPARE(transient->parentWindow().data(), parentWindow);
497
498 // now let's try to unmap the parent
499 m_windowInterface->deleteLater();
500 m_window = nullptr;
501 m_windowInterface = nullptr;
502 QVERIFY(parentWindowChangedSpy.wait());
503 QVERIFY(transient->parentWindow().isNull());

506void TestWindowManagement::testGeometry()
507{
508 QVERIFY(m_window);
509 QCOMPARE(m_window->geometry(), QRect());
510 QSignalSpy windowGeometryChangedSpy(m_window, &KWayland::Client::PlasmaWindow::geometryChanged);
511 m_windowInterface->setGeometry(QRect(20, -10, 30, 40));
512 QVERIFY(windowGeometryChangedSpy.wait());
513 QCOMPARE(m_window->geometry(), QRect(20, -10, 30, 40));
514 // setting an empty geometry should not be sent to the client
515 m_windowInterface->setGeometry(QRect());
516 QVERIFY(!windowGeometryChangedSpy.wait(10));
517 // setting to the geometry which the client still has should not trigger signal
518 m_windowInterface->setGeometry(QRect(20, -10, 30, 40));
519 QVERIFY(!windowGeometryChangedSpy.wait(10));
520 // setting another geometry should work, though
521 m_windowInterface->setGeometry(QRect(0, 0, 35, 45));
522 QVERIFY(windowGeometryChangedSpy.wait());
523 QCOMPARE(windowGeometryChangedSpy.count(), 2);
524 QCOMPARE(m_window->geometry(), QRect(0, 0, 35, 45));
525
526 // let's bind a second PlasmaWindowManagement to verify the initial setting
527 std::unique_ptr<KWayland::Client::PlasmaWindowManagement> pm(
528 m_registry->createPlasmaWindowManagement(m_registry->interface(KWayland::Client::Registry::Interface::PlasmaWindowManagement).name,
529 m_registry->interface(KWayland::Client::Registry::Interface::PlasmaWindowManagement).version));
530 QVERIFY(pm != nullptr);
531 QSignalSpy windowAddedSpy(pm.get(), &KWayland::Client::PlasmaWindowManagement::windowCreated);
532 QVERIFY(windowAddedSpy.wait());
533 auto window = pm->windows().first();
534 QCOMPARE(window->geometry(), QRect(0, 0, 35, 45));
535}
