25bool EffectsMouseInterceptionX11Filter::event(xcb_generic_event_t *event)
26{
27 const uint8_t eventType = event->response_type & ~0x80;
28 if (eventType == XCB_BUTTON_PRESS || eventType == XCB_BUTTON_RELEASE) {
29 auto *me = reinterpret_cast<xcb_button_press_event_t *>(event);
30 if (m_window == me->event) {
31 const bool isWheel = me->detail >= 4 && me->detail <= 7;
32 if (isWheel) {
33 if (eventType != XCB_BUTTON_PRESS) {
34 return false;
35 }
36 QPoint angleDelta;
37 switch (me->detail) {
38 case 4:
39 angleDelta.setY(120);
40 break;
41 case 5:
42 angleDelta.setY(-120);
43 break;
44 case 6:
45 angleDelta.setX(120);
46 break;
47 case 7:
48 angleDelta.setX(-120);
49 break;
50 }
51
52 const Qt::MouseButtons buttons = x11ToQtMouseButtons(me->state);
53 const Qt::KeyboardModifiers modifiers = x11ToQtKeyboardModifiers(me->state);
54
55 if (modifiers & Qt::AltModifier) {
56 int x = angleDelta.x();
57 int y = angleDelta.y();
58
59 angleDelta.setX(y);
60 angleDelta.setY(x);
61 // After Qt > 5.14 simplify to
62 // angleDelta = angleDelta.transposed();
63 }
64
65 QWheelEvent ev(QPoint(me->event_x, me->event_y), QCursor::pos(), QPoint(), angleDelta, buttons, modifiers, Qt::NoScrollPhase, false);
66 ev.setTimestamp(me->time);
67 return m_effects->checkInputWindowEvent(&ev);
68 }
69 const Qt::MouseButton button = x11ToQtMouseButton(me->detail);
70 Qt::MouseButtons buttons = x11ToQtMouseButtons(me->state);
71 const QEvent::Type type = (eventType == XCB_BUTTON_PRESS) ? QEvent::MouseButtonPress : QEvent::MouseButtonRelease;
72 if (type == QEvent::MouseButtonPress) {
73 buttons |= button;
74 } else {
75 buttons &= ~button;
76 }
77 QMouseEvent ev(type, QPoint(me->event_x, me->event_y), QPoint(me->root_x, me->root_y),
78 button, buttons, x11ToQtKeyboardModifiers(me->state));
79 ev.setTimestamp(me->time);
80 return m_effects->checkInputWindowEvent(&ev);
81 }
82 } else if (eventType == XCB_MOTION_NOTIFY) {
83 const auto *me = reinterpret_cast<xcb_motion_notify_event_t *>(event);
84 if (m_window == me->event) {
85 QMouseEvent ev(QEvent::MouseMove, QPoint(me->event_x, me->event_y), QPoint(me->root_x, me->root_y),
86 Qt::NoButton, x11ToQtMouseButtons(me->state), x11ToQtKeyboardModifiers(me->state));
87 ev.setTimestamp(me->time);
88 return m_effects->checkInputWindowEvent(&ev);
89 }
90 }
91 return false;
92}
