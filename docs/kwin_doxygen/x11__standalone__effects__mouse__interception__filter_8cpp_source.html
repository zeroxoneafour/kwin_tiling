26bool EffectsMouseInterceptionX11Filter::event(xcb_generic_event_t *event)
27{
28 const uint8_t eventType = event->response_type & ~0x80;
29 if (eventType == XCB_BUTTON_PRESS || eventType == XCB_BUTTON_RELEASE) {
30 auto *me = reinterpret_cast<xcb_button_press_event_t *>(event);
31 if (m_window == me->event) {
32 const bool isWheel = me->detail >= 4 && me->detail <= 7;
33 if (isWheel) {
34 if (eventType != XCB_BUTTON_PRESS) {
35 return false;
36 }
37 QPoint angleDelta;
38 switch (me->detail) {
39 case 4:
40 angleDelta.setY(120);
41 break;
42 case 5:
43 angleDelta.setY(-120);
44 break;
45 case 6:
46 angleDelta.setX(120);
47 break;
48 case 7:
49 angleDelta.setX(-120);
50 break;
51 }
52
53 const Qt::MouseButtons buttons = x11ToQtMouseButtons(me->state);
54 const Qt::KeyboardModifiers modifiers = x11ToQtKeyboardModifiers(me->state);
55
56 if (modifiers & Qt::AltModifier) {
57 int x = angleDelta.x();
58 int y = angleDelta.y();
59
60 angleDelta.setX(y);
61 angleDelta.setY(x);
62 // After Qt > 5.14 simplify to
63 // angleDelta = angleDelta.transposed();
64 }
65
66 QWheelEvent ev(QPoint(me->event_x, me->event_y), QCursor::pos(), QPoint(), angleDelta, buttons, modifiers, Qt::NoScrollPhase, false);
67 ev.setTimestamp(me->time);
68 return m_effects->checkInputWindowEvent(&ev);
69 }
70 const Qt::MouseButton button = x11ToQtMouseButton(me->detail);
71 Qt::MouseButtons buttons = x11ToQtMouseButtons(me->state);
72 const QEvent::Type type = (eventType == XCB_BUTTON_PRESS) ? QEvent::MouseButtonPress : QEvent::MouseButtonRelease;
73 if (type == QEvent::MouseButtonPress) {
74 buttons |= button;
75 } else {
76 buttons &= ~button;
77 }
78 QMouseEvent ev(type, QPoint(me->event_x, me->event_y), QPoint(me->root_x, me->root_y),
79 button, buttons, x11ToQtKeyboardModifiers(me->state));
80 ev.setTimestamp(me->time);
81 return m_effects->checkInputWindowEvent(&ev);
82 }
83 } else if (eventType == XCB_MOTION_NOTIFY) {
84 const auto *me = reinterpret_cast<xcb_motion_notify_event_t *>(event);
85 if (m_window == me->event) {
86 QMouseEvent ev(QEvent::MouseMove, QPoint(me->event_x, me->event_y), QPoint(me->root_x, me->root_y),
87 Qt::NoButton, x11ToQtMouseButtons(me->state), x11ToQtKeyboardModifiers(me->state));
88 ev.setTimestamp(me->time);
89 return m_effects->checkInputWindowEvent(&ev);
90 }
91 }
92 return false;
93}
