47WaylandEglLayerBuffer::WaylandEglLayerBuffer(const QSize &size, uint32_t format, const QVector<uint64_t> &modifiers, WaylandEglBackend *backend)
48 : m_backend(backend)
49{
50 gbm_device *gbmDevice = backend->backend()->gbmDevice();
51
52 if (!modifiers.isEmpty()) {
53 m_bo = gbm_bo_create_with_modifiers(gbmDevice,
54 size.width(),
55 size.height(),
56 format,
57 modifiers.constData(),
58 modifiers.size());
59 }
60
61 if (!m_bo) {
62 m_bo = gbm_bo_create(gbmDevice,
63 size.width(),
64 size.height(),
65 format,
66 GBM_BO_USE_SCANOUT | GBM_BO_USE_RENDERING);
67 }
68
69 if (!m_bo) {
70 qCCritical(KWIN_WAYLAND_BACKEND) << "Failed to allocate a buffer for an output layer";
71 return;
72 }
73
74 DmaBufAttributes attributes = dmaBufAttributesForBo(m_bo);
75
76 zwp_linux_buffer_params_v1 *params = zwp_linux_dmabuf_v1_create_params(backend->backend()->display()->linuxDmabuf()->handle());
77 for (int i = 0; i < attributes.planeCount; ++i) {
78 zwp_linux_buffer_params_v1_add(params,
79 attributes.fd[i].get(),
80 i,
81 attributes.offset[i],
82 attributes.pitch[i],
83 attributes.modifier >> 32,
84 attributes.modifier & 0xffffffff);
85 }
86
87 m_buffer = zwp_linux_buffer_params_v1_create_immed(params, size.width(), size.height(), format, ZWP_LINUX_BUFFER_PARAMS_V1_FLAGS_Y_INVERT);
88 zwp_linux_buffer_params_v1_destroy(params);
89
90 m_texture = backend->importDmaBufAsTexture(std::move(attributes));
91 m_texture->setYInverted(false);
92 m_framebuffer = std::make_unique<GLFramebuffer>(m_texture.get());
93}

185std::optional<OutputLayerBeginFrameInfo> WaylandEglPrimaryLayer::beginFrame()
186{
187 if (eglMakeCurrent(m_backend->eglDisplay(), EGL_NO_SURFACE, EGL_NO_SURFACE, m_backend->context()) == EGL_FALSE) {
188 qCCritical(KWIN_WAYLAND_BACKEND) << "Make Context Current failed";
189 return std::nullopt;
190 }
191
192 const QSize nativeSize = m_waylandOutput->pixelSize();
193 if (!m_swapchain || m_swapchain->size() != nativeSize) {
194 const WaylandLinuxDmabufV1 *dmabuf = m_backend->backend()->display()->linuxDmabuf();
195 const uint32_t format = DRM_FORMAT_XRGB8888;
196 if (!dmabuf->formats().contains(format)) {
197 qCCritical(KWIN_WAYLAND_BACKEND) << "DRM_FORMAT_XRGB8888 is unsupported";
198 return std::nullopt;
199 }
200 const QVector<uint64_t> modifiers = dmabuf->formats().value(format);
201 m_swapchain = std::make_unique<WaylandEglLayerSwapchain>(nativeSize, format, modifiers, m_backend);
202 }
203
204 m_buffer = m_swapchain->acquire();
205
206 QRegion repair;
207 if (m_backend->supportsBufferAge()) {
208 repair = m_damageJournal.accumulate(m_buffer->age(), infiniteRegion());
209 }
210
211 return OutputLayerBeginFrameInfo{
212 .renderTarget = RenderTarget(m_buffer->framebuffer()),
213 .repaint = repair,
214 };
215}

326bool WaylandEglBackend::initializeEgl()
327{
328 initClientExtensions();
329 EGLDisplay display = m_backend->sceneEglDisplay();
330
331 // Use eglGetPlatformDisplayEXT() to get the display pointer
332 // if the implementation supports it.
333 if (display == EGL_NO_DISPLAY) {
334 m_havePlatformBase = hasClientExtension(QByteArrayLiteral("EGL_EXT_platform_base"));
335 if (m_havePlatformBase) {
336 // Make sure that the wayland platform is supported
337 if (!hasClientExtension(QByteArrayLiteral("EGL_EXT_platform_wayland"))) {
338 return false;
339 }
340
341 display = eglGetPlatformDisplayEXT(EGL_PLATFORM_WAYLAND_EXT, m_backend->display()->nativeDisplay(), nullptr);
342 } else {
343 display = eglGetDisplay(m_backend->display()->nativeDisplay());
344 }
345 }
346
347 if (display == EGL_NO_DISPLAY) {
348 return false;
349 }
350 setEglDisplay(display);
351 return initEglAPI();
352}

370bool WaylandEglBackend::initRenderingContext()
371{
372 initBufferConfigs();
373
374 if (!createContext()) {
375 return false;
376 }
377
378 auto waylandOutputs = m_backend->waylandOutputs();
379
380 // we only allow to start with at least one output
381 if (waylandOutputs.isEmpty()) {
382 return false;
383 }
384
385 for (auto *out : waylandOutputs) {
386 if (!createEglWaylandOutput(out)) {
387 return false;
388 }
389 }
390
391 if (m_outputs.empty()) {
392 qCCritical(KWIN_WAYLAND_BACKEND) << "Create Window Surfaces failed";
393 return false;
394 }
395
396 return makeCurrent();
397}
