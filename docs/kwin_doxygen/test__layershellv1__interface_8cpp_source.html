121void TestLayerShellV1Interface::initTestCase()
122{
123 m_display.addSocketName(s_socketName);
124 m_display.start();
125 QVERIFY(m_display.isRunning());
126
127 m_serverLayerShell = new LayerShellV1Interface(&m_display, this);
128 m_serverXdgShell = new XdgShellInterface(&m_display, this);
129 m_serverCompositor = new CompositorInterface(&m_display, this);
130
131 m_connection = new KWayland::Client::ConnectionThread;
132 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
133 m_connection->setSocketName(s_socketName);
134
135 m_thread = new QThread(this);
136 m_connection->moveToThread(m_thread);
137 m_thread->start();
138
139 m_connection->initConnection();
140 QVERIFY(connectedSpy.wait());
141 QVERIFY(!m_connection->connections().isEmpty());
142
143 m_queue = new KWayland::Client::EventQueue(this);
144 QVERIFY(!m_queue->isValid());
145 m_queue->setup(m_connection);
146 QVERIFY(m_queue->isValid());
147
148 auto registry = new KWayland::Client::Registry(this);
149 connect(registry, &KWayland::Client::Registry::interfaceAnnounced, this, [this, registry](const QByteArray &interface, quint32 id, quint32 version) {
150 if (interface == QByteArrayLiteral("zwlr_layer_shell_v1")) {
151 m_clientLayerShell = new LayerShellV1();
152 m_clientLayerShell->init(*registry, id, version);
153 }
154 if (interface == QByteArrayLiteral("xdg_wm_base")) {
155 m_clientXdgShell = new XdgShell();
156 m_clientXdgShell->init(*registry, id, version);
157 }
158 });
159 QSignalSpy allAnnouncedSpy(registry, &KWayland::Client::Registry::interfaceAnnounced);
160 QSignalSpy compositorSpy(registry, &KWayland::Client::Registry::compositorAnnounced);
161 QSignalSpy shmSpy(registry, &KWayland::Client::Registry::shmAnnounced);
162 registry->setEventQueue(m_queue);
163 registry->create(m_connection->display());
164 QVERIFY(registry->isValid());
165 registry->setup();
166 QVERIFY(allAnnouncedSpy.wait());
167
168 m_clientCompositor = registry->createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
169 QVERIFY(m_clientCompositor->isValid());
170}

444void TestLayerShellV1Interface::testPopup()
445{
446 // Create a test wl_surface object for the panel.
447 QSignalSpy serverPanelSurfaceCreatedSpy(m_serverCompositor, &CompositorInterface::surfaceCreated);
448 std::unique_ptr<KWayland::Client::Surface> clientPanelSurface(m_clientCompositor->createSurface(this));
449 QVERIFY(serverPanelSurfaceCreatedSpy.wait());
450 SurfaceInterface *serverPanelSurface = serverPanelSurfaceCreatedSpy.last().first().value<SurfaceInterface *>();
451 QVERIFY(serverPanelSurface);
452
453 // Create a test wlr_layer_surface_v1 object for the panel..
454 std::unique_ptr<LayerSurfaceV1> clientPanelShellSurface(new LayerSurfaceV1);
455 clientPanelShellSurface->init(m_clientLayerShell->get_layer_surface(*clientPanelSurface, nullptr, LayerShellV1::layer_top, QStringLiteral("panel")));
456 clientPanelShellSurface->set_size(100, 50);
457 QSignalSpy layerSurfaceCreatedSpy(m_serverLayerShell, &LayerShellV1Interface::surfaceCreated);
458 QVERIFY(layerSurfaceCreatedSpy.wait());
459 auto serverPanelShellSurface = layerSurfaceCreatedSpy.last().first().value<LayerSurfaceV1Interface *>();
460 QVERIFY(serverPanelShellSurface);
461
462 // Create a wl_surface object for the popup.
463 std::unique_ptr<KWayland::Client::Surface> clientPopupSurface(m_clientCompositor->createSurface(this));
464 QVERIFY(serverPanelSurfaceCreatedSpy.wait());
465 SurfaceInterface *serverPopupSurface = serverPanelSurfaceCreatedSpy.last().first().value<SurfaceInterface *>();
466 QVERIFY(serverPopupSurface);
467
468 // Create an xdg_surface object for the popup.
469 std::unique_ptr<XdgSurface> clientXdgSurface(new XdgSurface);
470 clientXdgSurface->init(m_clientXdgShell->get_xdg_surface(*clientPopupSurface));
471
472 // Create an xdg_positioner object for the popup.
473 std::unique_ptr<::XdgPositioner> positioner(new ::XdgPositioner);
474 positioner->init(m_clientXdgShell->create_positioner());
475 positioner->set_size(100, 100);
476 positioner->set_anchor_rect(0, 0, 10, 10);
477
478 // Create an xdg_popup surface.
479 std::unique_ptr<XdgPopup> clientXdgPopup(new XdgPopup);
480 clientXdgPopup->init(clientXdgSurface->get_popup(nullptr, positioner->object()));
481
482 // Wait for the server side to catch up.
483 QSignalSpy popupCreatedSpy(m_serverXdgShell, &XdgShellInterface::popupCreated);
484 QVERIFY(popupCreatedSpy.wait());
485 XdgPopupInterface *serverPopupShellSurface = popupCreatedSpy.last().first().value<XdgPopupInterface *>();
486 QVERIFY(serverPopupShellSurface);
487 QCOMPARE(serverPopupShellSurface->parentSurface(), nullptr);
488
489 // Make the xdg_popup surface a child of the panel.
490 clientPanelShellSurface->get_popup(clientXdgPopup->object());
491
492 // Commit the initial state of the xdg_popup surface.
493 clientPopupSurface->commit(KWayland::Client::Surface::CommitFlag::None);
494 QSignalSpy initializeRequestedSpy(serverPopupShellSurface, &XdgPopupInterface::initializeRequested);
495 QVERIFY(initializeRequestedSpy.wait());
496
497 // The popup should be a transient for the panel.
498 QCOMPARE(serverPopupShellSurface->parentSurface(), serverPanelSurface);
499}
