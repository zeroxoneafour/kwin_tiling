60static GLTexture *bindSurfaceTexture(SurfaceItem *surfaceItem)
61{
62 SurfacePixmap *surfacePixmap = surfaceItem->pixmap();
63 auto platformSurfaceTexture =
64 static_cast<OpenGLSurfaceTexture *>(surfacePixmap->texture());
65 if (surfacePixmap->isDiscarded()) {
66 return platformSurfaceTexture->texture();
67 }
68
69 if (platformSurfaceTexture->texture()) {
70 const QRegion region = surfaceItem->damage();
71 if (!region.isEmpty()) {
72 platformSurfaceTexture->update(region);
73 surfaceItem->resetDamage();
74 }
75 } else {
76 if (!surfacePixmap->isValid()) {
77 return nullptr;
78 }
79 if (!platformSurfaceTexture->create()) {
80 qCDebug(KWIN_OPENGL) << "Failed to bind window";
81 return nullptr;
82 }
83 surfaceItem->resetDamage();
84 }
85
86 return platformSurfaceTexture->texture();
87}

95static RenderGeometry clipQuads(const Item *item, const ItemRendererOpenGL::RenderContext *context)
96{
97 const WindowQuadList quads = item->quads();
98
99 // Item to world translation.
100 const QPointF worldTranslation = context->transformStack.top().map(QPointF(0., 0.));
101 const qreal scale = context->renderTargetScale;
102
103 RenderGeometry geometry;
104 geometry.reserve(quads.count() * 6);
105
106 // split all quads in bounding rect with the actual rects in the region
107 for (const WindowQuad &quad : std::as_const(quads)) {
108 if (context->clip != infiniteRegion() && !context->hardwareClipping) {
109 // Scale to device coordinates, rounding as needed.
110 QRectF deviceBounds = logicalRectToDeviceRect(quad.bounds(), scale);
111
112 for (const QRect &clipRect : std::as_const(context->clip)) {
113 QRectF deviceClipRect = logicalRectToDeviceRect(clipRect, scale).translated(-worldTranslation);
114
115 const QRectF &intersected = deviceClipRect.intersected(deviceBounds);
116 if (intersected.isValid()) {
117 if (deviceBounds == intersected) {
118 // case 1: completely contains, include and do not check other rects
119 geometry.appendWindowQuad(quad, scale);
120 break;
121 }
122 // case 2: intersection
123 geometry.appendSubQuad(quad, intersected, scale);
124 }
125 }
126 } else {
127 geometry.appendWindowQuad(quad, scale);
128 }
129 }
130
131 return geometry;
132}

134void ItemRendererOpenGL::createRenderNode(Item *item, RenderContext *context)
135{
136 const QList<Item *> sortedChildItems = item->sortedChildItems();
137
138 QMatrix4x4 matrix;
139 const auto logicalPosition = QVector2D(item->position().x(), item->position().y());
140 const auto scale = context->renderTargetScale;
141 matrix.translate(roundVector(logicalPosition * scale).toVector3D());
142 matrix *= item->transform();
143 context->transformStack.push(context->transformStack.top() * matrix);
144
145 context->opacityStack.push(context->opacityStack.top() * item->opacity());
146
147 for (Item *childItem : sortedChildItems) {
148 if (childItem->z() >= 0) {
149 break;
150 }
151 if (childItem->explicitVisible()) {
152 createRenderNode(childItem, context);
153 }
154 }
155
156 item->preprocess();
157
158 RenderGeometry geometry = clipQuads(item, context);
159
160 if (auto shadowItem = qobject_cast<ShadowItem *>(item)) {
161 if (!geometry.isEmpty()) {
162 OpenGLShadowTextureProvider *textureProvider = static_cast<OpenGLShadowTextureProvider *>(shadowItem->textureProvider());
163 context->renderNodes.append(RenderNode{
164 .texture = textureProvider->shadowTexture(),
165 .geometry = geometry,
166 .transformMatrix = context->transformStack.top(),
167 .opacity = context->opacityStack.top(),
168 .hasAlpha = true,
169 .coordinateType = UnnormalizedCoordinates,
170 .scale = scale,
171 });
172 }
173 } else if (auto decorationItem = qobject_cast<DecorationItem *>(item)) {
174 if (!geometry.isEmpty()) {
175 auto renderer = static_cast<const SceneOpenGLDecorationRenderer *>(decorationItem->renderer());
176 context->renderNodes.append(RenderNode{
177 .texture = renderer->texture(),
178 .geometry = geometry,
179 .transformMatrix = context->transformStack.top(),
180 .opacity = context->opacityStack.top(),
181 .hasAlpha = true,
182 .coordinateType = UnnormalizedCoordinates,
183 .scale = scale,
184 });
185 }
186 } else if (auto surfaceItem = qobject_cast<SurfaceItem *>(item)) {
187 SurfacePixmap *pixmap = surfaceItem->pixmap();
188 if (pixmap) {
189 if (!geometry.isEmpty()) {
190 context->renderNodes.append(RenderNode{
191 .texture = bindSurfaceTexture(surfaceItem),
192 .geometry = geometry,
193 .transformMatrix = context->transformStack.top(),
194 .opacity = context->opacityStack.top(),
195 .hasAlpha = pixmap->hasAlphaChannel(),
196 .coordinateType = NormalizedCoordinates,
197 .scale = scale,
198 });
199 }
200 }
201 } else if (auto imageItem = qobject_cast<ImageItemOpenGL *>(item)) {
202 if (!geometry.isEmpty()) {
203 context->renderNodes.append(RenderNode{
204 .texture = imageItem->texture(),
205 .geometry = geometry,
206 .transformMatrix = context->transformStack.top(),
207 .opacity = context->opacityStack.top(),
208 .hasAlpha = imageItem->image().hasAlphaChannel(),
209 .coordinateType = NormalizedCoordinates,
210 .scale = scale,
211 });
212 }
213 }
214
215 for (Item *childItem : sortedChildItems) {
216 if (childItem->z() < 0) {
217 continue;
218 }
219 if (childItem->explicitVisible()) {
220 createRenderNode(childItem, context);
221 }
222 }
223
224 context->transformStack.pop();
225 context->opacityStack.pop();
226}

228void ItemRendererOpenGL::renderBackground(const QRegion &region)
229{
230 if (region == infiniteRegion() || (region.rectCount() == 1 && (*region.begin()) == renderTargetRect())) {
231 glClearColor(0, 0, 0, 0);
232 glClear(GL_COLOR_BUFFER_BIT);
233 } else if (!region.isEmpty()) {
234 glClearColor(0, 0, 0, 0);
235 glEnable(GL_SCISSOR_TEST);
236
237 const auto scale = renderTargetScale();
238 const auto targetRect = scaledRect(renderTargetRect(), scale).toRect();
239
240 for (const QRect &r : region) {
241 auto deviceRect = scaledRect(r, scale).toAlignedRect();
242 glScissor(deviceRect.x(), targetRect.height() - (deviceRect.y() + deviceRect.height()), deviceRect.width(), deviceRect.height());
243 glClear(GL_COLOR_BUFFER_BIT);
244 }
245
246 glDisable(GL_SCISSOR_TEST);
247 }

250void ItemRendererOpenGL::renderItem(Item *item, int mask, const QRegion &region, const WindowPaintData &data)
251{
252 if (region.isEmpty()) {
253 return;
254 }
255
256 RenderContext renderContext{
257 .clip = region,
258 .hardwareClipping = region != infiniteRegion() && ((mask & Scene::PAINT_WINDOW_TRANSFORMED) || (mask & Scene::PAINT_SCREEN_TRANSFORMED)),
259 .renderTargetScale = data.renderTargetScale().value_or(renderTargetScale()),
260 };
261
262 renderContext.transformStack.push(QMatrix4x4());
263 renderContext.opacityStack.push(data.opacity());
264
265 item->setTransform(data.toMatrix(renderTargetScale()));
266
267 createRenderNode(item, &renderContext);
268
269 int totalVertexCount = 0;
270 for (const RenderNode &node : std::as_const(renderContext.renderNodes)) {
271 totalVertexCount += node.geometry.count();
272 }
273 if (totalVertexCount == 0) {
274 return;
275 }
276
277 const size_t size = totalVertexCount * sizeof(GLVertex2D);
278
279 ShaderTraits shaderTraits = ShaderTrait::MapTexture;
280
281 if (data.brightness() != 1.0) {
282 shaderTraits |= ShaderTrait::Modulate;
283 }
284 if (data.saturation() != 1.0) {
285 shaderTraits |= ShaderTrait::AdjustSaturation;
286 }
287
288 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
289 vbo->reset();
290 vbo->setAttribLayout(GLVertexBuffer::GLVertex2DLayout, 2, sizeof(GLVertex2D));
291
292 GLVertex2D *map = (GLVertex2D *)vbo->map(size);
293
294 for (int i = 0, v = 0; i < renderContext.renderNodes.count(); i++) {
295 RenderNode &renderNode = renderContext.renderNodes[i];
296 if (renderNode.geometry.isEmpty() || !renderNode.texture) {
297 continue;
298 }
299
300 if (renderNode.opacity != 1.0) {
301 shaderTraits |= ShaderTrait::Modulate;
302 }
303
304 renderNode.firstVertex = v;
305 renderNode.vertexCount = renderNode.geometry.count();
306
307 renderNode.geometry.postProcessTextureCoordinates(renderNode.texture->matrix(renderNode.coordinateType));
308
309 renderNode.geometry.copy(std::span(&map[v], renderNode.geometry.count()));
310 v += renderNode.geometry.count();
311 }
312
313 vbo->unmap();
314 vbo->bindArrays();
315
316 GLShader *shader = ShaderManager::instance()->pushShader(shaderTraits);
317 shader->setUniform(GLShader::Saturation, data.saturation());
318
319 if (renderContext.hardwareClipping) {
320 glEnable(GL_SCISSOR_TEST);
321 }
322
323 // Make sure the blend function is set up correctly in case we will be doing blending
324 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
325
326 float opacity = -1.0;
327
328 // The scissor region must be in the render target local coordinate system.
329 QRegion scissorRegion = infiniteRegion();
330 if (renderContext.hardwareClipping) {
331 scissorRegion = mapToRenderTarget(region);
332 }
333
334 const QMatrix4x4 projectionMatrix = data.projectionMatrix();
335 for (int i = 0; i < renderContext.renderNodes.count(); i++) {
336 const RenderNode &renderNode = renderContext.renderNodes[i];
337 if (renderNode.vertexCount == 0) {
338 continue;
339 }
340
341 setBlendEnabled(renderNode.hasAlpha || renderNode.opacity < 1.0);
342
343 shader->setUniform(GLShader::ModelViewProjectionMatrix, projectionMatrix * renderNode.transformMatrix);
344 if (opacity != renderNode.opacity) {
345 shader->setUniform(GLShader::ModulationConstant,
346 modulate(renderNode.opacity, data.brightness()));
347 opacity = renderNode.opacity;
348 }
349
350 renderNode.texture->setFilter(GL_LINEAR);
351 renderNode.texture->setWrapMode(GL_CLAMP_TO_EDGE);
352 renderNode.texture->bind();
353
354 vbo->draw(scissorRegion, GL_TRIANGLES, renderNode.firstVertex,
355 renderNode.vertexCount, renderContext.hardwareClipping);
356 }
357
358 vbo->unbindArrays();
359
360 setBlendEnabled(false);
361
362 ShaderManager::instance()->popShader();
363
364 if (renderContext.hardwareClipping) {
365 glDisable(GL_SCISSOR_TEST);
366 }
367}
