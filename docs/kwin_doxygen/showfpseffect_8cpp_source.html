49void ShowFpsEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
50{
51 effects->prePaintScreen(data, presentTime);
52
53 m_newFps += 1;
54
55 m_paintDurationTimer.restart();
56 m_paintAmount = 0;
57
58 // detect highest monitor refresh rate
59 int maximumFps = 0;
60 const auto screens = effects->screens();
61 for (auto screen : screens) {
62 maximumFps = std::max(screen->refreshRate(), maximumFps);
63 }
64 maximumFps /= 1000; // Convert from mHz to Hz.
65
66 if (maximumFps != m_maximumFps) {
67 m_maximumFps = maximumFps;
68 Q_EMIT maximumFpsChanged();
69 }
70
71 if (!m_scene) {
72 m_window = std::make_unique<QWindow>();
73 m_window->create();
74 m_scene = std::make_unique<OffscreenQuickScene>(nullptr, m_window.get());
75 const auto url = QUrl::fromLocalFile(QStandardPaths::locate(QStandardPaths::GenericDataLocation, QStringLiteral("kwin/effects/showfps/qml/main.qml")));
76 m_scene->setSource(url, {{QStringLiteral("effect"), QVariant::fromValue(this)}});
77 }
78
79 const auto rect = effects->renderTargetRect();
80 m_scene->setGeometry(QRect(rect.x() + rect.width() - 300, 0, 300, 150));
81}

83void ShowFpsEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
84{
85 effects->paintScreen(mask, region, data);
86
87 auto now = std::chrono::steady_clock::now();
88 if ((now - m_lastFpsTime) >= std::chrono::milliseconds(1000)) {
89 m_fps = m_newFps;
90 m_newFps = 0;
91 m_lastFpsTime = now;
92 Q_EMIT fpsChanged();
93 }
94
95 effects->renderOffscreenQuickView(m_scene.get());
96}

98void ShowFpsEffect::paintWindow(EffectWindow *w, int mask, QRegion region, WindowPaintData &data)
99{
100 effects->paintWindow(w, mask, region, data);
101
102 // Take intersection of region and actual window's rect, minus the fps area
103 // (since we keep repainting it) and count the pixels.
104 QRegion repaintRegion = region & w->frameGeometry().toRect();
105 repaintRegion -= m_scene->geometry();
106 for (const QRect &rect : repaintRegion) {
107 m_paintAmount += rect.width() * rect.height();
108 }
109}
