73inline QRegion buildClipRegion(const QPoint &pos, int w, int h)
74{
75 const QSize screenSize = effects->virtualScreenSize();
76 QRegion r = QRect(pos, screenSize);
77 if (effects->optionRollOverDesktops()) {
78 r |= (r & QRect(-w, 0, w, h)).translated(w, 0); // W
79 r |= (r & QRect(w, 0, w, h)).translated(-w, 0); // E
80
81 r |= (r & QRect(0, -h, w, h)).translated(0, h); // N
82 r |= (r & QRect(0, h, w, h)).translated(0, -h); // S
83
84 r |= (r & QRect(-w, -h, w, h)).translated(w, h); // NW
85 r |= (r & QRect(w, -h, w, h)).translated(-w, h); // NE
86 r |= (r & QRect(w, h, w, h)).translated(-w, -h); // SE
87 r |= (r & QRect(-w, h, w, h)).translated(w, -h); // SW
88 }
89 return r;
90}

92void SlideEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
93{
94 std::chrono::milliseconds timeDelta = std::chrono::milliseconds::zero();
95 if (m_lastPresentTime.count()) {
96 timeDelta = presentTime - m_lastPresentTime;
97 }
98 m_lastPresentTime = presentTime;
99
100 if (m_state == State::ActiveAnimation) {
101 m_motionX.advance(timeDelta);
102 m_motionY.advance(timeDelta);
103 const QSize virtualSpaceSize = effects->virtualScreenSize();
104 m_currentPosition.setX(m_motionX.position() / virtualSpaceSize.width());
105 m_currentPosition.setY(m_motionY.position() / virtualSpaceSize.height());
106 }
107
108 const int w = effects->desktopGridWidth();
109 const int h = effects->desktopGridHeight();
110
111 // Clipping
112 m_paintCtx.visibleDesktops.clear();
113 m_paintCtx.visibleDesktops.reserve(4); // 4 - maximum number of visible desktops
114 bool includedX = false, includedY = false;
115 for (int i = 1; i <= effects->numberOfDesktops(); i++) {
116 if (effects->desktopGridCoords(i).x() % w == (int)(m_currentPosition.x()) % w) {
117 includedX = true;
118 } else if (effects->desktopGridCoords(i).x() % w == ((int)(m_currentPosition.x()) + 1) % w) {
119 includedX = true;
120 }
121 if (effects->desktopGridCoords(i).y() % h == (int)(m_currentPosition.y()) % h) {
122 includedY = true;
123 } else if (effects->desktopGridCoords(i).y() % h == ((int)(m_currentPosition.y()) + 1) % h) {
124 includedY = true;
125 }
126
127 if (includedX && includedY) {
128 m_paintCtx.visibleDesktops << i;
129 }
130 }
131
132 data.mask |= PAINT_SCREEN_TRANSFORMED | PAINT_SCREEN_BACKGROUND_FIRST;
133
134 effects->prePaintScreen(data, presentTime);
135}

193void SlideEffect::paintWindow(EffectWindow *w, int mask, QRegion region, WindowPaintData &data)
194{
195 if (!willBePainted(w)) {
196 return;
197 }
198
199 if (!isTranslated(w)) {
200 effects->paintWindow(
201 w,
202 mask,
203 region,
204 data);
205 return;
206 }
207
208 const int gridWidth = effects->desktopGridWidth();
209 const int gridHeight = effects->desktopGridHeight();
210
211 QPointF drawPosition = forcePositivePosition(m_currentPosition);
212 drawPosition = m_paintCtx.wrap ? constrainToDrawableRange(drawPosition) : drawPosition;
213
214 // If we're wrapping, draw the desktop in the second position.
215 const bool wrappingX = drawPosition.x() > gridWidth - 1;
216 const bool wrappingY = drawPosition.y() > gridHeight - 1;
217
218 const auto screens = effects->screens();
219
220 for (int desktop : std::as_const(m_paintCtx.visibleDesktops)) {
221 if (!w->isOnDesktop(desktop)) {
222 continue;
223 }
224 QPointF desktopTranslation = QPointF(effects->desktopGridCoords(desktop)) - drawPosition;
225 // Decide if that first desktop should be drawn at 0 or the higher position used for wrapping.
226 if (effects->desktopGridCoords(desktop).x() == 0 && wrappingX) {
227 desktopTranslation = QPointF(desktopTranslation.x() + gridWidth, desktopTranslation.y());
228 }
229 if (effects->desktopGridCoords(desktop).y() == 0 && wrappingY) {
230 desktopTranslation = QPointF(desktopTranslation.x(), desktopTranslation.y() + gridHeight);
231 }
232
233 for (EffectScreen *screen : screens) {
234 QPoint drawTranslation = getDrawCoords(desktopTranslation, screen);
235 data += drawTranslation;
236
237 const QRect screenArea = screen->geometry();
238 const QRect damage = screenArea.translated(drawTranslation).intersected(screenArea);
239
240 effects->paintWindow(
241 w,
242 mask,
243 // Only paint the region that intersects the current screen and desktop.
244 region.intersected(damage),
245 data);
246
247 // Undo the translation for the next screen. I know, it hurts me too.
248 data += QPoint(-drawTranslation.x(), -drawTranslation.y());
249 }
250 }
251}

266QPointF SlideEffect::forcePositivePosition(QPointF p) const
267{
268 if (p.x() < 0) {
269 p.setX(p.x() + std::ceil(-p.x() / effects->desktopGridWidth()) * effects->desktopGridWidth());
270 }
271 if (p.y() < 0) {
272 p.setY(p.y() + std::ceil(-p.y() / effects->desktopGridHeight()) * effects->desktopGridHeight());
273 }
274 return p;

286 * This function is called when the desktop changes.
287 * Called AFTER the gesture is released.

290void SlideEffect::startAnimation(int old, int current, EffectWindow *movingWindow)
291{
292 if (m_state == State::Inactive) {
293 prepareSwitching();
294 }
295
296 m_state = State::ActiveAnimation;
297 m_movingWindow = movingWindow;
298
299 m_startPos = m_currentPosition;
300 m_endPos = effects->desktopGridCoords(current);
301 if (effects->optionRollOverDesktops()) {
302 optimizePath();
303 }
304
305 const QSize virtualSpaceSize = effects->virtualScreenSize();
306 m_motionX.setAnchor(m_endPos.x() * virtualSpaceSize.width());
307 m_motionX.setPosition(m_startPos.x() * virtualSpaceSize.width());
308 m_motionY.setAnchor(m_endPos.y() * virtualSpaceSize.height());
309 m_motionY.setPosition(m_startPos.y() * virtualSpaceSize.height());
310
311 effects->setActiveFullScreenEffect(this);
312 effects->addRepaintFull();
313}

315void SlideEffect::prepareSwitching()
316{
317 const auto windows = effects->stackingOrder();
318 m_windowData.reserve(windows.count());
319
320 for (EffectWindow *w : windows) {
321 m_windowData[w] = WindowData{
322 .visibilityRef = EffectWindowVisibleRef(w, EffectWindow::PAINT_DISABLED_BY_DESKTOP),
323 };
324
325 if (shouldElevate(w)) {
326 effects->setElevatedWindow(w, true);
327 m_elevatedWindows << w;
328 }
329 w->setData(WindowForceBackgroundContrastRole, QVariant(true));
330 w->setData(WindowForceBlurRole, QVariant(true));
331 }
332}

334void SlideEffect::finishedSwitching()
335{
336 if (m_state == State::Inactive) {
337 return;
338 }
339 const EffectWindowList windows = effects->stackingOrder();
340 for (EffectWindow *w : windows) {
341 w->setData(WindowForceBackgroundContrastRole, QVariant());
342 w->setData(WindowForceBlurRole, QVariant());
343 }
344
345 for (EffectWindow *w : std::as_const(m_elevatedWindows)) {
346 effects->setElevatedWindow(w, false);
347 }
348 m_elevatedWindows.clear();
349
350 m_windowData.clear();
351 m_movingWindow = nullptr;
352 m_state = State::Inactive;
353 m_lastPresentTime = std::chrono::milliseconds::zero();
354 effects->setActiveFullScreenEffect(nullptr);
355 m_currentPosition = effects->desktopGridCoords(effects->currentDesktop());
356}

368void SlideEffect::desktopChanging(uint old, QPointF desktopOffset, EffectWindow *with)
369{
370 if (effects->hasActiveFullScreenEffect() && effects->activeFullScreenEffect() != this) {
371 return;
372 }
373
374 if (m_state == State::Inactive) {
375 prepareSwitching();
376 }
377
378 m_state = State::ActiveGesture;
379 m_movingWindow = with;
380
381 // Find desktop position based on animationDelta
382 QPoint gridPos = effects->desktopGridCoords(old);
383 m_currentPosition.setX(gridPos.x() + desktopOffset.x());
384 m_currentPosition.setY(gridPos.y() + desktopOffset.y());
385
386 if (effects->optionRollOverDesktops()) {
387 m_currentPosition = forcePositivePosition(m_currentPosition);
388 } else {
389 m_currentPosition = moveInsideDesktopGrid(m_currentPosition);
390 }
391
392 effects->setActiveFullScreenEffect(this);
393 effects->addRepaintFull();
394}

451/*
452 * Find the fastest path between two desktops.
453 * This function decides when it's better to wrap around the grid or not.
454 * Only call if wrapping is enabled.
455 */
456void SlideEffect::optimizePath()
457{
458 int w = effects->desktopGridWidth();
459 int h = effects->desktopGridHeight();
460
461 // Keep coordinates as low as possible
462 if (m_startPos.x() >= w && m_endPos.x() >= w) {
463 m_startPos.setX(fmod(m_startPos.x(), w));
464 m_endPos.setX(fmod(m_endPos.x(), w));
465 }
466 if (m_startPos.y() >= h && m_endPos.y() >= h) {
467 m_startPos.setY(fmod(m_startPos.y(), h));
468 m_endPos.setY(fmod(m_endPos.y(), h));
469 }
470
471 // Is there is a shorter possible route?
472 // If the x distance to be traveled is more than half the grid width, it's faster to wrap.
473 // To avoid negative coordinates, take the lower coordinate and raise.
474 if (std::abs((m_startPos.x() - m_endPos.x())) > w / 2.0) {
475 if (m_startPos.x() < m_endPos.x()) {
476 while (m_startPos.x() < m_endPos.x()) {
477 m_startPos.setX(m_startPos.x() + w);
478 }
479 } else {
480 while (m_endPos.x() < m_startPos.x()) {
481 m_endPos.setX(m_endPos.x() + w);
482 }
483 }
484 // Keep coordinates as low as possible
485 if (m_startPos.x() >= w && m_endPos.x() >= w) {
486 m_startPos.setX(fmod(m_startPos.x(), w));
487 m_endPos.setX(fmod(m_endPos.x(), w));
488 }
489 }
490
491 // Same for y
492 if (std::abs((m_endPos.y() - m_startPos.y())) > (double)h / (double)2) {
493 if (m_startPos.y() < m_endPos.y()) {
494 while (m_startPos.y() < m_endPos.y()) {
495 m_startPos.setY(m_startPos.y() + h);
496 }
497 } else {
498 while (m_endPos.y() < m_startPos.y()) {
499 m_endPos.setY(m_endPos.y() + h);
500 }
501 }
502 // Keep coordinates as low as possible
503 if (m_startPos.y() >= h && m_endPos.y() >= h) {
504 m_startPos.setY(fmod(m_startPos.y(), h));
505 m_endPos.setY(fmod(m_endPos.y(), h));
506 }
507 }
508}

510/*
511 * Takes the point and uses modulus to keep draw position within [0, desktopGridWidth]
512 * The render loop will draw the first desktop (0) after the last one (at position desktopGridWidth) for the wrap animation.
513 * This function finds the true fastest path, regardless of which direction the animation is already going;
514 * I was a little upset about this limitation until I realized that MacOS can't even wrap desktops :)
515 */
516QPointF SlideEffect::constrainToDrawableRange(QPointF p)
517{
518 p.setX(fmod(p.x(), effects->desktopGridWidth()));
519 p.setY(fmod(p.y(), effects->desktopGridHeight()));
520 return p;
521}
