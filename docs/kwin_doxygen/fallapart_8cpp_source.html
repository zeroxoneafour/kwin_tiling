49void FallApartEffect::prePaintWindow(EffectWindow *w, WindowPrePaintData &data, std::chrono::milliseconds presentTime)
50{
51 auto animationIt = windows.find(w);
52 if (animationIt != windows.end() && isRealWindow(w)) {
53 int time = 0;
54 if (animationIt->lastPresentTime.count()) {
55 time = (presentTime - animationIt->lastPresentTime).count();
56 }
57 animationIt->lastPresentTime = presentTime;
58
59 animationIt->progress += time / animationTime(1000.);
60 data.setTransformed();
61 }
62 effects->prePaintWindow(w, data, presentTime);
63}

65void FallApartEffect::apply(EffectWindow *w, int mask, WindowPaintData &data, WindowQuadList &quads)
66{
67 auto animationIt = windows.constFind(w);
68 if (animationIt != windows.constEnd() && isRealWindow(w)) {
69 const qreal t = animationIt->progress;
70 // Request the window to be divided into cells
71 quads = quads.makeGrid(blockSize);
72 int cnt = 0;
73 for (WindowQuad &quad : quads) {
74 // make fragments move in various directions, based on where
75 // they are (left pieces generally move to the left, etc.)
76 QPointF p1(quad[0].x(), quad[0].y());
77 double xdiff = 0;
78 if (p1.x() < w->width() / 2) {
79 xdiff = -(w->width() / 2 - p1.x()) / w->width() * 100;
80 }
81 if (p1.x() > w->width() / 2) {
82 xdiff = (p1.x() - w->width() / 2) / w->width() * 100;
83 }
84 double ydiff = 0;
85 if (p1.y() < w->height() / 2) {
86 ydiff = -(w->height() / 2 - p1.y()) / w->height() * 100;
87 }
88 if (p1.y() > w->height() / 2) {
89 ydiff = (p1.y() - w->height() / 2) / w->height() * 100;
90 }
91 double modif = t * t * 64;
92 srandom(cnt); // change direction randomly but consistently
93 xdiff += (rand() % 21 - 10);
94 ydiff += (rand() % 21 - 10);
95 for (int j = 0;
96 j < 4;
97 ++j) {
98 quad[j].move(quad[j].x() + xdiff * modif, quad[j].y() + ydiff * modif);
99 }
100 // also make the fragments rotate around their center
101 QPointF center((quad[0].x() + quad[1].x() + quad[2].x() + quad[3].x()) / 4,
102 (quad[0].y() + quad[1].y() + quad[2].y() + quad[3].y()) / 4);
103 double adiff = (rand() % 720 - 360) / 360. * 2 * M_PI; // spin randomly
104 for (int j = 0;
105 j < 4;
106 ++j) {
107 double x = quad[j].x() - center.x();
108 double y = quad[j].y() - center.y();
109 double angle = atan2(y, x);
110 angle += animationIt->progress * adiff;
111 double dist = sqrt(x * x + y * y);
112 x = dist * cos(angle);
113 y = dist * sin(angle);
114 quad[j].move(center.x() + x, center.y() + y);
115 }
116 ++cnt;
117 }
118 data.multiplyOpacity(interpolate(1.0, 0.0, t));
119 }
120}

137bool FallApartEffect::isRealWindow(EffectWindow *w)
138{
139 // TODO: isSpecialWindow is rather generic, maybe tell windowtypes separately?
140 /*
141 qCDebug(KWIN_FALLAPART) << "--" << w->caption() << "--------------------------------";
142 qCDebug(KWIN_FALLAPART) << "Tooltip:" << w->isTooltip();

151 if (w->isPopupWindow()) {
152 return false;

154 if (w->isX11Client() && !w->isManaged()) {

157 if (!w->isNormalWindow()) {
158 return false;
159 }
160 return true;

163void FallApartEffect::slotWindowClosed(EffectWindow *c)
164{
165 if (effects->activeFullScreenEffect()) {
166 return;
167 }
168 if (!isRealWindow(c)) {
169 return;
170 }
171 if (!c->isVisible()) {
172 return;
173 }
174 const void *e = c->data(WindowClosedGrabRole).value<void *>();
175 if (e && e != this) {
176 return;
177 }
178 c->setData(WindowClosedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
179
180 FallApartAnimation &animation = windows[c];
181 animation.progress = 0;
182 animation.deletedRef = EffectWindowDeletedRef(c);
183 animation.visibleRef = EffectWindowVisibleRef(c, EffectWindow::PAINT_DISABLED_BY_DELETE);
184
185 redirect(c);
186}

193void FallApartEffect::slotWindowDataChanged(EffectWindow *w, int role)
194{

196 return;
197 }
198
199 if (w->data(role).value<void *>() == this) {
200 return;
201 }
202

204 if (it != windows.end()) {
205 unredirect(it.key());
206 windows.erase(it);
207 }
