50 m_attemptedThisFrame = true;

60 Q_ASSERT(buffer);

62 if (!m_attemptedFormats[dmabufAttrs.format].contains(dmabufAttrs.modifier)) {
63 m_attemptedFormats[dmabufAttrs.format] << dmabufAttrs.modifier;
64 QVector<KWaylandServer::LinuxDmaBufV1Feedback::Tranche> scanoutTranches;
65 const auto tranches = m_eglBackend->dmabuf()->tranches();
66 for (const auto &tranche : tranches) {

68 for (auto it = tranche.formatTable.constBegin(); it != tranche.formatTable.constEnd(); it++) {

72 for (const auto &mod : trancheModifiers) {
73 if (drmModifiers.contains(mod) && !m_attemptedFormats[format].contains(mod)) {
74 scanoutTranche.formatTable[format] << mod;

76 }
