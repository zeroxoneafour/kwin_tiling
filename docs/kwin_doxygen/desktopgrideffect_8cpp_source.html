64 if (!effects->hasActiveFullScreenEffect() || effects->activeFullScreenEffect() == this) {

88void DesktopGridEffect::reconfigure(ReconfigureFlags)
89{
90 DesktopGridConfig::self()->read();
91 setLayout(DesktopGridConfig::layoutMode());
92 setAnimationDuration(animationTime(300));
93
94 for (const ElectricBorder &border : std::as_const(m_borderActivate)) {
95 effects->unreserveElectricBorder(border, this);
96 }
97
98 for (const ElectricBorder &border : std::as_const(m_touchBorderActivate)) {
99 effects->unregisterTouchBorder(border, m_toggleAction);
100 }
101
102 m_borderActivate.clear();
103 m_touchBorderActivate.clear();
104
105 const QList<int> activateBorders = DesktopGridConfig::borderActivate();
106 for (const int &border : activateBorders) {
107 m_borderActivate.append(ElectricBorder(border));
108 effects->reserveElectricBorder(ElectricBorder(border), this);
109 }
110
111 const QList<int> touchActivateBorders = DesktopGridConfig::touchBorderActivate();
112 for (const int &border : touchActivateBorders) {
113 m_touchBorderActivate.append(ElectricBorder(border));
114 effects->registerRealtimeTouchBorder(ElectricBorder(border), m_realtimeToggleAction, [this](ElectricBorder border, const QPointF &deltaProgress, const EffectScreen *screen) {
115 if (m_status == Status::Active) {
116 return;
117 }
118 const int rows = gridRows();
119 const int columns = gridColumns();
120 if (border == ElectricTop || border == ElectricBottom) {
121 const int maxDelta = (screen->geometry().height() / rows) * (rows - (effects->currentDesktop() % rows));
122 partialActivate(std::min(1.0, std::abs(deltaProgress.y()) / maxDelta));
123 } else {
124 const int maxDelta = (screen->geometry().width() / columns) * (columns - (effects->currentDesktop() % columns));
125 partialActivate(std::min(1.0, std::abs(deltaProgress.x()) / maxDelta));
126 }
127 });
128 }
129
130 Q_EMIT showAddRemoveChanged();
131 Q_EMIT desktopNameAlignmentChanged();
132 Q_EMIT desktopLayoutModeChanged();
133 Q_EMIT customLayoutRowsChanged();
134}

186void DesktopGridEffect::swapDesktops(int from, int to)
187{
188 QList<EffectWindow *> fromList;
189 QList<EffectWindow *> toList;
190 for (auto *w : effects->stackingOrder()) {
191 if (!w->isNormalWindow() || !w->isOnCurrentActivity() ) {
192 continue;
193 }
194 if (w->isOnDesktop(from)) {
195 fromList << w;
196 } else if (w->isOnDesktop(to)) {
197 toList << w;
198 }
199 }
200 for (auto *w : fromList) {
201 effects->windowToDesktop(w, to);
202 }
203 for (auto *w : toList) {
204 effects->windowToDesktop(w, from);
205 }
206}

208int DesktopGridEffect::gridRows() const
209{
210 switch (desktopLayoutMode()) {
211 case DesktopLayoutMode::LayoutAutomatic:
212 return ceil(sqrt(effects->numberOfDesktops()));
213 case DesktopLayoutMode::LayoutCustom:
214 return std::clamp(customLayoutRows(), 1, effects->numberOfDesktops());
215 case DesktopLayoutMode::LayoutPager:
216 default:
217 return effects->desktopGridSize().height();
218 }
219}
220
221int DesktopGridEffect::gridColumns() const
222{
223 switch (desktopLayoutMode()) {
224 case DesktopLayoutMode::LayoutAutomatic:
225 return ceil(sqrt(effects->numberOfDesktops()));
226 case DesktopLayoutMode::LayoutCustom:
227 return std::max(1.0, ceil(qreal(effects->numberOfDesktops()) / customLayoutRows()));
228 case DesktopLayoutMode::LayoutPager:
229 default:
230 return effects->desktopGridSize().width();
231 }
232}
