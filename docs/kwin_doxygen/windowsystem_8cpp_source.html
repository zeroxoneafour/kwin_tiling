196void WindowSystem::setType(WId win, NET::WindowType windowType)
197{
198 const auto windows = qApp->allWindows();
199 auto it = std::find_if(windows.begin(), windows.end(), [win](QWindow *w) {
200 return w->winId() == win;
201 });
202 if (it == windows.end()) {
203 return;
204 }
205
206 (*it)->setProperty("kwin_windowType", QVariant::fromValue(windowType));
207}

257void WindowSystem::requestToken(QWindow *win, uint32_t serial, const QString &appId)
258{
259 auto seat = KWin::waylandServer()->seat();
260 auto token = KWin::waylandServer()->xdgActivationIntegration()->requestPrivilegedToken(nullptr, seat->display()->serial(), seat, appId);
261 // Ensure that xdgActivationTokenArrived is always emitted asynchronously
262 QTimer::singleShot(0, [serial, token] {
263 Q_EMIT KWindowSystem::self()->xdgActivationTokenArrived(serial, token);
264 });
265}
