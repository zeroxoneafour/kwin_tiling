236template<typename Data>
237class AbstractWrapper
238{
239public:
240 typedef typename Data::cookie_type Cookie;
241 typedef typename Data::reply_type Reply;
242 virtual ~AbstractWrapper()
243 {
244 cleanup();
245 }
246 inline AbstractWrapper &operator=(const AbstractWrapper &other)
247 {
248 if (this != &other) {
249 // if we had managed a reply, free it
250 cleanup();
251 // copy members
252 m_retrieved = other.m_retrieved;
253 m_cookie = other.m_cookie;
254 m_window = other.m_window;
255 m_reply = other.m_reply;
256 // take over the responsibility for the reply pointer
257 takeFromOther(const_cast<AbstractWrapper &>(other));
258 }
259 return *this;
260 }
261
262 inline const Reply *operator->()
263 {
264 getReply();
265 return m_reply;
266 }
267 inline bool isNull()
268 {
269 getReply();
270 return m_reply == nullptr;
271 }
272 inline bool isNull() const
273 {
274 const_cast<AbstractWrapper *>(this)->getReply();
275 return m_reply == NULL;
276 }
277 inline operator bool()
278 {
279 return !isNull();
280 }
281 inline operator bool() const
282 {
283 return !isNull();
284 }
285 inline const Reply *data()
286 {
287 getReply();
288 return m_reply;
289 }
290 inline const Reply *data() const
291 {
292 const_cast<AbstractWrapper *>(this)->getReply();
293 return m_reply;
294 }
295 inline WindowId window() const
296 {
297 return m_window;
298 }
299 inline bool isRetrieved() const
300 {
301 return m_retrieved;
302 }
310 inline Reply *take()
311 {
312 getReply();
313 Reply *ret = m_reply;
314 m_reply = nullptr;
315 m_window = XCB_WINDOW_NONE;
316 return ret;
317 }
318
319protected:
320 AbstractWrapper()
321 : m_retrieved(false)
322 , m_window(XCB_WINDOW_NONE)
323 , m_reply(nullptr)
324 {
325 m_cookie.sequence = 0;
326 }
327 explicit AbstractWrapper(WindowId window, Cookie cookie)
328 : m_retrieved(false)
329 , m_cookie(cookie)
330 , m_window(window)
331 , m_reply(nullptr)
332 {
333 }
334 explicit AbstractWrapper(const AbstractWrapper &other)
335 : m_retrieved(other.m_retrieved)
336 , m_cookie(other.m_cookie)
337 , m_window(other.m_window)
338 , m_reply(nullptr)
339 {
340 takeFromOther(const_cast<AbstractWrapper &>(other));
341 }
342 void getReply()
343 {
344 if (m_retrieved || !m_cookie.sequence) {
345 return;
346 }
347 m_reply = Data::replyFunc(connection(), m_cookie, nullptr);
348 m_retrieved = true;
349 }
350
351private:
352 inline void cleanup()
353 {
354 if (!m_retrieved && m_cookie.sequence) {
355 xcb_discard_reply(connection(), m_cookie.sequence);
356 } else if (m_reply) {
357 free(m_reply);
358 }
359 }
360 inline void takeFromOther(AbstractWrapper &other)
361 {
362 if (m_retrieved) {
363 m_reply = other.take();
364 } else {
365 // ensure that other object doesn't try to get the reply or discards it in the dtor
366 other.m_retrieved = true;
367 other.m_window = XCB_WINDOW_NONE;
368 }
369 }
370 bool m_retrieved;
371 Cookie m_cookie;
372 WindowId m_window;
373 Reply *m_reply;
