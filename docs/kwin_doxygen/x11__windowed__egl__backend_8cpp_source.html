88std::optional<OutputLayerBeginFrameInfo> X11WindowedEglCursorLayer::beginFrame()
89{
90 eglMakeCurrent(m_backend->eglDisplay(), EGL_NO_SURFACE, EGL_NO_SURFACE, m_backend->context());
91
92 const auto tmp = size().expandedTo(QSize(64, 64));
93 const QSize bufferSize(std::ceil(tmp.width()), std::ceil(tmp.height()));
94 if (!m_texture || m_texture->size() != bufferSize) {
95 m_texture = std::make_unique<GLTexture>(GL_RGBA8, bufferSize);
96 m_framebuffer = std::make_unique<GLFramebuffer>(m_texture.get());

99 return OutputLayerBeginFrameInfo{
100 .renderTarget = RenderTarget(m_framebuffer.get()),
101 .repaint = infiniteRegion(),
102 };

105bool X11WindowedEglCursorLayer::endFrame(const QRegion &renderedRegion, const QRegion &damagedRegion)
106{
107 QImage buffer(m_framebuffer->size(), QImage::Format_RGBA8888_Premultiplied);
108
109 GLFramebuffer::pushFramebuffer(m_framebuffer.get());
110 glReadPixels(0, 0, buffer.width(), buffer.height(), GL_RGBA, GL_UNSIGNED_BYTE, buffer.bits());
111 GLFramebuffer::popFramebuffer();
112
113 m_output->cursor()->update(buffer.mirrored(false, true), hotspot());
114
115 return true;
116}

143bool X11WindowedEglBackend::createSurfaces()
144{
145 const auto &outputs = m_backend->outputs();
146 for (const auto &output : outputs) {
147 X11WindowedOutput *x11Output = static_cast<X11WindowedOutput *>(output);
148 EGLSurface s = createSurface(x11Output->window());
149 if (s == EGL_NO_SURFACE) {
150 return false;
151 }
152 m_outputs[output] = Layers{
153 .primaryLayer = std::make_unique<X11WindowedEglPrimaryLayer>(this, x11Output, s),
154 .cursorLayer = std::make_unique<X11WindowedEglCursorLayer>(this, x11Output),
155 };
156 }
157 if (m_outputs.empty()) {
158 return false;
159 }
160 return true;
161}

170void X11WindowedEglBackend::presentSurface(EGLSurface surface, const QRegion &damage, const QRect &screenGeometry)
171{
172 const bool fullRepaint = supportsBufferAge() || (damage == screenGeometry);
173
174 if (fullRepaint || !havePostSubBuffer()) {
175 // the entire screen changed, or we cannot do partial updates (which implies we enabled surface preservation)
176 eglSwapBuffers(eglDisplay(), surface);
177 } else {
178 // a part of the screen changed, and we can use eglPostSubBufferNV to copy the updated area
179 for (const QRect &r : damage) {
180 eglPostSubBufferNV(eglDisplay(), surface, r.left(), screenGeometry.height() - r.bottom() - 1, r.width(), r.height());
181 }
182 }
183}

205std::shared_ptr<GLTexture> X11WindowedEglBackend::textureForOutput(Output *output) const
206{
207 auto it = m_outputs.find(output);
208 if (it == m_outputs.end()) {
209 return nullptr;
210 }
211
212 GLFramebuffer::pushFramebuffer(it->second.primaryLayer->fbo());
213 auto ret = AbstractEglBackend::textureForOutput(output);
214 GLFramebuffer::popFramebuffer();
215 return ret;
