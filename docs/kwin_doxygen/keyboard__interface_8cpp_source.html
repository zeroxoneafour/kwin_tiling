29void KeyboardInterfacePrivate::keyboard_bind_resource(Resource *resource)
30{
31 const ClientConnection *focusedClient = focusedSurface ? focusedSurface->client() : nullptr;
32
33 if (resource->version() >= WL_KEYBOARD_REPEAT_INFO_SINCE_VERSION) {
34 send_repeat_info(resource->handle, keyRepeat.charactersPerSecond, keyRepeat.delay);
35 }
36 if (!keymap.isNull()) {
37 sendKeymap(resource);
38 }
39
40 if (focusedClient && focusedClient->client() == resource->client()) {
41 const QVector<quint32> keys = pressedKeys();
42 const QByteArray keysData = QByteArray::fromRawData(reinterpret_cast<const char *>(keys.data()), sizeof(quint32) * keys.count());
43 const quint32 serial = seat->display()->nextSerial();
44
45 send_enter(resource->handle, serial, focusedSurface->resource(), keysData);
46 send_modifiers(resource->handle, serial, modifiers.depressed, modifiers.latched, modifiers.locked, modifiers.group);
47 }
48}

74void KeyboardInterfacePrivate::sendKeymap(Resource *resource)
75{
76 // From version 7 on, keymaps must be mapped privately, so that
77 // we can seal the fd and reuse it between clients.
78 if (resource->version() >= 7 && sharedKeymapFile.effectiveFlags().testFlag(KWin::RamFile::Flag::SealWrite)) {
79 send_keymap(resource->handle, keymap_format::keymap_format_xkb_v1, sharedKeymapFile.fd(), sharedKeymapFile.size());
80 // otherwise give each client its own unsealed copy.
81 } else {
82 KWin::RamFile keymapFile("kwin-xkb-keymap", keymap.constData(), keymap.size() + 1); // Include QByteArray null-terminator.
83 send_keymap(resource->handle, keymap_format::keymap_format_xkb_v1, keymapFile.fd(), keymapFile.size());
84 }
85}

111bool KeyboardInterfacePrivate::updateKey(quint32 key, KeyboardKeyState state)
112{
113 auto it = states.find(key);
114 if (it == states.end()) {
115 states.insert(key, state);
116 return true;
117 }
118 if (it.value() == state) {
119 return false;
120 }
121 it.value() = state;
122 return true;
123}

137void KeyboardInterface::setFocusedSurface(SurfaceInterface *surface, quint32 serial)
138{
139 if (d->focusedSurface == surface) {
140 return;
141 }
142
143 if (d->focusedSurface) {
144 d->sendLeave(d->focusedSurface, serial);
145 disconnect(d->destroyConnection);
146 }
147
148 d->focusedSurface = surface;
149 if (!d->focusedSurface) {
150 return;
151 }
152 d->destroyConnection = connect(d->focusedSurface, &SurfaceInterface::aboutToBeDestroyed, this, [this] {
153 d->sendLeave(d->focusedSurface, d->seat->display()->nextSerial());
154 d->focusedSurface = nullptr;
155 });
156
157 d->sendEnter(d->focusedSurface, serial);
158 d->sendModifiers();
159}

194void KeyboardInterface::sendModifiers(quint32 depressed, quint32 latched, quint32 locked, quint32 group)
195{
196 bool changed = false;
197 if (d->modifiers.depressed != depressed) {
198 d->modifiers.depressed = depressed;
199 changed = true;
200 }
201 if (d->modifiers.latched != latched) {
202 d->modifiers.latched = latched;
203 changed = true;
204 }
205 if (d->modifiers.locked != locked) {
206 d->modifiers.locked = locked;
207 changed = true;
208 }
209 if (d->modifiers.group != group) {
210 d->modifiers.group = group;
211 changed = true;
212 }
213 if (!changed || !d->focusedSurface) {
214 return;
215 }
216
217 d->modifiers.serial = d->seat->display()->nextSerial();
218 d->sendModifiers(depressed, latched, locked, group, d->modifiers.serial);
219}
