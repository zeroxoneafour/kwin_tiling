77void DataDeviceInterfacePrivate::data_device_start_drag(Resource *resource,
78 wl_resource *sourceResource,
79 wl_resource *originResource,
80 wl_resource *iconResource,
81 uint32_t serial)
82{
83 SurfaceInterface *iconSurface = SurfaceInterface::get(iconResource);
84
85 const SurfaceRole *surfaceRole = SurfaceRole::get(iconSurface);
86 if (surfaceRole) {
87 wl_resource_post_error(resource->handle, error_role, "the icon surface already has a role assigned %s", surfaceRole->name().constData());
88 return;
89 }
90
91 SurfaceInterface *focusSurface = SurfaceInterface::get(originResource);
92 DataSourceInterface *dataSource = nullptr;
93 if (sourceResource) {
94 dataSource = DataSourceInterface::get(sourceResource);
95 }
96
97 const bool pointerGrab = seat->hasImplicitPointerGrab(serial) && seat->focusedPointerSurface() == focusSurface;
98 if (!pointerGrab) {
99 // Client doesn't have pointer grab.
100 const bool touchGrab = seat->hasImplicitTouchGrab(serial) && seat->focusedTouchSurface() == focusSurface;
101 if (!touchGrab) {
102 // Client neither has pointer nor touch grab. No drag start allowed.
103 return;
104 }
105 }
106
107 DragAndDropIcon *dragIcon = nullptr;
108 if (iconSurface) {
109 // drag icon lifespan is mapped to surface lifespan
110 dragIcon = new DragAndDropIcon(iconSurface);
111 }
112 drag.serial = serial;
113 Q_EMIT q->dragStarted(dataSource, focusSurface, serial, dragIcon);
114}

116void DataDeviceInterfacePrivate::data_device_set_selection(Resource *resource, wl_resource *source, uint32_t serial)
117{
118 DataSourceInterface *dataSource = DataSourceInterface::get(source);
119
120 if (dataSource && dataSource->supportedDragAndDropActions() && wl_resource_get_version(dataSource->resource()) >= WL_DATA_SOURCE_ACTION_SINCE_VERSION) {
121 wl_resource_post_error(dataSource->resource(), QtWaylandServer::wl_data_source::error_invalid_source, "Data source is for drag and drop");
122 return;
123 }
124
125 if (selection == dataSource) {
126 return;
127 }
128 if (selection) {
129 selection->cancel();
130 }
131 selection = dataSource;
132 Q_EMIT q->selectionChanged(selection);
133}

140DataOfferInterface *DataDeviceInterfacePrivate::createDataOffer(AbstractDataSource *source)
141{
142 if (!source) {
143 // a data offer can only exist together with a source
144 return nullptr;
145 }
146
147 wl_resource *data_offer_resource = wl_resource_create(resource()->client(), &wl_data_offer_interface, resource()->version(), 0);
148 if (!data_offer_resource) {
149 wl_resource_post_no_memory(resource()->handle);
150 return nullptr;
151 }
152
153 DataOfferInterface *offer = new DataOfferInterface(source, data_offer_resource);
154 send_data_offer(offer->resource());
155 offer->sendAllOffers();
156 offer->sendSourceActions();
157 return offer;
158}

192void DataDeviceInterface::drop()
193{
194 d->send_drop();
195 d->drag.surface = nullptr; // prevent sending wl_data_device.leave event
196
197 disconnect(d->drag.posConnection);
198 d->drag.posConnection = QMetaObject::Connection();
199 disconnect(d->drag.destroyConnection);
200 d->drag.destroyConnection = QMetaObject::Connection();
201
202 if (d->seat->dragSource()->selectedDndAction() != DataDeviceManagerInterface::DnDAction::Ask) {
203 disconnect(d->drag.sourceActionConnection);
204 d->drag.sourceActionConnection = QMetaObject::Connection();
205 disconnect(d->drag.targetActionConnection);
206 d->drag.targetActionConnection = QMetaObject::Connection();
207 }
208}

210static DataDeviceManagerInterface::DnDAction chooseDndAction(AbstractDataSource *source, DataOfferInterface *offer)
211{
212 if (offer->preferredDragAndDropAction().has_value()) {
213 if (source->supportedDragAndDropActions().testFlag(*offer->preferredDragAndDropAction())) {
214 return *offer->preferredDragAndDropAction();
215 }
216 }
217
218 if (offer->supportedDragAndDropActions().has_value()) {
219 for (const auto &action : {DataDeviceManagerInterface::DnDAction::Copy, DataDeviceManagerInterface::DnDAction::Move, DataDeviceManagerInterface::DnDAction::Ask}) {
220 if (source->supportedDragAndDropActions().testFlag(action) && offer->supportedDragAndDropActions()->testFlag(action)) {
221 return action;
222 }
223 }
224 }
225
226 return DataDeviceManagerInterface::DnDAction::None;
227}

229void DataDeviceInterface::updateDragTarget(SurfaceInterface *surface, quint32 serial)
230{
231 if (d->drag.surface == surface) {
232 return;
233 }
234
235 if (d->drag.surface) {
236 if (d->drag.surface->resource()) {
237 d->send_leave();
238 }
239 if (d->drag.posConnection) {
240 disconnect(d->drag.posConnection);
241 d->drag.posConnection = QMetaObject::Connection();
242 }
243 disconnect(d->drag.destroyConnection);
244 d->drag.destroyConnection = QMetaObject::Connection();
245 d->drag.surface = nullptr;
246 if (d->drag.sourceActionConnection) {
247 disconnect(d->drag.sourceActionConnection);
248 d->drag.sourceActionConnection = QMetaObject::Connection();
249 }
250 if (d->drag.targetActionConnection) {
251 disconnect(d->drag.targetActionConnection);
252 d->drag.targetActionConnection = QMetaObject::Connection();
253 }
254 // don't update serial, we need it
255 }
256 auto dragSource = d->seat->dragSource();
257 if (!surface || !dragSource) {
258 if (auto s = dragSource) {
259 s->dndAction(DataDeviceManagerInterface::DnDAction::None);
260 }
261 return;
262 }
263
264 if (dragSource) {
265 dragSource->accept(QString());
266 }
267 DataOfferInterface *offer = d->createDataOffer(dragSource);
268 d->drag.surface = surface;
269 if (d->seat->isDragPointer()) {
270 d->drag.posConnection = connect(d->seat, &SeatInterface::pointerPosChanged, this, [this] {
271 const QPointF pos = d->seat->dragSurfaceTransformation().map(d->seat->pointerPos());
272 d->send_motion(d->seat->timestamp().count(), wl_fixed_from_double(pos.x()), wl_fixed_from_double(pos.y()));
273 });
274 } else if (d->seat->isDragTouch()) {
275 // When dragging from one window to another, we may end up in a data_device
276 // that didn't get "data_device_start_drag". In that case, the internal
277 // touch point serial will be incorrect and we need to update it to the
278 // serial from the seat.
279 SeatInterfacePrivate *seatPrivate = SeatInterfacePrivate::get(seat());
280 if (seatPrivate->drag.dragImplicitGrabSerial != d->drag.serial) {
281 d->drag.serial = seatPrivate->drag.dragImplicitGrabSerial.value();
282 }
283
284 d->drag.posConnection = connect(d->seat, &SeatInterface::touchMoved, this, [this](qint32 id, quint32 serial, const QPointF &globalPosition) {
285 if (serial != d->drag.serial) {
286 // different touch down has been moved
287 return;
288 }
289 const QPointF pos = d->seat->dragSurfaceTransformation().map(globalPosition);
290 d->send_motion(d->seat->timestamp().count(), wl_fixed_from_double(pos.x()), wl_fixed_from_double(pos.y()));
291 });
292 }
293 d->drag.destroyConnection = connect(d->drag.surface, &QObject::destroyed, this, [this] {
294 d->send_leave();
295 if (d->drag.posConnection) {
296 disconnect(d->drag.posConnection);
297 }
298 d->drag = DataDeviceInterfacePrivate::Drag();
299 });
300
301 QPointF pos;
302 if (d->seat->isDragPointer()) {
303 pos = d->seat->dragSurfaceTransformation().map(d->seat->pointerPos());
304 } else if (d->seat->isDragTouch()) {
305 pos = d->seat->dragSurfaceTransformation().map(d->seat->firstTouchPointPosition());
306 }
307 d->send_enter(serial, surface->resource(), wl_fixed_from_double(pos.x()), wl_fixed_from_double(pos.y()), offer ? offer->resource() : nullptr);
308 if (offer) {
309 auto matchOffers = [dragSource, offer] {
310 const DataDeviceManagerInterface::DnDAction action = chooseDndAction(dragSource, offer);
311 offer->dndAction(action);
312 dragSource->dndAction(action);
313 };
314 d->drag.targetActionConnection = connect(offer, &DataOfferInterface::dragAndDropActionsChanged, dragSource, matchOffers);
315 d->drag.sourceActionConnection = connect(dragSource, &AbstractDataSource::supportedDragAndDropActionsChanged, dragSource, matchOffers);
316 }
317}
