46bool BasicEGLSurfaceTextureInternal::updateFromFramebuffer()
47{
48 const QOpenGLFramebufferObject *fbo = m_pixmap->fbo();
49 if (!fbo) {
50 return false;
51 }
52 m_texture.reset(new GLTexture(fbo->texture(), 0, fbo->size()));
53 m_texture->setWrapMode(GL_CLAMP_TO_EDGE);
54 m_texture->setFilter(GL_LINEAR);
55 m_texture->setYInverted(false);
56 return true;
57}

59static QRegion scale(const QRegion &region, qreal scaleFactor)
60{
61 if (scaleFactor == 1) {
62 return region;
63 }
64
65 QRegion scaled;
66 for (const QRect &rect : region) {
67 scaled += QRect(rect.topLeft() * scaleFactor, rect.size() * scaleFactor);
68 }
69 return scaled;
70}

72bool BasicEGLSurfaceTextureInternal::updateFromImage(const QRegion &region)
73{
74 const QImage image = m_pixmap->image();
75 if (image.isNull()) {
76 return false;
77 }
78
79 if (!m_texture) {
80 m_texture.reset(new GLTexture(image));
81 } else {
82 const QRegion nativeRegion = scale(region, image.devicePixelRatio());
83 for (const QRect &rect : nativeRegion) {
84 m_texture->update(image, rect.topLeft(), rect);
85 }
86 }
87
88 return true;
89}
