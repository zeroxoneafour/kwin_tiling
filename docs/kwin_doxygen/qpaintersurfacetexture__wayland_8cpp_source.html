25bool QPainterSurfaceTextureWayland::create()
26{
27 auto buffer = qobject_cast<KWaylandServer::ShmClientBuffer *>(m_pixmap->buffer());
28 if (Q_LIKELY(buffer)) {
29 // The buffer data is copied as the buffer interface returns a QImage
30 // which doesn't own the data of the underlying wl_shm_buffer object.
31 m_image = buffer->data().copy();
32 }
33 return !m_image.isNull();
34}

36void QPainterSurfaceTextureWayland::update(const QRegion &region)
37{
38 auto buffer = qobject_cast<KWaylandServer::ShmClientBuffer *>(m_pixmap->buffer());
39 if (Q_UNLIKELY(!buffer)) {
40 return;
41 }
42
43 const QImage image = buffer->data();
44 const QRegion dirtyRegion = mapRegion(m_pixmap->item()->surfaceToBufferMatrix(), region);
45 QPainter painter(&m_image);
46 painter.setCompositionMode(QPainter::CompositionMode_Source);
47
48 // The buffer data is copied as the buffer interface returns a QImage
49 // which doesn't own the data of the underlying wl_shm_buffer object.
50 for (const QRect &rect : dirtyRegion) {
51 painter.drawImage(rect, image, rect);
52 }
53}
