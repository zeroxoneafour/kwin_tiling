121void UserActionsMenu::show(const QRect &pos, Window *window)
122{
123 Q_ASSERT(window);
124 QPointer<Window> windowPtr(window);
125 // Presumably window will never be nullptr,
126 // but play it safe and make sure not to crash.
127 if (windowPtr.isNull()) {
128 return;
129 }
130 if (isShown()) { // recursion
131 return;
132 }
133 if (windowPtr->isDesktop() || windowPtr->isDock()) {
134 return;
135 }
136 if (!KAuthorized::authorizeAction(QStringLiteral("kwin_rmb"))) {
137 return;
138 }
139 m_window = windowPtr;
140 init();
141 m_menu->popup(pos.bottomLeft());
142}

150void UserActionsMenu::helperDialog(const QString &message, Window *window)
151{
152 QStringList args;
153 QString type;
154 auto shortcut = [](const QString &name) {
155 QAction *action = Workspace::self()->findChild<QAction *>(name);
156 Q_ASSERT(action != nullptr);
157 const auto shortcuts = KGlobalAccel::self()->shortcut(action);
158 return QStringLiteral("%1 (%2)").arg(action->text(), shortcuts.isEmpty() ? QString() : shortcuts.first().toString(QKeySequence::NativeText));
159 };
160 if (message == QStringLiteral("noborderaltf3")) {
161 args << QStringLiteral("--msgbox") << i18n("You have selected to show a window without its border.\n"
162 "Without the border, you will not be able to enable the border "
163 "again using the mouse: use the window operations menu instead, "
164 "activated using the %1 keyboard shortcut.",
165 shortcut(QStringLiteral("Window Operations Menu")));
166 type = QStringLiteral("altf3warning");
167 } else if (message == QLatin1String("fullscreenaltf3")) {
168 args << QStringLiteral("--msgbox") << i18n("You have selected to show a window in fullscreen mode.\n"
169 "If the application itself does not have an option to turn the fullscreen "
170 "mode off you will not be able to disable it "
171 "again using the mouse: use the window operations menu instead, "
172 "activated using the %1 keyboard shortcut.",
173 shortcut(QStringLiteral("Window Operations Menu")));
174 type = QStringLiteral("altf3warning");
175 } else {
176 Q_UNREACHABLE();
177 }
178 if (!type.isEmpty()) {
179 KConfig cfg(QStringLiteral("kwin_dialogsrc"));
180 KConfigGroup cg(&cfg, "Notification Messages"); // Depends on KMessageBox
181 if (!cg.readEntry(type, true)) {
182 return;
183 }
184 args << QStringLiteral("--dontagain") << QLatin1String("kwin_dialogsrc:") + type;
185 }
186 if (window) {
187 args << QStringLiteral("--embed") << QString::number(window->window());
188 }
189 QtConcurrent::run([args]() {
190 KProcess::startDetached(QStringLiteral("kdialog"), args);
191 });
192}

357void UserActionsMenu::menuAboutToShow()
358{
359 if (m_window.isNull() || !m_menu) {
360 return;
361 }
362
363 m_window->blockActivityUpdates(true);
364
365 if (VirtualDesktopManager::self()->count() == 1) {
366 delete m_desktopMenu;
367 m_desktopMenu = nullptr;
368 delete m_multipleDesktopsMenu;
369 m_multipleDesktopsMenu = nullptr;
370 } else {
371 initDesktopPopup();
372 }
373 if (workspace()->outputs().count() == 1 || (!m_window->isMovable() && !m_window->isMovableAcrossScreens())) {
374 delete m_screenMenu;
375 m_screenMenu = nullptr;
376 } else {
377 initScreenPopup();
378 }
379
380 m_menu->setPalette(m_window->palette());
381 m_resizeOperation->setEnabled(m_window->isResizable());
382 m_moveOperation->setEnabled(m_window->isMovableAcrossScreens());
383 m_maximizeOperation->setEnabled(m_window->isMaximizable());
384 m_maximizeOperation->setChecked(m_window->maximizeMode() == MaximizeFull);
385 m_shadeOperation->setEnabled(m_window->isShadeable());
386 m_shadeOperation->setChecked(m_window->shadeMode() != ShadeNone);
387 m_keepAboveOperation->setChecked(m_window->keepAbove());
388 m_keepBelowOperation->setChecked(m_window->keepBelow());
389 m_fullScreenOperation->setEnabled(m_window->userCanSetFullScreen());
390 m_fullScreenOperation->setChecked(m_window->isFullScreen());
391 m_noBorderOperation->setEnabled(m_window->userCanSetNoBorder());
392 m_noBorderOperation->setChecked(m_window->noBorder());
393 m_minimizeOperation->setEnabled(m_window->isMinimizable());
394 m_closeOperation->setEnabled(m_window->isCloseable());
395 m_shortcutOperation->setEnabled(m_window->rules()->checkShortcut(QString()).isNull());
396
397 // drop the existing scripts menu
398 delete m_scriptsMenu;
399 m_scriptsMenu = nullptr;
400 // ask scripts whether they want to add entries for the given window
401 QList<QAction *> scriptActions = Scripting::self()->actionsForUserActionMenu(m_window.data(), m_scriptsMenu);
402 if (!scriptActions.isEmpty()) {
403 m_scriptsMenu = new QMenu(m_menu);
404 m_scriptsMenu->setPalette(m_window->palette());
405 m_scriptsMenu->addActions(scriptActions);
406
407 QAction *action = m_scriptsMenu->menuAction();
408 // set it as the first item after desktop
409 m_menu->insertAction(m_closeOperation, action);
410 action->setText(i18n("&Extensions"));
411 }
412
413 if (m_rulesOperation) {
414 m_rulesOperation->setEnabled(m_window->supportsWindowRules());
415 }
416 if (m_applicationRulesOperation) {
417 m_applicationRulesOperation->setEnabled(m_window->supportsWindowRules());
418 }
419
420 showHideActivityMenu();
421}

1210void Workspace::performWindowOperation(Window *window, Options::WindowOperation op)
1211{
1212 if (!window) {
1213 return;
1214 }
1215 if (op == Options::MoveOp || op == Options::UnrestrictedMoveOp) {
1216 Cursors::self()->mouse()->setPos(window->frameGeometry().center());
1217 }
1218 if (op == Options::ResizeOp || op == Options::UnrestrictedResizeOp) {
1219 Cursors::self()->mouse()->setPos(window->frameGeometry().bottomRight());
1220 }

1710bool Workspace::switchWindow(Window *window, Direction direction, QPoint curPos, VirtualDesktop *desktop)
1711{
1712 Window *switchTo = nullptr;
1713 int bestScore = 0;
1714
1715 QList<Window *> clist = stackingOrder();
1716 for (auto i = clist.rbegin(); i != clist.rend(); ++i) {
1717 auto other = *i;
1718 if (!other->isClient()) {
1719 continue;
1720 }
1721 if (other->wantsTabFocus() && *i != window && other->isOnDesktop(desktop) && !other->isMinimized() && (*i)->isOnCurrentActivity()) {
1722 // Centre of the other window
1723 const QPoint otherCenter(other->x() + other->width() / 2, other->y() + other->height() / 2);
1724
1725 int distance;
1726 int offset;
1727 switch (direction) {
1728 case DirectionNorth:
1729 distance = curPos.y() - otherCenter.y();
1730 offset = std::abs(otherCenter.x() - curPos.x());
1731 break;
1732 case DirectionEast:
1733 distance = otherCenter.x() - curPos.x();
1734 offset = std::abs(otherCenter.y() - curPos.y());
1735 break;
1736 case DirectionSouth:
1737 distance = otherCenter.y() - curPos.y();
1738 offset = std::abs(otherCenter.x() - curPos.x());
1739 break;
1740 case DirectionWest:
1741 distance = curPos.x() - otherCenter.x();
1742 offset = std::abs(otherCenter.y() - curPos.y());
1743 break;
1744 default:
1745 distance = -1;
1746 offset = -1;
1747 }
1748
1749 if (distance > 0) {
1750 // Inverse score
1751 int score = distance + offset + ((offset * offset) / distance);
1752 if (score < bestScore || !switchTo) {
1753 switchTo = other;
1754 bestScore = score;
1755 }
1756 }
1757 }
1758 }
1759 if (switchTo) {
1760 activateWindow(switchTo);
1761 }
1762
1763 return switchTo;
1764}

1824void Window::setShortcut(const QString &_cut)
1825{
1826 QString cut = rules()->checkShortcut(_cut);
1827 auto updateShortcut = [this](const QKeySequence &cut = QKeySequence()) {
1828 if (_shortcut == cut) {
1829 return;
1830 }
1831 _shortcut = cut;
1832 setShortcutInternal();
1833 };
1834 if (cut.isEmpty()) {
1835 updateShortcut();
1836 return;
1837 }
1838 if (cut == shortcut().toString()) {
1839 return; // no change
1840 }
1841 // Format:
1842 // base+(abcdef)<space>base+(abcdef)
1843 // E.g. Alt+Ctrl+(ABCDEF);Meta+X,Meta+(ABCDEF)
1844 if (!cut.contains(QLatin1Char('(')) && !cut.contains(QLatin1Char(')')) && !cut.contains(QLatin1String(" - "))) {
1845 if (workspace()->shortcutAvailable(cut, this)) {
1846 updateShortcut(QKeySequence(cut));
1847 } else {
1848 updateShortcut();
1849 }
1850 return;
1851 }
1852 const QRegularExpression reg(QStringLiteral("(.*\\+)\\((.*)\\)"));
1853 QList<QKeySequence> keys;
1854 QStringList groups = cut.split(QStringLiteral(" - "));
1855 for (QStringList::ConstIterator it = groups.constBegin();
1856 it != groups.constEnd();
1857 ++it) {
1858 const QRegularExpressionMatch match = reg.match(*it);
1859 if (match.hasMatch()) {
1860 const QString base = match.captured(1);
1861 const QString list = match.captured(2);
1862 for (int i = 0;
1863 i < list.length();
1864 ++i) {
1865 QKeySequence c(base + list[i]);
1866 if (!c.isEmpty()) {
1867 keys.append(c);
1868 }
1869 }
1870 } else {
1871 // regexp doesn't match, so it should be a normal shortcut
1872 QKeySequence c(*it);
1873 if (!c.isEmpty()) {
1874 keys.append(c);
1875 }
1876 }
1877 }
1878 for (auto it = keys.constBegin();
1879 it != keys.constEnd();
1880 ++it) {
1881 if (_shortcut == *it) { // current one is in the list
1882 return;
1883 }
1884 }
1885 for (auto it = keys.constBegin();
1886 it != keys.constEnd();
1887 ++it) {
1888 if (workspace()->shortcutAvailable(*it, this)) {
1889 updateShortcut(*it);
1890 return;
1891 }
1892 }
1893 updateShortcut();
1894}

1916{
