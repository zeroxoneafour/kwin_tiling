80static DrmPlane::Transformations invertAndConvertTransform(Output::Transform transform)
81{
82 DrmPlane::Transformations ret;
83 switch (transform) {
84 case Output::Transform::Normal:
85 case Output::Transform::Flipped:
86 ret = DrmPlane::Transformation::Rotate0;
87 break;
88 case Output::Transform::Rotated90:
89 case Output::Transform::Flipped90:
90 ret = DrmPlane::Transformation::Rotate270;
91 break;
92 case Output::Transform::Rotated180:
93 case Output::Transform::Flipped180:
94 ret = DrmPlane::Transformation::Rotate180;
95 break;
96 case Output::Transform::Rotated270:
97 case Output::Transform::Flipped270:
98 ret = DrmPlane::Transformation::Rotate90;
99 break;
100 }
101 if (transform == Output::Transform::Flipped || transform == Output::Transform::Flipped180) {
102 ret |= DrmPlane::Transformation::ReflectX;
103 } else if (transform == Output::Transform::Flipped90 || transform == Output::Transform::Flipped270) {
104 ret |= DrmPlane::Transformation::ReflectY;
105 }
106 return ret;
107}

109bool EglGbmLayer::scanout(SurfaceItem *surfaceItem)
110{
111 static bool valid;
112 static const bool directScanoutDisabled = qEnvironmentVariableIntValue("KWIN_DRM_NO_DIRECT_SCANOUT", &valid) == 1 && valid;
113 if (directScanoutDisabled) {
114 return false;
115 }
116
117 SurfaceItemWayland *item = qobject_cast<SurfaceItemWayland *>(surfaceItem);
118 if (!item || !item->surface()) {
119 return false;
120 }
121 const auto surface = item->surface();
122 if (invertAndConvertTransform(surface->bufferTransform()) != m_pipeline->bufferOrientation()) {
123 return false;
124 }
125 const auto buffer = qobject_cast<KWaylandServer::LinuxDmaBufV1ClientBuffer *>(surface->buffer());
126 if (!buffer) {
127 return false;
128 }
129
130 const auto formats = m_pipeline->formats();
131 if (!formats.contains(buffer->format())) {
132 m_dmabufFeedback.scanoutFailed(surface, formats);
133 return false;
134 }
135 if (buffer->attributes().modifier == DRM_FORMAT_MOD_INVALID && m_pipeline->gpu()->platform()->gpuCount() > 1) {
136 // importing a buffer from another GPU without an explicit modifier can mess up the buffer format
137 return false;
138 }
139 if (!formats[buffer->format()].contains(buffer->attributes().modifier)) {
140 return false;
141 }
142 const auto gbmBuffer = GbmBuffer::importBuffer(m_pipeline->gpu(), buffer);
143 if (!gbmBuffer) {
144 m_dmabufFeedback.scanoutFailed(surface, formats);
145 return false;
146 }
147 m_scanoutBuffer = DrmFramebuffer::createFramebuffer(gbmBuffer);
148 if (m_scanoutBuffer && m_pipeline->testScanout()) {
149 m_dmabufFeedback.scanoutSuccessful(surface);
150 m_currentDamage = surfaceItem->damage();
151 surfaceItem->resetDamage();
152 // ensure the pixmap is updated when direct scanout ends
153 surfaceItem->destroyPixmap();
154 return true;
155 } else {
156 m_dmabufFeedback.scanoutFailed(surface, formats);
157 m_scanoutBuffer.reset();
158 return false;
159 }
160}
