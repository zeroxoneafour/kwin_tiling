95void TestWaylandServerSeat::testPointerButton()
96{
97 KWaylandServer::Display display;
98 display.addSocketName(s_socketName);
99 display.start();
100 SeatInterface *seat = new SeatInterface(&display, &display);
101 seat->setHasPointer(true);
102
103 // no button pressed yet, should be released and no serial
104 QVERIFY(!seat->isPointerButtonPressed(0));
105 QVERIFY(!seat->isPointerButtonPressed(1));
106 QCOMPARE(seat->pointerButtonSerial(0), quint32(0));
107 QCOMPARE(seat->pointerButtonSerial(1), quint32(0));
108
109 // mark the button as pressed
110 seat->notifyPointerButton(0, PointerButtonState::Pressed);
111 seat->notifyPointerFrame();
112 QVERIFY(seat->isPointerButtonPressed(0));
113 QCOMPARE(seat->pointerButtonSerial(0), display.serial());
114
115 // other button should still be unpressed
116 QVERIFY(!seat->isPointerButtonPressed(1));
117 QCOMPARE(seat->pointerButtonSerial(1), quint32(0));
118
119 // release it again
120 seat->notifyPointerButton(0, PointerButtonState::Released);
121 seat->notifyPointerFrame();
122 QVERIFY(!seat->isPointerButtonPressed(0));
123 QCOMPARE(seat->pointerButtonSerial(0), display.serial());
124}
