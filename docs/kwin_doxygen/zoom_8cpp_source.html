179void ZoomEffect::hideCursor()
180{
181 if (mouseTracking == MouseTrackingProportional && mousePointer == MousePointerKeep) {
182 return; // don't replace the actual cursor by a static image for no reason.
183 }
184 if (!isMouseHidden) {
185 // try to load the cursor-theme into a OpenGL texture and if successful then hide the mouse-pointer
186 GLTexture *texture = nullptr;
187 if (effects->isOpenGLCompositing()) {
188 texture = ensureCursorTexture();
189 }
190 if (texture) {
191 effects->hideCursor();
192 connect(effects, &EffectsHandler::cursorShapeChanged, this, &ZoomEffect::markCursorTextureDirty);
193 isMouseHidden = true;
194 }
195 }

198void ZoomEffect::reconfigure(ReconfigureFlags)
199{
200 ZoomConfig::self()->read();
201 // On zoom-in and zoom-out change the zoom by the defined zoom-factor.
202 zoomFactor = std::max(0.1, ZoomConfig::zoomFactor());
203 // Visibility of the mouse-pointer.
204 mousePointer = MousePointerType(ZoomConfig::mousePointer());
205 // Track moving of the mouse.
206 mouseTracking = MouseTrackingType(ZoomConfig::mouseTracking());
207#if HAVE_ACCESSIBILITY
208 // Enable tracking of the focused location.
209 m_accessibilityIntegration->setFocusTrackingEnabled(ZoomConfig::enableFocusTracking());
210 // Enable tracking of the text caret.
211 m_accessibilityIntegration->setTextCaretTrackingEnabled(ZoomConfig::enableTextCaretTracking());
212#endif
213 // The time in milliseconds to wait before a focus-event takes away a mouse-move.
214 focusDelay = std::max(uint(0), ZoomConfig::focusDelay());
215 // The factor the zoom-area will be moved on touching an edge on push-mode or using the navigation KAction's.
216 moveFactor = std::max(0.1, ZoomConfig::moveFactor());
217 if (source_zoom < 0) {
218 // Load the saved zoom value.
219 source_zoom = 1.0;
220 target_zoom = ZoomConfig::initialZoom();
221 if (target_zoom > 1.0) {
222 zoomIn(target_zoom);
223 }
224 } else {
225 source_zoom = 1.0;
226 }
227}

229void ZoomEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
230{
231 if (zoom != target_zoom) {
232 int time = 0;
233 if (lastPresentTime.count()) {
234 time = (presentTime - lastPresentTime).count();
235 }
236 lastPresentTime = presentTime;
237
238 const float zoomDist = std::abs(target_zoom - source_zoom);
239 if (target_zoom > zoom) {
240 zoom = std::min(zoom + ((zoomDist * time) / animationTime(150 * zoomFactor)), target_zoom);
241 } else {
242 zoom = std::max(zoom - ((zoomDist * time) / animationTime(150 * zoomFactor)), target_zoom);
243 }
244 }
245
246 if (zoom == 1.0) {
247 showCursor();
248 } else {
249 hideCursor();
250 data.mask |= PAINT_SCREEN_TRANSFORMED;
251 }
252
253 effects->prePaintScreen(data, presentTime);
254}

256ZoomEffect::OffscreenData *ZoomEffect::ensureOffscreenData(EffectScreen *screen)
257{
258 const QRect rect = effects->renderTargetRect();
259 const qreal devicePixelRatio = effects->renderTargetScale();
260 const QSize nativeSize = rect.size() * devicePixelRatio;
261
262 OffscreenData &data = m_offscreenData[effects->waylandDisplay() ? screen : nullptr];
263 if (!data.texture || data.texture->size() != nativeSize) {
264 data.texture.reset(new GLTexture(GL_RGBA8, nativeSize));
265 data.texture->setFilter(GL_LINEAR);
266 data.texture->setWrapMode(GL_CLAMP_TO_EDGE);
267 data.framebuffer = std::make_unique<GLFramebuffer>(data.texture.get());
268 }
269 if (!data.vbo || data.viewport != rect) {
270 data.vbo.reset(new GLVertexBuffer(GLVertexBuffer::Static));
271 data.viewport = scaledRect(rect, devicePixelRatio).toRect();
272
273 QVector<float> verts;
274 QVector<float> texcoords;
275
276 // The v-coordinate is flipped because projection matrix is "flipped."
277 texcoords << 1.0 << 1.0;
278 verts << (rect.x() + rect.width()) * devicePixelRatio << rect.y() * devicePixelRatio;
279 texcoords << 0.0 << 1.0;
280 verts << rect.x() * devicePixelRatio << rect.y() * devicePixelRatio;
281 texcoords << 0.0 << 0.0;
282 verts << rect.x() * devicePixelRatio << (rect.y() + rect.height()) * devicePixelRatio;
283
284 texcoords << 1.0 << 0.0;
285 verts << (rect.x() + rect.width()) * devicePixelRatio << (rect.y() + rect.height()) * devicePixelRatio;
286 texcoords << 1.0 << 1.0;
287 verts << (rect.x() + rect.width()) * devicePixelRatio << rect.y() * devicePixelRatio;
288 texcoords << 0.0 << 0.0;
289 verts << rect.x() * devicePixelRatio << (rect.y() + rect.height()) * devicePixelRatio;
290
291 data.vbo->setData(6, 2, verts.constData(), texcoords.constData());
292 }
293
294 return &data;
295}

297void ZoomEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
298{
299 OffscreenData *offscreenData = ensureOffscreenData(data.screen());
300
301 // Render the scene in an offscreen texture and then upscale it.
302 GLFramebuffer::pushFramebuffer(offscreenData->framebuffer.get());
303 effects->paintScreen(mask, region, data);
304 GLFramebuffer::popFramebuffer();
305
306 const QSize screenSize = effects->virtualScreenSize();
307 const auto scale = effects->renderTargetScale();
308
309 // mouse-tracking allows navigation of the zoom-area using the mouse.
310 qreal xTranslation = 0;
311 qreal yTranslation = 0;
312 switch (mouseTracking) {
313 case MouseTrackingProportional:
314 xTranslation = -int(cursorPoint.x() * (zoom - 1.0));
315 yTranslation = -int(cursorPoint.y() * (zoom - 1.0));
316 prevPoint = cursorPoint;
317 break;
318 case MouseTrackingCentred:
319 prevPoint = cursorPoint;
320 // fall through
321 case MouseTrackingDisabled:
322 xTranslation = std::min(0, std::max(int(screenSize.width() - screenSize.width() * zoom), int(screenSize.width() / 2 - prevPoint.x() * zoom)));
323 yTranslation = std::min(0, std::max(int(screenSize.height() - screenSize.height() * zoom), int(screenSize.height() / 2 - prevPoint.y() * zoom)));
324 break;
325 case MouseTrackingPush: {
326 // touching an edge of the screen moves the zoom-area in that direction.
327 int x = cursorPoint.x() * zoom - prevPoint.x() * (zoom - 1.0);
328 int y = cursorPoint.y() * zoom - prevPoint.y() * (zoom - 1.0);
329 int threshold = 4;
330 xMove = yMove = 0;
331 if (x < threshold) {
332 xMove = (x - threshold) / zoom;
333 } else if (x + threshold > screenSize.width()) {
334 xMove = (x + threshold - screenSize.width()) / zoom;
335 }
336 if (y < threshold) {
337 yMove = (y - threshold) / zoom;
338 } else if (y + threshold > screenSize.height()) {
339 yMove = (y + threshold - screenSize.height()) / zoom;
340 }
341 if (xMove) {
342 prevPoint.setX(std::max(0, std::min(screenSize.width(), prevPoint.x() + xMove)));
343 }
344 if (yMove) {
345 prevPoint.setY(std::max(0, std::min(screenSize.height(), prevPoint.y() + yMove)));
346 }
347 xTranslation = -int(prevPoint.x() * (zoom - 1.0));
348 yTranslation = -int(prevPoint.y() * (zoom - 1.0));
349 break;
350 }
351 }
352
353 // use the focusPoint if focus tracking is enabled
354 if (isFocusTrackingEnabled() || isTextCaretTrackingEnabled()) {
355 bool acceptFocus = true;
356 if (mouseTracking != MouseTrackingDisabled && focusDelay > 0) {
357 // Wait some time for the mouse before doing the switch. This serves as threshold
358 // to prevent the focus from jumping around to much while working with the mouse.
359 const int msecs = lastMouseEvent.msecsTo(lastFocusEvent);
360 acceptFocus = msecs > focusDelay;
361 }
362 if (acceptFocus) {
363 xTranslation = -int(focusPoint.x() * (zoom - 1.0));
364 yTranslation = -int(focusPoint.y() * (zoom - 1.0));
365 prevPoint = focusPoint;
366 }
367 }
368
369 // Render transformed offscreen texture.
370 glClearColor(0.0, 0.0, 0.0, 0.0);
371 glClear(GL_COLOR_BUFFER_BIT);
372
373 QMatrix4x4 matrix;
374 matrix.translate(xTranslation * scale, yTranslation * scale);
375 matrix.scale(zoom, zoom);
376
377 auto shader = ShaderManager::instance()->pushShader(ShaderTrait::MapTexture);
378 shader->setUniform(GLShader::ModelViewProjectionMatrix, data.projectionMatrix() * matrix);
379 for (auto &[screen, data] : m_offscreenData) {
380 data.texture->bind();
381 data.vbo->render(GL_TRIANGLES);
382 data.texture->unbind();
383 }
384 ShaderManager::instance()->popShader();
385
386 if (mousePointer != MousePointerHide) {
387 // Draw the mouse-texture at the position matching to zoomed-in image of the desktop. Hiding the
388 // previous mouse-cursor and drawing our own fake mouse-cursor is needed to be able to scale the
389 // mouse-cursor up and to re-position those mouse-cursor to match to the chosen zoom-level.
390
391 GLTexture *cursorTexture = ensureCursorTexture();
392 if (cursorTexture) {
393 const auto cursor = effects->cursorImage();
394 QSizeF cursorSize = QSizeF(cursor.image().size()) / cursor.image().devicePixelRatio();
395 if (mousePointer == MousePointerScale) {
396 cursorSize *= zoom;
397 }
398
399 const QPointF p = (effects->cursorPos() - cursor.hotSpot()) * zoom + QPoint(xTranslation, yTranslation);
400
401 cursorTexture->bind();
402 glEnable(GL_BLEND);
403 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
404 auto s = ShaderManager::instance()->pushShader(ShaderTrait::MapTexture);
405 QMatrix4x4 mvp = data.projectionMatrix();
406 mvp.translate(p.x() * scale, p.y() * scale);
407 s->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
408 cursorTexture->render(cursorSize, scale);
409 ShaderManager::instance()->popShader();
410 cursorTexture->unbind();
411 glDisable(GL_BLEND);
412 }
413 }

430void ZoomEffect::zoomIn(double to)
431{
432 source_zoom = zoom;
433 if (to < 0.0) {
434 target_zoom *= zoomFactor;
435 } else {
436 target_zoom = to;
437 }
438 if (!polling) {
439 polling = true;
440 effects->startMousePolling();
441 }
442 cursorPoint = effects->cursorPos().toPoint();
443 if (mouseTracking == MouseTrackingDisabled) {
444 prevPoint = cursorPoint;
445 }
446 effects->addRepaintFull();
447}

449void ZoomEffect::zoomOut()
450{
451 source_zoom = zoom;
452 target_zoom /= zoomFactor;
453 if ((zoomFactor > 1 && target_zoom < 1.01) || (zoomFactor < 1 && target_zoom > 0.99)) {
454 target_zoom = 1;
455 if (polling) {
456 polling = false;
457 effects->stopMousePolling();
458 }
459 }
460 if (mouseTracking == MouseTrackingDisabled) {
461 prevPoint = effects->cursorPos().toPoint();
462 }
463 effects->addRepaintFull();
464}

486void ZoomEffect::moveZoom(int x, int y)
487{
488 if (timeline.state() == QTimeLine::Running) {
489 timeline.stop();
490 }
491
492 const QSize screenSize = effects->virtualScreenSize();
493 if (x < 0) {
494 xMove = -std::max(1.0, screenSize.width() / zoom / moveFactor);
495 } else if (x > 0) {
496 xMove = std::max(1.0, screenSize.width() / zoom / moveFactor);
497 } else {
498 xMove = 0;
499 }
500
501 if (y < 0) {
502 yMove = -std::max(1.0, screenSize.height() / zoom / moveFactor);
503 } else if (y > 0) {
504 yMove = std::max(1.0, screenSize.height() / zoom / moveFactor);
505 } else {
506 yMove = 0;
507 }
508
509 timeline.start();
510}
