94void TouchPointsEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
95{
96 int time = 0;
97 if (m_lastPresentTime.count()) {
98 time = (presentTime - m_lastPresentTime).count();
99 }
100
101 auto it = m_points.begin();
102 while (it != m_points.end()) {
103 it->time += time;
104 if (it->time > m_ringLife) {
105 it = m_points.erase(it);
106 } else {
107 it++;
108 }
109 }
110
111 if (m_points.isEmpty()) {
112 m_lastPresentTime = std::chrono::milliseconds::zero();
113 } else {
114 m_lastPresentTime = presentTime;
115 }
116
117 effects->prePaintScreen(data, presentTime);
118}

120void TouchPointsEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
121{
122 effects->paintScreen(mask, region, data);
123
124 paintScreenSetup(mask, region, data);
125 for (auto it = m_points.constBegin(), end = m_points.constEnd(); it != end; ++it) {
126 for (int i = 0; i < m_ringCount; ++i) {
127 float alpha = computeAlpha(it->time, i);
128 float size = computeRadius(it->time, it->press, i);
129 if (size > 0 && alpha > 0) {
130 QColor color = it->color;
131 color.setAlphaF(alpha);
132 drawCircle(color, it->pos.x(), it->pos.y(), size);
133 }
134 }
135 }
136 paintScreenFinish(mask, region, data);
137}

145float TouchPointsEffect::computeRadius(int time, bool press, int ring)
146{
147 float ringDistance = m_ringLife / (m_ringCount * 3);
148 if (press) {
149 return ((time - ringDistance * ring) / m_ringLife) * m_ringMaxSize;
150 }
151 return ((m_ringLife - time - ringDistance * ring) / m_ringLife) * m_ringMaxSize;
152}
153
154float TouchPointsEffect::computeAlpha(int time, int ring)
155{
156 float ringDistance = m_ringLife / (m_ringCount * 3);
157 return (m_ringLife - (float)time - ringDistance * (ring)) / m_ringLife;
158}

200void TouchPointsEffect::drawCircleGl(const QColor &color, float cx, float cy, float r)
201{
202 static const int num_segments = 80;
203 static const float theta = 2 * 3.1415926 / float(num_segments);
204 static const float c = cosf(theta); // precalculate the sine and cosine
205 static const float s = sinf(theta);
206 const auto scale = effects->renderTargetScale();
207 float t;
208
209 float x = r; // we start at angle = 0
210 float y = 0;
211
212 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
213 vbo->reset();
214 vbo->setUseColor(true);
215 vbo->setColor(color);
216 QVector<float> verts;
217 verts.reserve(num_segments * 2);
218
219 for (int ii = 0; ii < num_segments; ++ii) {
220 verts << (x + cx) * scale << (y + cy) * scale; // output vertex
221 // apply the rotation matrix
222 t = x;
223 x = c * x - s * y;
224 y = s * t + c * y;
225 }
226 vbo->setData(verts.size() / 2, 2, verts.data(), nullptr);
227 vbo->render(GL_LINE_LOOP);
228}
