74void PlasmaShellInterfacePrivate::org_kde_plasma_shell_get_surface(QtWaylandServer::org_kde_plasma_shell::Resource *resource,
75 uint32_t id,
76 struct ::wl_resource *surface)
77{
78 SurfaceInterface *s = SurfaceInterface::get(surface);
79 if (!s) {
80 wl_resource_post_error(resource->handle, 0, "Invalid surface");
81 return;
82 }
83
84 if (PlasmaShellSurfaceInterface::get(s)) {
85 wl_resource_post_error(resource->handle, 0, "org_kde_plasma_shell_surface already exists");
86 return;
87 }
88
89 wl_resource *shell_resource = wl_resource_create(resource->client(), &org_kde_plasma_surface_interface, resource->version(), id);
90
91 auto shellSurface = new PlasmaShellSurfaceInterface(s, shell_resource);
92 s_shellSurfaces.append(shellSurface);
93
94 QObject::connect(shellSurface, &QObject::destroyed, [shellSurface]() {
95 s_shellSurfaces.removeOne(shellSurface);
96 });
97
98 Q_EMIT q->surfaceCreated(shellSurface);
99}

158void PlasmaShellSurfaceInterfacePrivate::org_kde_plasma_surface_set_role(Resource *resource, uint32_t role)
159{
160 PlasmaShellSurfaceInterface::Role r = PlasmaShellSurfaceInterface::Role::Normal;
161 switch (role) {
162 case role_desktop:
163 r = PlasmaShellSurfaceInterface::Role::Desktop;
164 break;
165 case role_panel:
166 r = PlasmaShellSurfaceInterface::Role::Panel;
167 break;
168 case role_onscreendisplay:
169 r = PlasmaShellSurfaceInterface::Role::OnScreenDisplay;
170 break;
171 case role_notification:
172 r = PlasmaShellSurfaceInterface::Role::Notification;
173 break;
174 case role_tooltip:
175 r = PlasmaShellSurfaceInterface::Role::ToolTip;
176 break;
177 case role_criticalnotification:
178 r = PlasmaShellSurfaceInterface::Role::CriticalNotification;
179 break;
180 case role_appletpopup:
181 r = PlasmaShellSurfaceInterface::Role::AppletPopup;
182 break;
183 case role_normal:
184 default:
185 r = PlasmaShellSurfaceInterface::Role::Normal;
186 break;
187 }
188 if (r == m_role) {
189 return;
190 }
191 m_role = r;
192 Q_EMIT q->roleChanged();
193}
