45void ShowPaintEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
46{
47 m_painted = QRegion();
48 effects->paintScreen(mask, region, data);
49 if (effects->isOpenGLCompositing()) {
50 paintGL(data.projectionMatrix(), effects->renderTargetScale());
51 } else if (effects->compositingType() == QPainterCompositing) {
52 paintQPainter();
53 }
54 if (++m_colorIndex == s_colors.count()) {
55 m_colorIndex = 0;
56 }
57}

65void ShowPaintEffect::paintGL(const QMatrix4x4 &projection, qreal scale)
66{
67 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
68 vbo->reset();
69 vbo->setUseColor(true);
70 ShaderBinder binder(ShaderTrait::UniformColor);
71 binder.shader()->setUniform(GLShader::ModelViewProjectionMatrix, projection);
72 glEnable(GL_BLEND);
73 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
74 QColor color = s_colors[m_colorIndex];
75 color.setAlphaF(s_alpha);
76 vbo->setColor(color);
77 QVector<float> verts;
78 verts.reserve(m_painted.rectCount() * 12);
79 for (const QRect &r : m_painted) {
80 const auto deviceRect = scaledRect(r, scale);
81 verts << deviceRect.x() + deviceRect.width() << deviceRect.y();
82 verts << deviceRect.x() << deviceRect.y();
83 verts << deviceRect.x() << deviceRect.y() + deviceRect.height();
84 verts << deviceRect.x() << deviceRect.y() + deviceRect.height();
85 verts << deviceRect.x() + deviceRect.width() << deviceRect.y() + deviceRect.height();
86 verts << deviceRect.x() + deviceRect.width() << deviceRect.y();
87 }
88 vbo->setData(verts.count() / 2, 2, verts.data(), nullptr);
89 vbo->render(GL_TRIANGLES);
90 glDisable(GL_BLEND);
91}

93void ShowPaintEffect::paintQPainter()
94{
