63bool EglGbmBackend::initializeEgl()
64{
65 initClientExtensions();
66 EGLDisplay display = m_backend->primaryGpu()->eglDisplay();
67
68 // Use eglGetPlatformDisplayEXT() to get the display pointer
69 // if the implementation supports it.
70 if (display == EGL_NO_DISPLAY) {
71 const bool hasMesaGBM = hasClientExtension(QByteArrayLiteral("EGL_MESA_platform_gbm"));
72 const bool hasKHRGBM = hasClientExtension(QByteArrayLiteral("EGL_KHR_platform_gbm"));
73 const GLenum platform = hasMesaGBM ? EGL_PLATFORM_GBM_MESA : EGL_PLATFORM_GBM_KHR;
74
75 if (!hasClientExtension(QByteArrayLiteral("EGL_EXT_platform_base")) || (!hasMesaGBM && !hasKHRGBM)) {
76 setFailed("Missing one or more extensions between EGL_EXT_platform_base, "
77 "EGL_MESA_platform_gbm, EGL_KHR_platform_gbm");
78 return false;
79 }
80
81 if (!m_backend->primaryGpu()->gbmDevice()) {
82 setFailed("Could not create gbm device");
83 return false;
84 }
85
86 display = eglGetPlatformDisplayEXT(platform, m_backend->primaryGpu()->gbmDevice(), nullptr);
87 m_backend->primaryGpu()->setEglDisplay(display);
88 }
89
90 if (display == EGL_NO_DISPLAY) {
91 return false;
92 }
93 setEglDisplay(display);
94 return initEglAPI();
95}

124bool EglGbmBackend::initBufferConfigs()
125{
126 const EGLint config_attribs[] = {
127 EGL_SURFACE_TYPE,
128 EGL_WINDOW_BIT,
129 EGL_RED_SIZE,
130 1,
131 EGL_GREEN_SIZE,
132 1,
133 EGL_BLUE_SIZE,
134 1,
135 EGL_ALPHA_SIZE,
136 0,
137 EGL_RENDERABLE_TYPE,
138 isOpenGLES() ? EGL_OPENGL_ES2_BIT : EGL_OPENGL_BIT,
139 EGL_CONFIG_CAVEAT,
140 EGL_NONE,
141 EGL_NONE,
142 };
143
144 EGLint count;
145 EGLConfig configs[1024];
146 if (!eglChooseConfig(eglDisplay(), config_attribs, configs,
147 sizeof(configs) / sizeof(EGLConfig),
148 &count)) {
149 qCCritical(KWIN_DRM) << "eglChooseConfig failed:" << getEglErrorString();
150 return false;
151 }
152
153 setConfig(EGL_NO_CONFIG_KHR);
154
155 // Loop through all configs, choosing the first one that has suitable format.
156 for (EGLint i = 0; i < count; i++) {
157 EGLint gbmFormat;
158 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_NATIVE_VISUAL_ID, &gbmFormat);
159
160 GbmFormat format;
161 format.drmFormat = gbmFormat;
162 EGLint red, green, blue;
163 // Query number of bits for color channel
164 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_RED_SIZE, &red);
165 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_GREEN_SIZE, &green);
166 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_BLUE_SIZE, &blue);
167 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_ALPHA_SIZE, &format.alphaSize);
168 format.bpp = red + green + blue;
169 if (m_formats.contains(gbmFormat)) {
170 continue;
171 }
172 m_formats[gbmFormat] = format;
173 m_configs[format.drmFormat] = configs[i];
174 }
175 if (!m_formats.isEmpty()) {
176 return true;
177 }
178
179 qCCritical(KWIN_DRM, "Choosing EGL config did not return a supported config. There were %u configs", count);
180 for (EGLint i = 0; i < count; i++) {
181 EGLint gbmFormat, blueSize, redSize, greenSize, alphaSize;
182 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_NATIVE_VISUAL_ID, &gbmFormat);
183 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_RED_SIZE, &redSize);
184 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_GREEN_SIZE, &greenSize);
185 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_BLUE_SIZE, &blueSize);
186 eglGetConfigAttrib(eglDisplay(), configs[i], EGL_ALPHA_SIZE, &alphaSize);
187 gbm_format_name_desc name;
188 gbm_format_get_name(gbmFormat, &name);
189 qCCritical(KWIN_DRM, "EGL config %d has format %s with %d,%d,%d,%d bits for r,g,b,a", i, name.name, redSize, greenSize, blueSize, alphaSize);
190 }
191 return false;
192}
