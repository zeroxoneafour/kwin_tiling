118void XdgOutputV1Interface::update()
119{
120 if (!output || output->isRemoved()) {
121 return;
122 }
123
124 const QRectF geometry = output->handle()->fractionalGeometry();
125 const auto resources = resourceMap();
126
127 if (pos != geometry.topLeft()) {
128 pos = geometry.topLeft();
129 for (auto resource : resources) {
130 sendLogicalPosition(resource);
131 }
132 }
133
134 if (size != geometry.size()) {
135 size = geometry.size();
136 for (auto resource : resources) {
137 sendLogicalSize(resource);
138 }
139 }
140
141 for (auto resource : resources) {
142 if (wl_resource_get_version(resource->handle) < 3) {
143 send_done(resource->handle);
144 }
145 }
146
147 output->scheduleDone();
148}

155void XdgOutputV1Interface::zxdg_output_v1_bind_resource(Resource *resource)
156{
157 if (!output || output->isRemoved()) {
158 return;
159 }
160
161 sendLogicalPosition(resource);
162 sendLogicalSize(resource);
163 if (resource->version() >= ZXDG_OUTPUT_V1_NAME_SINCE_VERSION) {
164 send_name(resource->handle, name);
165 }
166 if (resource->version() >= ZXDG_OUTPUT_V1_DESCRIPTION_SINCE_VERSION) {
167 send_description(resource->handle, description);
168 }
169
170 sendDone(resource);
171
172 ClientConnection *connection = output->display()->getConnection(resource->client());
173 connect(connection, &ClientConnection::scaleOverrideChanged, this, &XdgOutputV1Interface::resend, Qt::UniqueConnection);
174}

192void XdgOutputV1Interface::sendDone(Resource *resource)
193{
194 if (wl_resource_get_version(resource->handle) >= 3) {

201void XdgOutputV1Interface::resend()
202{
203 if (!output || output->isRemoved()) {
204 return;
205 }
206
207 auto changedConnection = qobject_cast<ClientConnection *>(sender());
208 const auto outputResources = resourceMap();
209 for (auto resource : outputResources) {
210 ClientConnection *connection = output->display()->getConnection(resource->client());
211 if (connection == changedConnection) {
212 sendLogicalPosition(resource);
213 sendLogicalSize(resource);
214 sendDone(resource);
215 }
216 }
217}
