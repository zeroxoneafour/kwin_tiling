189void InputMethodContextV1Interface::sendContentType(TextInputContentHints hint, TextInputContentPurpose purpose)
190{
191 quint32 contentHint = QtWaylandServer::zwp_text_input_v1::content_hint_none;
192 quint32 contentPurpose;
193
194 if (hint.testFlag(TextInputContentHint::AutoCapitalization)) {
195 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_auto_capitalization;
196 }
197 if (hint.testFlag(TextInputContentHint::AutoCorrection)) {
198 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_auto_correction;
199 }
200 if (hint.testFlag(TextInputContentHint::AutoCompletion)) {
201 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_auto_completion;
202 }
203 if (hint.testFlag(TextInputContentHint::LowerCase)) {
204 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_lowercase;
205 }
206 if (hint.testFlag(TextInputContentHint::UpperCase)) {
207 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_uppercase;
208 }
209 if (hint.testFlag(TextInputContentHint::TitleCase)) {
210 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_titlecase;
211 }
212 if (hint.testFlag(TextInputContentHint::HiddenText)) {
213 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_hidden_text;
214 }
215 if (hint.testFlag(TextInputContentHint::SensitiveData)) {
216 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_lowercase;
217 }
218 if (hint.testFlag(TextInputContentHint::Latin)) {
219 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_latin;
220 }
221 if (hint.testFlag(TextInputContentHint::MultiLine)) {
222 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_multiline;
223 }
224 if (hint.testFlag(TextInputContentHint::None)) {
225 contentHint |= QtWaylandServer::zwp_text_input_v1::content_hint_none;
226 }
227
228 switch (purpose) {
229 case TextInputContentPurpose::Alpha:
230 contentPurpose = QtWaylandServer::zwp_text_input_v1::content_purpose_alpha;
231 break;
232 case TextInputContentPurpose::Digits:
233 contentPurpose = QtWaylandServer::zwp_text_input_v1::content_purpose_digits;

235 case TextInputContentPurpose::Number:
236 contentPurpose = QtWaylandServer::zwp_text_input_v1::content_purpose_number;

238 case TextInputContentPurpose::Phone:

240 break;
241 case TextInputContentPurpose::Url:
242 contentPurpose = QtWaylandServer::zwp_text_input_v1::content_purpose_url;
243 break;
244 case TextInputContentPurpose::Email:
245 contentPurpose = QtWaylandServer::zwp_text_input_v1::content_purpose_email;

247 case TextInputContentPurpose::Name:

249 break;

253 case TextInputContentPurpose::Date:
