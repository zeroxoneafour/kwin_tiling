71static QByteArray readWindowProperty(xcb_window_t win, xcb_atom_t atom, xcb_atom_t type, int format)
72{
73 if (win == XCB_WINDOW_NONE) {
74 return QByteArray();
75 }
76 uint32_t len = 32768;
77 for (;;) {
78 Xcb::Property prop(false, win, atom, XCB_ATOM_ANY, 0, len);
79 if (prop.isNull()) {
80 // get property failed
81 return QByteArray();
82 }
83 if (prop->bytes_after > 0) {
84 len *= 2;
85 continue;
86 }
87 return prop.toByteArray(format, type);
88 }
89}

99static xcb_atom_t registerSupportProperty(const QByteArray &propertyName)
100{
101 auto c = kwinApp()->x11Connection();
102 if (!c) {
103 return XCB_ATOM_NONE;
104 }
105 // get the atom for the propertyName
106 UniqueCPtr<xcb_intern_atom_reply_t> atomReply(xcb_intern_atom_reply(c,
107 xcb_intern_atom_unchecked(c, false, propertyName.size(), propertyName.constData()),
108 nullptr));
109 if (!atomReply) {
110 return XCB_ATOM_NONE;
111 }
112 // announce property on root window
113 unsigned char dummy = 0;
114 xcb_change_property(c, XCB_PROP_MODE_REPLACE, kwinApp()->x11RootWindow(), atomReply->atom, atomReply->atom, 8, 1, &dummy);
115 // TODO: add to _NET_SUPPORTED
116 return atomReply->atom;
117}

672void EffectsHandlerImpl::doStartMouseInterception(Qt::CursorShape shape)
673{
674 input()->pointer()->setEffectsOverrideCursor(shape);
675
676 // We want to allow global shortcuts to be triggered when moving a
677 // window so it is possible to pick up a window and then move it to a

852xcb_atom_t EffectsHandlerImpl::announceSupportProperty(const QByteArray &propertyName, Effect *effect)
853{
854 PropertyEffectMap::iterator it = m_propertiesForEffects.find(propertyName);
855 if (it != m_propertiesForEffects.end()) {
856 // property has already been registered for an effect
857 // just append Effect and return the atom stored in m_managedProperties
858 if (!it.value().contains(effect)) {
859 it.value().append(effect);
860 }
861 return m_managedProperties.value(propertyName, XCB_ATOM_NONE);
862 }
863 m_propertiesForEffects.insert(propertyName, QList<Effect *>() << effect);
864 const auto atom = registerSupportProperty(propertyName);
865 if (atom == XCB_ATOM_NONE) {
866 return atom;
867 }
868 m_compositor->keepSupportProperty(atom);
869 m_managedProperties.insert(propertyName, atom);
870 registerPropertyType(atom, true);
871 return atom;
872}
873
874void EffectsHandlerImpl::removeSupportProperty(const QByteArray &propertyName, Effect *effect)
875{
876 PropertyEffectMap::iterator it = m_propertiesForEffects.find(propertyName);
877 if (it == m_propertiesForEffects.end()) {
878 // property is not registered - nothing to do
879 return;
880 }
881 if (!it.value().contains(effect)) {
882 // property is not registered for given effect - nothing to do
883 return;
884 }
885 it.value().removeAll(effect);
886 if (!it.value().isEmpty()) {
887 // property still registered for another effect - nothing further to do
888 return;
889 }
890 const xcb_atom_t atom = m_managedProperties.take(propertyName);
891 registerPropertyType(atom, false);
892 m_propertiesForEffects.remove(propertyName);
893 m_compositor->removeSupportProperty(atom); // delayed removal
894}

1739void EffectsHandlerImpl::renderOffscreenQuickView(OffscreenQuickView *w) const
1740{
1741 if (!w->isVisible()) {
1742 return;
1743 }
1744 if (compositingType() == OpenGLCompositing) {
1745 GLTexture *t = w->bufferAsTexture();
1746 if (!t) {
1747 return;
1748 }
1749
1750 ShaderTraits traits = ShaderTrait::MapTexture;
1751 const qreal a = w->opacity();
1752 if (a != 1.0) {
1753 traits |= ShaderTrait::Modulate;
1754 }
1755
1756 GLShader *shader = ShaderManager::instance()->pushShader(traits);
1757 const QRectF rect = scaledRect(w->geometry(), m_scene->renderer()->renderTargetScale());
1758
1759 QMatrix4x4 mvp(m_scene->renderer()->renderTargetProjectionMatrix());
1760 mvp.translate(rect.x(), rect.y());
1761 shader->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
1762
1763 if (a != 1.0) {
1764 shader->setUniform(GLShader::ModulationConstant, QVector4D(a, a, a, a));
1765 }
1766
1767 glEnable(GL_BLEND);
1768 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
1769 t->bind();
1770 t->render(w->size(), m_scene->renderer()->renderTargetScale());
1771 t->unbind();
1772 glDisable(GL_BLEND);
1773
1774 ShaderManager::instance()->popShader();
1775 } else if (compositingType() == QPainterCompositing) {
1776 QPainter *painter = effects->scenePainter();
1777 const QImage buffer = w->bufferAsImage();
1778 if (buffer.isNull()) {
1779 return;
1780 }
1781 painter->save();
1782 painter->setOpacity(w->opacity());
1783 painter->drawImage(w->geometry(), buffer);
1784 painter->restore();
1785 }
1786}
