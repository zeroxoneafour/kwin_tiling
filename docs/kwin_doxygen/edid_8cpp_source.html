22static bool verifyHeader(const uint8_t *data)
23{
24 if (data[0] != 0x0 || data[7] != 0x0) {
25 return false;
26 }
27
28 return std::all_of(data + 1, data + 7,
29 [](uint8_t byte) {
30 return byte == 0xff;
31 });
32}

40static QByteArray parsePnpId(const uint8_t *data)
41{
42 // Decode PNP ID from three 5 bit words packed into 2 bytes:
43 //
44 // | Byte | Bit |
45 // | | 7 | 6 | 5 | 4 | 3 | 2 | 1 | 0 |
46 // ----------------------------------------
47 // | 1 | 0)| (4| 3 | 2 | 1 | 0)| (4| 3 |
48 // | | * | Character 1 | Char 2|
49 // ----------------------------------------
50 // | 2 | 2 | 1 | 0)| (4| 3 | 2 | 1 | 0)|
51 // | | Character2| Character 3 |
52 // ----------------------------------------
53 const uint offset = 0x8;
54
55 char pnpId[4];
56 pnpId[0] = 'A' + ((data[offset + 0] >> 2) & 0x1f) - 1;
57 pnpId[1] = 'A' + (((data[offset + 0] & 0x3) << 3) | ((data[offset + 1] >> 5) & 0x7)) - 1;
58 pnpId[2] = 'A' + (data[offset + 1] & 0x1f) - 1;
59 pnpId[3] = '\0';
60
61 return QByteArray(pnpId);
62}

64static QByteArray parseEisaId(const uint8_t *data)
65{
66 for (int i = 72; i <= 108; i += 18) {
67 // Skip the block if it isn't used as monitor descriptor.
68 if (data[i]) {
69 continue;
70 }
71 if (data[i + 1]) {
72 continue;
73 }
74
75 // We have found the EISA ID, it's stored as ASCII.
76 if (data[i + 3] == 0xfe) {
77 return QByteArray(reinterpret_cast<const char *>(&data[i + 5]), 12).trimmed();
78 }
79 }
80
81 // If there isn't an ASCII EISA ID descriptor, try to decode PNP ID
82 return parsePnpId(data);
83}

105static QByteArray parseSerialNumber(const uint8_t *data)
106{
107 for (int i = 72; i <= 108; i += 18) {
108 // Skip the block if it isn't used as monitor descriptor.
109 if (data[i]) {
110 continue;
111 }
112 if (data[i + 1]) {
113 continue;
114 }
115
116 // We have found the serial number, it's stored as ASCII.
117 if (data[i + 3] == 0xff) {
118 return QByteArray(reinterpret_cast<const char *>(&data[i + 5]), 12).trimmed();
119 }
120 }
121
122 // Maybe there isn't an ASCII serial number descriptor, so use this instead.
123 const uint32_t offset = 0xc;
124
125 uint32_t serialNumber = data[offset + 0];
126 serialNumber |= uint32_t(data[offset + 1]) << 8;
127 serialNumber |= uint32_t(data[offset + 2]) << 16;
128 serialNumber |= uint32_t(data[offset + 3]) << 24;
129 if (serialNumber) {
130 return QByteArray::number(serialNumber);
131 }
132
133 return QByteArray();
134}

136static QByteArray parseVendor(const uint8_t *data)
137{
138 const auto pnpId = parsePnpId(data);
139
140 // Map to vendor name
141 QFile pnpFile(QStandardPaths::locate(QStandardPaths::GenericDataLocation, QStringLiteral("hwdata/pnp.ids")));
142 if (pnpFile.exists() && pnpFile.open(QIODevice::ReadOnly)) {
143 while (!pnpFile.atEnd()) {
144 const auto line = pnpFile.readLine();
145 if (line.startsWith(pnpId)) {
146 return line.mid(4).trimmed();
147 }
148 }
149 }
150
151 return {};
152}

158Edid::Edid(const void *data, uint32_t size)
159{
160 m_raw.resize(size);
161 memcpy(m_raw.data(), data, size);
162
163 const uint8_t *bytes = static_cast<const uint8_t *>(data);
164
165 if (size < 128) {
166 return;
167 }
168
169 if (!verifyHeader(bytes)) {
170 return;
171 }
172
173 m_physicalSize = parsePhysicalSize(bytes);
174 m_eisaId = parseEisaId(bytes);
175 m_monitorName = parseMonitorName(bytes);
176 m_serialNumber = parseSerialNumber(bytes);
177 m_vendor = parseVendor(bytes);
178
179 m_isValid = true;
180}
