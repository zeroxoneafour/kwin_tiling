87static bool keyboard_grabbed = false;
88
89bool grabXKeyboard(xcb_window_t w)
90{
91 if (QWidget::keyboardGrabber() != nullptr) {
92 return false;
93 }
94 if (keyboard_grabbed) {
95 qCDebug(KWIN_CORE) << "Failed to grab X Keyboard: already grabbed by us";
96 return false;
97 }
98 if (qApp->activePopupWidget() != nullptr) {
99 qCDebug(KWIN_CORE) << "Failed to grab X Keyboard: no popup widget";
100 return false;
101 }
102 if (w == XCB_WINDOW_NONE) {
103 w = rootWindow();
104 }
105 const xcb_grab_keyboard_cookie_t c = xcb_grab_keyboard_unchecked(connection(), false, w, xTime(),
106 XCB_GRAB_MODE_ASYNC, XCB_GRAB_MODE_ASYNC);
107 UniqueCPtr<xcb_grab_keyboard_reply_t> grab(xcb_grab_keyboard_reply(connection(), c, nullptr));
108 if (!grab) {
109 qCDebug(KWIN_CORE) << "Failed to grab X Keyboard: grab null";
110 return false;
111 }
112 if (grab->status != XCB_GRAB_STATUS_SUCCESS) {
113 qCDebug(KWIN_CORE) << "Failed to grab X Keyboard: grab failed with status" << grab->status;
114 return false;
115 }
116 keyboard_grabbed = true;
117 return true;
118}
119
120void ungrabXKeyboard()
121{
122 if (!keyboard_grabbed) {
123 // grabXKeyboard() may fail sometimes, so don't fail, but at least warn anyway
124 qCDebug(KWIN_CORE) << "ungrabXKeyboard() called but keyboard not grabbed!";
125 }
126 keyboard_grabbed = false;
127 xcb_ungrab_keyboard(connection(), XCB_TIME_CURRENT_TIME);
128}

134Qt::MouseButton x11ToQtMouseButton(int button)
135{
136 if (button == XCB_BUTTON_INDEX_1) {
137 return Qt::LeftButton;
138 }
139 if (button == XCB_BUTTON_INDEX_2) {
140 return Qt::MiddleButton;
141 }
142 if (button == XCB_BUTTON_INDEX_3) {
143 return Qt::RightButton;
144 }
145 if (button == XCB_BUTTON_INDEX_4) {
146 return Qt::XButton1;
147 }
148 if (button == XCB_BUTTON_INDEX_5) {
149 return Qt::XButton2;
150 }
151 return Qt::NoButton;
152}

154Qt::MouseButtons x11ToQtMouseButtons(int state)
155{
156 Qt::MouseButtons ret = {};
157 if (state & XCB_KEY_BUT_MASK_BUTTON_1) {
158 ret |= Qt::LeftButton;
159 }
160 if (state & XCB_KEY_BUT_MASK_BUTTON_2) {
161 ret |= Qt::MiddleButton;
162 }
163 if (state & XCB_KEY_BUT_MASK_BUTTON_3) {
164 ret |= Qt::RightButton;
165 }
166 if (state & XCB_KEY_BUT_MASK_BUTTON_4) {
167 ret |= Qt::XButton1;
168 }
169 if (state & XCB_KEY_BUT_MASK_BUTTON_5) {
170 ret |= Qt::XButton2;
171 }
172 return ret;
173}

175Qt::KeyboardModifiers x11ToQtKeyboardModifiers(int state)
176{
177 Qt::KeyboardModifiers ret = {};
178 if (state & XCB_KEY_BUT_MASK_SHIFT) {
179 ret |= Qt::ShiftModifier;
180 }
181 if (state & XCB_KEY_BUT_MASK_CONTROL) {
182 ret |= Qt::ControlModifier;
183 }
184 if (state & KKeyServer::modXAlt()) {
185 ret |= Qt::AltModifier;
186 }
187 if (state & KKeyServer::modXMeta()) {
188 ret |= Qt::MetaModifier;
189 }
190 return ret;
191}

193QPointF popupOffset(const QRectF &anchorRect, const Qt::Edges anchorEdge, const Qt::Edges gravity, const QSizeF popupSize)
194{
195 QPointF anchorPoint;
196 switch (anchorEdge & (Qt::LeftEdge | Qt::RightEdge)) {
197 case Qt::LeftEdge:
198 anchorPoint.setX(anchorRect.x());
199 break;
200 case Qt::RightEdge:
201 anchorPoint.setX(anchorRect.x() + anchorRect.width());
202 break;
203 default:
204 anchorPoint.setX(qRound(anchorRect.x() + anchorRect.width() / 2.0));
205 }
206 switch (anchorEdge & (Qt::TopEdge | Qt::BottomEdge)) {
207 case Qt::TopEdge:
208 anchorPoint.setY(anchorRect.y());
209 break;
210 case Qt::BottomEdge:
211 anchorPoint.setY(anchorRect.y() + anchorRect.height());
212 break;
213 default:
214 anchorPoint.setY(qRound(anchorRect.y() + anchorRect.height() / 2.0));
215 }
216
217 // calculate where the top left point of the popup will end up with the applied gravity
218 // gravity indicates direction. i.e if gravitating towards the top the popup's bottom edge
219 // will next to the anchor point
220 QPointF popupPosAdjust;
221 switch (gravity & (Qt::LeftEdge | Qt::RightEdge)) {
222 case Qt::LeftEdge:
223 popupPosAdjust.setX(-popupSize.width());
224 break;
225 case Qt::RightEdge:
226 popupPosAdjust.setX(0);
227 break;
228 default:
229 popupPosAdjust.setX(qRound(-popupSize.width() / 2.0));
230 }
231 switch (gravity & (Qt::TopEdge | Qt::BottomEdge)) {
232 case Qt::TopEdge:
233 popupPosAdjust.setY(-popupSize.height());
234 break;
235 case Qt::BottomEdge:
236 popupPosAdjust.setY(0);
237 break;
238 default:
239 popupPosAdjust.setY(qRound(-popupSize.height() / 2.0));
240 }
241
242 return anchorPoint + popupPosAdjust;
243}
