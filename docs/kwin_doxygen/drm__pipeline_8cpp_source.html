48bool DrmPipeline::testScanout()
49{
50 // TODO make the modeset check only be tested at most once per scanout cycle
51 if (gpu()->needsModeset()) {
52 return false;
53 }
54 if (gpu()->atomicModeSetting()) {
55 return commitPipelines({this}, CommitMode::Test) == Error::None;
56 } else {
57 if (m_pending.layer->currentBuffer()->buffer()->size() != m_pending.mode->size()) {
58 // scaling isn't supported with the legacy API
59 return false;
60 }
61 // no other way to test than to do it.
62 // As we only have a maximum of one test per scanout cycle, this is fine
63 return presentLegacy() == Error::None;
64 }

97DrmPipeline::Error DrmPipeline::commitPipelinesAtomic(const QVector<DrmPipeline *> &pipelines, CommitMode mode, const QVector<DrmObject *> &unusedObjects)
98{
99 const auto &failed = [&pipelines, &unusedObjects]() {
100 for (const auto &pipeline : pipelines) {
101 pipeline->printDebugInfo();
102 pipeline->atomicCommitFailed();
103 }
104 for (const auto &obj : unusedObjects) {
105 obj->printProps(DrmObject::PrintMode::OnlyChanged);
106 obj->rollbackPending();
107 }
108 };
109
110 DrmUniquePtr<drmModeAtomicReq> req{drmModeAtomicAlloc()};
111 if (!req) {
112 qCCritical(KWIN_DRM) << "Failed to allocate drmModeAtomicReq!" << strerror(errno);
113 return Error::OutofMemory;
114 }
115 for (const auto &pipeline : pipelines) {
116 pipeline->checkHardwareRotation();
117 if (pipeline->activePending()) {
118 if (!pipeline->m_pending.layer->checkTestBuffer()) {
119 qCWarning(KWIN_DRM) << "Checking test buffer failed for" << mode;
120 failed();
121 return Error::TestBufferFailed;
122 }
123 pipeline->prepareAtomicPresentation();
124 if (mode == CommitMode::TestAllowModeset || mode == CommitMode::CommitModeset) {
125 pipeline->prepareAtomicModeset();
126 }
127 } else {
128 pipeline->prepareAtomicDisable();
129 }
130 if (!pipeline->populateAtomicValues(req.get())) {
131 failed();
132 return errnoToError();
133 }
134 }
135 for (const auto &unused : unusedObjects) {
136 unused->disable();
137 if (!unused->atomicPopulate(req.get())) {
138 qCWarning(KWIN_DRM) << "Populating atomic values failed for unused resource" << unused;
139 failed();
140 return errnoToError();
141 }
142 }
143 const auto gpu = pipelines[0]->gpu();
144 switch (mode) {
145 case CommitMode::TestAllowModeset: {
146 bool withModeset = drmModeAtomicCommit(gpu->fd(), req.get(), DRM_MODE_ATOMIC_ALLOW_MODESET | DRM_MODE_ATOMIC_TEST_ONLY, nullptr) == 0;
147 if (!withModeset) {
148 qCDebug(KWIN_DRM) << "Atomic modeset test failed!" << strerror(errno);
149 failed();
150 return errnoToError();
151 }
152 bool withoutModeset = drmModeAtomicCommit(gpu->fd(), req.get(), DRM_MODE_ATOMIC_TEST_ONLY, nullptr) == 0;
153 for (const auto &pipeline : pipelines) {
154 pipeline->m_pending.needsModeset = !withoutModeset;
155 }
156 std::for_each(pipelines.begin(), pipelines.end(), std::mem_fn(&DrmPipeline::atomicTestSuccessful));
157 std::for_each(unusedObjects.begin(), unusedObjects.end(), std::mem_fn(&DrmObject::commitPending));
158 return Error::None;
159 }
160 case CommitMode::CommitModeset: {
161 // The kernel fails commits with DRM_MODE_PAGE_FLIP_EVENT when a crtc is disabled in the commit
162 // and already was disabled before, to work around some quirks in old userspace.
163 // Instead of using DRM_MODE_PAGE_FLIP_EVENT | DRM_MODE_ATOMIC_NONBLOCK, do the modeset in a blocking
164 // fashion without page flip events and directly call the pageFlipped method afterwards
165 bool commit = drmModeAtomicCommit(gpu->fd(), req.get(), DRM_MODE_ATOMIC_ALLOW_MODESET, nullptr) == 0;
166 if (!commit) {
167 qCCritical(KWIN_DRM) << "Atomic modeset commit failed!" << strerror(errno);
168 failed();
169 return errnoToError();
170 }
171 std::for_each(pipelines.begin(), pipelines.end(), std::mem_fn(&DrmPipeline::atomicModesetSuccessful));
172 for (const auto &obj : unusedObjects) {
173 obj->commitPending();
174 obj->commit();
175 if (auto crtc = dynamic_cast<DrmCrtc *>(obj)) {
176 crtc->flipBuffer();
177 } else if (auto plane = dynamic_cast<DrmPlane *>(obj)) {
178 plane->flipBuffer();
179 }
180 }
181 return Error::None;
182 }
183 case CommitMode::Test: {
184 bool test = drmModeAtomicCommit(pipelines[0]->gpu()->fd(), req.get(), DRM_MODE_ATOMIC_TEST_ONLY, nullptr) == 0;
185 if (!test) {
186 qCDebug(KWIN_DRM) << "Atomic test failed!" << strerror(errno);
187 failed();
188 return errnoToError();
189 }
190 std::for_each(pipelines.begin(), pipelines.end(), std::mem_fn(&DrmPipeline::atomicTestSuccessful));
191 Q_ASSERT(unusedObjects.isEmpty());
192 return Error::None;
193 }
194 case CommitMode::Commit: {
195 bool commit = drmModeAtomicCommit(pipelines[0]->gpu()->fd(), req.get(), DRM_MODE_ATOMIC_NONBLOCK | DRM_MODE_PAGE_FLIP_EVENT, gpu) == 0;
196 if (!commit) {
197 qCCritical(KWIN_DRM) << "Atomic commit failed!" << strerror(errno);
198 failed();
199 return errnoToError();
200 }
201 std::for_each(pipelines.begin(), pipelines.end(), std::mem_fn(&DrmPipeline::atomicCommitSuccessful));
202 Q_ASSERT(unusedObjects.isEmpty());
203 return Error::None;
204 }
205 default:
206 Q_UNREACHABLE();
207 }
208}

219static QRect centerBuffer(const QSize &bufferSize, const QSize &modeSize)
220{
221 const double widthScale = bufferSize.width() / double(modeSize.width());
222 const double heightScale = bufferSize.height() / double(modeSize.height());
223 if (widthScale > heightScale) {
224 const QSize size = bufferSize / widthScale;
225 const uint32_t yOffset = (modeSize.height() - size.height()) / 2;
226 return QRect(QPoint(0, yOffset), size);
227 } else {
228 const QSize size = bufferSize / heightScale;
229 const uint32_t xOffset = (modeSize.width() - size.width()) / 2;
230 return QRect(QPoint(xOffset, 0), size);
231 }

234void DrmPipeline::prepareAtomicPresentation()
235{
236 if (const auto contentType = m_connector->getProp(DrmConnector::PropertyIndex::ContentType)) {
237 contentType->setEnum(m_pending.contentType);
238 }
239
240 m_pending.crtc->setPending(DrmCrtc::PropertyIndex::VrrEnabled, m_pending.syncMode == RenderLoopPrivate::SyncMode::Adaptive || m_pending.syncMode == RenderLoopPrivate::SyncMode::AdaptiveAsync);
241 m_pending.crtc->setPending(DrmCrtc::PropertyIndex::Gamma_LUT, m_pending.gamma ? m_pending.gamma->blobId() : 0);
242 const auto fb = m_pending.layer->currentBuffer().get();
243 m_pending.crtc->primaryPlane()->set(QPoint(0, 0), fb->buffer()->size(), centerBuffer(orientateSize(fb->buffer()->size(), m_pending.bufferOrientation), m_pending.mode->size()));
244 m_pending.crtc->primaryPlane()->setBuffer(fb);
245
246 if (m_pending.crtc->cursorPlane()) {
247 const auto layer = cursorLayer();
248 m_pending.crtc->cursorPlane()->set(QPoint(0, 0), gpu()->cursorSize(), QRect(layer->position(), gpu()->cursorSize()));
249 m_pending.crtc->cursorPlane()->setBuffer(layer->isVisible() ? layer->currentBuffer().get() : nullptr);
250 m_pending.crtc->cursorPlane()->setPending(DrmPlane::PropertyIndex::CrtcId, layer->isVisible() ? m_pending.crtc->id() : 0);
251 }
252}

266void DrmPipeline::prepareAtomicModeset()
267{
268 m_connector->setPending(DrmConnector::PropertyIndex::CrtcId, m_pending.crtc->id());
269 if (const auto &prop = m_connector->getProp(DrmConnector::PropertyIndex::Broadcast_RGB)) {
270 prop->setEnum(m_pending.rgbRange);
271 }
272 if (const auto &prop = m_connector->getProp(DrmConnector::PropertyIndex::LinkStatus)) {
273 prop->setEnum(DrmConnector::LinkStatus::Good);
274 }
275 if (const auto overscan = m_connector->getProp(DrmConnector::PropertyIndex::Overscan)) {
276 overscan->setPending(m_pending.overscan);
277 } else if (const auto underscan = m_connector->getProp(DrmConnector::PropertyIndex::Underscan)) {
278 const uint32_t hborder = calculateUnderscan();
279 underscan->setEnum(m_pending.overscan != 0 ? DrmConnector::UnderscanOptions::On : DrmConnector::UnderscanOptions::Off);
280 m_connector->getProp(DrmConnector::PropertyIndex::Underscan_vborder)->setPending(m_pending.overscan);
281 m_connector->getProp(DrmConnector::PropertyIndex::Underscan_hborder)->setPending(hborder);
282 }
283 if (const auto bpc = m_connector->getProp(DrmConnector::PropertyIndex::MaxBpc)) {
284 uint64_t preferred = 8;
285 if (auto backend = dynamic_cast<EglGbmBackend *>(gpu()->platform()->renderBackend()); backend && backend->prefer10bpc()) {
286 preferred = 10;
287 }
288 bpc->setPending(std::min(bpc->maxValue(), preferred));
289 }
290 if (const auto hdr = m_connector->getProp(DrmConnector::PropertyIndex::HdrMetadata)) {
291 hdr->setPending(0);
292 }
293
294 m_pending.crtc->setPending(DrmCrtc::PropertyIndex::Active, 1);
295 m_pending.crtc->setPending(DrmCrtc::PropertyIndex::ModeId, m_pending.mode->blobId());
296
297 m_pending.crtc->primaryPlane()->setPending(DrmPlane::PropertyIndex::CrtcId, m_pending.crtc->id());
298 if (const auto rotation = m_pending.crtc->primaryPlane()->getProp(DrmPlane::PropertyIndex::Rotation)) {
299 rotation->setEnum(m_pending.bufferOrientation);
300 }
301 if (m_pending.crtc->cursorPlane()) {
302 if (const auto rotation = m_pending.crtc->cursorPlane()->getProp(DrmPlane::PropertyIndex::Rotation)) {
303 rotation->setEnum(DrmPlane::Transformation::Rotate0);
304 }
305 }
306}

308bool DrmPipeline::populateAtomicValues(drmModeAtomicReq *req)
309{
310 if (!m_connector->atomicPopulate(req)) {
311 return false;
312 }
313 if (m_pending.crtc) {
314 if (!m_pending.crtc->atomicPopulate(req)) {
315 return false;
316 }
317 if (!m_pending.crtc->primaryPlane()->atomicPopulate(req)) {
318 return false;
319 }
320 if (m_pending.crtc->cursorPlane() && !m_pending.crtc->cursorPlane()->atomicPopulate(req)) {
321 return false;
322 }
323 }
324 return true;
325}

339uint32_t DrmPipeline::calculateUnderscan()
340{
341 const auto size = m_pending.mode->size();
342 const float aspectRatio = size.width() / static_cast<float>(size.height());
343 uint32_t hborder = m_pending.overscan * aspectRatio;
344 if (hborder > 128) {
345 // overscan only goes from 0-100 so we cut off the 101-128 value range of underscan_vborder
346 hborder = 128;
347 m_pending.overscan = 128 / aspectRatio;
348 }
349 return hborder;
350}

392void DrmPipeline::atomicCommitSuccessful()
393{
394 atomicTestSuccessful();
395 m_pending.needsModeset = false;
396 if (activePending()) {
397 m_pageflipPending = true;
398 }
399 m_connector->commit();
400 if (m_pending.crtc) {
401 m_pending.crtc->commit();
402 m_pending.crtc->primaryPlane()->setNext(m_pending.layer->currentBuffer());
403 m_pending.crtc->primaryPlane()->commit();
404 if (m_pending.crtc->cursorPlane()) {
405 m_pending.crtc->cursorPlane()->setNext(cursorLayer()->currentBuffer());
406 m_pending.crtc->cursorPlane()->commit();
407 }
408 }
409 m_current = m_pending;
410}

421bool DrmPipeline::setCursor(const QPoint &hotspot)
422{
423 bool result;
424 m_pending.cursorHotspot = hotspot;
425 // explicitly check for the cursor plane and not for AMS, as we might not always have one
426 if (m_pending.crtc->cursorPlane()) {
427 result = commitPipelines({this}, CommitMode::Test) == Error::None;
428 if (result && m_output) {
429 m_output->renderLoop()->scheduleRepaint();
430 }
431 } else {
432 result = setCursorLegacy();
433 }
434 if (result) {
435 m_next = m_pending;
436 } else {
437 m_pending = m_next;
438 }
439 return result;
440}

525bool DrmPipeline::pruneModifier()
526{
527 if (!m_pending.layer->currentBuffer()
528 || m_pending.layer->currentBuffer()->buffer()->modifier() == DRM_FORMAT_MOD_NONE
529 || m_pending.layer->currentBuffer()->buffer()->modifier() == DRM_FORMAT_MOD_INVALID) {
530 return false;
531 }
532 auto &modifiers = m_pending.formats[m_pending.layer->currentBuffer()->buffer()->format()];
533 if (modifiers.empty()) {
534 return false;
535 } else {
536 modifiers.clear();
537 return true;
538 }

576DrmGammaRamp::DrmGammaRamp(DrmCrtc *crtc, const std::shared_ptr<ColorTransformation> &transformation)
577 : m_gpu(crtc->gpu())
578 , m_lut(transformation, crtc->gammaRampSize())
579{
580 if (crtc->gpu()->atomicModeSetting()) {
581 QVector<drm_color_lut> atomicLut(m_lut.size());
582 for (uint32_t i = 0; i < m_lut.size(); i++) {
583 atomicLut[i].red = m_lut.red()[i];
584 atomicLut[i].green = m_lut.green()[i];
585 atomicLut[i].blue = m_lut.blue()[i];
586 }
587 if (drmModeCreatePropertyBlob(crtc->gpu()->fd(), atomicLut.data(), sizeof(drm_color_lut) * m_lut.size(), &m_blobId) != 0) {
588 qCWarning(KWIN_DRM) << "Failed to create gamma blob!" << strerror(errno);
589 }
590 }
591}

595 if (m_blobId != 0) {
596 drmModeDestroyPropertyBlob(m_gpu->fd(), m_blobId);
