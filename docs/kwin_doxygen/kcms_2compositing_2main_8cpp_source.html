104void KWinCompositingKCM::init()
105{
106 auto currentIndexChangedSignal = static_cast<void (QComboBox::*)(int)>(&QComboBox::currentIndexChanged);
107
108 // animation speed
109 m_form.animationDurationFactor->setMaximum(s_animationMultipliers.size() - 1);
110 connect(m_form.animationDurationFactor, &QSlider::valueChanged, this, [this]() {
111 updateUnmanagedItemStatus();
112 m_settings->setAnimationDurationFactor(s_animationMultipliers[m_form.animationDurationFactor->value()]);
113 });
114
115 if (isRunningPlasma()) {
116 m_form.animationSpeedLabel->hide();
117 m_form.animationSpeedControls->hide();
118 }
119
120 // windowThumbnail
121 connect(m_form.kcfg_HiddenPreviews, currentIndexChangedSignal, this, [this](int index) {
122 if (index == 2) {
123 m_form.windowThumbnailWarning->animatedShow();
124 } else {
125 m_form.windowThumbnailWarning->animatedHide();
126 }
127 });
128
129 if (m_settings->lastFailureTimestamp() > 0) {
130 m_form.glCrashedWarning->animatedShow();
131 }
132}

134void KWinCompositingKCM::updateUnmanagedItemStatus()
135{
136 const auto animationDuration = s_animationMultipliers[m_form.animationDurationFactor->value()];
137
138 const bool inPlasma = isRunningPlasma();
139
140 bool changed = false;
141 if (!inPlasma) {
142 changed |= (animationDuration != m_settings->animationDurationFactor());
143 }
144 unmanagedWidgetChangeState(changed);
145
146 bool defaulted = true;
147 if (!inPlasma) {
148 defaulted &= animationDuration == m_settings->defaultAnimationDurationFactorValue();
149 }
150
151 unmanagedWidgetDefaultState(defaulted);
152}

154void KWinCompositingKCM::load()
155{
156 KCModule::load();
157
158 // unmanaged items
159 m_settings->findItem("AnimationDurationFactor")->readConfig(m_settings->config());
160 const double multiplier = m_settings->animationDurationFactor();
161 auto const it = std::lower_bound(s_animationMultipliers.begin(), s_animationMultipliers.end(), multiplier, std::greater<qreal>());
162 const int index = static_cast<int>(std::distance(s_animationMultipliers.begin(), it));
163 m_form.animationDurationFactor->setValue(index);
164 m_form.animationDurationFactor->setDisabled(m_settings->isAnimationDurationFactorImmutable());
165}

178void KWinCompositingKCM::save()
179{
180 if (!isRunningPlasma()) {
181 const auto animationDuration = s_animationMultipliers[m_form.animationDurationFactor->value()];
182 m_settings->setAnimationDurationFactor(animationDuration);
183 }
184 m_settings->save();
185
186 KCModule::save();
187
188 // This clears up old entries that are now migrated to kdeglobals
189 KConfig("kwinrc", KConfig::NoGlobals).group("KDE").revertToDefault("AnimationDurationFactor");
190
191 // Send signal to all kwin instances
192 QDBusMessage message = QDBusMessage::createSignal(QStringLiteral("/Compositor"),
193 QStringLiteral("org.kde.kwin.Compositing"),
194 QStringLiteral("reinit"));
195 QDBusConnection::sessionBus().send(message);
196}
