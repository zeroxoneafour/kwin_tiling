46static void convertFromGLImage(QImage &img, int w, int h)
47{
48 // from QtOpenGL/qgl.cpp
49 // SPDX-FileCopyrightText: 2010 Nokia Corporation and /or its subsidiary(-ies)
50 // see https://github.com/qt/qtbase/blob/dev/src/opengl/qgl.cpp
51 if (QSysInfo::ByteOrder == QSysInfo::BigEndian) {
52 // OpenGL gives RGBA; Qt wants ARGB
53 uint *p = reinterpret_cast<uint *>(img.bits());
54 uint *end = p + w * h;
55 while (p < end) {
56 uint a = *p << 24;
57 *p = (*p >> 8) | a;
58 p++;
59 }
60 } else {
61 // OpenGL gives ABGR (i.e. RGBA backwards); Qt wants ARGB
62 for (int y = 0; y < h; y++) {
63 uint *q = reinterpret_cast<uint *>(img.scanLine(y));
64 for (int x = 0; x < w; ++x) {
65 const uint pixel = *q;
66 *q = ((pixel << 16) & 0xff0000) | ((pixel >> 16) & 0xff)
67 | (pixel & 0xff00ff00);
68
69 q++;
70 }
71 }
72 }
73 img = img.mirrored();
74}

116QFuture<QImage> ScreenShotEffect::scheduleScreenShot(const QRect &area, ScreenShotFlags flags)
117{
118 for (ScreenShotAreaData &data : m_areaScreenShots) {
119 if (data.area == area && data.flags == flags) {
120 return data.promise.future();
121 }
122 }
123
124 ScreenShotAreaData data;
125 data.area = area;
126 data.flags = flags;
127
128 const QList<EffectScreen *> screens = effects->screens();
129 for (EffectScreen *screen : screens) {
130 if (screen->geometry().intersects(area)) {
131 data.screens.append(screen);
132 }
133 }
134
135 qreal devicePixelRatio = 1.0;
136 if (flags & ScreenShotNativeResolution) {
137 for (const EffectScreen *screen : std::as_const(data.screens)) {
138 if (screen->devicePixelRatio() > devicePixelRatio) {
139 devicePixelRatio = screen->devicePixelRatio();
140 }
141 }
142 }
143
144 data.result = QImage(area.size() * devicePixelRatio, QImage::Format_ARGB32_Premultiplied);
145 data.result.fill(Qt::transparent);
146 data.result.setDevicePixelRatio(devicePixelRatio);
147
148 m_areaScreenShots.append(data);
149 effects->addRepaint(area);
150
151 data.promise.reportStarted();
152 return data.promise.future();
153}

221void ScreenShotEffect::takeScreenShot(ScreenShotWindowData *screenshot)
222{
223 EffectWindow *window = screenshot->window;
224
225 WindowPaintData d;
226 QRectF geometry = window->expandedGeometry();
227 qreal devicePixelRatio = 1;
228 if (window->hasDecoration() && !(screenshot->flags & ScreenShotIncludeDecoration)) {
229 geometry = window->clientGeometry();
230 }
231 if (screenshot->flags & ScreenShotNativeResolution) {
232 if (const EffectScreen *screen = window->screen()) {
233 devicePixelRatio = screen->devicePixelRatio();
234 }
235 }
236 bool validTarget = true;
237 std::unique_ptr<GLTexture> offscreenTexture;
238 std::unique_ptr<GLFramebuffer> target;
239 if (effects->isOpenGLCompositing()) {
240 offscreenTexture.reset(new GLTexture(GL_RGBA8, QSizeF(geometry.size() * devicePixelRatio).toSize()));
241 offscreenTexture->setFilter(GL_LINEAR);
242 offscreenTexture->setWrapMode(GL_CLAMP_TO_EDGE);
243 target.reset(new GLFramebuffer(offscreenTexture.get()));
244 validTarget = target->valid();
245 }
246 if (validTarget) {
247 d.setXTranslation(-geometry.x());
248 d.setYTranslation(-geometry.y());
249 d.setRenderTargetScale(devicePixelRatio);
250
251 // render window into offscreen texture
252 int mask = PAINT_WINDOW_TRANSFORMED | PAINT_WINDOW_TRANSLUCENT;
253 QImage img;
254 if (effects->isOpenGLCompositing()) {
255 GLFramebuffer::pushFramebuffer(target.get());
256 glClearColor(0.0, 0.0, 0.0, 0.0);
257 glClear(GL_COLOR_BUFFER_BIT);
258 glClearColor(0.0, 0.0, 0.0, 1.0);
259
260 QMatrix4x4 projection;
261 projection.ortho(QRect(0, 0, geometry.width() * devicePixelRatio, geometry.height() * devicePixelRatio));
262 d.setProjectionMatrix(projection);
263
264 effects->drawWindow(window, mask, infiniteRegion(), d);
265
266 // copy content from framebuffer into image
267 img = QImage(offscreenTexture->size(), QImage::Format_ARGB32);
268 img.setDevicePixelRatio(devicePixelRatio);
269 glReadnPixels(0, 0, img.width(), img.height(), GL_RGBA, GL_UNSIGNED_BYTE, img.sizeInBytes(),
270 static_cast<GLvoid *>(img.bits()));
271 GLFramebuffer::popFramebuffer();
272 convertFromGLImage(img, img.width(), img.height());
273 }
274
275 if (screenshot->flags & ScreenShotIncludeCursor) {
276 grabPointerImage(img, geometry.x(), geometry.y());
277 }
278
279 screenshot->promise.reportResult(img);
280 screenshot->promise.reportFinished();
281 } else {
282 screenshot->promise.reportCanceled();
283 }
284}

286bool ScreenShotEffect::takeScreenShot(ScreenShotAreaData *screenshot)
287{
288 if (!effects->waylandDisplay()) {
289 // On X11, all screens are painted simultaneously and there is no native HiDPI support.
290 QImage snapshot = blitScreenshot(screenshot->area);
291 if (screenshot->flags & ScreenShotIncludeCursor) {
292 grabPointerImage(snapshot, screenshot->area.x(), screenshot->area.y());
293 }
294 screenshot->promise.reportResult(snapshot);
295 screenshot->promise.reportFinished();
296 } else {
297 if (!screenshot->screens.contains(m_paintedScreen)) {
298 return false;
299 }
300 screenshot->screens.removeOne(m_paintedScreen);
301
302 const QRect sourceRect = screenshot->area & m_paintedScreen->geometry();
303 qreal sourceDevicePixelRatio = 1.0;
304 if (screenshot->flags & ScreenShotNativeResolution) {
305 sourceDevicePixelRatio = m_paintedScreen->devicePixelRatio();
306 }
307
308 const QImage snapshot = blitScreenshot(sourceRect, sourceDevicePixelRatio);
309 const QRect nativeArea(screenshot->area.topLeft(),
310 screenshot->area.size() * screenshot->result.devicePixelRatio());
311
312 QPainter painter(&screenshot->result);
313 painter.setWindow(nativeArea);
314 painter.drawImage(sourceRect, snapshot);
315 painter.end();
316
317 if (screenshot->screens.isEmpty()) {
318 if (screenshot->flags & ScreenShotIncludeCursor) {
319 grabPointerImage(screenshot->result, screenshot->area.x(), screenshot->area.y());
320 }
321 screenshot->promise.reportResult(screenshot->result);
322 screenshot->promise.reportFinished();
323 }
324 }
325
326 return screenshot->promise.isFinished();
327}

329bool ScreenShotEffect::takeScreenShot(ScreenShotScreenData *screenshot)
330{
331 if (!m_paintedScreen || screenshot->screen == m_paintedScreen) {
332 qreal devicePixelRatio = 1.0;
333 if (screenshot->flags & ScreenShotNativeResolution) {
334 devicePixelRatio = screenshot->screen->devicePixelRatio();
335 }
336
337 QImage snapshot = blitScreenshot(screenshot->screen->geometry(), devicePixelRatio);
338 if (screenshot->flags & ScreenShotIncludeCursor) {
339 const int xOffset = screenshot->screen->geometry().x();
340 const int yOffset = screenshot->screen->geometry().y();
341 grabPointerImage(snapshot, xOffset, yOffset);
342 }
343
344 screenshot->promise.reportResult(snapshot);
345 screenshot->promise.reportFinished();
346 }
347
348 return screenshot->promise.isFinished();

351QImage ScreenShotEffect::blitScreenshot(const QRect &geometry, qreal devicePixelRatio) const
352{
353 QImage image;
354
355 if (effects->isOpenGLCompositing()) {
356 const QSize nativeSize = geometry.size() * devicePixelRatio;
357
358 if (GLFramebuffer::blitSupported() && !GLPlatform::instance()->isGLES()) {
359 image = QImage(nativeSize.width(), nativeSize.height(), QImage::Format_ARGB32);
360 GLTexture texture(GL_RGBA8, nativeSize.width(), nativeSize.height());
361 GLFramebuffer target(&texture);
362 target.blitFromFramebuffer(effects->mapToRenderTarget(QRectF(geometry)).toRect());
363 // copy content from framebuffer into image
364 texture.bind();
365 glGetTexImage(GL_TEXTURE_2D, 0, GL_RGBA, GL_UNSIGNED_BYTE,
366 static_cast<GLvoid *>(image.bits()));
367 texture.unbind();
368 } else {
369 image = QImage(nativeSize.width(), nativeSize.height(), QImage::Format_ARGB32);
370 glReadPixels(0, 0, nativeSize.width(), nativeSize.height(), GL_RGBA,
371 GL_UNSIGNED_BYTE, static_cast<GLvoid *>(image.bits()));
372 }
373 convertFromGLImage(image, nativeSize.width(), nativeSize.height());
374 }
375
376 image.setDevicePixelRatio(devicePixelRatio);
377 return image;
378}

380void ScreenShotEffect::grabPointerImage(QImage &snapshot, int xOffset, int yOffset) const
381{
382 if (effects->isCursorHidden()) {
383 return;
384 }
385
386 const PlatformCursorImage cursor = effects->cursorImage();
387 if (cursor.image().isNull()) {
388 return;
389 }
390
391 QPainter painter(&snapshot);
392 painter.setRenderHint(QPainter::SmoothPixmapTransform);
393 painter.drawImage(effects->cursorPos() - cursor.hotSpot() - QPoint(xOffset, yOffset), cursor.image());
394}
