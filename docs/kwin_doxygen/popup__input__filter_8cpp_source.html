26void PopupInputFilter::handleWindowAdded(Window *window)
27{
28 if (m_popupWindows.contains(window)) {
29 return;
30 }
31 if (window->hasPopupGrab()) {
32 // TODO: verify that the Window is allowed as a popup

44bool PopupInputFilter::pointerEvent(MouseEvent *event, quint32 nativeButton)
45{
46 if (m_popupWindows.isEmpty()) {
47 return false;
48 }
49 if (event->type() == QMouseEvent::MouseButtonPress) {
50 auto pointerFocus = input()->findToplevel(event->globalPos());
51 if (!pointerFocus || !Window::belongToSameApplication(pointerFocus, m_popupWindows.constLast())) {
52 // a press on a window (or no window) not belonging to the popup window
53 cancelPopups();
54 // filter out this press
55 return true;
56 }
57 if (pointerFocus && pointerFocus->isDecorated()) {
58 // test whether it is on the decoration
59 if (!exclusiveContains(pointerFocus->clientGeometry(), event->globalPos())) {
60 cancelPopups();
61 return true;
62 }
63 }
64 }
65 return false;
66}

68bool PopupInputFilter::keyEvent(KeyEvent *event)
69{
70 if (m_popupWindows.isEmpty()) {
71 return false;
72 }
73
74 auto seat = waylandServer()->seat();
75
76 auto last = m_popupWindows.last();
77 if (last->surface() == nullptr) {
78 return false;
79 }
80
81 seat->setFocusedKeyboardSurface(last->surface());
82
83 if (!passToInputMethod(event)) {
84 passToWaylandServer(event);
85 }
86
87 return true;
88}
