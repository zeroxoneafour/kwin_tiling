25bool EffectsKeyboardInterceptionX11Filter::event(xcb_generic_event_t *event)
26{
27 switch (event->response_type & ~0x80) {
28 case XCB_KEY_PRESS: {
29 const auto keyEvent = reinterpret_cast<xcb_key_press_event_t *>(event);
30 processKey(true, keyEvent->detail, keyEvent->time);
31 return true;
32 }
33 case XCB_KEY_RELEASE: {
34 const auto keyEvent = reinterpret_cast<xcb_key_release_event_t *>(event);
35 processKey(false, keyEvent->detail, keyEvent->time);
36 return true;
37 }
38 default:
39 return false;
40 }
41}

43void EffectsKeyboardInterceptionX11Filter::processKey(bool press, xcb_keycode_t keycode, xcb_timestamp_t timestamp)
44{
45 const xkb_keysym_t keysym = xkb_state_key_get_one_sym(m_keyboard->xkbState(), keycode);
46
47 Qt::KeyboardModifiers modifiers = m_keyboard->modifiers();
48 if (QXkbCommon::isKeypad(keysym)) {
49 modifiers |= Qt::KeypadModifier;
50 }
51
52 const int qtKey = QXkbCommon::keysymToQtKey(keysym, modifiers, m_keyboard->xkbState(), keycode);
53 const QString text = QXkbCommon::lookupString(m_keyboard->xkbState(), keycode);
54
55 QKeyEvent keyEvent(press ? QEvent::KeyPress : QEvent::KeyRelease, qtKey, modifiers, text);
56 keyEvent.setTimestamp(timestamp);
57
58 m_effects->grabbedKeyboardEvent(&keyEvent);
59}
