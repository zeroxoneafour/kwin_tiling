29bool EffectsKeyboardInterceptionX11Filter::event(xcb_generic_event_t *event)
30{
31 switch (event->response_type & ~0x80) {
32 case XCB_KEY_PRESS: {
33 const auto keyEvent = reinterpret_cast<xcb_key_press_event_t *>(event);
34 processKey(true, keyEvent->detail, keyEvent->time);
35 return true;
36 }
37 case XCB_KEY_RELEASE: {
38 const auto keyEvent = reinterpret_cast<xcb_key_release_event_t *>(event);
39 processKey(false, keyEvent->detail, keyEvent->time);
40 return true;
41 }
42 default:
43 return false;
44 }
45}

47void EffectsKeyboardInterceptionX11Filter::processKey(bool press, xcb_keycode_t keycode, xcb_timestamp_t timestamp)
48{
49 const xkb_keysym_t keysym = xkb_state_key_get_one_sym(m_keyboard->xkbState(), keycode);
50
51 Qt::KeyboardModifiers modifiers = m_keyboard->modifiers();
52 if (QXkbCommon::isKeypad(keysym)) {
53 modifiers |= Qt::KeypadModifier;
54 }
55
56 const int qtKey = QXkbCommon::keysymToQtKey(keysym, modifiers, m_keyboard->xkbState(), keycode);
57 const QString text = QXkbCommon::lookupString(m_keyboard->xkbState(), keycode);
58
59 QKeyEvent keyEvent(press ? QEvent::KeyPress : QEvent::KeyRelease, qtKey, modifiers, text);
60 keyEvent.setTimestamp(timestamp);
61
62 m_effects->grabbedKeyboardEvent(&keyEvent);
63}
