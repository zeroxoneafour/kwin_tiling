35static quint64 refreshRateForMode(_drmModeModeInfo *m)
36{
37 // Calculate higher precision (mHz) refresh rate
38 // logic based on Weston, see compositor-drm.c
39 quint64 refreshRate = (m->clock * 1000000LL / m->htotal + m->vtotal / 2) / m->vtotal;
40 if (m->flags & DRM_MODE_FLAG_INTERLACE) {
41 refreshRate *= 2;
42 }
43 if (m->flags & DRM_MODE_FLAG_DBLSCAN) {
44 refreshRate /= 2;
45 }
46 if (m->vscan > 1) {
47 refreshRate /= m->vscan;
48 }
49 return refreshRate;
50}

241bool DrmConnector::updateProperties()
242{
243 if (auto connector = drmModeGetConnector(gpu()->fd(), id())) {
244 m_conn.reset(connector);
245 } else if (!m_conn) {
246 return false;
247 }
248 if (!DrmObject::updateProperties()) {
249 return false;
250 }
251 if (const auto &dpms = getProp(PropertyIndex::Dpms)) {
252 dpms->setLegacy();
253 }
254
255 auto &underscan = m_props[static_cast<uint32_t>(PropertyIndex::Underscan)];
256 auto &vborder = m_props[static_cast<uint32_t>(PropertyIndex::Underscan_vborder)];
257 auto &hborder = m_props[static_cast<uint32_t>(PropertyIndex::Underscan_hborder)];
258 if (underscan && vborder && hborder) {
259 underscan->setEnum(vborder->current() > 0 ? UnderscanOptions::On : UnderscanOptions::Off);
260 } else {
261 underscan.reset();
262 vborder.reset();
263 hborder.reset();
264 }
265
266 // parse edid
267 if (const auto edidProp = getProp(PropertyIndex::Edid); edidProp && edidProp->immutableBlob()) {
268 m_edid = Edid(edidProp->immutableBlob()->data, edidProp->immutableBlob()->length);
269 if (!m_edid.isValid()) {
270 qCWarning(KWIN_DRM) << "Couldn't parse EDID for connector" << this;
271 }
272 } else if (m_conn->connection == DRM_MODE_CONNECTED) {
273 qCDebug(KWIN_DRM) << "Could not find edid for connector" << this;
274 }
275
276 // check the physical size
277 if (m_edid.physicalSize().isEmpty()) {
278 m_physicalSize = QSize(m_conn->mmWidth, m_conn->mmHeight);
279 } else {
280 m_physicalSize = m_edid.physicalSize();
281 }
282
283 // update modes
284 bool equal = m_conn->count_modes == m_driverModes.count();
285 for (int i = 0; equal && i < m_conn->count_modes; i++) {
286 equal &= checkIfEqual(m_driverModes[i]->nativeMode(), &m_conn->modes[i]);
287 }
288 if (!equal && m_conn->count_modes > 0) {
289 // reload modes
290 m_driverModes.clear();
291 for (int i = 0; i < m_conn->count_modes; i++) {
292 m_driverModes.append(std::make_shared<DrmConnectorMode>(this, m_conn->modes[i]));
293 }
294 m_modes.clear();
295 m_modes.append(m_driverModes);
296 m_modes.append(generateCommonModes());
297 if (m_pipeline->mode()) {
298 if (const auto mode = findMode(*m_pipeline->mode()->nativeMode())) {
299 m_pipeline->setMode(mode);
300 } else {
301 m_pipeline->setMode(m_modes.constFirst());
302 }
303 } else {
304 m_pipeline->setMode(m_modes.constFirst());
305 }
306 m_pipeline->applyPendingChanges();
307 if (m_pipeline->output()) {
308 m_pipeline->output()->updateModes();
309 }
310 }
311
312 return true;
313}

370QList<std::shared_ptr<DrmConnectorMode>> DrmConnector::generateCommonModes()
371{
372 QList<std::shared_ptr<DrmConnectorMode>> ret;
373 QSize maxSize;
374 uint32_t maxSizeRefreshRate = 0;
375 for (const auto &mode : std::as_const(m_driverModes)) {
376 if (mode->size().width() >= maxSize.width() && mode->size().height() >= maxSize.height() && mode->refreshRate() >= maxSizeRefreshRate) {
377 maxSize = mode->size();
378 maxSizeRefreshRate = mode->refreshRate();
379 }
380 }
381 const uint64_t maxBandwidthEstimation = maxSize.width() * maxSize.height() * uint64_t(maxSizeRefreshRate);
382 for (const auto &size : s_commonModes) {
383 const uint64_t bandwidthEstimation = size.width() * size.height() * 60000ull;
384 if (size.width() > maxSize.width() || size.height() > maxSize.height() || bandwidthEstimation > maxBandwidthEstimation) {
385 continue;
386 }
387 const auto generatedMode = generateMode(size, 60);
388 if (std::any_of(m_driverModes.cbegin(), m_driverModes.cend(), [generatedMode](const auto &mode) {
389 return mode->size() == generatedMode->size() && mode->refreshRate() == generatedMode->refreshRate();
390 })) {
391 continue;
392 }
393 ret << generatedMode;
394 }
395 return ret;
396}

398std::shared_ptr<DrmConnectorMode> DrmConnector::generateMode(const QSize &size, float refreshRate)
399{
400 auto modeInfo = libxcvt_gen_mode_info(size.width(), size.height(), refreshRate, false, false);
401
402 drmModeModeInfo mode{
403 .clock = uint32_t(modeInfo->dot_clock),
404 .hdisplay = uint16_t(modeInfo->hdisplay),
405 .hsync_start = modeInfo->hsync_start,
406 .hsync_end = modeInfo->hsync_end,
407 .htotal = modeInfo->htotal,
408 .vdisplay = uint16_t(modeInfo->vdisplay),
409 .vsync_start = modeInfo->vsync_start,
410 .vsync_end = modeInfo->vsync_end,
411 .vtotal = modeInfo->vtotal,
412 .vscan = 1,
413 .vrefresh = uint32_t(modeInfo->vrefresh),
414 .flags = modeInfo->mode_flags,
415 .type = DRM_MODE_TYPE_USERDEF,
416 };
417
418 sprintf(mode.name, "%dx%d@%d", size.width(), size.height(), mode.vrefresh);
419
420 free(modeInfo);
421 return std::make_shared<DrmConnectorMode>(this, mode);
422}
