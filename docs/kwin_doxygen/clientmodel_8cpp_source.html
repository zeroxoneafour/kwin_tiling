36QVariant ClientModel::data(const QModelIndex &index, int role) const
37{
38 if (!index.isValid()) {
39 return QVariant();
40 }
41
42 if (m_clientList.isEmpty()) {
43 return QVariant();
44 }
45
46 int clientIndex = index.row();
47 if (clientIndex >= m_clientList.count()) {
48 return QVariant();
49 }
50 QSharedPointer<TabBoxClient> client = m_clientList[clientIndex].toStrongRef();
51 if (!client) {
52 return QVariant();
53 }
54 switch (role) {
55 case Qt::DisplayRole:
56 case CaptionRole: {
57 QString caption = client->caption();
58 if (Qt::mightBeRichText(caption)) {
59 caption = caption.toHtmlEscaped();
60 }
61 return caption;
62 }
63 case ClientRole:
64 return QVariant::fromValue<void *>(client.data());
65 case DesktopNameRole: {
66 return tabBox->desktopName(client.data());
67 }
68 case WIdRole:
69 return client->internalId();
70 case MinimizedRole:
71 return client->isMinimized();
72 case CloseableRole:
73 // clients that claim to be first are not closeable
74 return client->isCloseable() && !client->isFirstInTabBox();
75 case IconRole:
76 return client->icon();
77 default:
78 return QVariant();
79 }
80}

82QString ClientModel::longestCaption() const
83{
84 QString caption;
85 for (const QWeakPointer<TabBoxClient> &clientPointer : std::as_const(m_clientList)) {
86 QSharedPointer<TabBoxClient> client = clientPointer.toStrongRef();
87 if (!client) {
88 continue;
89 }
90 if (client->caption().size() > caption.size()) {
91 caption = client->caption();

155void ClientModel::createFocusChainClientList(int desktop,
156 const QSharedPointer<TabBoxClient> &start, TabBoxClientList &stickyClients)
157{
158 auto c = start;
159 if (!tabBox->isInFocusChain(c.data())) {
160 QSharedPointer<TabBoxClient> firstClient = tabBox->firstClientFocusChain().toStrongRef();
161 if (firstClient) {
162 c = firstClient;
163 }
164 }
165 auto stop = c;
166 do {
167 QSharedPointer<TabBoxClient> add = tabBox->clientToAddToList(c.data(), desktop);
168 if (!add.isNull()) {
169 m_clientList += add;
170 if (add.data()->isFirstInTabBox()) {
171 stickyClients << add;
172 }
173 }
174 c = tabBox->nextClientFocusChain(c.data());
175 } while (c && c != stop);
176}

178void ClientModel::createStackingOrderClientList(int desktop,
179 const QSharedPointer<TabBoxClient> &start, TabBoxClientList &stickyClients)
180{
181 // TODO: needs improvement
182 const TabBoxClientList stacking = tabBox->stackingOrder();
183 auto c = stacking.first().toStrongRef();
184 auto stop = c;
185 int index = 0;
186 while (c) {
187 QSharedPointer<TabBoxClient> add = tabBox->clientToAddToList(c.data(), desktop);
188 if (!add.isNull()) {
189 if (start == add.data()) {
190 m_clientList.removeAll(add);
191 m_clientList.prepend(add);
192 } else {
193 m_clientList += add;
194 }
195 if (add.data()->isFirstInTabBox()) {
196 stickyClients << add;
197 }
198 }
199 if (index >= stacking.size() - 1) {
200 c = nullptr;
201 } else {
202 c = stacking[++index];
203 }
204
205 if (c == stop) {
206 break;
207 }
208 }
209}

211void ClientModel::createClientList(int desktop, bool partialReset)
212{
213 auto start = tabBox->activeClient().toStrongRef();
214 // TODO: new clients are not added at correct position
215 if (partialReset && !m_clientList.isEmpty()) {
216 QSharedPointer<TabBoxClient> firstClient = m_clientList.constFirst();
217 if (firstClient) {
218 start = firstClient;
219 }
220 }
221
222 beginResetModel();
223 m_clientList.clear();
224 TabBoxClientList stickyClients;
225
226 switch (tabBox->config().clientSwitchingMode()) {
227 case TabBoxConfig::FocusChainSwitching: {
228 createFocusChainClientList(desktop, start, stickyClients);
229 break;
230 }
231 case TabBoxConfig::StackingOrderSwitching: {
232 createStackingOrderClientList(desktop, start, stickyClients);
233 break;
234 }
235 }
236
237 if (tabBox->config().orderMinimizedMode() == TabBoxConfig::GroupByMinimized) {
238 // Put all non-minimized included clients first.
239 std::stable_partition(m_clientList.begin(), m_clientList.end(), [](const auto &client) {
240 return !client.toStrongRef()->isMinimized();
241 });
242 }
243
244 for (const QWeakPointer<TabBoxClient> &c : std::as_const(stickyClients)) {
245 m_clientList.removeAll(c);
246 m_clientList.prepend(c);
247 }
248 if (tabBox->config().clientApplicationsMode() != TabBoxConfig::AllWindowsCurrentApplication
249 && (tabBox->config().showDesktopMode() == TabBoxConfig::ShowDesktopClient || m_clientList.isEmpty())) {
250 QWeakPointer<TabBoxClient> desktopClient = tabBox->desktopClient();
251 if (!desktopClient.isNull()) {
252 m_clientList.append(desktopClient);
253 }
254 }
255 endResetModel();
256}
