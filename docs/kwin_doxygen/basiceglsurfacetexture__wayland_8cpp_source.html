36bool BasicEGLSurfaceTextureWayland::create()
37{
38 if (auto buffer = qobject_cast<KWaylandServer::LinuxDmaBufV1ClientBuffer *>(m_pixmap->buffer())) {
39 return loadDmabufTexture(buffer);

70bool BasicEGLSurfaceTextureWayland::loadShmTexture(KWaylandServer::ShmClientBuffer *buffer)
71{
72 const QImage &image = buffer->data();
73 if (Q_UNLIKELY(image.isNull())) {
74 return false;
75 }
76
77 m_texture.reset(new GLTexture(image));
78 m_texture->setFilter(GL_LINEAR);
79 m_texture->setWrapMode(GL_CLAMP_TO_EDGE);
80 m_texture->setYInverted(true);
81 m_bufferType = BufferType::Shm;
82
83 return true;
84}

86void BasicEGLSurfaceTextureWayland::updateShmTexture(KWaylandServer::ShmClientBuffer *buffer, const QRegion &region)
87{
88 if (Q_UNLIKELY(m_bufferType != BufferType::Shm)) {
89 destroy();
90 create();
91 return;
92 }
93
94 const QImage &image = buffer->data();
95 if (Q_UNLIKELY(image.isNull())) {
96 return;
97 }
98
99 const QRegion damage = mapRegion(m_pixmap->item()->surfaceToBufferMatrix(), region);
100 for (const QRect &rect : damage) {
101 m_texture->update(image, rect.topLeft(), rect);
102 }
103}

105bool BasicEGLSurfaceTextureWayland::loadEglTexture(KWaylandServer::DrmClientBuffer *buffer)
106{
107 const AbstractEglBackendFunctions *funcs = backend()->functions();
108 if (Q_UNLIKELY(!funcs->eglQueryWaylandBufferWL)) {
109 return false;
110 }
111 if (Q_UNLIKELY(!buffer->resource())) {
112 return false;
113 }
114
115 m_texture.reset(new GLTexture(GL_TEXTURE_2D));
116 m_texture->setSize(buffer->size());
117 m_texture->create();
118 m_texture->setWrapMode(GL_CLAMP_TO_EDGE);
119 m_texture->setFilter(GL_LINEAR);
120 m_texture->bind();
121 m_image = attach(buffer);
122 m_texture->unbind();
123 m_bufferType = BufferType::Egl;
124
125 if (EGL_NO_IMAGE_KHR == m_image) {
126 qCDebug(KWIN_OPENGL) << "failed to create egl image";
127 m_texture.reset();
128 return false;
129 }
130
131 return true;
132}

134void BasicEGLSurfaceTextureWayland::updateEglTexture(KWaylandServer::DrmClientBuffer *buffer)
135{
136 if (Q_UNLIKELY(m_bufferType != BufferType::Egl)) {
137 destroy();
138 create();
139 return;
140 }
141 if (Q_UNLIKELY(!buffer->resource())) {
142 return;
143 }
144
145 m_texture->bind();
146 EGLImageKHR image = attach(buffer);
147 m_texture->unbind();
148 if (image != EGL_NO_IMAGE_KHR) {
149 if (m_image != EGL_NO_IMAGE_KHR) {
150 eglDestroyImageKHR(backend()->eglDisplay(), m_image);
151 }
152 m_image = image;
153 }

156bool BasicEGLSurfaceTextureWayland::loadDmabufTexture(KWaylandServer::LinuxDmaBufV1ClientBuffer *buffer)
157{
158 auto dmabuf = static_cast<EglDmabufBuffer *>(buffer);
159 if (Q_UNLIKELY(dmabuf->images().constFirst() == EGL_NO_IMAGE_KHR)) {
160 qCritical(KWIN_OPENGL) << "Invalid dmabuf-based wl_buffer";
161 return false;
162 }
163
164 m_texture.reset(new GLTexture(GL_TEXTURE_2D));
165 m_texture->setSize(dmabuf->size());
166 m_texture->create();
167 m_texture->setWrapMode(GL_CLAMP_TO_EDGE);
168 m_texture->setFilter(GL_NEAREST);
169 m_texture->bind();
170 glEGLImageTargetTexture2DOES(GL_TEXTURE_2D, static_cast<GLeglImageOES>(dmabuf->images().constFirst()));
171 m_texture->unbind();
172 m_texture->setYInverted(dmabuf->origin() == KWaylandServer::ClientBuffer::Origin::TopLeft);
173 m_bufferType = BufferType::DmaBuf;
174
175 return true;
176}

178void BasicEGLSurfaceTextureWayland::updateDmabufTexture(KWaylandServer::LinuxDmaBufV1ClientBuffer *buffer)
179{
180 if (Q_UNLIKELY(m_bufferType != BufferType::DmaBuf)) {
181 destroy();
182 create();
183 return;
184 }
185
186 auto dmabuf = static_cast<EglDmabufBuffer *>(buffer);
187 m_texture->bind();
188 glEGLImageTargetTexture2DOES(GL_TEXTURE_2D, static_cast<GLeglImageOES>(dmabuf->images().constFirst()));
189 m_texture->unbind();
190 // The origin in a dmabuf-buffer is at the upper-left corner, so the meaning
191 // of Y-inverted is the inverse of OpenGL.
192 m_texture->setYInverted(dmabuf->origin() == KWaylandServer::ClientBuffer::Origin::TopLeft);
193}

195EGLImageKHR BasicEGLSurfaceTextureWayland::attach(KWaylandServer::DrmClientBuffer *buffer)
196{
197 if (buffer->textureFormat() != EGL_TEXTURE_RGB && buffer->textureFormat() != EGL_TEXTURE_RGBA) {
198 qCDebug(KWIN_OPENGL) << "Unsupported texture format: " << buffer->textureFormat();
199 return EGL_NO_IMAGE_KHR;
200 }
201
202 const EGLint attribs[] = {
203 EGL_WAYLAND_PLANE_WL, 0,
204 EGL_NONE};
205 EGLImageKHR image = eglCreateImageKHR(backend()->eglDisplay(), EGL_NO_CONTEXT,
206 EGL_WAYLAND_BUFFER_WL,
207 static_cast<EGLClientBuffer>(buffer->resource()), attribs);
208 if (image != EGL_NO_IMAGE_KHR) {
209 glEGLImageTargetTexture2DOES(GL_TEXTURE_2D, static_cast<GLeglImageOES>(image));
210 m_texture->setYInverted(buffer->origin() == KWaylandServer::ClientBuffer::Origin::TopLeft);
211 }
212 return image;
213}
