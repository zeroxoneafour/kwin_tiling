235void InputEventFilter::passToWaylandServer(QKeyEvent *event)
236{
237 Q_ASSERT(waylandServer());
238 if (event->isAutoRepeat()) {
239 return;
240 }
241
242 KWaylandServer::SeatInterface *seat = waylandServer()->seat();
243 const int keyCode = event->nativeScanCode();
244 switch (event->type()) {
245 case QEvent::KeyPress:
246 seat->notifyKeyboardKey(keyCode, KWaylandServer::KeyboardKeyState::Pressed);
247 break;
248 case QEvent::KeyRelease:
249 seat->notifyKeyboardKey(keyCode, KWaylandServer::KeyboardKeyState::Released);
250 break;
251 default:
252 break;
253 }
254}

256bool InputEventFilter::passToInputMethod(QKeyEvent *event)
257{
258 if (!kwinApp()->inputMethod()) {
259 return false;
260 }
261 if (auto keyboardGrab = kwinApp()->inputMethod()->keyboardGrab()) {
262 if (event->isAutoRepeat()) {
263 return true;
264 }
265 auto newState = event->type() == QEvent::KeyPress ? KWaylandServer::KeyboardKeyState::Pressed : KWaylandServer::KeyboardKeyState::Released;
266 keyboardGrab->sendKey(waylandServer()->display()->nextSerial(), event->timestamp(), event->nativeScanCode(), newState);
267 return true;
268 } else {
269 return false;
270 }

306class LockScreenFilter : public InputEventFilter
307{
308public:
309 bool pointerEvent(MouseEvent *event, quint32 nativeButton) override
310 {
311 if (!waylandServer()->isScreenLocked()) {
312 return false;
313 }
314
315 auto window = input()->findToplevel(event->globalPos());
316 if (window && window->isClient() && window->isLockScreen()) {
317 workspace()->activateWindow(window);
318 }
319
320 auto seat = waylandServer()->seat();
321 seat->setTimestamp(event->timestamp());
322 if (event->type() == QEvent::MouseMove) {
323 if (pointerSurfaceAllowed()) {
324 // TODO: should the pointer position always stay in sync, i.e. not do the check?
325 seat->notifyPointerMotion(event->screenPos());
326 seat->notifyPointerFrame();
327 }
328 } else if (event->type() == QEvent::MouseButtonPress || event->type() == QEvent::MouseButtonRelease) {
329 if (pointerSurfaceAllowed()) {
330 // TODO: can we leak presses/releases here when we move the mouse in between from an allowed surface to
331 // disallowed one or vice versa?
332 const auto state = event->type() == QEvent::MouseButtonPress
333 ? KWaylandServer::PointerButtonState::Pressed
334 : KWaylandServer::PointerButtonState::Released;
335 seat->notifyPointerButton(nativeButton, state);
336 seat->notifyPointerFrame();
337 }
338 }
339 return true;
340 }
341 bool wheelEvent(WheelEvent *event) override
342 {
343 if (!waylandServer()->isScreenLocked()) {
344 return false;
345 }
346 auto seat = waylandServer()->seat();
347 if (pointerSurfaceAllowed()) {
348 const WheelEvent *wheelEvent = static_cast<WheelEvent *>(event);
349 seat->setTimestamp(wheelEvent->timestamp());
350 seat->notifyPointerAxis(wheelEvent->orientation(), wheelEvent->delta(),
351 wheelEvent->deltaV120(),
352 kwinAxisSourceToKWaylandAxisSource(wheelEvent->axisSource()));
353 seat->notifyPointerFrame();
354 }
355 return true;
356 }
357 bool keyEvent(KeyEvent *event) override
358 {
359 if (!waylandServer()->isScreenLocked()) {
360 return false;
361 }
362 if (event->isAutoRepeat()) {
363 // wayland client takes care of it
364 return true;
365 }
366 // send event to KSldApp for global accel
367 // if event is set to accepted it means a whitelisted shortcut was triggered
368 // in that case we filter it out and don't process it further
369#if KWIN_BUILD_SCREENLOCKER
370 event->setAccepted(false);
371 QCoreApplication::sendEvent(ScreenLocker::KSldApp::self(), event);
372 if (event->isAccepted()) {
373 return true;
374 }
375#endif
376
377 // continue normal processing
378 input()->keyboard()->update();
379 auto seat = waylandServer()->seat();
380 seat->setTimestamp(event->timestamp());
381 if (!keyboardSurfaceAllowed()) {
382 // don't pass event to seat
383 return true;
384 }
385 switch (event->type()) {
386 case QEvent::KeyPress:
387 seat->notifyKeyboardKey(event->nativeScanCode(), KWaylandServer::KeyboardKeyState::Pressed);
388 break;
389 case QEvent::KeyRelease:
390 seat->notifyKeyboardKey(event->nativeScanCode(), KWaylandServer::KeyboardKeyState::Released);
391 break;
392 default:
393 break;
394 }
395 return true;
396 }

705class WindowSelectorFilter : public InputEventFilter
706{
707public:
708 bool pointerEvent(MouseEvent *event, quint32 nativeButton) override
709 {
710 if (!m_active) {
711 return false;
712 }
713 switch (event->type()) {
714 case QEvent::MouseButtonRelease:
715 if (event->buttons() == Qt::NoButton) {
716 if (event->button() == Qt::RightButton) {
717 cancel();
718 } else {
719 accept(event->globalPos());
720 }
721 }
722 break;
723 default:
724 break;
725 }
726 return true;
727 }
728 bool wheelEvent(WheelEvent *event) override
729 {
730 // filter out while selecting a window
731 return m_active;
732 }
733 bool keyEvent(KeyEvent *event) override
734 {
735 if (!m_active) {
736 return false;
737 }
738 waylandServer()->seat()->setFocusedKeyboardSurface(nullptr);
739 passToWaylandServer(event);
740
741 if (event->type() == QEvent::KeyPress) {
742 // x11 variant does this on key press, so do the same
743 if (event->key() == Qt::Key_Escape) {
744 cancel();
745 } else if (event->key() == Qt::Key_Enter || event->key() == Qt::Key_Return || event->key() == Qt::Key_Space) {
746 accept(input()->globalPointer());
747 }
748 if (input()->supportsPointerWarping()) {
749 int mx = 0;
750 int my = 0;
751 if (event->key() == Qt::Key_Left) {
752 mx = -10;
753 }
754 if (event->key() == Qt::Key_Right) {
755 mx = 10;
756 }
757 if (event->key() == Qt::Key_Up) {
758 my = -10;
759 }
760 if (event->key() == Qt::Key_Down) {
761 my = 10;
762 }
763 if (event->modifiers() & Qt::ControlModifier) {
764 mx /= 10;
765 my /= 10;
766 }
767 input()->warpPointer(input()->globalPointer() + QPointF(mx, my));
768 }
769 }
770 // filter out while selecting a window
771 return true;
772 }

1559 }
1560 if (event->type() == QEvent::TabletRelease) {
1561 decoration->window()->processDecorationButtonRelease(&e);

1564 }
1565 case QEvent::TabletLeaveProximity: {
1566 QHoverEvent leaveEvent(QEvent::HoverLeave, QPointF(), QPointF());
