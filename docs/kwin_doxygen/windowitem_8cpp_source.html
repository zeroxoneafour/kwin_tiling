129bool WindowItem::computeVisibility() const
130{
131 if (!m_window->readyForPainting()) {
132 return false;
133 }
134 if (waylandServer() && waylandServer()->isScreenLocked()) {
135 return m_window->isLockScreen() || m_window->isInputMethod() || m_window->isLockScreenOverlay();
136 }
137 if (m_window->isDeleted()) {
138 if (m_forceVisibleByDeleteCount == 0) {
139 return false;
140 }
141 }
142 if (!m_window->isOnCurrentDesktop()) {
143 if (m_forceVisibleByDesktopCount == 0) {
144 return false;
145 }
146 }
147 if (!m_window->isOnCurrentActivity()) {
148 if (m_forceVisibleByActivityCount == 0) {
149 return false;
150 }
151 }
152 if (m_window->isMinimized()) {
153 if (m_forceVisibleByMinimizeCount == 0) {
154 return false;
155 }
156 }
157 if (m_window->isHiddenInternal()) {
158 if (m_forceVisibleByHiddenCount == 0) {
159 return false;
160 }
161 }
162 return true;
163}

175void WindowItem::updateSurfaceItem(SurfaceItem *surfaceItem)
176{
177 m_surfaceItem.reset(surfaceItem);
178
179 if (m_surfaceItem) {
180 connect(m_window, &Window::shadeChanged, this, &WindowItem::updateSurfaceVisibility);
181 connect(m_window, &Window::bufferGeometryChanged, this, &WindowItem::updateSurfacePosition);
182 connect(m_window, &Window::frameGeometryChanged, this, &WindowItem::updateSurfacePosition);
183
184 connect(surfaceItem, &SurfaceItem::damaged, this, &WindowItem::markDamaged);
185 connect(surfaceItem, &SurfaceItem::childAdded, this, [this](Item *item) {
186 auto surfaceItem = static_cast<SurfaceItem *>(item);
187 connect(surfaceItem, &SurfaceItem::damaged, this, &WindowItem::markDamaged);
188 });
189
190 updateSurfacePosition();
191 updateSurfaceVisibility();
192 } else {
193 disconnect(m_window, &Window::shadeChanged, this, &WindowItem::updateSurfaceVisibility);
194 disconnect(m_window, &Window::bufferGeometryChanged, this, &WindowItem::updateSurfacePosition);
195 disconnect(m_window, &Window::frameGeometryChanged, this, &WindowItem::updateSurfacePosition);
196 }
197}

212void WindowItem::updateShadowItem()
213{
214 Shadow *shadow = m_window->shadow();
215 if (shadow) {
216 if (!m_shadowItem || m_shadowItem->shadow() != shadow) {
217 m_shadowItem.reset(new ShadowItem(shadow, m_window, scene(), this));
218 }
219 if (m_decorationItem) {
220 m_shadowItem->stackBefore(m_decorationItem.get());
221 } else if (m_surfaceItem) {
222 m_shadowItem->stackBefore(m_surfaceItem.get());
223 }
224 markDamaged();
225 } else {
226 m_shadowItem.reset();
227 }
228}

230void WindowItem::updateDecorationItem()
231{
232 if (m_window->isDeleted() || m_window->isZombie()) {
233 return;
234 }
235 if (m_window->decoration()) {
236 m_decorationItem.reset(new DecorationItem(m_window->decoration(), m_window, scene(), this));
237 if (m_shadowItem) {
238 m_decorationItem->stackAfter(m_shadowItem.get());
239 } else if (m_surfaceItem) {
240 m_decorationItem->stackBefore(m_surfaceItem.get());
241 }
242 connect(m_window->decoration(), &KDecoration2::Decoration::damaged, this, &WindowItem::markDamaged);
243 markDamaged();
244 } else {
245 m_decorationItem.reset();
246 }
247}

268void WindowItemX11::initialize()
269{
270 switch (kwinApp()->operationMode()) {
271 case Application::OperationModeX11:
272 updateSurfaceItem(new SurfaceItemX11(window(), scene(), this));
273 break;
274 case Application::OperationModeXwayland:
275 if (!window()->surface()) {
276 updateSurfaceItem(nullptr);
277 } else {
278 updateSurfaceItem(new SurfaceItemXwayland(window(), scene(), this));
279 }
280 break;
281 case Application::OperationModeWaylandOnly:
282 Q_UNREACHABLE();
283 }
284}
