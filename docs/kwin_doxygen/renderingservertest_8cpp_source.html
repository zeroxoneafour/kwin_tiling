30static int startXServer()
31{
32 const QByteArray process = QByteArrayLiteral("Xwayland");
33 int pipeFds[2];
34 if (pipe(pipeFds) != 0) {
35 std::cerr << "FATAL ERROR failed to create pipe to start X Server " << process.constData() << std::endl;
36 exit(1);
37 }
38
39 pid_t pid = fork();
40 if (pid == 0) {
41 // child process - should be turned into Xwayland
42 // writes to pipe, closes read side
43 close(pipeFds[0]);
44 char fdbuf[16];
45 sprintf(fdbuf, "%d", pipeFds[1]);
46 execlp(process.constData(), process.constData(), "-displayfd", fdbuf, "-rootless", (char *)nullptr);
47 close(pipeFds[1]);
48 exit(20);
49 }
50 // parent process - this is the wayland server
51 // reads from pipe, closes write side
52 close(pipeFds[1]);
53 return pipeFds[0];
54}

56static void readDisplayFromPipe(int pipe)
57{
58 QFile readPipe;
59 if (!readPipe.open(pipe, QIODevice::ReadOnly)) {
60 std::cerr << "FATAL ERROR failed to open pipe to start X Server XWayland" << std::endl;
61 exit(1);
62 }
63 QByteArray displayNumber = readPipe.readLine();
64
65 displayNumber.prepend(QByteArray(":"));
66 displayNumber.remove(displayNumber.size() - 1, 1);
67 std::cout << "X-Server started on display " << displayNumber.constData() << std::endl;
68
69 setenv("DISPLAY", displayNumber.constData(), true);
70
71 // close our pipe
72 close(pipe);
73}

109void CompositorWindow::surfaceCreated(KWaylandServer::XdgToplevelInterface *surface)
110{
111 using namespace KWaylandServer;
112 surface->sendConfigure(QSize(), XdgToplevelInterface::States());
113 m_stackingOrder << surface;
114 connect(surface->surface(), &SurfaceInterface::damaged, this, static_cast<void (CompositorWindow::*)()>(&CompositorWindow::update));
115 connect(surface, &XdgToplevelInterface::destroyed, this, [surface, this] {
116 m_stackingOrder.removeAll(surface);
117 updateFocus();
118 update();
119 });
120 updateFocus();
121}

123void CompositorWindow::updateFocus()
124{
125 using namespace KWaylandServer;
126 if (!m_seat || m_stackingOrder.isEmpty()) {
127 return;
128 }
129 auto it = std::find_if(m_stackingOrder.constBegin(), m_stackingOrder.constEnd(), [](XdgToplevelInterface *toplevel) {
130 return toplevel->surface()->isMapped();
131 });
132 if (it == m_stackingOrder.constEnd()) {
133 return;
134 }
135 m_seat->notifyPointerEnter((*it)->surface(), m_seat->pointerPos());
136 m_seat->setFocusedKeyboardSurface((*it)->surface());
137}

144void CompositorWindow::paintEvent(QPaintEvent *event)
145{
146 QWidget::paintEvent(event);
147 QPainter p(this);
148 for (auto window : m_stackingOrder) {
149 KWaylandServer::SurfaceInterface *surface = window->surface();
150 if (!surface || !surface->isMapped()) {
151 continue;
152 }
153 auto clientBuffer = qobject_cast<KWaylandServer::ShmClientBuffer *>(surface->buffer());
154 if (clientBuffer) {
155 p.drawImage(QPoint(0, 0), clientBuffer->data());
156 }
157 surface->frameRendered(QDateTime::currentMSecsSinceEpoch());
158 }
159}

230int main(int argc, char **argv)
231{
232 using namespace KWaylandServer;
233 QApplication app(argc, argv);
234
235 QCommandLineParser parser;
236 parser.addHelpOption();
237 QCommandLineOption xwaylandOption(QStringList{QStringLiteral("x"), QStringLiteral("xwayland")}, QStringLiteral("Start a rootless Xwayland server"));
238 parser.addOption(xwaylandOption);
239 parser.process(app);
240
241 KWaylandServer::Display display;
242 display.start();
243 new DataDeviceManagerInterface(&display);
244 new CompositorInterface(&display, &display);
245 XdgShellInterface *shell = new XdgShellInterface(&display);
246 display.createShm();
247
248 const QSize windowSize(1024, 768);
249
250 auto outputHandle = std::make_unique<FakeOutput>();
251 outputHandle->setPhysicalSize(QSize(269, 202));
252 outputHandle->setMode(windowSize, 60000);
253
254 auto outputInterface = std::make_unique<OutputInterface>(&display, outputHandle.get());
255
256 SeatInterface *seat = new SeatInterface(&display);
257 seat->setHasKeyboard(true);
258 seat->setHasPointer(true);
259 seat->setName(QStringLiteral("testSeat0"));
260
261 CompositorWindow compositorWindow;
262 compositorWindow.setSeat(seat);
263 compositorWindow.setMinimumSize(windowSize);
264 compositorWindow.setMaximumSize(windowSize);
265 compositorWindow.setGeometry(QRect(QPoint(0, 0), windowSize));
266 compositorWindow.show();
267 QObject::connect(shell, &XdgShellInterface::toplevelCreated, &compositorWindow, &CompositorWindow::surfaceCreated);
268
269 // start XWayland
270 if (parser.isSet(xwaylandOption)) {
271 // starts XWayland by forking and opening a pipe
272 const int pipe = startXServer();
273 if (pipe == -1) {
274 exit(1);
275 }
276
277 QtConcurrent::run([pipe] {
278 readDisplayFromPipe(pipe);
279 });
280 }
281
282 return app.exec();
283}
