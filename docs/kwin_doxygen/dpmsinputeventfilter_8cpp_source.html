52bool DpmsInputEventFilter::touchDown(qint32 id, const QPointF &pos, std::chrono::microseconds time)
53{
54 if (m_enableDoubleTap) {
55 if (m_touchPoints.isEmpty()) {
56 if (!m_doubleTapTimer.isValid()) {
57 // this is the first tap
58 m_doubleTapTimer.start();
59 } else {
60 if (m_doubleTapTimer.elapsed() < qApp->doubleClickInterval()) {
61 m_secondTap = true;
62 } else {
63 // took too long. Let's consider it a new click
64 m_doubleTapTimer.restart();
65 }
66 }
67 } else {
68 // not a double tap
69 m_doubleTapTimer.invalidate();
70 m_secondTap = false;
71 }
72 m_touchPoints << id;
73 }
74 return true;
75}

77bool DpmsInputEventFilter::touchUp(qint32 id, std::chrono::microseconds time)
78{
79 if (m_enableDoubleTap) {
80 m_touchPoints.removeAll(id);
81 if (m_touchPoints.isEmpty() && m_doubleTapTimer.isValid() && m_secondTap) {
82 if (m_doubleTapTimer.elapsed() < qApp->doubleClickInterval()) {
83 waylandServer()->seat()->setTimestamp(std::chrono::duration_cast<std::chrono::milliseconds>(time));
84 notify();
85 }
86 m_doubleTapTimer.invalidate();
87 m_secondTap = false;
88 }
89 }
90 return true;
91}
