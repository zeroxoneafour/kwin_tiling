285void ExpoLayout::updatePolish()
286{
287 if (!m_cells.isEmpty()) {
288 switch (m_mode) {
289 case LayoutClosest:
290 calculateWindowTransformationsClosest();
291 break;
292 case LayoutNatural:
293 calculateWindowTransformationsNatural();
294 break;
295 case LayoutNone:
296 resetTransformations();
297 break;
298 }
299 }
300
301 setReady();

333static int distance(const QPoint &a, const QPoint &b)
334{
335 const int xdiff = a.x() - b.x();
336 const int ydiff = a.y() - b.y();
337 return int(std::sqrt(qreal(xdiff * xdiff + ydiff * ydiff)));
338}

351void ExpoLayout::calculateWindowTransformationsClosest()
352{
353 QRect area = QRect(0, 0, width(), height());
354 const int columns = int(std::ceil(std::sqrt(qreal(m_cells.count()))));
355 const int rows = int(std::ceil(m_cells.count() / qreal(columns)));
356
357 // Assign slots
358 const int slotWidth = area.width() / columns;
359 const int slotHeight = area.height() / rows;
360 QVector<ExpoCell *> takenSlots;
361 takenSlots.resize(rows * columns);
362 takenSlots.fill(nullptr);
363
364 // precalculate all slot centers
365 QVector<QPoint> slotCenters;
366 slotCenters.resize(rows * columns);
367 for (int x = 0; x < columns; ++x) {
368 for (int y = 0; y < rows; ++y) {
369 slotCenters[x + y * columns] = QPoint(area.x() + slotWidth * x + slotWidth / 2,
370 area.y() + slotHeight * y + slotHeight / 2);
371 }
372 }
373
374 // Assign each window to the closest available slot
375 QList<ExpoCell *> tmpList = m_cells; // use a QLinkedList copy instead?
376 while (!tmpList.isEmpty()) {
377 ExpoCell *cell = tmpList.first();
378 int slotCandidate = -1, slotCandidateDistance = INT_MAX;
379 const QPoint pos = cell->naturalRect().center();
380
381 for (int i = 0; i < columns * rows; ++i) { // all slots
382 const int dist = distance(pos, slotCenters[i]);
383 if (dist < slotCandidateDistance) { // window is interested in this slot
384 ExpoCell *occupier = takenSlots[i];
385 Q_ASSERT(occupier != cell);
386 if (!occupier || dist < distance(occupier->naturalRect().center(), slotCenters[i])) {
387 // either nobody lives here, or we're better - takeover the slot if it's our best
388 slotCandidate = i;
389 slotCandidateDistance = dist;
390 }
391 }
392 }
393 Q_ASSERT(slotCandidate != -1);
394 if (takenSlots[slotCandidate]) {
395 tmpList << takenSlots[slotCandidate]; // occupier needs a new home now :p
396 }
397 tmpList.removeAll(cell);
398 takenSlots[slotCandidate] = cell; // ...and we rumble in =)
399 }
400
401 for (int slot = 0; slot < columns * rows; ++slot) {
402 ExpoCell *cell = takenSlots[slot];
403 if (!cell) { // some slots might be empty
404 continue;
405 }
406
407 // Work out where the slot is
408 QRect target(area.x() + (slot % columns) * slotWidth,
409 area.y() + (slot / columns) * slotHeight,
410 slotWidth, slotHeight);
411 target.adjust(m_spacing, m_spacing, -m_spacing, -m_spacing); // Borders
412 target = target.marginsRemoved(cell->margins());
413
414 qreal scale;
415 if (target.width() / qreal(cell->naturalWidth()) < target.height() / qreal(cell->naturalHeight())) {
416 // Center vertically
417 scale = target.width() / qreal(cell->naturalWidth());
418 target.moveTop(target.top() + (target.height() - int(cell->naturalHeight() * scale)) / 2);
419 target.setHeight(int(cell->naturalHeight() * scale));
420 } else {
421 // Center horizontally
422 scale = target.height() / qreal(cell->naturalHeight());
423 target.moveLeft(target.left() + (target.width() - int(cell->naturalWidth() * scale)) / 2);
424 target.setWidth(int(cell->naturalWidth() * scale));
425 }
426 // Don't scale the windows too much
427 if (scale > 2.0 || (scale > 1.0 && (cell->naturalWidth() > 300 || cell->naturalHeight() > 300))) {
428 scale = (cell->naturalWidth() > 300 || cell->naturalHeight() > 300) ? 1.0 : 2.0;
429 target = QRect(
430 target.center().x() - int(cell->naturalWidth() * scale) / 2,
431 target.center().y() - int(cell->naturalHeight() * scale) / 2,
432 scale * cell->naturalWidth(), scale * cell->naturalHeight());
433 }
434
435 cell->setX(target.x());
436 cell->setY(target.y());
437 cell->setWidth(target.width());
438 cell->setHeight(target.height());
439 }
440}

447static bool isOverlappingAny(ExpoCell *w, const QHash<ExpoCell *, QRect> &targets, const QRegion &border, int spacing)
448{
449 QHash<ExpoCell *, QRect>::const_iterator winTarget = targets.find(w);
450 if (winTarget == targets.constEnd()) {
451 return false;
452 }
453 if (border.intersects(*winTarget)) {
454 return true;
455 }
456 const QMargins halfSpacing(spacing / 2, spacing / 2, spacing / 2, spacing / 2);
457
458 // Is there a better way to do this?
459 QHash<ExpoCell *, QRect>::const_iterator target;
460 for (target = targets.constBegin(); target != targets.constEnd(); ++target) {
461 if (target == winTarget) {
462 continue;
463 }
464 if (winTarget->marginsAdded(halfSpacing).intersects(target->marginsAdded(halfSpacing))) {
465 return true;
466 }
467 }
468 return false;
469}

471void ExpoLayout::calculateWindowTransformationsNatural()
472{
473 const QRect area = QRect(0, 0, width(), height());
474
475 // As we are using pseudo-random movement (See "slot") we need to make sure the list
476 // is always sorted the same way no matter which window is currently active.
477 std::sort(m_cells.begin(), m_cells.end(), [](const ExpoCell *a, const ExpoCell *b) {
478 return a->persistentKey() < b->persistentKey();
479 });
480
481 QRect bounds;
482 int direction = 0;
483 QHash<ExpoCell *, QRect> targets;
484 QHash<ExpoCell *, int> directions;
485
486 for (ExpoCell *cell : std::as_const(m_cells)) {
487 const QRect cellRect(cell->naturalX(), cell->naturalY(), cell->naturalWidth(), cell->naturalHeight());
488 targets[cell] = cellRect;
489 // Reuse the unused "slot" as a preferred direction attribute. This is used when the window
490 // is on the edge of the screen to try to use as much screen real estate as possible.
491 directions[cell] = direction;
492 bounds = bounds.united(cellRect);
493 direction++;
494 if (direction == 4) {
495 direction = 0;
496 }
497 }
498
499 // Iterate over all windows, if two overlap push them apart _slightly_ as we try to
500 // brute-force the most optimal positions over many iterations.
501 const int halfSpacing = m_spacing / 2;
502 bool overlap;
503 do {
504 overlap = false;
505 for (ExpoCell *cell : std::as_const(m_cells)) {
506 QRect *target_w = &targets[cell];
507 for (ExpoCell *e : std::as_const(m_cells)) {
508 if (cell == e) {
509 continue;
510 }
511
512 QRect *target_e = &targets[e];
513 if (target_w->adjusted(-halfSpacing, -halfSpacing, halfSpacing, halfSpacing)
514 .intersects(target_e->adjusted(-halfSpacing, -halfSpacing, halfSpacing, halfSpacing))) {
515 overlap = true;
516
517 // Determine pushing direction
518 QPoint diff(target_e->center() - target_w->center());
519 // Prevent dividing by zero and non-movement
520 if (diff.x() == 0 && diff.y() == 0) {
521 diff.setX(1);
522 }
523 // Try to keep screen aspect ratio
524 // if (bounds.height() / bounds.width() > area.height() / area.width())
525 // diff.setY(diff.y() / 2);
526 // else
527 // diff.setX(diff.x() / 2);
528 // Approximate a vector of between 10px and 20px in magnitude in the same direction
529 diff *= m_accuracy / qreal(diff.manhattanLength());
530 // Move both windows apart
531 target_w->translate(-diff);
532 target_e->translate(diff);
533
534 // Try to keep the bounding rect the same aspect as the screen so that more
535 // screen real estate is utilised. We do this by splitting the screen into nine
536 // equal sections, if the window center is in any of the corner sections pull the
537 // window towards the outer corner. If it is in any of the other edge sections
538 // alternate between each corner on that edge. We don't want to determine it
539 // randomly as it will not produce consistant locations when using the filter.
540 // Only move one window so we don't cause large amounts of unnecessary zooming
541 // in some situations. We need to do this even when expanding later just in case
542 // all windows are the same size.
543 // (We are using an old bounding rect for this, hopefully it doesn't matter)
544 int xSection = (target_w->x() - bounds.x()) / (bounds.width() / 3);
545 int ySection = (target_w->y() - bounds.y()) / (bounds.height() / 3);
546 diff = QPoint(0, 0);
547 if (xSection != 1 || ySection != 1) { // Remove this if you want the center to pull as well
548 if (xSection == 1) {
549 xSection = (directions[cell] / 2 ? 2 : 0);
550 }
551 if (ySection == 1) {
552 ySection = (directions[cell] % 2 ? 2 : 0);
553 }
554 }
555 if (xSection == 0 && ySection == 0) {
556 diff = QPoint(bounds.topLeft() - target_w->center());
557 }
558 if (xSection == 2 && ySection == 0) {
559 diff = QPoint(bounds.topRight() - target_w->center());
560 }
561 if (xSection == 2 && ySection == 2) {
562 diff = QPoint(bounds.bottomRight() - target_w->center());
563 }
564 if (xSection == 0 && ySection == 2) {
565 diff = QPoint(bounds.bottomLeft() - target_w->center());
566 }
567 if (diff.x() != 0 || diff.y() != 0) {
568 diff *= m_accuracy / qreal(diff.manhattanLength());
569 target_w->translate(diff);
570 }
571
572 // Update bounding rect
573 bounds = bounds.united(*target_w);
574 bounds = bounds.united(*target_e);
575 }
576 }
577 }
578 } while (overlap);
579
580 // Compute the scale factor so the bounding rect fits the target area.
581 qreal scale;
582 if (bounds.width() <= area.width() && bounds.height() <= area.height()) {
583 scale = 1.0;
584 } else if (area.width() / qreal(bounds.width()) < area.height() / qreal(bounds.height())) {
585 scale = area.width() / qreal(bounds.width());
586 } else {
587 scale = area.height() / qreal(bounds.height());
588 }
589 // Make bounding rect fill the screen size for later steps
590 bounds = QRect(bounds.x() - (area.width() / scale - bounds.width()) / 2,
591 bounds.y() - (area.height() / scale - bounds.height()) / 2,
592 area.width() / scale,
593 area.height() / scale);
594
595 // Move all windows back onto the screen and set their scale
596 QHash<ExpoCell *, QRect>::iterator target = targets.begin();
597 while (target != targets.end()) {
598 target->setRect((target->x() - bounds.x()) * scale + area.x(),
599 (target->y() - bounds.y()) * scale + area.y(),
600 target->width() * scale,
601 target->height() * scale);
602 ++target;
603 }
604
605 // Try to fill the gaps by enlarging windows if they have the space
606 if (m_fillGaps) {
607 // Don't expand onto or over the border
608 QRegion borderRegion(area.adjusted(-200, -200, 200, 200));
609 borderRegion ^= area;
610
611 bool moved;
612 do {
613 moved = false;
614 for (ExpoCell *cell : std::as_const(m_cells)) {
615 QRect oldRect;
616 QRect *target = &targets[cell];
617 // This may cause some slight distortion if the windows are enlarged a large amount
618 int widthDiff = m_accuracy;
619 int heightDiff = heightForWidth(cell, target->width() + widthDiff) - target->height();
620 int xDiff = widthDiff / 2; // Also move a bit in the direction of the enlarge, allows the
621 int yDiff = heightDiff / 2; // center windows to be enlarged if there is gaps on the side.
622
623 // heightDiff (and yDiff) will be re-computed after each successful enlargement attempt
624 // so that the error introduced in the window's aspect ratio is minimized
625
626 // Attempt enlarging to the top-right
627 oldRect = *target;
628 target->setRect(target->x() + xDiff,
629 target->y() - yDiff - heightDiff,
630 target->width() + widthDiff,
631 target->height() + heightDiff);
632 if (isOverlappingAny(cell, targets, borderRegion, m_spacing)) {
633 *target = oldRect;
634 } else {
635 moved = true;
636 heightDiff = heightForWidth(cell, target->width() + widthDiff) - target->height();
637 yDiff = heightDiff / 2;
638 }
639
640 // Attempt enlarging to the bottom-right
641 oldRect = *target;
642 target->setRect(target->x() + xDiff,
643 target->y() + yDiff,
644 target->width() + widthDiff,
645 target->height() + heightDiff);
646 if (isOverlappingAny(cell, targets, borderRegion, m_spacing)) {
647 *target = oldRect;
648 } else {
649 moved = true;
650 heightDiff = heightForWidth(cell, target->width() + widthDiff) - target->height();
651 yDiff = heightDiff / 2;
652 }
653
654 // Attempt enlarging to the bottom-left
655 oldRect = *target;
656 target->setRect(target->x() - xDiff - widthDiff,
657 target->y() + yDiff,
658 target->width() + widthDiff,
659 target->height() + heightDiff);
660 if (isOverlappingAny(cell, targets, borderRegion, m_spacing)) {
661 *target = oldRect;
662 } else {
663 moved = true;
664 heightDiff = heightForWidth(cell, target->width() + widthDiff) - target->height();
665 yDiff = heightDiff / 2;
666 }
667
668 // Attempt enlarging to the top-left
669 oldRect = *target;
670 target->setRect(target->x() - xDiff - widthDiff,
671 target->y() - yDiff - heightDiff,
672 target->width() + widthDiff,
673 target->height() + heightDiff);
674 if (isOverlappingAny(cell, targets, borderRegion, m_spacing)) {
675 *target = oldRect;
676 } else {
677 moved = true;
678 }
679 }
680 } while (moved);
681
682 // The expanding code above can actually enlarge windows over 1.0/2.0 scale, we don't like this
683 // We can't add this to the loop above as it would cause a never-ending loop so we have to make
684 // do with the less-than-optimal space usage with using this method.
685 for (ExpoCell *cell : std::as_const(m_cells)) {
686 QRect *target = &targets[cell];
687 qreal scale = target->width() / qreal(cell->naturalWidth());
688 if (scale > 2.0 || (scale > 1.0 && (cell->naturalWidth() > 300 || cell->naturalHeight() > 300))) {
689 scale = (cell->naturalWidth() > 300 || cell->naturalHeight() > 300) ? 1.0 : 2.0;
690 target->setRect(target->center().x() - int(cell->naturalWidth() * scale) / 2,
691 target->center().y() - int(cell->naturalHeight() * scale) / 2,
692 cell->naturalWidth() * scale,
693 cell->naturalHeight() * scale);
694 }
695 }
696 }
697
698 for (ExpoCell *cell : std::as_const(m_cells)) {
699 const QRect rect = centered(cell, targets.value(cell).marginsRemoved(cell->margins()));
700
701 cell->setX(rect.x());
702 cell->setY(rect.y());
703 cell->setWidth(rect.width());
704 cell->setHeight(rect.height());
705 }
706}
