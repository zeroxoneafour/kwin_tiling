81{

112bool InternalWindow::eventFilter(QObject *watched, QEvent *event)
113{
114 if (watched == m_handle && event->type() == QEvent::DynamicPropertyChange) {
115 QDynamicPropertyChangeEvent *pe = static_cast<QDynamicPropertyChangeEvent *>(event);
116 if (pe->propertyName() == s_skipClosePropertyName) {
117 setSkipCloseAnimation(m_handle->property(s_skipClosePropertyName).toBool());
118 }
119 if (pe->propertyName() == s_shadowEnabledPropertyName) {
120 updateShadow();
121 }
122 if (pe->propertyName() == "kwin_windowType") {
123 m_windowType = m_handle->property("kwin_windowType").value<NET::WindowType>();
124 workspace()->updateClientArea();
125 }
126 }
127 return false;
128}

329{

360{

363 leaveInteractiveMoveResize();
364 Q_EMIT clientFinishUserMovedResized(this);
365 }
366
367 Deleted *deleted = Deleted::create(this);
368 Q_EMIT windowClosed(this, deleted);

447void InternalWindow::updateCaption()
448{
449 const QString oldSuffix = m_captionSuffix;
450 const auto shortcut = shortcutCaptionSuffix();
451 m_captionSuffix = shortcut;
452 if ((!isSpecialWindow() || isToolbar()) && findWindowWithSameCaption()) {
453 int i = 2;
454 do {
455 m_captionSuffix = shortcut + QLatin1String(" <") + QString::number(i) + QLatin1Char('>');
456 i++;
457 } while (findWindowWithSameCaption());
458 }
459 if (m_captionSuffix != oldSuffix) {
460 Q_EMIT captionChanged();
461 }
462}

471void InternalWindow::commitGeometry(const QRectF &rect)
472{
473 // The client geometry and the buffer geometry are the same.
474 const QRectF oldClientGeometry = m_clientGeometry;
475 const QRectF oldFrameGeometry = m_frameGeometry;
476 const Output *oldOutput = m_output;
477
478 Q_EMIT frameGeometryAboutToChange(this);
479
480 m_clientGeometry = frameRectToClientRect(rect);
481 m_frameGeometry = rect;
482 m_bufferGeometry = m_clientGeometry;
483
484 if (oldClientGeometry == m_clientGeometry && oldFrameGeometry == m_frameGeometry) {
485 return;
486 }
487
488 m_output = workspace()->outputAt(rect.center());
489 syncGeometryToInternalWindow();
490
491 if (oldClientGeometry != m_clientGeometry) {
492 Q_EMIT bufferGeometryChanged(this, oldClientGeometry);
493 Q_EMIT clientGeometryChanged(this, oldClientGeometry);
494 }
495 if (oldFrameGeometry != m_frameGeometry) {
496 Q_EMIT frameGeometryChanged(this, oldFrameGeometry);
497 }
498 if (oldOutput != m_output) {
499 Q_EMIT outputChanged();
500 }
501 Q_EMIT geometryShapeChanged(this, oldFrameGeometry);
502}
