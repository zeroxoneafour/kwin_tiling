44uint32_t spaVideoFormatToDrmFormat(spa_video_format spa_format)
45{
46 switch (spa_format) {
47 case SPA_VIDEO_FORMAT_RGBA:
48 return DRM_FORMAT_ABGR8888;
49 case SPA_VIDEO_FORMAT_RGBx:
50 return DRM_FORMAT_XBGR8888;
51 case SPA_VIDEO_FORMAT_BGRA:
52 return DRM_FORMAT_ARGB8888;
53 case SPA_VIDEO_FORMAT_BGRx:
54 return DRM_FORMAT_XRGB8888;
55 case SPA_VIDEO_FORMAT_BGR:
56 return DRM_FORMAT_BGR888;
57 case SPA_VIDEO_FORMAT_RGB:
58 return DRM_FORMAT_RGB888;
59 default:
60 qCDebug(KWIN_SCREENCAST) << "unknown format" << spa_format;
61 return DRM_FORMAT_INVALID;
62 }
63}

65void ScreenCastStream::onStreamStateChanged(void *data, pw_stream_state old, pw_stream_state state, const char *error_message)
66{
67 ScreenCastStream *pw = static_cast<ScreenCastStream *>(data);
68 qCDebug(KWIN_SCREENCAST) << "state changed" << pw_stream_state_as_string(old) << " -> " << pw_stream_state_as_string(state) << error_message;
69
70 switch (state) {
71 case PW_STREAM_STATE_ERROR:
72 qCWarning(KWIN_SCREENCAST) << "Stream error: " << error_message;
73 break;
74 case PW_STREAM_STATE_PAUSED:
75 if (pw->nodeId() == 0 && pw->pwStream) {
76 pw->pwNodeId = pw_stream_get_node_id(pw->pwStream);
77 Q_EMIT pw->streamReady(pw->nodeId());
78 }
79 break;
80 case PW_STREAM_STATE_STREAMING:
81 Q_EMIT pw->startStreaming();
82 break;
83 case PW_STREAM_STATE_CONNECTING:
84 break;
85 case PW_STREAM_STATE_UNCONNECTED:
86 if (!pw->m_stopped) {
87 Q_EMIT pw->stopStreaming();
88 }
89 break;
90 }
91}

97void ScreenCastStream::newStreamParams()
98{
99 qCDebug(KWIN_SCREENCAST) << "announcing stream params. with dmabuf:" << m_dmabufParams.has_value();
100 uint8_t paramsBuffer[1024];
101 spa_pod_builder pod_builder = SPA_POD_BUILDER_INIT(paramsBuffer, sizeof(paramsBuffer));
102 const int buffertypes = m_dmabufParams ? (1 << SPA_DATA_DmaBuf) : (1 << SPA_DATA_MemFd);
103 const int bpp = videoFormat.format == SPA_VIDEO_FORMAT_RGB || videoFormat.format == SPA_VIDEO_FORMAT_BGR ? 3 : 4;
104 const int stride = SPA_ROUND_UP_N(m_resolution.width() * bpp, 4);
105
106 struct spa_pod_frame f;
107 spa_pod_builder_push_object(&pod_builder, &f, SPA_TYPE_OBJECT_ParamBuffers, SPA_PARAM_Buffers);
108 spa_pod_builder_add(&pod_builder,
109 SPA_PARAM_BUFFERS_size, SPA_POD_Int(stride * m_resolution.height()),
110 SPA_PARAM_BUFFERS_buffers, SPA_POD_CHOICE_RANGE_Int(16, 2, 16),
111 SPA_PARAM_BUFFERS_stride, SPA_POD_Int(stride),
112 SPA_PARAM_BUFFERS_dataType, SPA_POD_CHOICE_FLAGS_Int(buffertypes), 0);
113 if (!m_dmabufParams) {
114 spa_pod_builder_add(&pod_builder,
115 SPA_PARAM_BUFFERS_blocks, SPA_POD_Int(1),
116 SPA_PARAM_BUFFERS_align, SPA_POD_Int(16), 0);
117 } else {
118 spa_pod_builder_add(&pod_builder,
119 SPA_PARAM_BUFFERS_blocks, SPA_POD_Int(m_dmabufParams->planeCount), 0);
120 }
121 spa_pod *bufferPod = (spa_pod *)spa_pod_builder_pop(&pod_builder, &f);
122
123 QVarLengthArray<const spa_pod *> params = {
124 bufferPod,
125 (spa_pod *)spa_pod_builder_add_object(&pod_builder,
126 SPA_TYPE_OBJECT_ParamMeta, SPA_PARAM_Meta,
127 SPA_PARAM_META_type, SPA_POD_Id(SPA_META_Cursor),
128 SPA_PARAM_META_size, SPA_POD_Int(CURSOR_META_SIZE(m_cursor.bitmapSize.width(), m_cursor.bitmapSize.height()))),
129 (spa_pod *)spa_pod_builder_add_object(&pod_builder,
130 SPA_TYPE_OBJECT_ParamMeta, SPA_PARAM_Meta,
131 SPA_PARAM_META_type, SPA_POD_Id(SPA_META_VideoDamage),
132 SPA_PARAM_META_size, SPA_POD_CHOICE_RANGE_Int(sizeof(struct spa_meta_region) * videoDamageRegionCount, sizeof(struct spa_meta_region) * 1, sizeof(struct spa_meta_region) * videoDamageRegionCount)),
133 (spa_pod *)spa_pod_builder_add_object(&pod_builder,
134 SPA_TYPE_OBJECT_ParamMeta, SPA_PARAM_Meta,
135 SPA_PARAM_META_type, SPA_POD_Id(SPA_META_Header),
136 SPA_PARAM_META_size, SPA_POD_Int(sizeof(struct spa_meta_header))),
137 };
138
139 pw_stream_update_params(pwStream, params.data(), params.count());
140}

142void ScreenCastStream::onStreamParamChanged(void *data, uint32_t id, const struct spa_pod *format)
143{
144 if (!format || id != SPA_PARAM_Format) {
145 return;
146 }
147
148 ScreenCastStream *pw = static_cast<ScreenCastStream *>(data);
149 spa_format_video_raw_parse(format, &pw->videoFormat);
150 auto modifierProperty = spa_pod_find_prop(format, nullptr, SPA_FORMAT_VIDEO_modifier);
151 QVector<uint64_t> receivedModifiers;
152 if (modifierProperty) {
153 const struct spa_pod *modifierPod = &modifierProperty->value;
154
155 uint32_t modifiersCount = SPA_POD_CHOICE_N_VALUES(modifierPod);
156 uint64_t *modifiers = (uint64_t *)SPA_POD_CHOICE_VALUES(modifierPod);
157 receivedModifiers = QVector<uint64_t>(modifiers, modifiers + modifiersCount);
158 }
159 if (modifierProperty && (!pw->m_dmabufParams || !receivedModifiers.contains(pw->m_dmabufParams->modifier))) {
160 if (modifierProperty->flags & SPA_POD_PROP_FLAG_DONT_FIXATE) {
161 pw->m_dmabufParams = kwinApp()->outputBackend()->testCreateDmaBuf(pw->m_resolution, spaVideoFormatToDrmFormat(pw->videoFormat.format), receivedModifiers);
162 } else {
163 pw->m_dmabufParams = kwinApp()->outputBackend()->testCreateDmaBuf(pw->m_resolution, spaVideoFormatToDrmFormat(pw->videoFormat.format), {DRM_FORMAT_MOD_INVALID});
164 }
165
166 qCDebug(KWIN_SCREENCAST) << "Stream dmabuf modifiers received, offering our best suited modifier" << pw->m_dmabufParams.has_value();
167 char buffer[2048];
168 auto params = pw->buildFormats(pw->m_dmabufParams.has_value(), buffer);
169 pw_stream_update_params(pw->pwStream, params.data(), params.count());
170 return;
171 }
172
173 qCDebug(KWIN_SCREENCAST) << "Stream format found, defining buffers";
174 pw->newStreamParams();
175}

177void ScreenCastStream::onStreamAddBuffer(void *data, pw_buffer *buffer)
178{
179 ScreenCastStream *stream = static_cast<ScreenCastStream *>(data);
180 struct spa_data *spa_data = buffer->buffer->datas;
181
182 spa_data->mapoffset = 0;
183 spa_data->flags = SPA_DATA_FLAG_READWRITE;
184
185 std::shared_ptr<DmaBufTexture> dmabuff;
186
187 if (spa_data[0].type != SPA_ID_INVALID && spa_data[0].type & (1 << SPA_DATA_DmaBuf)) {
188 Q_ASSERT(stream->m_dmabufParams);
189 dmabuff = kwinApp()->outputBackend()->createDmaBufTexture(*stream->m_dmabufParams);
190 }
191
192 if (dmabuff) {
193 spa_data->maxsize = dmabuff->attributes().pitch[0] * stream->m_resolution.height();
194
195 const DmaBufAttributes &dmabufAttribs = dmabuff->attributes();
196 Q_ASSERT(buffer->buffer->n_datas >= uint(dmabufAttribs.planeCount));
197 for (int i = 0; i < dmabufAttribs.planeCount; ++i) {
198 buffer->buffer->datas[i].type = SPA_DATA_DmaBuf;
199 buffer->buffer->datas[i].fd = dmabufAttribs.fd[i].get();
200 buffer->buffer->datas[i].data = nullptr;
201 }
202 stream->m_dmabufDataForPwBuffer.insert(buffer, dmabuff);
203#ifdef F_SEAL_SEAL // Disable memfd on systems that don't have it, like BSD < 12
204 } else {
205 if (!(spa_data[0].type & (1 << SPA_DATA_MemFd))) {
206 qCCritical(KWIN_SCREENCAST) << "memfd: Client doesn't support memfd buffer data type";
207 return;
208 }
209
210 const int bytesPerPixel = stream->m_source->hasAlphaChannel() ? 4 : 3;
211 const int stride = SPA_ROUND_UP_N(stream->m_resolution.width() * bytesPerPixel, 4);
212 spa_data->maxsize = stride * stream->m_resolution.height();
213 spa_data->type = SPA_DATA_MemFd;
214 spa_data->fd = memfd_create("kwin-screencast-memfd", MFD_CLOEXEC | MFD_ALLOW_SEALING);
215 if (spa_data->fd == -1) {
216 qCCritical(KWIN_SCREENCAST) << "memfd: Can't create memfd";
217 return;
218 }
219 spa_data->mapoffset = 0;
220
221 if (ftruncate(spa_data->fd, spa_data->maxsize) < 0) {
222 qCCritical(KWIN_SCREENCAST) << "memfd: Can't truncate to" << spa_data->maxsize;
223 return;
224 }
225
226 unsigned int seals = F_SEAL_GROW | F_SEAL_SHRINK | F_SEAL_SEAL;
227 if (fcntl(spa_data->fd, F_ADD_SEALS, seals) == -1) {
228 qCWarning(KWIN_SCREENCAST) << "memfd: Failed to add seals";
229 }
230
231 spa_data->data = mmap(nullptr,
232 spa_data->maxsize,
233 PROT_READ | PROT_WRITE,
234 MAP_SHARED,
235 spa_data->fd,
236 spa_data->mapoffset);
237 if (spa_data->data == MAP_FAILED) {
238 qCCritical(KWIN_SCREENCAST) << "memfd: Failed to mmap memory";
239 } else {
240 qCDebug(KWIN_SCREENCAST) << "memfd: created successfully" << spa_data->data << spa_data->maxsize;
241 }
242#endif
243 }
244
245 stream->m_waitForNewBuffers = false;
246}

248void ScreenCastStream::onStreamRemoveBuffer(void *data, pw_buffer *buffer)
249{
250 ScreenCastStream *stream = static_cast<ScreenCastStream *>(data);
251 stream->m_dmabufDataForPwBuffer.remove(buffer);
252
253 struct spa_buffer *spa_buffer = buffer->buffer;
254 struct spa_data *spa_data = spa_buffer->datas;
255 if (spa_data && spa_data->type == SPA_DATA_MemFd) {
256 munmap(spa_data->data, spa_data->maxsize);
257 close(spa_data->fd);
258 } else if (spa_data && spa_data->type == SPA_DATA_DmaBuf) {
259 for (int i = 0, c = buffer->buffer->n_datas; i < c; ++i) {
260 close(buffer->buffer->datas[i].fd);
261 }
262 }

331bool ScreenCastStream::createStream()
332{
333 const QByteArray objname = "kwin-screencast-" + objectName().toUtf8();
334 pwStream = pw_stream_new(pwCore->pwCore, objname, nullptr);
335
336 // it could make sense to offer the same format as the source
337 const auto format = m_source->hasAlphaChannel() ? SPA_VIDEO_FORMAT_BGRA : SPA_VIDEO_FORMAT_BGR;
338 const int drmFormat = spaVideoFormatToDrmFormat(format);
339 m_hasDmaBuf = kwinApp()->outputBackend()->testCreateDmaBuf(m_resolution, drmFormat, {DRM_FORMAT_MOD_INVALID}).has_value();
340 m_modifiers = Compositor::self()->backend()->supportedFormats().value(drmFormat);
341
342 char buffer[2048];
343 QVector<const spa_pod *> params = buildFormats(false, buffer);
344
345 pw_stream_add_listener(pwStream, &streamListener, &pwStreamEvents, this);
346 auto flags = pw_stream_flags(PW_STREAM_FLAG_DRIVER | PW_STREAM_FLAG_ALLOC_BUFFERS);
347
348 if (pw_stream_connect(pwStream, PW_DIRECTION_OUTPUT, SPA_ID_INVALID, flags, params.data(), params.count()) != 0) {
349 qCWarning(KWIN_SCREENCAST) << "Could not connect to stream";
350 pw_stream_destroy(pwStream);
351 pwStream = nullptr;
352 return false;
353 }
354
355 if (m_cursor.mode == KWaylandServer::ScreencastV1Interface::Embedded) {
356 connect(Cursors::self(), &Cursors::positionChanged, this, [this] {
357 recordFrame({});
358 });
359 } else if (m_cursor.mode == KWaylandServer::ScreencastV1Interface::Metadata) {
360 connect(Cursors::self(), &Cursors::positionChanged, this, &ScreenCastStream::recordCursor);
361 }
362
363 return true;

377void ScreenCastStream::recordFrame(const QRegion &_damagedRegion)
378{
379 QRegion damagedRegion = _damagedRegion;
380 Q_ASSERT(!m_stopped);
381
382 if (m_pendingBuffer) {
383 qCWarning(KWIN_SCREENCAST) << "Dropping a screencast frame because the compositor is slow";
384 return;
385 }
386
387 if (m_waitForNewBuffers) {
388 qCWarning(KWIN_SCREENCAST) << "Waiting for new buffers to be created";
389 return;
390 }
391
392 const auto size = m_source->textureSize();
393 if (size != m_resolution) {
394 m_resolution = size;
395 m_waitForNewBuffers = true;
396 m_dmabufParams = std::nullopt;
397 pw_loop_signal_event(pwCore.get()->pwMainLoop, pwRenegotiate);
398 return;
399 }
400
401 const char *error = "";
402 auto state = pw_stream_get_state(pwStream, &error);
403 if (state != PW_STREAM_STATE_STREAMING) {
404 if (error) {
405 qCWarning(KWIN_SCREENCAST) << "Failed to record frame: stream is not active" << error;
406 }
407 return;
408 }
409
410 struct pw_buffer *buffer = pw_stream_dequeue_buffer(pwStream);
411
412 if (!buffer) {
413 return;
414 }
415
416 struct spa_buffer *spa_buffer = buffer->buffer;
417 struct spa_data *spa_data = spa_buffer->datas;
418
419 uint8_t *data = (uint8_t *)spa_data->data;
420 if (!data && spa_buffer->datas->type != SPA_DATA_DmaBuf) {
421 qCWarning(KWIN_SCREENCAST) << "Failed to record frame: invalid buffer data";
422 pw_stream_queue_buffer(pwStream, buffer);
423 return;
424 }
425
426 spa_data->chunk->offset = 0;
427 static_cast<OpenGLBackend *>(Compositor::self()->backend())->makeCurrent();
428 if (data || spa_data[0].type == SPA_DATA_MemFd) {
429 const bool hasAlpha = m_source->hasAlphaChannel();
430 const int bpp = data && !hasAlpha ? 3 : 4;
431 const uint stride = SPA_ROUND_UP_N(size.width() * bpp, 4);
432
433 QImage dest(data, size.width(), size.height(), stride, hasAlpha ? QImage::Format_RGBA8888_Premultiplied : QImage::Format_RGB888);
434 if (dest.sizeInBytes() > spa_data->maxsize) {
435 qCDebug(KWIN_SCREENCAST) << "Failed to record frame: frame is too big";
436 pw_stream_queue_buffer(pwStream, buffer);
437 return;
438 }
439
440 spa_data->chunk->size = dest.sizeInBytes();
441 spa_data->chunk->stride = dest.bytesPerLine();
442
443 m_source->render(&dest);
444
445 auto cursor = Cursors::self()->currentCursor();
446 if (m_cursor.mode == KWaylandServer::ScreencastV1Interface::Embedded && exclusiveContains(m_cursor.viewport, cursor->pos())) {
447 QPainter painter(&dest);
448 const auto position = (cursor->pos() - m_cursor.viewport.topLeft() - cursor->hotspot()) * m_cursor.scale;
449 painter.drawImage(QRect{position.toPoint(), cursor->image().size()}, cursor->image());
450 }
451 } else {
452 auto &buf = m_dmabufDataForPwBuffer[buffer];
453 Q_ASSERT(buf);
454
455 const DmaBufAttributes &dmabufAttribs = buf->attributes();
456 Q_ASSERT(buffer->buffer->n_datas >= uint(dmabufAttribs.planeCount));
457 for (int i = 0; i < dmabufAttribs.planeCount; ++i) {
458 buffer->buffer->datas[i].chunk->stride = dmabufAttribs.pitch[i];
459 buffer->buffer->datas[i].chunk->offset = dmabufAttribs.offset[i];
460 }
461 spa_data->chunk->size = spa_data->maxsize;
462
463 m_source->render(buf->framebuffer());
464
465 auto cursor = Cursors::self()->currentCursor();
466 if (m_cursor.mode == KWaylandServer::ScreencastV1Interface::Embedded && exclusiveContains(m_cursor.viewport, cursor->pos())) {
467 if (!cursor->image().isNull()) {
468 GLFramebuffer::pushFramebuffer(buf->framebuffer());
469
470 QRect r(QPoint(), size);
471 auto shader = ShaderManager::instance()->pushShader(ShaderTrait::MapTexture);
472
473 QMatrix4x4 mvp;
474 mvp.ortho(r);
475 shader->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
476
477 if (!m_cursor.texture || m_cursor.lastKey != cursor->image().cacheKey()) {
478 m_cursor.texture.reset(new GLTexture(cursor->image()));
479 }
480
481 m_cursor.texture->setYInverted(false);
482 m_cursor.texture->bind();
483 const auto cursorRect = cursorGeometry(cursor);
484 mvp.translate(cursorRect.left(), r.height() - cursorRect.top() - cursor->image().height());
485 shader->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
486
487 glEnable(GL_BLEND);
488 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
489 m_cursor.texture->render(cursorRect.size(), m_cursor.scale);
490 glDisable(GL_BLEND);
491 m_cursor.texture->unbind();
492
493 ShaderManager::instance()->popShader();
494 GLFramebuffer::popFramebuffer();
495
496 damagedRegion += QRegion{m_cursor.lastRect.toAlignedRect()} | cursorRect.toAlignedRect();
497 m_cursor.lastRect = cursorRect;
498 } else {
499 damagedRegion |= m_cursor.lastRect.toAlignedRect();
500 m_cursor.lastRect = {};
501 }
502 }
503 }
504
505 if (m_cursor.mode == KWaylandServer::ScreencastV1Interface::Metadata) {
506 sendCursorData(Cursors::self()->currentCursor(),
507 (spa_meta_cursor *)spa_buffer_find_meta_data(spa_buffer, SPA_META_Cursor, sizeof(spa_meta_cursor)));
508 }
509
510 addDamage(spa_buffer, damagedRegion);
511 addHeader(spa_buffer);
512 tryEnqueue(buffer);
513}

515void ScreenCastStream::addHeader(spa_buffer *spaBuffer)
516{
517 spa_meta_header *spaHeader = (spa_meta_header *)spa_buffer_find_meta_data(spaBuffer, SPA_META_Header, sizeof(spaHeader));
518 if (spaHeader) {
519 spaHeader->flags = 0;
520 spaHeader->dts_offset = 0;
521 spaHeader->seq = m_sequential++;
522

524 if (!m_start) {
525 m_start = timestamp;
526 }
527 spaHeader->pts = (timestamp - m_start.value()).count();

531void ScreenCastStream::addDamage(spa_buffer *spaBuffer, const QRegion &damagedRegion)
532{
533 if (spa_meta *vdMeta = spa_buffer_find_meta(spaBuffer, SPA_META_VideoDamage)) {
534 struct spa_meta_region *r = (spa_meta_region *)spa_meta_first(vdMeta);
535
536 // If there's too many rectangles, we just send the bounding rect
537 if (damagedRegion.rectCount() > videoDamageRegionCount - 1) {
538 if (spa_meta_check(r, vdMeta)) {
539 auto rect = damagedRegion.boundingRect();
540 r->region = SPA_REGION(rect.x(), rect.y(), quint32(rect.width()), quint32(rect.height()));
541 r++;
542 }
543 } else {
544 for (const QRect &rect : damagedRegion) {
545 if (spa_meta_check(r, vdMeta)) {
546 r->region = SPA_REGION(rect.x(), rect.y(), quint32(rect.width()), quint32(rect.height()));
547 r++;
548 }
549 }
550 }
551
552 if (spa_meta_check(r, vdMeta)) {
553 r->region = SPA_REGION(0, 0, 0, 0);
554 }
555 }

558void ScreenCastStream::recordCursor()
559{
560 Q_ASSERT(!m_stopped);
561
562 if (m_pendingBuffer) {
563 qCWarning(KWIN_SCREENCAST) << "Dropping a screencast cursor update because the compositor is slow";
564 return;
565 }
566
567 const char *error = "";
568 auto state = pw_stream_get_state(pwStream, &error);
569 if (state != PW_STREAM_STATE_STREAMING) {
570 if (error) {
571 qCWarning(KWIN_SCREENCAST) << "Failed to record cursor position: stream is not active" << error;
572 }
573 return;
574 }
575
576 if (!exclusiveContains(m_cursor.viewport, Cursors::self()->currentCursor()->pos()) && !m_cursor.visible) {
577 return;
578 }
579
580 m_pendingBuffer = pw_stream_dequeue_buffer(pwStream);
581 if (!m_pendingBuffer) {
582 return;
583 }
584
585 struct spa_buffer *spa_buffer = m_pendingBuffer->buffer;
586 spa_buffer->datas[0].chunk->size = 0;
587
588 sendCursorData(Cursors::self()->currentCursor(),
589 (spa_meta_cursor *)spa_buffer_find_meta_data(spa_buffer, SPA_META_Cursor, sizeof(spa_meta_cursor)));
590 addHeader(spa_buffer);
591 addDamage(spa_buffer, {});
592 enqueue();
593}
594
595void ScreenCastStream::tryEnqueue(pw_buffer *buffer)
596{
597 m_pendingBuffer = buffer;
598
599 // The GPU doesn't necessarily process draw commands as soon as they are issued. Thus,
600 // we need to insert a fence into the command stream and enqueue the pipewire buffer
601 // only after the fence is signaled; otherwise stream consumers will most likely see
602 // a corrupted buffer.
603 if (Compositor::self()->scene()->supportsNativeFence()) {
604 Q_ASSERT_X(eglGetCurrentContext(), "tryEnqueue", "no current context");
605 m_pendingFence = std::make_unique<EGLNativeFence>(kwinApp()->outputBackend()->sceneEglDisplay());
606 if (!m_pendingFence->isValid()) {
607 qCWarning(KWIN_SCREENCAST) << "Failed to create a native EGL fence";
608 glFinish();
609 enqueue();
610 } else {
611 m_pendingNotifier = std::make_unique<QSocketNotifier>(m_pendingFence->fileDescriptor(), QSocketNotifier::Read);
612 connect(m_pendingNotifier.get(), &QSocketNotifier::activated, this, &ScreenCastStream::enqueue);
613 }
614 } else {
615 // The compositing backend doesn't support native fences. We don't have any other choice
616 // but stall the graphics pipeline. Otherwise stream consumers may see an incomplete buffer.
617 glFinish();
618 enqueue();
619 }
620}
621
622void ScreenCastStream::enqueue()
623{
624 Q_ASSERT_X(m_pendingBuffer, "enqueue", "pending buffer must be valid");
625
626 m_pendingFence.reset();
627 m_pendingNotifier.reset();
628
629 pw_stream_queue_buffer(pwStream, m_pendingBuffer);
630
631 m_pendingBuffer = nullptr;
632}

634QVector<const spa_pod *> ScreenCastStream::buildFormats(bool fixate, char buffer[2048])
635{
636 const auto format = m_source->hasAlphaChannel() ? SPA_VIDEO_FORMAT_BGRA : SPA_VIDEO_FORMAT_BGR;
637 spa_pod_builder podBuilder = SPA_POD_BUILDER_INIT(buffer, 2048);
638 spa_fraction minFramerate = SPA_FRACTION(1, 1);
639 spa_fraction maxFramerate = SPA_FRACTION(25, 1);
640 spa_fraction defaultFramerate = SPA_FRACTION(0, 1);
641
642 spa_rectangle resolution = SPA_RECTANGLE(uint32_t(m_resolution.width()), uint32_t(m_resolution.height()));
643
644 QVector<const spa_pod *> params;
645 params.reserve(fixate + m_hasDmaBuf + 1);
646 if (fixate) {
647 params.append(buildFormat(&podBuilder, SPA_VIDEO_FORMAT_BGRA, &resolution, &defaultFramerate, &minFramerate, &maxFramerate, {m_dmabufParams->modifier}, SPA_POD_PROP_FLAG_MANDATORY));
648 }
649 if (m_hasDmaBuf) {
650 params.append(buildFormat(&podBuilder, SPA_VIDEO_FORMAT_BGRA, &resolution, &defaultFramerate, &minFramerate, &maxFramerate, m_modifiers, SPA_POD_PROP_FLAG_MANDATORY | SPA_POD_PROP_FLAG_DONT_FIXATE));
651 }
652 params.append(buildFormat(&podBuilder, format, &resolution, &defaultFramerate, &minFramerate, &maxFramerate, {}, SPA_POD_PROP_FLAG_MANDATORY | SPA_POD_PROP_FLAG_DONT_FIXATE));
653 return params;
654}

656spa_pod *ScreenCastStream::buildFormat(struct spa_pod_builder *b, enum spa_video_format format, struct spa_rectangle *resolution,
657 struct spa_fraction *defaultFramerate, struct spa_fraction *minFramerate, struct spa_fraction *maxFramerate,
658 const QVector<uint64_t> &modifiers, quint32 modifiersFlags)
659{
660 struct spa_pod_frame f[2];
661 spa_pod_builder_push_object(b, &f[0], SPA_TYPE_OBJECT_Format, SPA_PARAM_EnumFormat);
662 spa_pod_builder_add(b, SPA_FORMAT_mediaType, SPA_POD_Id(SPA_MEDIA_TYPE_video), 0);
663 spa_pod_builder_add(b, SPA_FORMAT_mediaSubtype, SPA_POD_Id(SPA_MEDIA_SUBTYPE_raw), 0);
664 spa_pod_builder_add(b, SPA_FORMAT_VIDEO_size, SPA_POD_Rectangle(resolution), 0);
665 spa_pod_builder_add(b, SPA_FORMAT_VIDEO_framerate, SPA_POD_Fraction(defaultFramerate), 0);
666 spa_pod_builder_add(b, SPA_FORMAT_VIDEO_maxFramerate,
667 SPA_POD_CHOICE_RANGE_Fraction(
668 SPA_POD_Fraction(maxFramerate),
669 SPA_POD_Fraction(minFramerate),
670 SPA_POD_Fraction(maxFramerate)),
671 0);
672 if (format == SPA_VIDEO_FORMAT_BGRA) {
673 /* announce equivalent format without alpha */
674 spa_pod_builder_add(b, SPA_FORMAT_VIDEO_format, SPA_POD_CHOICE_ENUM_Id(3, format, format, SPA_VIDEO_FORMAT_BGRx), 0);
675 } else {
676 spa_pod_builder_add(b, SPA_FORMAT_VIDEO_format, SPA_POD_Id(format), 0);
677 }
678
679 if (!modifiers.isEmpty()) {
680 spa_pod_builder_prop(b, SPA_FORMAT_VIDEO_modifier, modifiersFlags);
681 spa_pod_builder_push_choice(b, &f[1], SPA_CHOICE_Enum, 0);
682
683 int c = 0;
684 for (auto modifier : modifiers) {
685 spa_pod_builder_long(b, modifier);
686 if (c++ == 0) {
687 spa_pod_builder_long(b, modifier);
688 }
689 }
690 spa_pod_builder_pop(b, &f[1]);
691 }
692 return (spa_pod *)spa_pod_builder_pop(b, &f[0]);
693}

705void ScreenCastStream::sendCursorData(Cursor *cursor, spa_meta_cursor *spa_meta_cursor)
706{
707 if (!cursor || !spa_meta_cursor) {
708 return;
709 }
710
711 if (!exclusiveContains(m_cursor.viewport, cursor->pos())) {
712 spa_meta_cursor->id = 0;
713 spa_meta_cursor->position.x = -1;
714 spa_meta_cursor->position.y = -1;
715 spa_meta_cursor->hotspot.x = -1;
716 spa_meta_cursor->hotspot.y = -1;
717 spa_meta_cursor->bitmap_offset = 0;
718 m_cursor.visible = false;
719 return;
720 }
721 m_cursor.visible = true;
722 const auto position = (cursor->pos() - m_cursor.viewport.topLeft()) * m_cursor.scale;
723
724 spa_meta_cursor->id = 1;
725 spa_meta_cursor->position.x = position.x();
726 spa_meta_cursor->position.y = position.y();
727 spa_meta_cursor->hotspot.x = cursor->hotspot().x() * m_cursor.scale;
728 spa_meta_cursor->hotspot.y = cursor->hotspot().y() * m_cursor.scale;
729 spa_meta_cursor->bitmap_offset = 0;
730
731 const QImage image = cursor->image();
732 if (image.cacheKey() == m_cursor.lastKey) {
733 return;
734 }
735
736 m_cursor.lastKey = image.cacheKey();
737 spa_meta_cursor->bitmap_offset = sizeof(struct spa_meta_cursor);
738
739 const QSize targetSize = (cursor->rect().size() * m_cursor.scale).toSize();
740
741 struct spa_meta_bitmap *spa_meta_bitmap = SPA_MEMBER(spa_meta_cursor,
742 spa_meta_cursor->bitmap_offset,
743 struct spa_meta_bitmap);
744 spa_meta_bitmap->format = SPA_VIDEO_FORMAT_RGBA;
745 spa_meta_bitmap->offset = sizeof(struct spa_meta_bitmap);
746 spa_meta_bitmap->size.width = std::min(m_cursor.bitmapSize.width(), targetSize.width());
747 spa_meta_bitmap->size.height = std::min(m_cursor.bitmapSize.height(), targetSize.height());
748 spa_meta_bitmap->stride = spa_meta_bitmap->size.width * 4;
749
750 uint8_t *bitmap_data = SPA_MEMBER(spa_meta_bitmap, spa_meta_bitmap->offset, uint8_t);
751 QImage dest(bitmap_data,
752 spa_meta_bitmap->size.width,
753 spa_meta_bitmap->size.height,
754 spa_meta_bitmap->stride,
755 QImage::Format_RGBA8888_Premultiplied);
756 dest.fill(Qt::transparent);
757
758 if (!image.isNull()) {
759 QPainter painter(&dest);
760 painter.drawImage(QRect({0, 0}, targetSize), image);
761 }
762}
