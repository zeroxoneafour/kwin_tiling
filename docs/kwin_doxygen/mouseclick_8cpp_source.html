67void MouseClickEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
68{
69 const int time = m_lastPresentTime.count() ? (presentTime - m_lastPresentTime).count() : 0;
70
71 for (auto &click : m_clicks) {
72 click->m_time += time;
73 }
74
75 for (int i = 0; i < BUTTON_COUNT; ++i) {
76 if (m_buttons[i]->m_isPressed) {
77 m_buttons[i]->m_time += time;
78 }
79 }
80
81 while (m_clicks.size() > 0) {
82 if (m_clicks.front()->m_time <= m_ringLife) {
83 break;
84 }
85 m_clicks.pop_front();
86 }
87
88 if (isActive()) {
89 m_lastPresentTime = presentTime;
90 } else {
91 m_lastPresentTime = std::chrono::milliseconds::zero();
92 }
93
94 effects->prePaintScreen(data, presentTime);
95}

97void MouseClickEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
98{
99 effects->paintScreen(mask, region, data);
100
101 paintScreenSetup(mask, region, data);
102 for (const auto &click : m_clicks) {
103 for (int i = 0; i < m_ringCount; ++i) {
104 float alpha = computeAlpha(click.get(), i);
105 float size = computeRadius(click.get(), i);
106 if (size > 0 && alpha > 0) {
107 QColor color = m_colors[click->m_button];
108 color.setAlphaF(alpha);
109 drawCircle(color, click->m_pos.x(), click->m_pos.y(), size);
110 }
111 }
112
113 if (m_showText && click->m_frame) {
114 float frameAlpha = (click->m_time * 2.0f - m_ringLife) / m_ringLife;
115 frameAlpha = frameAlpha < 0 ? 1 : -(frameAlpha * frameAlpha) + 1;
116 click->m_frame->render(infiniteRegion(), frameAlpha, frameAlpha);
117 }
118 }
119 for (const auto &tool : std::as_const(m_tabletTools)) {
120 const int step = m_ringMaxSize * (1. - tool.m_pressure);
121 for (qreal size = m_ringMaxSize; size > 0; size -= step) {
122 drawCircle(tool.m_color, tool.m_globalPosition.x(), tool.m_globalPosition.y(), size);
123 }
124 }
125 paintScreenFinish(mask, region, data);
126}

134float MouseClickEffect::computeRadius(const MouseEvent *click, int ring)
135{
136 float ringDistance = m_ringLife / (m_ringCount * 3);
137 if (click->m_press) {
138 return ((click->m_time - ringDistance * ring) / m_ringLife) * m_ringMaxSize;
139 }
140 return ((m_ringLife - click->m_time - ringDistance * ring) / m_ringLife) * m_ringMaxSize;
141}

143float MouseClickEffect::computeAlpha(const MouseEvent *click, int ring)
144{
145 float ringDistance = m_ringLife / (m_ringCount * 3);
146 return (m_ringLife - (float)click->m_time - ringDistance * (ring)) / m_ringLife;
147}

149void MouseClickEffect::slotMouseChanged(const QPointF &pos, const QPointF &,
150 Qt::MouseButtons buttons, Qt::MouseButtons oldButtons,
151 Qt::KeyboardModifiers, Qt::KeyboardModifiers)
152{
153 if (buttons == oldButtons) {
154 return;
155 }
156
157 std::unique_ptr<MouseEvent> m;
158 int i = BUTTON_COUNT;
159 while (--i >= 0) {
160 MouseButton *b = m_buttons[i].get();
161 if (isPressed(b->m_button, buttons, oldButtons)) {
162 m = std::make_unique<MouseEvent>(i, pos.toPoint(), 0, createEffectFrame(pos.toPoint(), b->m_labelDown), true);
163 break;
164 } else if (isReleased(b->m_button, buttons, oldButtons) && (!b->m_isPressed || b->m_time > m_ringLife)) {
165 // we might miss a press, thus also check !b->m_isPressed, bug #314762
166 m = std::make_unique<MouseEvent>(i, pos.toPoint(), 0, createEffectFrame(pos.toPoint(), b->m_labelUp), false);
167 break;
168 }
169 b->setPressed(b->m_button & buttons);
170 }
171
172 if (m) {
173 m_clicks.push_back(std::move(m));
174 }
175 repaint();
176}

190void MouseClickEffect::repaint()
191{
192 if (m_clicks.size() > 0) {
193 QRegion dirtyRegion;
194 const int radius = m_ringMaxSize + m_lineWidth;
195 for (auto &click : m_clicks) {
196 dirtyRegion |= QRect(click->m_pos.x() - radius, click->m_pos.y() - radius, 2 * radius, 2 * radius);
197 if (click->m_frame) {
198 // we grant the plasma style 32px padding for stuff like shadows...
199 dirtyRegion |= click->m_frame->geometry().adjusted(-32, -32, 32, 32);
200 }
201 }
202 effects->addRepaint(dirtyRegion);
203 }
204 if (!m_tabletTools.isEmpty()) {
205 QRegion dirtyRegion;
206 const int radius = m_ringMaxSize + m_lineWidth;
207 for (const auto &event : std::as_const(m_tabletTools)) {
208 dirtyRegion |= QRect(event.m_globalPosition.x() - radius, event.m_globalPosition.y() - radius, 2 * radius, 2 * radius);
209 }
210 effects->addRepaint(dirtyRegion);
211 }

230 effects->startMousePolling();

233 effects->stopMousePolling();

238

273void MouseClickEffect::drawCircleGl(const QColor &color, float cx, float cy, float r)
274{
275 static const int num_segments = 80;
276 static const float theta = 2 * 3.1415926 / float(num_segments);
277 static const float c = cosf(theta); // precalculate the sine and cosine
278 static const float s = sinf(theta);
279 const float scale = effects->renderTargetScale();
280 float t;
281
282 float x = r; // we start at angle = 0
283 float y = 0;
284
285 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
286 vbo->reset();
287 vbo->setUseColor(true);
288 vbo->setColor(color);
289 QVector<float> verts;
290 verts.reserve(num_segments * 2);
291
292 for (int ii = 0; ii < num_segments; ++ii) {
293 verts << (x + cx) * scale << (y + cy) * scale; // output vertex
294 // apply the rotation matrix
295 t = x;
296 x = c * x - s * y;
297 y = s * t + c * y;
298 }
299 vbo->setData(verts.size() / 2, 2, verts.data(), nullptr);
300 vbo->render(GL_LINE_LOOP);
301}

329bool MouseClickEffect::tabletToolEvent(QTabletEvent *event)
330{
331 auto &tabletEvent = m_tabletTools[event->uniqueId()];
332 if (!tabletEvent.m_color.isValid()) {
333 switch (event->pointerType()) {
334#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
335 case QTabletEvent::UnknownPointer:
336 case QTabletEvent::Pen:
337 tabletEvent.m_color = MouseClickConfig::color1();
338 break;
339 case QTabletEvent::Eraser:
340 tabletEvent.m_color = MouseClickConfig::color2();
341 break;
342 case QTabletEvent::Cursor:
343 tabletEvent.m_color = MouseClickConfig::color3();
344 break;
345#else
346 case QPointingDevice::PointerType::Unknown:
347 case QPointingDevice::PointerType::Generic:
348 case QPointingDevice::PointerType::Finger:
349 case QPointingDevice::PointerType::Pen:
350 tabletEvent.m_color = MouseClickConfig::color1();
351 break;
352 case QPointingDevice::PointerType::Eraser:
353 tabletEvent.m_color = MouseClickConfig::color2();
354 break;
355 case QPointingDevice::PointerType::Cursor:
356 tabletEvent.m_color = MouseClickConfig::color3();
357 break;
358#endif
359 }
360 }
361 switch (event->type()) {
362 case QEvent::TabletPress:
363 tabletEvent.m_pressed = true;
364 break;
365 case QEvent::TabletRelease:
366 tabletEvent.m_pressed = false;
367 break;
368 case QEvent::TabletLeaveProximity:
369 m_tabletTools.remove(event->uniqueId());
370 return false;
371 default:
372 break;
373 }
374 tabletEvent.m_globalPosition = event->globalPos();
375 tabletEvent.m_pressure = event->pressure();
376
377 return false;
378}
