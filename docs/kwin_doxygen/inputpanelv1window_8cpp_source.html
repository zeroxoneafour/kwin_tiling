81void KWin::InputPanelV1Window::reposition()
82{
83 if (!readyForPainting()) {
84 return;
85 }
86
87 switch (m_mode) {
88 case Mode::None: {
89 // should never happen
90 }; break;
91 case Mode::VirtualKeyboard: {
92 QSizeF panelSize = surface()->size();
93 if (!panelSize.isValid() || panelSize.isEmpty()) {
94 return;
95 }
96
97 const auto activeOutput = workspace()->activeOutput();
98 const QRectF outputArea = activeOutput->geometry();
99 QRectF availableArea;
100 if (waylandServer()->isScreenLocked()) {
101 availableArea = outputArea;
102 } else {
103 availableArea = workspace()->clientArea(MaximizeArea, this, activeOutput);
104 }
105
106 panelSize = panelSize.boundedTo(availableArea.size());
107 QRectF geo(QPointF(availableArea.left(), availableArea.top() + availableArea.height() - panelSize.height()), panelSize);
108 geo.translate((availableArea.width() - panelSize.width()) / 2, availableArea.height() - outputArea.height());
109 moveResize(geo);
110 } break;
111 case Mode::Overlay: {
112 auto textInputSurface = waylandServer()->seat()->focusedTextInputSurface();
113 auto textWindow = waylandServer()->findWindow(textInputSurface);
114 QRect cursorRectangle;
115 auto textInputV1 = waylandServer()->seat()->textInputV1();
116 if (textInputV1 && textInputV1->isEnabled() && textInputV1->surface() == textInputSurface) {
117 cursorRectangle = textInputV1->cursorRectangle();
118 }
119 auto textInputV2 = waylandServer()->seat()->textInputV2();
120 if (textInputV2 && textInputV2->isEnabled() && textInputV2->surface() == textInputSurface) {
121 cursorRectangle = textInputV2->cursorRectangle();
122 }
123 auto textInputV3 = waylandServer()->seat()->textInputV3();
124 if (textInputV3 && textInputV3->isEnabled() && textInputV3->surface() == textInputSurface) {
125 cursorRectangle = textInputV3->cursorRectangle();
126 }
127 if (textWindow) {
128 cursorRectangle.translate(textWindow->bufferGeometry().topLeft().toPoint());
129 const QRectF screen = Workspace::self()->clientArea(PlacementArea, this, cursorRectangle.bottomLeft());
130
131 // Reuse the similar logic like xdg popup
132 QRectF popupRect(popupOffset(cursorRectangle, Qt::BottomEdge | Qt::LeftEdge, Qt::RightEdge | Qt::BottomEdge, surface()->size()), surface()->size());
133
134 if (popupRect.left() < screen.left()) {
135 popupRect.moveLeft(screen.left());
136 }
137 if (popupRect.right() > screen.right()) {
138 popupRect.moveRight(screen.right());
139 }
140 if (popupRect.top() < screen.top() || popupRect.bottom() > screen.bottom()) {
141 auto flippedPopupRect =
142 QRectF(popupOffset(cursorRectangle, Qt::TopEdge | Qt::LeftEdge, Qt::RightEdge | Qt::TopEdge, surface()->size()), surface()->size());
143
144 // if it still doesn't fit we should continue with the unflipped version
145 if (flippedPopupRect.top() >= screen.top() || flippedPopupRect.bottom() <= screen.bottom()) {
146 popupRect.moveTop(flippedPopupRect.top());
147 }
148 }
149 moveResize(popupRect);
150 }
151 } break;
152 }
153}

155void InputPanelV1Window::destroyWindow()
156{
157 markAsZombie();
158
159 Deleted *deleted = Deleted::create(this);
160 Q_EMIT windowClosed(this, deleted);
161 StackingUpdatesBlocker blocker(workspace());
162 waylandServer()->removeWindow(this);
163 deleted->unrefWindow();
164
165 delete this;
166}

189void InputPanelV1Window::maybeShow()
190{
191 const bool shouldShow = m_mode == Mode::Overlay || (m_mode == Mode::VirtualKeyboard && m_allowed && m_virtualKeyboardShouldBeShown);
192 if (shouldShow && !isZombie() && surface()->isMapped()) {
193 setReadyForPainting();
194 reposition();
195 showClient();
196 }
197}
