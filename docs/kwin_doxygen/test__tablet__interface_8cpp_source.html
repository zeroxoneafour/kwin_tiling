177void TestTabletInterface::initTestCase()
178{
179 m_display.addSocketName(s_socketName);
180 m_display.start();
181 QVERIFY(m_display.isRunning());
182
183 m_seat = new SeatInterface(&m_display, this);
184 m_serverCompositor = new CompositorInterface(&m_display, this);
185 m_tabletManager = new TabletManagerV2Interface(&m_display, this);
186
187 connect(m_serverCompositor, &CompositorInterface::surfaceCreated, this, [this](SurfaceInterface *surface) {
188 m_surfaces += surface;
189 });
190
191 // setup connection
192 m_connection = new KWayland::Client::ConnectionThread;
193 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
194 m_connection->setSocketName(s_socketName);
195
196 m_thread = new QThread(this);
197 m_connection->moveToThread(m_thread);
198 m_thread->start();
199
200 m_connection->initConnection();
201 QVERIFY(connectedSpy.wait());
202 QVERIFY(!m_connection->connections().isEmpty());
203
204 m_queue = new KWayland::Client::EventQueue(this);
205 QVERIFY(!m_queue->isValid());
206 m_queue->setup(m_connection);
207 QVERIFY(m_queue->isValid());
208
209 auto registry = new KWayland::Client::Registry(this);
210 connect(registry, &KWayland::Client::Registry::interfaceAnnounced, this, [this, registry](const QByteArray &interface, quint32 name, quint32 version) {
211 if (interface == "zwp_tablet_manager_v2") {
212 auto tabletClient = new QtWayland::zwp_tablet_manager_v2(registry->registry(), name, version);
213 auto _seat = tabletClient->get_tablet_seat(*m_clientSeat);
214 m_tabletSeatClient = new TabletSeat(_seat);
215 }
216 });
217 connect(registry, &KWayland::Client::Registry::seatAnnounced, this, [this, registry](quint32 name, quint32 version) {
218 m_clientSeat = registry->createSeat(name, version);
219 });
220 registry->setEventQueue(m_queue);
221 QSignalSpy compositorSpy(registry, &KWayland::Client::Registry::compositorAnnounced);
222 registry->create(m_connection->display());
223 QVERIFY(registry->isValid());
224 registry->setup();
225 wl_display_flush(m_connection->display());
226
227 QVERIFY(compositorSpy.wait());
228 m_clientCompositor = registry->createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
229 QVERIFY(m_clientCompositor->isValid());
230
231 QSignalSpy surfaceSpy(m_serverCompositor, &CompositorInterface::surfaceCreated);
232 for (int i = 0; i < 3; ++i) {
233 m_surfacesClient += m_clientCompositor->createSurface(this);
234 }
235 QVERIFY(surfaceSpy.count() < 3 && surfaceSpy.wait(200));
236 QVERIFY(m_surfaces.count() == 3);
237 QVERIFY(m_tabletSeatClient);
238}

257void TestTabletInterface::testAdd()
258{
259 TabletSeatV2Interface *seatInterface = m_tabletManager->seat(m_seat);
260 QVERIFY(seatInterface);
261
262 QSignalSpy tabletSpy(m_tabletSeatClient, &TabletSeat::tabletAdded);
263 m_tablet = seatInterface->addTablet(1, 2, QStringLiteral("event33"), QStringLiteral("my tablet"), {QStringLiteral("/test/event33")});
264 QVERIFY(m_tablet);
265 QVERIFY(tabletSpy.wait() || tabletSpy.count() == 1);
266 QCOMPARE(m_tabletSeatClient->m_tablets.count(), 1);
267
268 QSignalSpy toolSpy(m_tabletSeatClient, &TabletSeat::toolAdded);
269 m_tool = seatInterface->addTool(KWaylandServer::TabletToolV2Interface::Pen, 0, 0, {TabletToolV2Interface::Tilt, TabletToolV2Interface::Pressure}, "my tablet");
270 QVERIFY(m_tool);
271 QVERIFY(toolSpy.wait() || toolSpy.count() == 1);
272 QCOMPARE(m_tabletSeatClient->m_tools.count(), 1);
273
274 QVERIFY(!m_tool->isClientSupported()); // There's no surface in it yet
275 m_tool->setCurrentSurface(nullptr);
276 QVERIFY(!m_tool->isClientSupported()); // There's no surface in it
277
278 QCOMPARE(m_surfaces.count(), 3);
279 for (SurfaceInterface *surface : m_surfaces) {
280 m_tool->setCurrentSurface(surface);
281 }
282 m_tool->setCurrentSurface(nullptr);

285void TestTabletInterface::testAddPad()
286{
287 TabletSeatV2Interface *seatInterface = m_tabletManager->seat(m_seat);
288 QVERIFY(seatInterface);
289
290 QSignalSpy tabletPadSpy(m_tabletSeatClient, &TabletSeat::padAdded);

293 QVERIFY(m_tabletPad);
294 QVERIFY(tabletPadSpy.wait() || tabletPadSpy.count() == 1);
295 QCOMPARE(m_tabletSeatClient->m_pads.count(), 1);
296 QVERIFY(m_tabletSeatClient->m_pads[0]);
297
298 QVERIFY(m_tabletPad->ring(0));
299 QVERIFY(m_tabletPad->strip(0));
300
301 QCOMPARE(m_surfaces.count(), 3);
302 QVERIFY(m_tabletSeatClient->m_pads[0]->buttonStates.isEmpty());
303 QSignalSpy buttonSpy(m_tabletSeatClient->m_pads[0], &TabletPad::buttonReceived);
304 m_tabletPad->setCurrentSurface(m_surfaces[0], m_tablet);
305 m_tabletPad->sendButton(123ms, 0, QtWayland::zwp_tablet_pad_v2::button_state_pressed);
306 QVERIFY(buttonSpy.count() || buttonSpy.wait(100));
307 QCOMPARE(m_tabletSeatClient->m_pads[0]->doneCalled, true);
308 QCOMPARE(m_tabletSeatClient->m_pads[0]->buttonStates.count(), 1);
309 QCOMPARE(m_tabletSeatClient->m_pads[0]->buttonStates[*m_surfacesClient[0]][0], QtWayland::zwp_tablet_pad_v2::button_state_pressed);
310}
