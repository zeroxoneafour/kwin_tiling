20static QString outputName(const EffectScreen *screen)
21{
22 const auto screens = effects->screens();
23 const bool shouldShowSerialNumber = std::any_of(screens.cbegin(), screens.cend(), [screen](const EffectScreen *other) {
24 return other != screen && other->manufacturer() == screen->manufacturer() && other->model() == screen->model();
25 });
26 const bool shouldShowConnector = shouldShowSerialNumber && std::any_of(screens.cbegin(), screens.cend(), [screen](const EffectScreen *other) {
27 return other != screen && other->serialNumber() == screen->serialNumber();
28 });
29
30 QStringList parts;
31 if (!screen->manufacturer().isEmpty()) {
32 parts.append(screen->manufacturer());
33 }
34
35 if (!screen->model().isEmpty()) {
36 parts.append(screen->model());
37 }
38
39 if (shouldShowSerialNumber && !screen->serialNumber().isEmpty()) {
40 parts.append(screen->serialNumber());
41 }
42
43 if (shouldShowConnector) {
44 parts.append(screen->name());
45 }
46
47 if (parts.isEmpty()) {
48 return i18nc("@label", "Unknown");
49 } else {
50 return parts.join(QLatin1Char(' '));
51 }
52}

70void OutputLocatorEffect::show()
71{
72 if (isActive()) {
73 m_showTimer.start(std::chrono::milliseconds(2500));
74 return;
75 }
76
77 // Needed until Qt6 https://codereview.qt-project.org/c/qt/qtdeclarative/+/361506
78 m_dummyWindow = std::make_unique<QWindow>();
79 m_dummyWindow->create();
80
81 const auto screens = effects->screens();
82 for (const auto screen : screens) {
83 auto scene = new OffscreenQuickScene(this, m_dummyWindow.get());
84 scene->setSource(m_qmlUrl, {{QStringLiteral("outputName"), outputName(screen)}, {QStringLiteral("resolution"), screen->geometry().size()}, {QStringLiteral("scale"), screen->devicePixelRatio()}});
85 QRectF geometry(0, 0, scene->rootItem()->implicitWidth(), scene->rootItem()->implicitHeight());
86 geometry.moveCenter(screen->geometry().center());
87 scene->setGeometry(geometry.toRect());
88 connect(scene, &OffscreenQuickView::repaintNeeded, this, [scene] {
89 effects->addRepaint(scene->geometry());
90 });
91 m_scenesByScreens.insert(screen, scene);
92 }
93
94 m_showTimer.start(std::chrono::milliseconds(2500));
95}

97void OutputLocatorEffect::hide()
98{
99 m_showTimer.stop();
100 const QRegion repaintRegion = std::accumulate(m_scenesByScreens.cbegin(), m_scenesByScreens.cend(), QRegion(), [](QRegion region, OffscreenQuickScene *scene) {
101 return region |= scene->geometry();
102 });
103 qDeleteAll(m_scenesByScreens);
104 m_scenesByScreens.clear();
105 effects->addRepaint(repaintRegion);
106}
