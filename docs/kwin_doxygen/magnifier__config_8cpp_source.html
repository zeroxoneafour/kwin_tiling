40MagnifierEffectConfig::MagnifierEffectConfig(QWidget *parent, const QVariantList &args)
41 : KCModule(parent, args)
42 , m_ui(this)
43{
44 QVBoxLayout *layout = new QVBoxLayout(this);
45 layout->addWidget(&m_ui);
46
47 MagnifierConfig::instance(KWIN_CONFIG);
48 addConfig(MagnifierConfig::self(), &m_ui);
49
50 connect(m_ui.editor, &KShortcutsEditor::keyChange, this, &MagnifierEffectConfig::markAsChanged);
51
52 // Shortcut config. The shortcut belongs to the component "kwin"!
53 m_actionCollection = new KActionCollection(this, QStringLiteral("kwin"));
54
55 m_actionCollection->setComponentDisplayName(i18n("KWin"));
56 m_actionCollection->setConfigGroup(QStringLiteral("Magnifier"));
57 m_actionCollection->setConfigGlobal(true);
58
59 QAction *a;
60 a = m_actionCollection->addAction(KStandardAction::ZoomIn);
61 a->setProperty("isConfigurationAction", true);
62 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Equal));
63 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Equal));
64
65 a = m_actionCollection->addAction(KStandardAction::ZoomOut);
66 a->setProperty("isConfigurationAction", true);
67 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Minus));
68 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Minus));
69
70 a = m_actionCollection->addAction(KStandardAction::ActualSize);
71 a->setProperty("isConfigurationAction", true);
72 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_0));
73 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_0));
74
75 m_ui.editor->addCollection(m_actionCollection);
76}
