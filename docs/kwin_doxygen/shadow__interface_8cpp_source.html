43void ShadowManagerInterfacePrivate::org_kde_kwin_shadow_manager_create(Resource *resource, uint32_t id, wl_resource *surface)
44{
45 SurfaceInterface *s = SurfaceInterface::get(surface);
46 if (!s) {
47 wl_resource_post_error(resource->handle, 0, "Invalid surface");
48 return;
49 }
50
51 wl_resource *shadow_resource = wl_resource_create(resource->client(), &org_kde_kwin_shadow_interface, resource->version(), id);
52 if (!shadow_resource) {
53 wl_client_post_no_memory(resource->client());
54 return;
55 }
56
57 auto shadow = new ShadowInterface(q, shadow_resource);
58
59 SurfaceInterfacePrivate *surfacePrivate = SurfaceInterfacePrivate::get(s);
60 surfacePrivate->setShadow(QPointer<ShadowInterface>(shadow));
61}

145void ShadowInterfacePrivate::org_kde_kwin_shadow_commit(Resource *resource)
146{
147#define BUFFER(__FLAG__, __PART__) \
148 if (pending.flags & State::Flags::__FLAG__##Buffer) { \
149 if (current.__PART__) { \
150 current.__PART__->unref(); \
151 } \
152 if (pending.__PART__) { \
153 pending.__PART__->ref(); \
154 } \
155 current.__PART__ = pending.__PART__; \
156 }
157 BUFFER(Left, left)
158 BUFFER(TopLeft, topLeft)
159 BUFFER(Top, top)
160 BUFFER(TopRight, topRight)
161 BUFFER(Right, right)
162 BUFFER(BottomRight, bottomRight)
163 BUFFER(Bottom, bottom)
164 BUFFER(BottomLeft, bottomLeft)
165#undef BUFFER
166
167 if (pending.flags & State::Offset) {
168 current.offset = pending.offset;
169 }
170 pending = State();
171}

173void ShadowInterfacePrivate::attach(ShadowInterfacePrivate::State::Flags flag, wl_resource *buffer)
174{
175 ClientBuffer *b = manager->display()->clientBufferForResource(buffer);
176 switch (flag) {
177 case State::LeftBuffer:
178 pending.left = b;
179 break;
180 case State::TopLeftBuffer:
181 pending.topLeft = b;
182 break;
183 case State::TopBuffer:
184 pending.top = b;
185 break;
186 case State::TopRightBuffer:
187 pending.topRight = b;
188 break;
189 case State::RightBuffer:
190 pending.right = b;
191 break;
192 case State::BottomRightBuffer:
193 pending.bottomRight = b;
194 break;
195 case State::BottomBuffer:
196 pending.bottom = b;
197 break;
198 case State::BottomLeftBuffer:
199 pending.bottomLeft = b;
200 break;
201 default:
202 Q_UNREACHABLE();
203 break;
204 }
205 pending.flags = State::Flags(pending.flags | flag);
206}
