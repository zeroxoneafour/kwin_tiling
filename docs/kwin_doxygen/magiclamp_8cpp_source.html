47void MagicLampEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
48{
49 auto animationIt = m_animations.begin();
50 while (animationIt != m_animations.end()) {
51 (*animationIt).timeLine.advance(presentTime);
52 ++animationIt;
53 }
54
55 // We need to mark the screen windows as transformed. Otherwise the
56 // whole screen won't be repainted, resulting in artefacts.
57 data.mask |= PAINT_SCREEN_WITH_TRANSFORMED_WINDOWS;
58
59 effects->prePaintScreen(data, presentTime);
60}

74void MagicLampEffect::apply(EffectWindow *w, int mask, WindowPaintData &data, WindowQuadList &quads)
75{
76 auto animationIt = m_animations.constFind(w);
77 if (animationIt != m_animations.constEnd()) {
78 // 0 = not minimized, 1 = fully minimized
79 const qreal progress = (*animationIt).timeLine.value();
80
81 QRect geo = w->frameGeometry().toRect();
82 QRect icon = w->iconGeometry().toRect();
83 IconPosition position = Top;
84 // If there's no icon geometry, minimize to the center of the screen
85 if (!icon.isValid()) {
86 QRect extG = geo;
87 QPoint pt = cursorPos().toPoint();
88 // focussing inside the window is no good, leads to ugly artefacts, find nearest border
89 if (extG.contains(pt)) {
90 const int d[2][2] = {{pt.x() - extG.x(), extG.right() - pt.x()},
91 {pt.y() - extG.y(), extG.bottom() - pt.y()}};
92 int di = d[1][0];
93 position = Top;
94 if (d[0][0] < di) {
95 di = d[0][0];
96 position = Left;
97 }
98 if (d[1][1] < di) {
99 di = d[1][1];
100 position = Bottom;
101 }
102 if (d[0][1] < di) {
103 position = Right;
104 }
105 switch (position) {
106 case Top:
107 pt.setY(extG.y());
108 break;
109 case Left:
110 pt.setX(extG.x());
111 break;
112 case Bottom:
113 pt.setY(extG.bottom());
114 break;
115 case Right:
116 pt.setX(extG.right());
117 break;
118 }
119 } else {
120 if (pt.y() < geo.y()) {
121 position = Top;
122 } else if (pt.x() < geo.x()) {
123 position = Left;
124 } else if (pt.y() > geo.bottom()) {
125 position = Bottom;
126 } else if (pt.x() > geo.right()) {
127 position = Right;
128 }
129 }
130 icon = QRect(pt, QSize(0, 0));
131 } else {
132 // Assumption: there is a panel containing the icon position
133 EffectWindow *panel = nullptr;
134 const auto stackingOrder = effects->stackingOrder();
135 for (EffectWindow *window : stackingOrder) {
136 if (!window->isDock()) {
137 continue;
138 }
139 // we have to use intersects as there seems to be a Plasma bug
140 // the published icon geometry might be bigger than the panel
141 if (window->frameGeometry().intersects(icon)) {
142 panel = window;
143 break;
144 }
145 }
146 if (panel) {
147 // Assumption: width of horizonal panel is greater than its height and vice versa
148 // The panel has to border one screen edge, so get it's screen area
149 QRectF panelScreen = effects->clientArea(ScreenArea, panel);
150 if (panel->width() >= panel->height()) {
151 // horizontal panel
152 if (panel->y() <= panelScreen.height() / 2) {
153 position = Top;
154 } else {
155 position = Bottom;
156 }
157 } else {
158 // vertical panel
159 if (panel->x() <= panelScreen.width() / 2) {
160 position = Left;
161 } else {
162 position = Right;
163 }
164 }
165 } else {
166 // we did not find a panel, so it might be autohidden
167 QRectF iconScreen = effects->clientArea(ScreenArea, icon.topLeft(), effects->currentDesktop());
168 // as the icon geometry could be overlap a screen edge we use an intersection
169 QRectF rect = iconScreen.intersected(icon);
170 // here we need a different assumption: icon geometry borders one screen edge
171 // this assumption might be wrong for e.g. task applet being the only applet in panel
172 // in this case the icon borders two screen edges
173 // there might be a wrong animation, but not distorted
174 if (rect.x() == iconScreen.x()) {
175 position = Left;
176 } else if (rect.x() + rect.width() == iconScreen.x() + iconScreen.width()) {
177 position = Right;
178 } else if (rect.y() == iconScreen.y()) {
179 position = Top;
180 } else {
181 position = Bottom;
182 }
183 }
184 }
185
186#define SET_QUADS(_SET_A_, _A_, _DA_, _SET_B_, _B_, _O0_, _O1_, _O2_, _O3_) \
187 quad[0]._SET_A_((icon._A_() + icon._DA_() * (quad[0]._A_() / geo._DA_()) - (quad[0]._A_() + geo._A_())) * p_progress[_O0_] + quad[0]._A_()); \
188 quad[1]._SET_A_((icon._A_() + icon._DA_() * (quad[1]._A_() / geo._DA_()) - (quad[1]._A_() + geo._A_())) * p_progress[_O1_] + quad[1]._A_()); \
189 quad[2]._SET_A_((icon._A_() + icon._DA_() * (quad[2]._A_() / geo._DA_()) - (quad[2]._A_() + geo._A_())) * p_progress[_O2_] + quad[2]._A_()); \
190 quad[3]._SET_A_((icon._A_() + icon._DA_() * (quad[3]._A_() / geo._DA_()) - (quad[3]._A_() + geo._A_())) * p_progress[_O3_] + quad[3]._A_()); \
191 \
192 quad[0]._SET_B_(quad[0]._B_() + offset[_O0_]); \
193 quad[1]._SET_B_(quad[1]._B_() + offset[_O1_]); \
194 quad[2]._SET_B_(quad[2]._B_() + offset[_O2_]); \
195 quad[3]._SET_B_(quad[3]._B_() + offset[_O3_])
196
197 quads = quads.makeGrid(40);
198 float quadFactor; // defines how fast a quad is vertically moved: y coordinates near to window top are slowed down
199 // it is used as quadFactor^3/windowHeight^3
200 // quadFactor is the y position of the quad but is changed towards becomming the window height
201 // by that the factor becomes 1 and has no influence any more
202 float offset[2] = {0, 0}; // how far has a quad to be moved? Distance between icon and window multiplied by the progress and by the quadFactor
203 float p_progress[2] = {0, 0}; // the factor which defines how far the x values have to be changed
204 // factor is the current moved y value diveded by the distance between icon and window
205 WindowQuad lastQuad;
206 lastQuad[0].setX(-1);
207 lastQuad[0].setY(-1);
208 lastQuad[1].setX(-1);
209 lastQuad[1].setY(-1);
210 lastQuad[2].setX(-1);
211 lastQuad[2].setY(-1);
212
213 if (position == Bottom) {
214 float height_cube = float(geo.height()) * float(geo.height()) * float(geo.height());
215 for (WindowQuad &quad : quads) {
216
217 if (quad[0].y() != lastQuad[0].y() || quad[2].y() != lastQuad[2].y()) {
218 quadFactor = quad[0].y() + (geo.height() - quad[0].y()) * progress;
219 offset[0] = (icon.y() + quad[0].y() - geo.y()) * progress * ((quadFactor * quadFactor * quadFactor) / height_cube);
220 quadFactor = quad[2].y() + (geo.height() - quad[2].y()) * progress;
221 offset[1] = (icon.y() + quad[2].y() - geo.y()) * progress * ((quadFactor * quadFactor * quadFactor) / height_cube);
222 p_progress[1] = std::min(offset[1] / (icon.y() + icon.height() - geo.y() - float(quad[2].y())), 1.0f);
223 p_progress[0] = std::min(offset[0] / (icon.y() + icon.height() - geo.y() - float(quad[0].y())), 1.0f);
224 } else {
225 lastQuad = quad;
226 }
227
228 p_progress[0] = std::abs(p_progress[0]);
229 p_progress[1] = std::abs(p_progress[1]);
230 // x values are moved towards the center of the icon
231 SET_QUADS(setX, x, width, setY, y, 0, 0, 1, 1);
232 }
233 } else if (position == Top) {
234 float height_cube = float(geo.height()) * float(geo.height()) * float(geo.height());
235 for (WindowQuad &quad : quads) {
236
237 if (quad[0].y() != lastQuad[0].y() || quad[2].y() != lastQuad[2].y()) {
238 quadFactor = geo.height() - quad[0].y() + (quad[0].y()) * progress;
239 offset[0] = (geo.y() - icon.height() + geo.height() + quad[0].y() - icon.y()) * progress * ((quadFactor * quadFactor * quadFactor) / height_cube);
240 quadFactor = geo.height() - quad[2].y() + (quad[2].y()) * progress;
241 offset[1] = (geo.y() - icon.height() + geo.height() + quad[2].y() - icon.y()) * progress * ((quadFactor * quadFactor * quadFactor) / height_cube);
242 p_progress[0] = std::min(offset[0] / (geo.y() - icon.height() + geo.height() - icon.y() - float(geo.height() - quad[0].y())), 1.0f);
243 p_progress[1] = std::min(offset[1] / (geo.y() - icon.height() + geo.height() - icon.y() - float(geo.height() - quad[2].y())), 1.0f);
244 } else {
245 lastQuad = quad;
246 }
247
248 offset[0] = -offset[0];
249 offset[1] = -offset[1];
250
251 p_progress[0] = std::abs(p_progress[0]);
252 p_progress[1] = std::abs(p_progress[1]);
253 // x values are moved towards the center of the icon
254 SET_QUADS(setX, x, width, setY, y, 0, 0, 1, 1);
255 }
256 } else if (position == Left) {
257 float width_cube = float(geo.width()) * float(geo.width()) * float(geo.width());
258 for (WindowQuad &quad : quads) {
259
260 if (quad[0].x() != lastQuad[0].x() || quad[1].x() != lastQuad[1].x()) {
261 quadFactor = geo.width() - quad[0].x() + (quad[0].x()) * progress;
262 offset[0] = (geo.x() - icon.width() + geo.width() + quad[0].x() - icon.x()) * progress * ((quadFactor * quadFactor * quadFactor) / width_cube);
263 quadFactor = geo.width() - quad[1].x() + (quad[1].x()) * progress;
264 offset[1] = (geo.x() - icon.width() + geo.width() + quad[1].x() - icon.x()) * progress * ((quadFactor * quadFactor * quadFactor) / width_cube);
265 p_progress[0] = std::min(offset[0] / (geo.x() - icon.width() + geo.width() - icon.x() - float(geo.width() - quad[0].x())), 1.0f);
266 p_progress[1] = std::min(offset[1] / (geo.x() - icon.width() + geo.width() - icon.x() - float(geo.width() - quad[1].x())), 1.0f);
267 } else {
268 lastQuad = quad;
269 }
270
271 offset[0] = -offset[0];
272 offset[1] = -offset[1];
273
274 p_progress[0] = std::abs(p_progress[0]);
275 p_progress[1] = std::abs(p_progress[1]);
276 // y values are moved towards the center of the icon
277 SET_QUADS(setY, y, height, setX, x, 0, 1, 1, 0);
278 }
279 } else if (position == Right) {
280 float width_cube = float(geo.width()) * float(geo.width()) * float(geo.width());
281 for (WindowQuad &quad : quads) {
282
283 if (quad[0].x() != lastQuad[0].x() || quad[1].x() != lastQuad[1].x()) {
284 quadFactor = quad[0].x() + (geo.width() - quad[0].x()) * progress;
285 offset[0] = (icon.x() + quad[0].x() - geo.x()) * progress * ((quadFactor * quadFactor * quadFactor) / width_cube);
286 quadFactor = quad[1].x() + (geo.width() - quad[1].x()) * progress;
287 offset[1] = (icon.x() + quad[1].x() - geo.x()) * progress * ((quadFactor * quadFactor * quadFactor) / width_cube);
288 p_progress[0] = std::min(offset[0] / (icon.x() + icon.width() - geo.x() - float(quad[0].x())), 1.0f);
289 p_progress[1] = std::min(offset[1] / (icon.x() + icon.width() - geo.x() - float(quad[1].x())), 1.0f);
290 } else {
291 lastQuad = quad;
292 }
293
294 p_progress[0] = std::abs(p_progress[0]);
295 p_progress[1] = std::abs(p_progress[1]);
296 // y values are moved towards the center of the icon
297 SET_QUADS(setY, y, height, setX, x, 0, 1, 1, 0);
298 }
299 }
300 }

303void MagicLampEffect::postPaintScreen()
304{
305 auto animationIt = m_animations.begin();
306 while (animationIt != m_animations.end()) {
307 if ((*animationIt).timeLine.done()) {
308 unredirect(animationIt.key());
309 animationIt = m_animations.erase(animationIt);
310 } else {
311 ++animationIt;
312 }
313 }
314
315 effects->addRepaintFull();
316
317 // Call the next effect.
318 effects->postPaintScreen();
319}

326void MagicLampEffect::slotWindowMinimized(EffectWindow *w)
327{
328 if (effects->activeFullScreenEffect()) {
329 return;
330 }
331
332 MagicLampAnimation &animation = m_animations[w];
333
334 if (animation.timeLine.running()) {
335 animation.timeLine.toggleDirection();
336 } else {
337 animation.visibleRef = EffectWindowVisibleRef(w, EffectWindow::PAINT_DISABLED_BY_MINIMIZE);
338 animation.timeLine.setDirection(TimeLine::Forward);
339 animation.timeLine.setDuration(m_duration);
340 animation.timeLine.setEasingCurve(QEasingCurve::Linear);
341 }
342
343 redirect(w);
344 effects->addRepaintFull();
345}
