61void ClientConnectionPrivate::destroyListenerCallback(wl_listener *listener, void *data)
62{
63 wl_client *client = reinterpret_cast<wl_client *>(data);
64 auto it = std::find_if(s_allClients.constBegin(), s_allClients.constEnd(), [client](ClientConnectionPrivate *c) {
65 return c->client == client;
66 });
67 Q_ASSERT(it != s_allClients.constEnd());
68 auto p = (*it);
69 auto q = p->q;
70 Q_EMIT q->aboutToBeDestroyed();
71 p->client = nullptr;
72 wl_list_remove(&p->listener.link);
73 Q_EMIT q->disconnected(q);
74 q->deleteLater();
75}
