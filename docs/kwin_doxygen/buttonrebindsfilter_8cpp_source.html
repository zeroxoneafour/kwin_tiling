125ButtonRebindsFilter::ButtonRebindsFilter()
126 : m_configWatcher(KConfigWatcher::create(KSharedConfig::openConfig("kcminputrc")))
127{
128 KWin::input()->addInputDevice(&m_inputDevice);
129 const QLatin1String groupName("ButtonRebinds");
130 connect(m_configWatcher.get(), &KConfigWatcher::configChanged, this, [this, groupName](const KConfigGroup &group) {
131 if (group.parent().name() == groupName) {
132 loadConfig(group.parent());
133 } else if (group.parent().parent().name() == groupName) {
134 loadConfig(group.parent().parent());
135 }
136 });
137 loadConfig(m_configWatcher->config()->group(groupName));
138}

140void ButtonRebindsFilter::loadConfig(const KConfigGroup &group)
141{
142 Q_ASSERT(QLatin1String("ButtonRebinds") == group.name());
143 KWin::input()->uninstallInputEventFilter(this);
144 for (auto &action : m_actions) {
145 action.clear();
146 }
147
148 bool foundActions = false;
149 const auto mouseButtonEnum = QMetaEnum::fromType<Qt::MouseButtons>();
150 const auto mouseGroup = group.group("Mouse");
151 static constexpr auto maximumQtExtraButton = 24;
152 for (int i = 1; i <= maximumQtExtraButton; ++i) {
153 const QByteArray buttonName = QByteArray("ExtraButton") + QByteArray::number(i);
154 if (mouseGroup.hasKey(buttonName.constData())) {
155 const auto entry = mouseGroup.readEntry(buttonName.constData(), QStringList());
156 const auto button = static_cast<quint32>(mouseButtonEnum.keyToValue(buttonName));
157 insert(Pointer, {QString(), button}, entry);
158 foundActions = true;
159 }
160 }
161
162 const auto tabletsGroup = group.group("Tablet");
163 const auto tablets = tabletsGroup.groupList();
164 for (const auto &tabletName : tablets) {
165 const auto tabletGroup = tabletsGroup.group(tabletName);
166 const auto tabletButtons = tabletGroup.keyList();
167 for (const auto &buttonName : tabletButtons) {
168 const auto entry = tabletGroup.readEntry(buttonName, QStringList());
169 bool ok = false;
170 const uint button = buttonName.toUInt(&ok);
171 if (ok) {
172 foundActions = true;
173 insert(TabletPad, {tabletName, button}, entry);
174 }
175 }
176 }
177
178 const auto tabletToolsGroup = group.group("TabletTool");
179 const auto tabletTools = tabletToolsGroup.groupList();
180 for (const auto &tabletToolName : tabletTools) {
181 const auto toolGroup = tabletToolsGroup.group(tabletToolName);
182 const auto tabletToolButtons = toolGroup.keyList();
183 for (const auto &buttonName : tabletToolButtons) {
184 const auto entry = toolGroup.readEntry(buttonName, QStringList());
185 bool ok = false;
186 const uint button = buttonName.toUInt(&ok);
187 if (ok) {
188 foundActions = true;
189 insert(TabletToolButtonType, {tabletToolName, button}, entry);
190 }
191 }
192 }
193
194 if (foundActions) {
195 KWin::input()->prependInputEventFilter(this);
196 }
197}

199bool ButtonRebindsFilter::pointerEvent(KWin::MouseEvent *event, quint32 nativeButton)
200{
201 if (event->type() != QEvent::MouseButtonPress && event->type() != QEvent::MouseButtonRelease) {
202 return false;
203 }
204 if (RebindScope::isRebinding()) {
205 return false;
206 }
207
208 return send(Pointer, {{}, event->button()}, event->type() == QEvent::MouseButtonPress, event->timestamp());

228void ButtonRebindsFilter::insert(TriggerType type, const Trigger &trigger, const QStringList &entry)
229{
230 if (entry.size() != 2) {
231 qCWarning(KWIN_BUTTONREBINDS) << "Failed to rebind to" << entry;
232 return;
233 }
234 if (entry.first() == QLatin1String("Key")) {
235 const auto keys = QKeySequence::fromString(entry.at(1), QKeySequence::PortableText);
236 if (!keys.isEmpty()) {
237 m_actions.at(type).insert(trigger, keys);
238 }
239 } else if (entry.first() == QLatin1String("MouseButton")) {
240 bool ok = false;
241 const MouseButton mb{entry.last().toUInt(&ok)};
242 if (ok) {
243 m_actions.at(type).insert(trigger, mb);
244 } else {
245 qCWarning(KWIN_BUTTONREBINDS) << "Could not convert" << entry << "into a mouse button";
246 }
247 } else if (entry.first() == QLatin1String("TabletToolButton")) {
248 bool ok = false;
249 const TabletToolButton tb{entry.last().toUInt(&ok)};
250 if (ok) {
251 m_actions.at(type).insert(trigger, tb);
252 } else {
253 qCWarning(KWIN_BUTTONREBINDS) << "Could not convert" << entry << "into a mouse button";
254 }
255 }
256}

258bool ButtonRebindsFilter::send(TriggerType type, const Trigger &trigger, bool pressed, std::chrono::microseconds timestamp)
259{
260 const auto &typeActions = m_actions.at(type);
261 if (typeActions.isEmpty()) {
262 return false;
263 }
264
265 const auto &action = typeActions[trigger];
266 if (const QKeySequence *seq = std::get_if<QKeySequence>(&action)) {
267 return sendKeySequence(*seq, pressed, timestamp);
268 }
269 if (const auto mb = std::get_if<MouseButton>(&action)) {
270 return sendMouseButton(mb->button, pressed, timestamp);
271 }
272 if (const auto tb = std::get_if<TabletToolButton>(&action)) {
273 return sendTabletToolButton(tb->button, pressed, timestamp);
274 }
275 return false;
276}

278bool ButtonRebindsFilter::sendKeySequence(const QKeySequence &keys, bool pressed, std::chrono::microseconds time)
279{
280 if (keys.isEmpty()) {
281 return false;
282 }
283 const auto &key = keys[0];
284
285 int sym = -1;
286 if (!KKeyServer::keyQtToSymX(keys[0], &sym)) {
287 qCWarning(KWIN_BUTTONREBINDS) << "Could not convert" << keys << "to keysym";
288 return false;
289 }
290 // KKeyServer returns upper case syms, lower it to not confuse modifiers handling
291 auto keyCode = KWin::input()->keyboard()->xkb()->keycodeFromKeysym(sym);
292 if (!keyCode) {
293 qCWarning(KWIN_BUTTONREBINDS) << "Could not convert" << keys << "sym: " << sym << "to keycode";
294 return false;
295 }
296
297 RebindScope scope;
298 auto sendKey = [this, pressed, time](xkb_keycode_t key) {
299 auto state = pressed ? KWin::InputRedirection::KeyboardKeyPressed : KWin::InputRedirection::KeyboardKeyReleased;
300 Q_EMIT m_inputDevice.keyChanged(key, state, time, &m_inputDevice);
301 };
302
303 if (key & Qt::ShiftModifier) {
304 sendKey(KEY_LEFTSHIFT);
305 }
306 if (key & Qt::ControlModifier) {
307 sendKey(KEY_LEFTCTRL);
308 }
309 if (key & Qt::AltModifier) {
310 sendKey(KEY_LEFTALT);
311 }
312 if (key & Qt::MetaModifier) {
313 sendKey(KEY_LEFTMETA);
314 }
315
316 sendKey(keyCode.value());
317 return true;
318}
