115void TestXdgOutput::testChanges()
116{
117 // verify the server modes
118 using namespace KWaylandServer;
119 KWayland::Client::Registry registry;
120 QSignalSpy announced(&registry, &KWayland::Client::Registry::outputAnnounced);
121 QSignalSpy xdgOutputAnnounced(&registry, &KWayland::Client::Registry::xdgOutputAnnounced);
122
123 registry.setEventQueue(m_queue);
124 registry.create(m_connection->display());
125 QVERIFY(registry.isValid());
126 registry.setup();
127 QVERIFY(announced.wait());
128 if (xdgOutputAnnounced.count() != 1) {
129 QVERIFY(xdgOutputAnnounced.wait());
130 }
131
132 KWayland::Client::Output output;
133 QSignalSpy outputChanged(&output, &KWayland::Client::Output::changed);
134
135 output.setup(registry.bindOutput(announced.first().first().value<quint32>(), announced.first().last().value<quint32>()));
136 QVERIFY(outputChanged.wait());
137
138 std::unique_ptr<KWayland::Client::XdgOutputManager> xdgOutputManager(
139 registry.createXdgOutputManager(xdgOutputAnnounced.first().first().value<quint32>(), xdgOutputAnnounced.first().last().value<quint32>(), this));
140
141 std::unique_ptr<KWayland::Client::XdgOutput> xdgOutput(xdgOutputManager->getXdgOutput(&output, this));
142 QSignalSpy xdgOutputChanged(xdgOutput.get(), &KWayland::Client::XdgOutput::changed);
143
144 // check details are sent on client bind
145 QVERIFY(xdgOutputChanged.wait());
146 xdgOutputChanged.clear();
147 QCOMPARE(xdgOutput->logicalPosition(), QPoint(11, 12));
148 QCOMPARE(xdgOutput->logicalSize(), QSize(1280, 720));
149 QCOMPARE(xdgOutput->name(), "testName");
150 QCOMPARE(xdgOutput->description(), "foo bar");
151
152 // change the logical position
153 m_outputHandle->moveTo(QPoint(1000, 2000));
154 QVERIFY(xdgOutputChanged.wait());
155 QCOMPARE(xdgOutputChanged.count(), 1);
156 QCOMPARE(xdgOutput->logicalPosition(), QPoint(1000, 2000));
157 QEXPECT_FAIL("", "KWayland::Client::XdgOutput incorrectly handles partial updates", Continue);
158 QCOMPARE(xdgOutput->logicalSize(), QSize(1280, 720));
159
160 // change the logical size
161 m_outputHandle->setScale(2);
162 QVERIFY(xdgOutputChanged.wait());
163 QCOMPARE(xdgOutputChanged.count(), 2);
164 QEXPECT_FAIL("", "KWayland::Client::XdgOutput incorrectly handles partial updates", Continue);
165 QCOMPARE(xdgOutput->logicalPosition(), QPoint(1000, 2000));
166 QCOMPARE(xdgOutput->logicalSize(), QSize(960, 540));
167}
