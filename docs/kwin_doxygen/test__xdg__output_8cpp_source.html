53void TestXdgOutput::init()
54{
55 using namespace KWin;
56 delete m_display;
57 m_display = new KWin::Display(this);
58 m_display->addSocketName(s_socketName);
59 m_display->start();
60 QVERIFY(m_display->isRunning());
61
62 m_outputHandle = std::make_unique<FakeOutput>();
63 m_outputHandle->setMode(QSize(1920, 1080), 60000);
64 m_outputHandle->moveTo(QPoint(11, 12)); // not a sensible value for one monitor, but works for this test
65 m_outputHandle->setScale(1.5);
66 m_outputHandle->setName("testName");
67 m_outputHandle->setManufacturer("foo");
68 m_outputHandle->setModel("bar");
69
70 m_serverOutput = new OutputInterface(m_display, m_outputHandle.get(), this);
71
72 m_serverXdgOutputManager = new XdgOutputManagerV1Interface(m_display, this);
73 m_serverXdgOutputManager->offer(m_serverOutput);
74
75 // setup connection
76 m_connection = new KWayland::Client::ConnectionThread;
77 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
78 m_connection->setSocketName(s_socketName);
79
80 m_thread = new QThread(this);
81 m_connection->moveToThread(m_thread);
82 m_thread->start();
83
84 m_connection->initConnection();
85 QVERIFY(connectedSpy.wait());
86
87 m_queue = new KWayland::Client::EventQueue(this);
88 QVERIFY(!m_queue->isValid());
89 m_queue->setup(m_connection);
90 QVERIFY(m_queue->isValid());

116void TestXdgOutput::testChanges()
117{
118 // verify the server modes
119 using namespace KWin;
120 KWayland::Client::Registry registry;
121 QSignalSpy announced(&registry, &KWayland::Client::Registry::outputAnnounced);
122 QSignalSpy xdgOutputAnnounced(&registry, &KWayland::Client::Registry::xdgOutputAnnounced);
123
124 registry.setEventQueue(m_queue);
125 registry.create(m_connection->display());
126 QVERIFY(registry.isValid());
127 registry.setup();
128 QVERIFY(announced.wait());
129 if (xdgOutputAnnounced.count() != 1) {
130 QVERIFY(xdgOutputAnnounced.wait());
131 }
132
133 KWayland::Client::Output output;
134 QSignalSpy outputChanged(&output, &KWayland::Client::Output::changed);
135
136 output.setup(registry.bindOutput(announced.first().first().value<quint32>(), announced.first().last().value<quint32>()));
137 QVERIFY(outputChanged.wait());
138
139 std::unique_ptr<KWayland::Client::XdgOutputManager> xdgOutputManager(
140 registry.createXdgOutputManager(xdgOutputAnnounced.first().first().value<quint32>(), xdgOutputAnnounced.first().last().value<quint32>(), this));
141
142 std::unique_ptr<KWayland::Client::XdgOutput> xdgOutput(xdgOutputManager->getXdgOutput(&output, this));
143 QSignalSpy xdgOutputChanged(xdgOutput.get(), &KWayland::Client::XdgOutput::changed);
144
145 // check details are sent on client bind
146 QVERIFY(xdgOutputChanged.wait());
147 xdgOutputChanged.clear();
148 QCOMPARE(xdgOutput->logicalPosition(), QPoint(11, 12));
149 QCOMPARE(xdgOutput->logicalSize(), QSize(1280, 720));
150 QCOMPARE(xdgOutput->name(), "testName");
151 QCOMPARE(xdgOutput->description(), "foo bar");
152
153 // change the logical position
154 m_outputHandle->moveTo(QPoint(1000, 2000));
155 QVERIFY(xdgOutputChanged.wait());
156 QCOMPARE(xdgOutputChanged.count(), 1);
157 QCOMPARE(xdgOutput->logicalPosition(), QPoint(1000, 2000));
158 QEXPECT_FAIL("", "KWayland::Client::XdgOutput incorrectly handles partial updates", Continue);
159 QCOMPARE(xdgOutput->logicalSize(), QSize(1280, 720));
160
161 // change the logical size
162 m_outputHandle->setScale(2);
163 QVERIFY(xdgOutputChanged.wait());
164 QCOMPARE(xdgOutputChanged.count(), 2);
165 QEXPECT_FAIL("", "KWayland::Client::XdgOutput incorrectly handles partial updates", Continue);
166 QCOMPARE(xdgOutput->logicalPosition(), QPoint(1000, 2000));
167 QCOMPARE(xdgOutput->logicalSize(), QSize(960, 540));
168}
