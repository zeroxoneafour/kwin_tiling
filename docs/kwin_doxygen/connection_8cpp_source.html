41class ConnectionAdaptor : public QObject
42{
43 Q_OBJECT
44 Q_CLASSINFO("D-Bus Interface", "org.kde.KWin.InputDeviceManager")
45 Q_PROPERTY(QStringList devicesSysNames READ devicesSysNames CONSTANT)
46
47private:
48 Connection *m_con;
49
50public:
51 ConnectionAdaptor(Connection *con)
52 : QObject(con)
53 , m_con(con)
54 {
55 connect(con, &Connection::deviceAdded, this, [this](LibInput::Device *inputDevice) {
56 Q_EMIT deviceAdded(inputDevice->sysName());
57 });
58 connect(con, &Connection::deviceRemoved, this, [this](LibInput::Device *inputDevice) {
59 Q_EMIT deviceRemoved(inputDevice->sysName());
60 });
61
62 QDBusConnection::sessionBus().registerObject(QStringLiteral("/org/kde/KWin/InputDevice"),
63 QStringLiteral("org.kde.KWin.InputDeviceManager"),
64 this,
65 QDBusConnection::ExportAllProperties | QDBusConnection::ExportAllSignals);
66 }
67
68 ~ConnectionAdaptor() override
69 {
70 QDBusConnection::sessionBus().unregisterObject(QStringLiteral("/org/kde/KWin/InputDeviceManager"));
71 }
72
73 QStringList devicesSysNames()
74 {
75 return m_con->devicesSysNames();
76 }
77
78Q_SIGNALS:
79 void deviceAdded(QString sysName);
80 void deviceRemoved(QString sysName);
81};

83std::unique_ptr<Connection> Connection::create(Session *session)
84{
85 std::unique_ptr<Udev> udev = std::make_unique<Udev>();
86 if (!udev->isValid()) {
87 qCWarning(KWIN_LIBINPUT) << "Failed to initialize udev";
88 return nullptr;
89 }
90 std::unique_ptr<Context> context = std::make_unique<Context>(session, std::move(udev));
91 if (!context->isValid()) {
92 qCWarning(KWIN_LIBINPUT) << "Failed to create context from udev";
93 return nullptr;
94 }
95 if (!context->initialize()) {
96 qCWarning(KWIN_LIBINPUT) << "Failed to initialize context";
97 return nullptr;
98 }
99 return std::unique_ptr<Connection>(new Connection(std::move(context)));
100}

120void Connection::doSetup()
121{
122 Q_ASSERT(!m_notifier);
123
124 gainRealTime();
125
126 m_notifier = std::make_unique<QSocketNotifier>(m_input->fileDescriptor(), QSocketNotifier::Read);
127 connect(m_notifier.get(), &QSocketNotifier::activated, this, &Connection::handleEvent);
128
129 connect(m_input->session(), &Session::activeChanged, this, [this](bool active) {
130 if (active) {
131 if (!m_input->isSuspended()) {
132 return;
133 }
134 m_input->resume();
135 } else {
136 deactivate();
137 }
138 });
139 handleEvent();
140}
141
142void Connection::deactivate()
143{
144 if (m_input->isSuspended()) {
145 return;
146 }
147 m_input->suspend();
148 handleEvent();
149}

151void Connection::handleEvent()
152{
153 QMutexLocker locker(&m_mutex);
154 const bool wasEmpty = m_eventQueue.empty();
155 do {
156 m_input->dispatch();
157 std::unique_ptr<Event> event = m_input->event();
158 if (!event) {
159 break;
160 }
161 m_eventQueue.push_back(std::move(event));
162 } while (true);
163 if (wasEmpty && !m_eventQueue.empty()) {
164 Q_EMIT eventsRead();
165 }
166}

169QPointF devicePointToGlobalPosition(const QPointF &devicePos, const Output *output)
170{
171 using Transform = Output::Transform;
172
173 QPointF pos = devicePos;
174 // TODO: Do we need to handle the flipped cases differently?
175 switch (output->transform()) {
176 case Transform::Normal:
177 case Transform::Flipped:
178 break;
179 case Transform::Rotated90:
180 case Transform::Flipped90:
181 pos = QPointF(output->modeSize().height() - devicePos.y(), devicePos.x());
182 break;
183 case Transform::Rotated180:
184 case Transform::Flipped180:
185 pos = QPointF(output->modeSize().width() - devicePos.x(),
186 output->modeSize().height() - devicePos.y());
187 break;
188 case Transform::Rotated270:
189 case Transform::Flipped270:
190 pos = QPointF(devicePos.y(), output->modeSize().width() - devicePos.x());
191 break;
192 default:
193 Q_UNREACHABLE();
194 }
195 return output->geometry().topLeft() + pos / output->scale();
196}

199KWin::TabletToolId createTabletId(libinput_tablet_tool *tool, Device *dev)
200{
201 auto serial = libinput_tablet_tool_get_serial(tool);
202 auto toolId = libinput_tablet_tool_get_tool_id(tool);
203 auto type = libinput_tablet_tool_get_type(tool);
204 InputRedirection::TabletToolType toolType;
205 switch (type) {
206 case LIBINPUT_TABLET_TOOL_TYPE_PEN:
207 toolType = InputRedirection::Pen;
208 break;
209 case LIBINPUT_TABLET_TOOL_TYPE_ERASER:
210 toolType = InputRedirection::Eraser;
211 break;
212 case LIBINPUT_TABLET_TOOL_TYPE_BRUSH:
213 toolType = InputRedirection::Brush;
214 break;
215 case LIBINPUT_TABLET_TOOL_TYPE_PENCIL:
216 toolType = InputRedirection::Pencil;
217 break;
218 case LIBINPUT_TABLET_TOOL_TYPE_AIRBRUSH:
219 toolType = InputRedirection::Airbrush;
220 break;
221 case LIBINPUT_TABLET_TOOL_TYPE_MOUSE:
222 toolType = InputRedirection::Mouse;
223 break;
224 case LIBINPUT_TABLET_TOOL_TYPE_LENS:
225 toolType = InputRedirection::Lens;
226 break;
227 case LIBINPUT_TABLET_TOOL_TYPE_TOTEM:
228 toolType = InputRedirection::Totem;
229 break;
230 }
231 QVector<InputRedirection::Capability> capabilities;
232 if (libinput_tablet_tool_has_pressure(tool)) {
233 capabilities << InputRedirection::Pressure;
234 }
235 if (libinput_tablet_tool_has_distance(tool)) {
236 capabilities << InputRedirection::Distance;
237 }
238 if (libinput_tablet_tool_has_rotation(tool)) {
239 capabilities << InputRedirection::Rotation;
240 }
241 if (libinput_tablet_tool_has_tilt(tool)) {
242 capabilities << InputRedirection::Tilt;
243 }
244 if (libinput_tablet_tool_has_slider(tool)) {
245 capabilities << InputRedirection::Slider;
246 }
247 if (libinput_tablet_tool_has_wheel(tool)) {
248 capabilities << InputRedirection::Wheel;
249 }
250 return {dev->sysName(), toolType, capabilities, serial, toolId, dev->groupUserData(), dev->name()};
251}

265void Connection::processEvents()
266{
267 QMutexLocker locker(&m_mutex);
268 while (m_eventQueue.size() != 0) {
269 std::unique_ptr<Event> event = std::move(m_eventQueue.front());
270 m_eventQueue.pop_front();
271 switch (event->type()) {
272 case LIBINPUT_EVENT_DEVICE_ADDED: {
273 auto device = new Device(event->nativeDevice());
274 device->moveToThread(thread());
275 m_devices << device;
276
277 applyDeviceConfig(device);
278 applyScreenToDevice(device);
279
280 connect(device, &Device::outputNameChanged, this, [this, device] {
281 // If the output name changes from something to empty we need to
282 // re-run the assignment heuristic so that an output is assinged
283 if (device->outputName().isEmpty()) {
284 applyScreenToDevice(device);
285 }
286 });
287
288 Q_EMIT deviceAdded(device);
289 break;
290 }
291 case LIBINPUT_EVENT_DEVICE_REMOVED: {
292 auto it = std::find_if(m_devices.begin(), m_devices.end(), [&event](Device *d) {
293 return event->device() == d;
294 });
295 if (it == m_devices.end()) {
296 // we don't know this device
297 break;
298 }
299 auto device = *it;
300 m_devices.erase(it);
301 Q_EMIT deviceRemoved(device);
302 device->deleteLater();
303 break;
304 }
305 case LIBINPUT_EVENT_KEYBOARD_KEY: {
306 KeyEvent *ke = static_cast<KeyEvent *>(event.get());
307 Q_EMIT ke->device()->keyChanged(ke->key(), ke->state(), ke->time(), ke->device());
308 break;
309 }
310 case LIBINPUT_EVENT_POINTER_SCROLL_WHEEL: {
311 const PointerEvent *pointerEvent = static_cast<PointerEvent *>(event.get());
312 const auto axes = pointerEvent->axis();
313 for (const InputRedirection::PointerAxis &axis : axes) {
314 Q_EMIT pointerEvent->device()->pointerAxisChanged(axis,
315 pointerEvent->scrollValue(axis),
316 pointerEvent->scrollValueV120(axis),
317 InputRedirection::PointerAxisSourceWheel,
318 pointerEvent->time(),
319 pointerEvent->device());
320 }
321 break;
322 }
323 case LIBINPUT_EVENT_POINTER_SCROLL_FINGER: {
324 const PointerEvent *pointerEvent = static_cast<PointerEvent *>(event.get());
325 const auto axes = pointerEvent->axis();
326 for (const InputRedirection::PointerAxis &axis : axes) {
327 Q_EMIT pointerEvent->device()->pointerAxisChanged(axis,
328 pointerEvent->scrollValue(axis),
329 0,
330 InputRedirection::PointerAxisSourceFinger,
331 pointerEvent->time(),
332 pointerEvent->device());
333 }
334 break;
335 }
336 case LIBINPUT_EVENT_POINTER_SCROLL_CONTINUOUS: {
337 const PointerEvent *pointerEvent = static_cast<PointerEvent *>(event.get());
338 const auto axes = pointerEvent->axis();
339 for (const InputRedirection::PointerAxis &axis : axes) {
340 Q_EMIT pointerEvent->device()->pointerAxisChanged(axis,
341 pointerEvent->scrollValue(axis),
342 0,
343 InputRedirection::PointerAxisSourceContinuous,
344 pointerEvent->time(),
345 pointerEvent->device());
346 }
347 break;
348 }
349 case LIBINPUT_EVENT_POINTER_BUTTON: {
350 PointerEvent *pe = static_cast<PointerEvent *>(event.get());
351 Q_EMIT pe->device()->pointerButtonChanged(pe->button(), pe->buttonState(), pe->time(), pe->device());
352 break;
353 }
354 case LIBINPUT_EVENT_POINTER_MOTION: {
355 PointerEvent *pe = static_cast<PointerEvent *>(event.get());
356 auto delta = pe->delta();
357 auto deltaNonAccel = pe->deltaUnaccelerated();
358 auto latestTime = pe->time();
359 auto it = m_eventQueue.begin();
360 while (it != m_eventQueue.end()) {
361 if ((*it)->type() == LIBINPUT_EVENT_POINTER_MOTION) {
362 std::unique_ptr<PointerEvent> p{static_cast<PointerEvent *>(it->release())};
363 delta += p->delta();
364 deltaNonAccel += p->deltaUnaccelerated();
365 latestTime = p->time();
366 it = m_eventQueue.erase(it);
367 } else {
368 break;
369 }
370 }
371 Q_EMIT pe->device()->pointerMotion(delta, deltaNonAccel, latestTime, pe->device());
372 break;
373 }
374 case LIBINPUT_EVENT_POINTER_MOTION_ABSOLUTE: {
375 PointerEvent *pe = static_cast<PointerEvent *>(event.get());
376 if (workspace()) {
377 Q_EMIT pe->device()->pointerMotionAbsolute(pe->absolutePos(workspace()->geometry().size()), pe->time(), pe->device());
378 }
379 break;
380 }
381 case LIBINPUT_EVENT_TOUCH_DOWN: {
382#ifndef KWIN_BUILD_TESTING
383 TouchEvent *te = static_cast<TouchEvent *>(event.get());
384 const auto *output = te->device()->output();
385 if (!output) {
386 qCWarning(KWIN_LIBINPUT) << "Touch down received for device with no output assigned";
387 break;
388 }
389 const QPointF globalPos = devicePointToGlobalPosition(te->absolutePos(output->modeSize()), output);
390 Q_EMIT te->device()->touchDown(te->id(), globalPos, te->time(), te->device());
391 break;
392#endif
393 }
394 case LIBINPUT_EVENT_TOUCH_UP: {
395 TouchEvent *te = static_cast<TouchEvent *>(event.get());
396 const auto *output = te->device()->output();
397 if (!output) {
398 break;
399 }
400 Q_EMIT te->device()->touchUp(te->id(), te->time(), te->device());
401 break;
402 }
403 case LIBINPUT_EVENT_TOUCH_MOTION: {
404#ifndef KWIN_BUILD_TESTING
405 TouchEvent *te = static_cast<TouchEvent *>(event.get());
406 const auto *output = te->device()->output();
407 if (!output) {
408 break;
409 }
410 const QPointF globalPos = devicePointToGlobalPosition(te->absolutePos(output->modeSize()), output);
411 Q_EMIT te->device()->touchMotion(te->id(), globalPos, te->time(), te->device());
412 break;
413#endif
414 }
415 case LIBINPUT_EVENT_TOUCH_CANCEL: {
416 Q_EMIT event->device()->touchCanceled(event->device());
417 break;
418 }
419 case LIBINPUT_EVENT_TOUCH_FRAME: {
420 Q_EMIT event->device()->touchFrame(event->device());
421 break;
422 }
423 case LIBINPUT_EVENT_GESTURE_PINCH_BEGIN: {
424 PinchGestureEvent *pe = static_cast<PinchGestureEvent *>(event.get());
425 Q_EMIT pe->device()->pinchGestureBegin(pe->fingerCount(), pe->time(), pe->device());
426 break;
427 }
428 case LIBINPUT_EVENT_GESTURE_PINCH_UPDATE: {
429 PinchGestureEvent *pe = static_cast<PinchGestureEvent *>(event.get());
430 Q_EMIT pe->device()->pinchGestureUpdate(pe->scale(), pe->angleDelta(), pe->delta(), pe->time(), pe->device());
431 break;
432 }
433 case LIBINPUT_EVENT_GESTURE_PINCH_END: {
434 PinchGestureEvent *pe = static_cast<PinchGestureEvent *>(event.get());
435 if (pe->isCancelled()) {
436 Q_EMIT pe->device()->pinchGestureCancelled(pe->time(), pe->device());
437 } else {
438 Q_EMIT pe->device()->pinchGestureEnd(pe->time(), pe->device());
439 }
440 break;
441 }
442 case LIBINPUT_EVENT_GESTURE_SWIPE_BEGIN: {
443 SwipeGestureEvent *se = static_cast<SwipeGestureEvent *>(event.get());
444 Q_EMIT se->device()->swipeGestureBegin(se->fingerCount(), se->time(), se->device());
445 break;
446 }
447 case LIBINPUT_EVENT_GESTURE_SWIPE_UPDATE: {
448 SwipeGestureEvent *se = static_cast<SwipeGestureEvent *>(event.get());
449 Q_EMIT se->device()->swipeGestureUpdate(se->delta(), se->time(), se->device());
450 break;
451 }
452 case LIBINPUT_EVENT_GESTURE_SWIPE_END: {
453 SwipeGestureEvent *se = static_cast<SwipeGestureEvent *>(event.get());
454 if (se->isCancelled()) {
455 Q_EMIT se->device()->swipeGestureCancelled(se->time(), se->device());
456 } else {
457 Q_EMIT se->device()->swipeGestureEnd(se->time(), se->device());
458 }
459 break;
460 }
461 case LIBINPUT_EVENT_GESTURE_HOLD_BEGIN: {
462 HoldGestureEvent *he = static_cast<HoldGestureEvent *>(event.get());
463 Q_EMIT he->device()->holdGestureBegin(he->fingerCount(), he->time(), he->device());
464 break;
465 }
466 case LIBINPUT_EVENT_GESTURE_HOLD_END: {
467 HoldGestureEvent *he = static_cast<HoldGestureEvent *>(event.get());
468 if (he->isCancelled()) {
469 Q_EMIT he->device()->holdGestureCancelled(he->time(), he->device());
470 } else {
471 Q_EMIT he->device()->holdGestureEnd(he->time(), he->device());
472 }
473 break;
474 }
475 case LIBINPUT_EVENT_SWITCH_TOGGLE: {
476 SwitchEvent *se = static_cast<SwitchEvent *>(event.get());
477 switch (se->state()) {
478 case SwitchEvent::State::Off:
479 Q_EMIT se->device()->switchToggledOff(se->time(), se->device());
480 break;
481 case SwitchEvent::State::On:
482 Q_EMIT se->device()->switchToggledOn(se->time(), se->device());
483 break;
484 default:
485 Q_UNREACHABLE();
486 }
487 break;
488 }
489 case LIBINPUT_EVENT_TABLET_TOOL_AXIS:
490 case LIBINPUT_EVENT_TABLET_TOOL_PROXIMITY:
491 case LIBINPUT_EVENT_TABLET_TOOL_TIP: {
492 auto *tte = static_cast<TabletToolEvent *>(event.get());
493
494 KWin::InputRedirection::TabletEventType tabletEventType;
495 switch (event->type()) {
496 case LIBINPUT_EVENT_TABLET_TOOL_AXIS:
497 tabletEventType = KWin::InputRedirection::Axis;
498 break;
499 case LIBINPUT_EVENT_TABLET_TOOL_PROXIMITY:
500 tabletEventType = KWin::InputRedirection::Proximity;
501 break;
502 case LIBINPUT_EVENT_TABLET_TOOL_TIP:
503 default:
504 tabletEventType = KWin::InputRedirection::Tip;
505 break;
506 }
507
508 if (workspace()) {
509#ifndef KWIN_BUILD_TESTING
510 Output *output = tte->device()->output();
511 if (!output && workspace()->activeWindow()) {
512 output = workspace()->activeWindow()->output();
513 }
514 if (!output) {
515 output = workspace()->activeOutput();
516 }
517 const QPointF globalPos =
518 devicePointToGlobalPosition(tte->transformedPosition(output->modeSize()),
519 output);
520#else
521 const QPointF globalPos;
522#endif
523 Q_EMIT event->device()->tabletToolEvent(tabletEventType,
524 globalPos, tte->pressure(),
525 tte->xTilt(), tte->yTilt(), tte->rotation(),
526 tte->isTipDown(), tte->isNearby(), createTabletId(tte->tool(), event->device()), tte->time());
527 }
528 break;
529 }
530 case LIBINPUT_EVENT_TABLET_TOOL_BUTTON: {
531 auto *tabletEvent = static_cast<TabletToolButtonEvent *>(event.get());
532 Q_EMIT event->device()->tabletToolButtonEvent(tabletEvent->buttonId(),
533 tabletEvent->isButtonPressed(),
534 createTabletId(tabletEvent->tool(), event->device()), tabletEvent->time());
535 break;
536 }
537 case LIBINPUT_EVENT_TABLET_PAD_BUTTON: {
538 auto *tabletEvent = static_cast<TabletPadButtonEvent *>(event.get());
539 Q_EMIT event->device()->tabletPadButtonEvent(tabletEvent->buttonId(),
540 tabletEvent->isButtonPressed(),
541 createTabletPadId(event->device()), tabletEvent->time());
542 break;
543 }
544 case LIBINPUT_EVENT_TABLET_PAD_RING: {
545 auto *tabletEvent = static_cast<TabletPadRingEvent *>(event.get());
546 tabletEvent->position();
547 Q_EMIT event->device()->tabletPadRingEvent(tabletEvent->number(),
548 tabletEvent->position(),
549 tabletEvent->source() == LIBINPUT_TABLET_PAD_RING_SOURCE_FINGER,
550 createTabletPadId(event->device()), tabletEvent->time());
551 break;
552 }
553 case LIBINPUT_EVENT_TABLET_PAD_STRIP: {
554 auto *tabletEvent = static_cast<TabletPadStripEvent *>(event.get());
555 Q_EMIT event->device()->tabletPadStripEvent(tabletEvent->number(),
556 tabletEvent->position(),
557 tabletEvent->source() == LIBINPUT_TABLET_PAD_STRIP_SOURCE_FINGER,
558 createTabletPadId(event->device()), tabletEvent->time());
559 break;
560 }
561 default:
562 // nothing
563 break;
564 }
565 }

576void Connection::applyScreenToDevice(Device *device)
577{
578#ifndef KWIN_BUILD_TESTING
579 QMutexLocker locker(&m_mutex);
580 if (!device->isTouch() && !device->isTabletTool()) {
581 return;
582 }
583
584 Output *deviceOutput = nullptr;
585 const QVector<Output *> outputs = kwinApp()->outputBackend()->outputs();
586
587 // let's try to find a screen for it
588 if (!device->outputName().isEmpty()) {
589 // we have an output name, try to find a screen with matching name
590 for (Output *output : outputs) {
591 if (!output->isEnabled()) {
592 continue;
593 }
594 if (output->name() == device->outputName()) {
595 deviceOutput = output;
596 break;
597 }
598 }
599 }
600 if (!deviceOutput && device->isTouch()) {
601 // do we have an internal screen?
602 Output *internalOutput = nullptr;
603 for (Output *output : outputs) {
604 if (!output->isEnabled()) {
605 continue;
606 }
607 if (output->isInternal()) {
608 internalOutput = output;
609 break;
610 }
611 }
612 auto testScreenMatches = [device](const Output *output) {
613 const auto &size = device->size();
614 const auto &screenSize = output->physicalSize();
615 return std::round(size.width()) == std::round(screenSize.width())
616 && std::round(size.height()) == std::round(screenSize.height());
617 };
618 if (internalOutput && testScreenMatches(internalOutput)) {
619 deviceOutput = internalOutput;
620 }
621 // let's compare all screens for size
622 for (Output *output : outputs) {
623 if (!output->isEnabled()) {
624 continue;
625 }
626 if (testScreenMatches(output)) {
627 deviceOutput = output;
628 break;
629 }
630 }
631 if (!deviceOutput) {
632 // still not found
633 if (internalOutput) {
634 // we have an internal id, so let's use that
635 deviceOutput = internalOutput;
636 } else {
637 for (Output *output : outputs) {
638 // just take first screen, we have no clue
639 if (output->isEnabled()) {
640 deviceOutput = output;
641 break;
642 }
643 }
644 }
645 }
646 }
647
648 device->setOutput(deviceOutput);
649
650 // TODO: this is currently non-functional even on DRM. Needs orientation() override there.
651 device->setOrientation(Qt::PrimaryOrientation);
652#endif
653}

655void Connection::applyDeviceConfig(Device *device)
656{
657 KConfigGroup defaults = m_config->group("Libinput").group("Defaults");
658 if (defaults.isValid()) {
659 if (device->isAlphaNumericKeyboard() && defaults.hasGroup("Keyboard")) {
660 defaults = defaults.group("Keyboard");
661 } else if (device->isTouchpad() && defaults.hasGroup("Touchpad")) {
662 // A Touchpad is a Pointer, so we need to check for it before Pointer.
663 defaults = defaults.group("Touchpad");
664 } else if (device->isPointer() && defaults.hasGroup("Pointer")) {
665 defaults = defaults.group("Pointer");
666 }
667
668 device->setDefaultConfig(defaults);
669 }
670
671 // pass configuration to Device
672 device->setConfig(m_config->group("Libinput").group(QString::number(device->vendor())).group(QString::number(device->product())).group(device->name()));
673 device->loadConfiguration();
674}
