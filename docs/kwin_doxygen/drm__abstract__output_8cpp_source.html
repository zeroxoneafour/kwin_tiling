40{
41 const int height = pixelSize().height();
42 const QMatrix4x4 matrix = Output::logicalToNativeMatrix(rect(), scale(), transform());
43 QVector<EGLint> rects;
44 rects.reserve(region.rectCount() * 4);
45 for (const QRect &_rect : region) {
46 const QRect rect = matrix.mapRect(_rect);
47 rects << rect.left();
48 rects << height - (rect.y() + rect.height());
