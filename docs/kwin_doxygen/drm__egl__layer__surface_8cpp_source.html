67std::optional<OutputLayerBeginFrameInfo> EglGbmLayerSurface::startRendering(const QSize &bufferSize, DrmPlane::Transformations renderOrientation, DrmPlane::Transformations bufferOrientation, const QMap<uint32_t, QVector<uint64_t>> &formats)
68{
69 if (!checkSurface(bufferSize, formats)) {
70 return std::nullopt;
71 }
72 if (!m_surface.gbmSurface->makeContextCurrent()) {
73 return std::nullopt;
74 }
75
76 // shadow buffer
77 const QSize renderSize = (renderOrientation & (DrmPlane::Transformation::Rotate90 | DrmPlane::Transformation::Rotate270)) ? m_surface.gbmSurface->size().transposed() : m_surface.gbmSurface->size();
78 if (doesShadowBufferFit(m_shadowBuffer.get(), renderSize, renderOrientation, bufferOrientation)) {
79 m_oldShadowBuffer.reset();
80 } else {
81 if (doesShadowBufferFit(m_oldShadowBuffer.get(), renderSize, renderOrientation, bufferOrientation)) {
82 m_shadowBuffer = m_oldShadowBuffer;
83 } else {
84 if (renderOrientation != bufferOrientation) {
85 const auto format = m_eglBackend->gbmFormatForDrmFormat(m_surface.gbmSurface->format());
86 if (!format.has_value()) {
87 return std::nullopt;
88 }
89 m_shadowBuffer = std::make_shared<ShadowBuffer>(renderSize, format.value());
90 if (!m_shadowBuffer->isComplete()) {
91 return std::nullopt;
92 }
93 } else {
94 m_shadowBuffer.reset();
95 }
96 }
97 }
98
99 if (m_shadowBuffer) {
100 // the blit after rendering will completely overwrite the back buffer anyways
101 return OutputLayerBeginFrameInfo{
102 .renderTarget = RenderTarget(m_shadowBuffer->fbo()),
103 .repaint = {},
104 };
105 } else {
106 return OutputLayerBeginFrameInfo{
107 .renderTarget = RenderTarget(m_surface.gbmSurface->fbo()),
108 .repaint = m_surface.gbmSurface->repaintRegion(),
109 };
110 }
111}

113void EglGbmLayerSurface::aboutToStartPainting(DrmOutput *output, const QRegion &damagedRegion)
114{
115 if (m_shadowBuffer) {
116 // with a shadow buffer, we always fully damage the surface
117 return;
118 }
119 if (m_surface.gbmSurface && m_surface.gbmSurface->bufferAge() > 0 && !damagedRegion.isEmpty() && m_eglBackend->supportsPartialUpdate()) {
120 QVector<EGLint> rects = output->regionToRects(damagedRegion);
121 const bool correct = eglSetDamageRegionKHR(m_eglBackend->eglDisplay(), m_surface.gbmSurface->eglSurface(), rects.data(), rects.count() / 4);
122 if (!correct) {
123 qCWarning(KWIN_DRM) << "eglSetDamageRegionKHR failed:" << getEglErrorString();
124 }
125 }

128bool EglGbmLayerSurface::endRendering(DrmPlane::Transformations renderOrientation, const QRegion &damagedRegion)
129{
130 if (m_shadowBuffer) {
131 GLFramebuffer::pushFramebuffer(m_surface.gbmSurface->fbo());
132 // TODO handle bufferOrientation != Rotate0
133 m_shadowBuffer->render(renderOrientation);
134 GLFramebuffer::popFramebuffer();
135 }
136 const auto gbmBuffer = m_surface.gbmSurface->swapBuffers(damagedRegion);
137 if (!gbmBuffer) {
138 return false;
139 }
140 const auto buffer = importBuffer(m_surface, gbmBuffer);
141 if (buffer) {
142 m_surface.currentBuffer = gbmBuffer;
143 m_surface.currentFramebuffer = buffer;
144 return true;
145 } else {
146 return false;
147 }

195bool EglGbmLayerSurface::checkSurface(const QSize &size, const QMap<uint32_t, QVector<uint64_t>> &formats)
196{
197 if (doesSurfaceFit(m_surface, size, formats)) {
198 return true;
199 }
200 if (doesSurfaceFit(m_oldSurface, size, formats)) {
201 m_surface = m_oldSurface;
202 return true;
203 }

205 m_oldSurface = m_surface;
206 m_surface = newSurface.value();
207 return true;
208 }
209 return false;

220std::optional<EglGbmLayerSurface::Surface> EglGbmLayerSurface::createSurface(const QSize &size, const QMap<uint32_t, QVector<uint64_t>> &formats) const
221{
222 QVector<GbmFormat> preferredFormats;
223 QVector<GbmFormat> fallbackFormats;
224 for (auto it = formats.begin(); it != formats.end(); it++) {
225 const auto format = m_eglBackend->gbmFormatForDrmFormat(it.key());
226 if (format.has_value() && format->bpp >= 24) {
227 if (format->bpp <= 32) {
228 preferredFormats.push_back(format.value());
229 } else {
230 fallbackFormats.push_back(format.value());
231 }
232 }
233 }
234 const auto sort = [this](const auto &lhs, const auto &rhs) {
235 if (lhs.drmFormat == rhs.drmFormat) {
236 // prefer having an alpha channel
237 return lhs.alphaSize > rhs.alphaSize;
238 } else if (m_eglBackend->prefer10bpc() && ((lhs.bpp == 30) != (rhs.bpp == 30))) {
239 // prefer 10bpc / 30bpp formats
240 return lhs.bpp == 30;
241 } else {
242 // fallback: prefer formats with lower bandwidth requirements
243 return lhs.bpp < rhs.bpp;
244 }
245 };
246 const auto testFormats = [this, &size, &formats](const QVector<GbmFormat> &gbmFormats, MultiGpuImportMode importMode) -> std::optional<Surface> {
247 for (const auto &format : gbmFormats) {
248 const auto surface = createSurface(size, format.drmFormat, formats[format.drmFormat], importMode);
249 if (surface.has_value()) {
250 return surface;
251 }
252 }
253 return std::nullopt;
254 };
255 std::sort(preferredFormats.begin(), preferredFormats.end(), sort);
256 if (const auto surface = testFormats(preferredFormats, MultiGpuImportMode::Dmabuf)) {
257 return surface;
258 }
259 if (m_gpu != m_eglBackend->gpu()) {
260 if (const auto surface = testFormats(preferredFormats, MultiGpuImportMode::DumbBuffer)) {
261 return surface;
262 }
263 }
264 std::sort(fallbackFormats.begin(), fallbackFormats.end(), sort);
265 if (const auto surface = testFormats(fallbackFormats, MultiGpuImportMode::Dmabuf)) {
266 return surface;
267 }
268 if (m_gpu != m_eglBackend->gpu()) {
269 if (const auto surface = testFormats(fallbackFormats, MultiGpuImportMode::DumbBuffer)) {
270 return surface;
271 }
272 }
273 return std::nullopt;
274}

276std::optional<EglGbmLayerSurface::Surface> EglGbmLayerSurface::createSurface(const QSize &size, uint32_t format, const QVector<uint64_t> &modifiers, MultiGpuImportMode importMode) const
277{
278 Surface ret;
279 ret.importMode = importMode;
280 ret.forceLinear = importMode == MultiGpuImportMode::DumbBuffer || m_bufferTarget != BufferTarget::Normal;
281 ret.gbmSurface = createGbmSurface(size, format, modifiers, ret.forceLinear);
282 if (!ret.gbmSurface) {
283 return std::nullopt;
284 }
285 if (importMode == MultiGpuImportMode::DumbBuffer || m_bufferTarget == BufferTarget::Dumb) {
286 ret.importSwapchain = std::make_shared<DumbSwapchain>(m_gpu, size, format);
287 if (ret.importSwapchain->isEmpty()) {
288 return std::nullopt;
289 }
290 }
291 if (!doRenderTestBuffer(ret)) {
292 return std::nullopt;
293 }
294 return ret;
295}

297std::shared_ptr<GbmSurface> EglGbmLayerSurface::createGbmSurface(const QSize &size, uint32_t format, const QVector<uint64_t> &modifiers, bool forceLinear) const
298{
299 static bool modifiersEnvSet = false;
300 static const bool modifiersEnv = qEnvironmentVariableIntValue("KWIN_DRM_USE_MODIFIERS", &modifiersEnvSet) != 0;
301 bool allowModifiers = m_gpu->addFB2ModifiersSupported() && (!modifiersEnvSet || (modifiersEnvSet && modifiersEnv)) && !modifiers.isEmpty();
302#if !HAVE_GBM_BO_GET_FD_FOR_PLANE
303 allowModifiers &= m_gpu == m_eglBackend->gpu();
304#endif
305 const auto config = m_eglBackend->config(format);
306 if (!config) {
307 return nullptr;
308 }
309
310 if (allowModifiers) {
311 const auto ret = GbmSurface::createSurface(m_eglBackend, size, format, forceLinear ? linearModifier : modifiers, config);
312 if (const auto surface = std::get_if<std::shared_ptr<GbmSurface>>(&ret)) {
313 return *surface;
314 } else if (std::get<GbmSurface::Error>(ret) != GbmSurface::Error::ModifiersUnsupported) {
315 return nullptr;
316 }
317 }
318 uint32_t gbmFlags = GBM_BO_USE_RENDERING;
319 if (m_gpu == m_eglBackend->gpu()) {
320 gbmFlags |= GBM_BO_USE_SCANOUT;
321 }
322 if (forceLinear || m_gpu != m_eglBackend->gpu()) {
323 gbmFlags |= GBM_BO_USE_LINEAR;
324 }
325 const auto ret = GbmSurface::createSurface(m_eglBackend, size, format, gbmFlags, config);
326 const auto surface = std::get_if<std::shared_ptr<GbmSurface>>(&ret);
327 return surface ? *surface : nullptr;
328}

330std::shared_ptr<DrmFramebuffer> EglGbmLayerSurface::doRenderTestBuffer(Surface &surface) const
331{
332 if (!surface.gbmSurface->makeContextCurrent()) {
333 return nullptr;

336 const auto buffer = surface.gbmSurface->swapBuffers(infiniteRegion());

349std::shared_ptr<DrmFramebuffer> EglGbmLayerSurface::importBuffer(Surface &surface, const std::shared_ptr<GbmBuffer> &sourceBuffer) const
350{
351 if (m_bufferTarget == BufferTarget::Dumb || surface.importMode == MultiGpuImportMode::DumbBuffer) {
352 return importWithCpu(surface, sourceBuffer.get());
353 } else if (m_gpu != m_eglBackend->gpu()) {
354 return importDmabuf(sourceBuffer.get());
355 } else {
356 const auto ret = DrmFramebuffer::createFramebuffer(sourceBuffer);
357 if (!ret) {
358 qCWarning(KWIN_DRM, "Failed to create %s framebuffer: %s", formatName(sourceBuffer->format()).name, strerror(errno));
359 }
360 return ret;
361 }

378std::shared_ptr<DrmFramebuffer> EglGbmLayerSurface::importWithCpu(Surface &surface, GbmBuffer *sourceBuffer) const
379{
380 Q_ASSERT(surface.importSwapchain && !surface.importSwapchain->isEmpty());
381 if (!sourceBuffer->map(GBM_BO_TRANSFER_READ)) {
382 qCWarning(KWIN_DRM, "mapping a %s gbm_bo failed: %s", formatName(sourceBuffer->format()).name, strerror(errno));
383 return nullptr;
384 }
385 const auto importBuffer = surface.importSwapchain->acquireBuffer();
386 if (sourceBuffer->planeCount() != 1 || sourceBuffer->strides()[0] != importBuffer->strides()[0]) {
387 qCCritical(KWIN_DRM, "stride of gbm_bo (%d) and dumb buffer (%d) with format %s don't match!",
388 sourceBuffer->strides()[0], importBuffer->strides()[0], formatName(sourceBuffer->format()).name);
389 return nullptr;
390 }
391 if (!memcpy(importBuffer->data(), sourceBuffer->mappedData(), importBuffer->size().height() * importBuffer->strides()[0])) {
392 return nullptr;
393 }
394 const auto ret = DrmFramebuffer::createFramebuffer(importBuffer);
395 if (!ret) {
396 qCWarning(KWIN_DRM, "Failed to create %s framebuffer for CPU import: %s", formatName(sourceBuffer->format()).name, strerror(errno));
397 }
398 return ret;
399}
