52WaylandInputDevice::WaylandInputDevice(KWayland::Client::Keyboard *keyboard, WaylandSeat *seat)
53 : m_seat(seat)
54 , m_keyboard(keyboard)
55{
56 connect(keyboard, &Keyboard::keyChanged, this, [this](quint32 key, Keyboard::KeyState nativeState, quint32 time) {
57 InputRedirection::KeyboardKeyState state;
58 switch (nativeState) {
59 case Keyboard::KeyState::Pressed:
60 if (key == KEY_RIGHTCTRL) {
61 m_seat->backend()->togglePointerLock();
62 }
63 state = InputRedirection::KeyboardKeyPressed;
64 break;
65 case Keyboard::KeyState::Released:
66 state = InputRedirection::KeyboardKeyReleased;
67 break;
68 default:
69 Q_UNREACHABLE();
70 }
71 Q_EMIT keyChanged(key, state, std::chrono::milliseconds(time), this);
72 });
73}

75WaylandInputDevice::WaylandInputDevice(KWayland::Client::Pointer *pointer, WaylandSeat *seat)
76 : m_seat(seat)
77 , m_pointer(pointer)
78{
79 connect(pointer, &Pointer::entered, this, [this](quint32 serial, const QPointF &relativeToSurface) {
80 WaylandOutput *output = m_seat->backend()->findOutput(m_pointer->enteredSurface());
81 Q_ASSERT(output);
82 output->cursor()->setPointer(m_pointer.get());
83 });
84 connect(pointer, &Pointer::left, this, [this]() {
85 // wl_pointer.leave carries the wl_surface, but KWayland::Client::Pointer::left does not.
86 const auto outputs = m_seat->backend()->outputs();
87 for (Output *output : outputs) {
88 WaylandOutput *waylandOutput = static_cast<WaylandOutput *>(output);
89 if (waylandOutput->cursor()->pointer()) {
90 waylandOutput->cursor()->setPointer(nullptr);
91 }
92 }
93 });
94 connect(pointer, &Pointer::motion, this, [this](const QPointF &relativeToSurface, quint32 time) {
95 WaylandOutput *output = m_seat->backend()->findOutput(m_pointer->enteredSurface());
96 Q_ASSERT(output);
97 const QPointF absolutePos = output->geometry().topLeft() + relativeToSurface;
98 Q_EMIT pointerMotionAbsolute(absolutePos, std::chrono::milliseconds(time), this);
99 });
100 connect(pointer, &Pointer::buttonStateChanged, this, [this](quint32 serial, quint32 time, quint32 button, Pointer::ButtonState nativeState) {
101 InputRedirection::PointerButtonState state;
102 switch (nativeState) {
103 case Pointer::ButtonState::Pressed:
104 state = InputRedirection::PointerButtonPressed;
105 break;
106 case Pointer::ButtonState::Released:
107 state = InputRedirection::PointerButtonReleased;
108 break;
109 default:
110 Q_UNREACHABLE();
111 }
112 Q_EMIT pointerButtonChanged(button, state, std::chrono::milliseconds(time), this);
113 });
114 // TODO: Send discreteDelta and source as well.
115 connect(pointer, &Pointer::axisChanged, this, [this](quint32 time, Pointer::Axis nativeAxis, qreal delta) {
116 InputRedirection::PointerAxis axis;
117 switch (nativeAxis) {
118 case Pointer::Axis::Horizontal:
119 axis = InputRedirection::PointerAxisHorizontal;
120 break;
121 case Pointer::Axis::Vertical:
122 axis = InputRedirection::PointerAxisVertical;
123 break;
124 default:
125 Q_UNREACHABLE();
126 }
127 Q_EMIT pointerAxisChanged(axis, delta, 0, InputRedirection::PointerAxisSourceUnknown, std::chrono::milliseconds(time), this);
128 });
129
130 KWayland::Client::PointerGestures *pointerGestures = m_seat->backend()->display()->pointerGestures();
131 if (pointerGestures) {
132 m_pinchGesture.reset(pointerGestures->createPinchGesture(m_pointer.get(), this));
133 connect(m_pinchGesture.get(), &PointerPinchGesture::started, this, [this](quint32 serial, quint32 time) {

438bool WaylandBackend::initialize()
439{
440 m_display = std::make_unique<WaylandDisplay>();
441 if (!m_display->initialize(m_options.socketName)) {
442 return false;
443 }
444
445 createOutputs();
446
447 m_seat = std::make_unique<WaylandSeat>(m_display->seat(), this);
448
449 QAbstractEventDispatcher *dispatcher = QAbstractEventDispatcher::instance();
450 QObject::connect(dispatcher, &QAbstractEventDispatcher::aboutToBlock, m_display.get(), &WaylandDisplay::flush);
451 QObject::connect(dispatcher, &QAbstractEventDispatcher::awake, m_display.get(), &WaylandDisplay::flush);
452
453 connect(this, &WaylandBackend::pointerLockChanged, this, [this](bool locked) {
454 if (locked) {
455 m_seat->createRelativePointer();
456 } else {
457 m_seat->destroyRelativePointer();
458 }
459 });
460
461 return true;
462}

464void WaylandBackend::createOutputs()
465{
466 // we need to multiply the initial window size with the scale in order to
467 // create an output window of this size in the end
468 const QSize pixelSize = m_options.outputSize * m_options.outputScale;
469 for (int i = 0; i < m_options.outputCount; i++) {
470 WaylandOutput *output = createOutput(QStringLiteral("WL-%1").arg(i), pixelSize, m_options.outputScale);
471 m_outputs << output;
472 Q_EMIT outputAdded(output);
473 output->updateEnabled(true);
474 }
475
476 Q_EMIT outputsQueried();
477}

479WaylandOutput *WaylandBackend::createOutput(const QString &name, const QSize &size, qreal scale)
480{
481 WaylandOutput *waylandOutput = new WaylandOutput(name, this);
482 waylandOutput->init(size, scale);
483
484 // Wait until the output window is configured by the host compositor.
485 while (!waylandOutput->isReady()) {
486 wl_display_roundtrip(m_display->nativeDisplay());
487 }
488
489 return waylandOutput;
490}

532void WaylandBackend::togglePointerLock()
533{
534 if (!supportsPointerLock()) {
535 return;
536 }
537 if (!m_seat) {
538 return;
539 }
540 auto pointer = m_seat->pointerDevice()->nativePointer();
541 if (!pointer) {
542 return;
543 }
544 if (m_outputs.isEmpty()) {
545 return;
546 }
547
548 for (auto output : std::as_const(m_outputs)) {
549 output->lockPointer(m_seat->pointerDevice()->nativePointer(), !m_pointerLockRequested);
550 }
551 m_pointerLockRequested = !m_pointerLockRequested;
552}
