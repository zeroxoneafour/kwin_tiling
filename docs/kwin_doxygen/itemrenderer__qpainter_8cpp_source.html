57void ItemRendererQPainter::renderItem(Item *item, int mask, const QRegion &_region, const WindowPaintData &data)
58{
59 QRegion region = _region;
60
61 const QRect boundingRect = item->mapToGlobal(item->boundingRect()).toAlignedRect();
62 if (!(mask & (Scene::PAINT_WINDOW_TRANSFORMED | Scene::PAINT_SCREEN_TRANSFORMED))) {
63 region &= boundingRect;
64 }
65
66 if (region.isEmpty()) {
67 return;
68 }
69
70 m_painter->save();
71 m_painter->setClipRegion(region);
72 m_painter->setClipping(true);
73 m_painter->setOpacity(data.opacity());
74
75 if (mask & Scene::PAINT_WINDOW_TRANSFORMED) {
76 m_painter->translate(data.xTranslation(), data.yTranslation());
77 m_painter->scale(data.xScale(), data.yScale());
78 }
79
80 renderItem(m_painter.get(), item);
81
82 m_painter->restore();
83}

85void ItemRendererQPainter::renderItem(QPainter *painter, Item *item) const
86{
87 const QList<Item *> sortedChildItems = item->sortedChildItems();
88
89 painter->save();
90 painter->translate(item->position());
91 painter->setOpacity(painter->opacity() * item->opacity());
92
93 for (Item *childItem : sortedChildItems) {
94 if (childItem->z() >= 0) {
95 break;
96 }
97 if (childItem->explicitVisible()) {
98 renderItem(painter, childItem);
99 }
100 }
101
102 item->preprocess();
103 if (auto surfaceItem = qobject_cast<SurfaceItem *>(item)) {
104 renderSurfaceItem(painter, surfaceItem);
105 } else if (auto decorationItem = qobject_cast<DecorationItem *>(item)) {
106 renderDecorationItem(painter, decorationItem);
107 } else if (auto imageItem = qobject_cast<ImageItem *>(item)) {
108 renderImageItem(painter, imageItem);
109 }
110
111 for (Item *childItem : sortedChildItems) {
112 if (childItem->z() < 0) {
113 continue;
114 }
115 if (childItem->explicitVisible()) {
116 renderItem(painter, childItem);
117 }
118 }
119
120 painter->restore();
121}

123void ItemRendererQPainter::renderSurfaceItem(QPainter *painter, SurfaceItem *surfaceItem) const
124{
125 const SurfacePixmap *surfaceTexture = surfaceItem->pixmap();
126 if (!surfaceTexture || !surfaceTexture->isValid()) {
127 return;
128 }
129
130 QPainterSurfaceTexture *platformSurfaceTexture =
131 static_cast<QPainterSurfaceTexture *>(surfaceTexture->texture());
132 if (!platformSurfaceTexture->isValid()) {
133 platformSurfaceTexture->create();
134 } else {
135 platformSurfaceTexture->update(surfaceItem->damage());
136 }
137 surfaceItem->resetDamage();
138
139 const QVector<QRectF> shape = surfaceItem->shape();
140 for (const QRectF rect : shape) {
141 const QMatrix4x4 matrix = surfaceItem->surfaceToBufferMatrix();
142 const QPointF bufferTopLeft = matrix.map(rect.topLeft());
143 const QPointF bufferBottomRight = matrix.map(rect.bottomRight());
144
145 painter->drawImage(rect, platformSurfaceTexture->image(),
146 QRectF(bufferTopLeft, bufferBottomRight));
147 }
148}

150void ItemRendererQPainter::renderDecorationItem(QPainter *painter, DecorationItem *decorationItem) const
151{
152 const auto renderer = static_cast<const SceneQPainterDecorationRenderer *>(decorationItem->renderer());
153 QRectF dtr, dlr, drr, dbr;
154 decorationItem->window()->layoutDecorationRects(dlr, dtr, drr, dbr);
155
156 painter->drawImage(dtr, renderer->image(SceneQPainterDecorationRenderer::DecorationPart::Top));
157 painter->drawImage(dlr, renderer->image(SceneQPainterDecorationRenderer::DecorationPart::Left));
158 painter->drawImage(drr, renderer->image(SceneQPainterDecorationRenderer::DecorationPart::Right));
159 painter->drawImage(dbr, renderer->image(SceneQPainterDecorationRenderer::DecorationPart::Bottom));
160}
