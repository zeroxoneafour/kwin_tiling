21{

23 QX11Info::appScreen());

36OMLSyncControlVsyncMonitorHelper::OMLSyncControlVsyncMonitorHelper()
37{
38 // Establish a new X11 connection to avoid locking up the main X11 connection.
39 m_display = XOpenDisplay(DisplayString(QX11Info::display()));
40 if (!m_display) {
41 qCDebug(KWIN_X11STANDALONE) << "Failed to establish vsync monitor X11 connection";
42 return;
43 }
44
45 ::Window rootWindow = DefaultRootWindow(m_display);
46
47 const int attribs[] = {
48 GLX_RENDER_TYPE, GLX_RGBA_BIT,
49 GLX_DRAWABLE_TYPE, GLX_WINDOW_BIT,
50 0};
51
52 GLXFBConfig config = chooseGlxFbConfig(m_display, attribs);
53 if (!config) {
54 qCDebug(KWIN_X11STANDALONE) << "Couldn't find any suitable FBConfig for vsync monitor";
55 return;
56 }
57
58 XVisualInfo *visualInfo = glXGetVisualFromFBConfig(m_display, config);
59 if (!visualInfo) {
60 return;
61 }
62
63 Visual *visual = visualInfo->visual;
64 const int depth = visualInfo->depth;
65 XFree(visualInfo);
66
67 Colormap colormap = XCreateColormap(m_display, rootWindow, visual, AllocNone);
68 XSetWindowAttributes attributes;
69 attributes.colormap = colormap;
70
71 m_dummyWindow = XCreateWindow(m_display, rootWindow, 0, 0, 1, 1, 0, depth,
72 InputOutput, visual, CWColormap, &attributes);
73 XFreeColormap(m_display, colormap);
74 if (!m_dummyWindow) {
75 qCDebug(KWIN_X11STANDALONE) << "Failed to create a dummy window for vsync monitor";
76 return;
77 }
78
79 m_drawable = glXCreateWindow(m_display, config, m_dummyWindow, nullptr);
80 if (!m_drawable) {
81 qCDebug(KWIN_X11STANDALONE) << "Failed to create GLXWindow for dummy window";
82 return;
83 }
84
85 m_localContext = glXCreateNewContext(m_display, config, GLX_RGBA_TYPE, 0, true);
86 if (!m_localContext) {
87 qCDebug(KWIN_X11STANDALONE) << "Failed to create opengl context for vsync monitor";
88 return;
89 }
90}

113void OMLSyncControlVsyncMonitorHelper::poll()
114{
115 if (!glXMakeCurrent(m_display, m_drawable, m_localContext)) {
116 qCDebug(KWIN_X11STANDALONE) << "Failed to make vsync monitor OpenGL context current";
117 return;
118 }
119
120 int64_t ust, msc, sbc;
121
122 glXGetSyncValuesOML(m_display, m_drawable, &ust, &msc, &sbc);
123 glXWaitForMscOML(m_display, m_drawable, 0, 2, (msc + 1) % 2, &ust, &msc, &sbc);
124
125 Q_EMIT vblankOccurred(std::chrono::microseconds(ust));
126}
