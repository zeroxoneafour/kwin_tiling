21static int startXServer()
22{
23 const QByteArray process = QByteArrayLiteral("Xwayland");
24 int pipeFds[2];
25 if (pipe(pipeFds) != 0) {
26 std::cerr << "FATAL ERROR failed to create pipe to start X Server " << process.constData() << std::endl;
27 exit(1);
28 }
29
30 pid_t pid = fork();
31 if (pid == 0) {
32 // child process - should be turned into Xwayland
33 // writes to pipe, closes read side
34 close(pipeFds[0]);
35 char fdbuf[16];
36 sprintf(fdbuf, "%d", pipeFds[1]);
37 execlp(process.constData(), process.constData(), "-displayfd", fdbuf, (char *)nullptr);
38 close(pipeFds[1]);
39 exit(20);
40 }
41 // parent process - this is the wayland server
42 // reads from pipe, closes write side
43 close(pipeFds[1]);
44 return pipeFds[0];
45}

47static void readDisplayFromPipe(int pipe)
48{
49 QFile readPipe;
50 if (!readPipe.open(pipe, QIODevice::ReadOnly)) {
51 std::cerr << "FATAL ERROR failed to open pipe to start X Server XWayland" << std::endl;
52 exit(1);
53 }
54 QByteArray displayNumber = readPipe.readLine();
55
56 displayNumber.prepend(QByteArray(":"));
57 displayNumber.remove(displayNumber.size() - 1, 1);
58 std::cout << "X-Server started on display " << displayNumber.constData() << std::endl;
59
60 setenv("DISPLAY", displayNumber.constData(), true);
61
62 // close our pipe
63 close(pipe);
64}
