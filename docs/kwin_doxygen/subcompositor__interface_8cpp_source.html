27void SubCompositorInterfacePrivate::subcompositor_get_subsurface(Resource *resource,
28 uint32_t id,
29 ::wl_resource *surface_resource,
30 ::wl_resource *parent_resource)
31{
32 SurfaceInterface *surface = SurfaceInterface::get(surface_resource);
33 SurfaceInterface *parent = SurfaceInterface::get(parent_resource);
34
35 if (!surface) {
36 wl_resource_post_error(resource->handle, error_bad_surface, "no surface");
37 return;
38 }
39 if (!parent) {
40 wl_resource_post_error(resource->handle, error_bad_surface, "no parent");
41 return;
42 }
43
44 const SurfaceRole *surfaceRole = SurfaceRole::get(surface);
45 if (surfaceRole) {
46 wl_resource_post_error(resource->handle, error_bad_surface, "the surface already has a role assigned %s", surfaceRole->name().constData());
47 return;
48 }
49
50 if (surface == parent) {
51 wl_resource_post_error(resource->handle, error_bad_surface, "wl_surface@%d cannot be its own parent", wl_resource_get_id(surface_resource));
52 return;
53 }
54 if (parent->subSurface() && parent->subSurface()->mainSurface() == surface) {
55 wl_resource_post_error(resource->handle, error_bad_surface, "wl_surface@%d is an ancestor of parent", wl_resource_get_id(surface_resource));
56 return;
57 }
58
59 wl_resource *subsurfaceResource = wl_resource_create(resource->client(), &wl_subsurface_interface, resource->version(), id);
60 if (!subsurfaceResource) {
61 wl_resource_post_no_memory(resource->handle);
62 return;
63 }
64
65 Q_EMIT q->subSurfaceCreated(new SubSurfaceInterface(surface, parent, subsurfaceResource));
66}

233bool SubSurfaceInterface::isSynchronized() const
234{
235 if (d->mode == Mode::Synchronized) {
236 return true;
237 }
238 if (!d->parent) {
239 // that shouldn't happen, but let's assume false
240 return false;
241 }
242 if (d->parent->subSurface()) {
243 // follow parent's mode
244 return d->parent->subSurface()->isSynchronized();
245 }
246 // parent is no subsurface, thus parent is in desync mode and this surface is in desync mode
247 return false;
248}

250SurfaceInterface *SubSurfaceInterface::mainSurface() const
251{
252 if (!d->parent) {
253 return nullptr;
254 }
255 SurfaceInterfacePrivate *parentPrivate = SurfaceInterfacePrivate::get(d->parent);
256 if (parentPrivate->subSurface) {
257 return parentPrivate->subSurface->mainSurface();
258 }
259 return d->parent;
260}
