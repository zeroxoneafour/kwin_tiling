190void TestInputMethodInterface::initTestCase()
191{
192 m_display.addSocketName(s_socketName);
193 m_display.start();
194 QVERIFY(m_display.isRunning());
195
196 m_seat = new SeatInterface(&m_display, this);
197 m_serverCompositor = new CompositorInterface(&m_display, this);
198 m_inputMethodIface = new InputMethodV1Interface(&m_display, this);
199 m_inputPanelIface = new InputPanelV1Interface(&m_display, this);
200
201 m_outputHandle = std::make_unique<FakeOutput>();
202 m_outputInterface = std::make_unique<OutputInterface>(&m_display, m_outputHandle.get());
203
204 connect(m_serverCompositor, &CompositorInterface::surfaceCreated, this, [this](SurfaceInterface *surface) {
205 m_surfaces += surface;
206 });
207
208 // setup connection
209 m_connection = new KWayland::Client::ConnectionThread;
210 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
211 m_connection->setSocketName(s_socketName);
212
213 m_thread = new QThread(this);
214 m_connection->moveToThread(m_thread);
215 m_thread->start();
216
217 m_connection->initConnection();
218 QVERIFY(connectedSpy.wait());
219 QVERIFY(!m_connection->connections().isEmpty());
220
221 m_queue = new KWayland::Client::EventQueue(this);
222 QVERIFY(!m_queue->isValid());
223 m_queue->setup(m_connection);
224 QVERIFY(m_queue->isValid());
225
226 auto registry = new KWayland::Client::Registry(this);
227 QSignalSpy interfacesSpy(registry, &KWayland::Client::Registry::interfacesAnnounced);
228 connect(registry, &KWayland::Client::Registry::outputAnnounced, this, [this, registry](quint32 name, quint32 version) {
229 m_output = new KWayland::Client::Output(this);
230 m_output->setup(registry->bindOutput(name, version));
231 });
232 connect(registry, &KWayland::Client::Registry::interfaceAnnounced, this, [this, registry](const QByteArray &interface, quint32 name, quint32 version) {
233 if (interface == "zwp_input_panel_v1") {
234 m_inputPanel = new InputPanel(registry->registry(), name, version);
235 } else if (interface == "zwp_input_method_v1") {
236 m_inputMethod = new InputMethodV1(registry->registry(), name, version);
237 }
238 });
239 connect(registry, &KWayland::Client::Registry::seatAnnounced, this, [this, registry](quint32 name, quint32 version) {
240 m_clientSeat = registry->createSeat(name, version);
241 });
242 registry->setEventQueue(m_queue);
243 QSignalSpy compositorSpy(registry, &KWayland::Client::Registry::compositorAnnounced);
244 registry->create(m_connection->display());
245 QVERIFY(registry->isValid());
246 registry->setup();
247 wl_display_flush(m_connection->display());
248
249 QVERIFY(compositorSpy.wait());
250 m_clientCompositor = registry->createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
251 QVERIFY(m_clientCompositor->isValid());
252
253 QVERIFY(interfacesSpy.count() || interfacesSpy.wait());
254
255 QSignalSpy surfaceSpy(m_serverCompositor, &CompositorInterface::surfaceCreated);
256 for (int i = 0; i < 3; ++i) {
257 m_clientCompositor->createSurface(this);
258 }
259 QVERIFY(surfaceSpy.count() < 3 && surfaceSpy.wait(200));
260 QVERIFY(m_surfaces.count() == 3);
261 QVERIFY(m_inputPanel);
262 QVERIFY(m_output);
263}

378 QVERIFY(preeditStringSpy.wait());
379 QCOMPARE(preeditStringSpy.count(), 1);
380 QCOMPARE(preeditStringSpy.last().at(0).value<quint32>(), serial);
381 QCOMPARE(preeditStringSpy.last().at(1).value<QString>(), "hello");

384
385 // delete surrounding text

387 imContext->delete_surrounding_text(0, 5);
388 QVERIFY(deleteSurroundingSpy.wait());
