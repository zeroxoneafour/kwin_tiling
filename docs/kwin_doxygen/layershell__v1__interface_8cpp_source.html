94void LayerShellV1InterfacePrivate::zwlr_layer_shell_v1_get_layer_surface(Resource *resource,
95 uint32_t id,
96 wl_resource *surface_resource,
97 wl_resource *output_resource,
98 uint32_t layer,
99 const QString &scope)
100{
101 SurfaceInterface *surface = SurfaceInterface::get(surface_resource);
102 OutputInterface *output = OutputInterface::get(output_resource);
103
104 if (surface->buffer()) {
105 wl_resource_post_error(resource->handle, error_already_constructed, "the wl_surface already has a buffer attached");
106 return;
107 }
108
109 if (layer > layer_overlay) {
110 wl_resource_post_error(resource->handle, error_invalid_layer, "invalid layer %d", layer);
111 return;
112 }
113
114 SurfaceRole *surfaceRole = SurfaceRole::get(surface);
115 if (surfaceRole) {
116 wl_resource_post_error(resource->handle, error_role, "the wl_surface already has a role assigned %s", surfaceRole->name().constData());
117 return;
118 }
119
120 wl_resource *layerSurfaceResource = wl_resource_create(resource->client(), &zwlr_layer_surface_v1_interface, resource->version(), id);
121 if (!layerSurfaceResource) {
122 wl_resource_post_no_memory(resource->handle);
123 return;
124 }
125
126 auto layerSurface = new LayerSurfaceV1Interface(q, surface, output, LayerSurfaceV1Interface::Layer(layer), scope, layerSurfaceResource);
127 Q_EMIT q->surfaceCreated(layerSurface);
128}

168void LayerSurfaceV1InterfacePrivate::zwlr_layer_surface_v1_set_anchor(Resource *resource, uint32_t anchor)
169{
170 const uint32_t anchorMask = anchor_top | anchor_left | anchor_right | anchor_bottom;
171 if (anchor > anchorMask) {
172 wl_resource_post_error(resource->handle, error_invalid_anchor, "invalid anchor %d", anchor);
173 return;
174 }

176 pending.anchor = Qt::Edges();

178 if (anchor & anchor_top) {
179 pending.anchor |= Qt::TopEdge;
180 }
181
182 if (anchor & anchor_right) {

184 }
185
186 if (anchor & anchor_bottom) {
187 pending.anchor |= Qt::BottomEdge;
188 }
189
190 if (anchor & anchor_left) {
191 pending.anchor |= Qt::LeftEdge;

223void LayerSurfaceV1InterfacePrivate::zwlr_layer_surface_v1_ack_configure(Resource *resource, uint32_t serial)
224{
225 if (!serials.contains(serial)) {
226 wl_resource_post_error(resource->handle, error_invalid_surface_state, "invalid configure serial %d", serial);
227 return;
228 }
229 while (!serials.isEmpty()) {
230 const quint32 head = serials.takeFirst();
231 if (head == serial) {
232 break;
233 }
234 }
235 if (!isClosed) {
236 pending.acknowledgedConfigure = serial;
237 pending.acknowledgedConfigureIsSet = true;
238 }
239}

255void LayerSurfaceV1InterfacePrivate::commit()
256{
257 if (isClosed) {
258 return;
259 }
260
261 if (pending.acknowledgedConfigureIsSet) {
262 current.acknowledgedConfigure = pending.acknowledgedConfigure;
263 pending.acknowledgedConfigureIsSet = false;
264 Q_EMIT q->configureAcknowledged(pending.acknowledgedConfigure);
265 }
266
267 if (Q_UNLIKELY(surface->isMapped() && !isConfigured)) {
268 wl_resource_post_error(resource()->handle,
269 error_invalid_surface_state,
270 "a buffer has been attached to a layer surface prior "
271 "to the first layer_surface.configure event");
272 return;
273 }
274
275 if (Q_UNLIKELY(pending.desiredSize.width() == 0 && (!(pending.anchor & Qt::LeftEdge) || !(pending.anchor & Qt::RightEdge)))) {
276 wl_resource_post_error(resource()->handle,
277 error_invalid_size,
278 "the layer surface has a width of 0 but its anchor "
279 "doesn't include the left and the right screen edge");
280 return;
281 }
282
283 if (Q_UNLIKELY(pending.desiredSize.height() == 0 && (!(pending.anchor & Qt::TopEdge) || !(pending.anchor & Qt::BottomEdge)))) {
284 wl_resource_post_error(resource()->handle,
285 error_invalid_size,
286 "the layer surface has a height of 0 but its anchor "
287 "doesn't include the top and the bottom screen edge");
288 return;
289 }
290
291 // detect reset
292 if (!surface->isMapped() && firstBufferAttached) {
293 isCommitted = false;
294 firstBufferAttached = false;
295 isConfigured = false;
296
297 current = LayerSurfaceV1State();
298 pending = LayerSurfaceV1State();
299
300 return;
301 }
302
303 const LayerSurfaceV1State previous = std::exchange(current, pending);
304
305 isCommitted = true; // Must set the committed state before emitting any signals.
306 if (surface->isMapped()) {
307 firstBufferAttached = true;
308 }
309
310 if (previous.acceptsFocus != current.acceptsFocus) {
311 Q_EMIT q->acceptsFocusChanged();
312 }
313 if (previous.layer != current.layer) {
314 Q_EMIT q->layerChanged();
315 }
316 if (previous.anchor != current.anchor) {
317 Q_EMIT q->anchorChanged();
318 }
319 if (previous.desiredSize != current.desiredSize) {
320 Q_EMIT q->desiredSizeChanged();
321 }
322 if (previous.exclusiveZone != current.exclusiveZone) {
323 Q_EMIT q->exclusiveZoneChanged();
324 }
325 if (previous.margins != current.margins) {
326 Q_EMIT q->marginsChanged();
327 }
328}

412Qt::Edge LayerSurfaceV1Interface::exclusiveEdge() const
413{
414 if (exclusiveZone() <= 0) {
415 return Qt::Edge();
416 }
417 if (anchor() == (Qt::LeftEdge | Qt::TopEdge | Qt::RightEdge) || anchor() == Qt::TopEdge) {
418 return Qt::TopEdge;
419 }
420 if (anchor() == (Qt::TopEdge | Qt::RightEdge | Qt::BottomEdge) || anchor() == Qt::RightEdge) {
421 return Qt::RightEdge;
422 }
423 if (anchor() == (Qt::RightEdge | Qt::BottomEdge | Qt::LeftEdge) || anchor() == Qt::BottomEdge) {
424 return Qt::BottomEdge;
425 }
426 if (anchor() == (Qt::BottomEdge | Qt::LeftEdge | Qt::TopEdge) || anchor() == Qt::LeftEdge) {
427 return Qt::LeftEdge;
428 }
429 return Qt::Edge();
430}

442quint32 LayerSurfaceV1Interface::sendConfigure(const QSize &size)
443{
444 if (d->isClosed) {
445 qCWarning(KWIN_CORE) << "Cannot configure a closed layer shell surface";
446 return 0;
447 }
448
449 const uint32_t serial = d->shell->display()->nextSerial();
450 d->serials << serial;
451
452 d->send_configure(serial, size.width(), size.height());
453 d->isConfigured = true;
454
455 return serial;
