44void ColorBlindnessCorrectionEffect::loadData()
45{
46 ensureResources();
47
48 QString fragPath;
49 switch (m_mode) {
50 case Deuteranopia:
51 fragPath = QStringLiteral(":/effects/colorblindnesscorrection/shaders/Deutranopia.frag");
52 break;
53 case Tritanopia:
54 fragPath = QStringLiteral(":/effects/colorblindnesscorrection/shaders/Tritanopia.frag");
55 break;
56 case Protanopia: // Most common, use it as fallback
57 default:
58 fragPath = QStringLiteral(":/effects/colorblindnesscorrection/shaders/Protanopia.frag");
59 break;
60 }
61
62 m_shader = ShaderManager::instance()->generateShaderFromFile(ShaderTrait::MapTexture, QString(), fragPath);
63
64 if (!m_shader->isValid()) {
65 qCCritical(KWIN_COLORBLINDNESS_CORRECTION) << "Failed to load the shader!";
66 return;
67 }
68
69 if (ShaderBinder binder{m_shader.get()}; !m_shader->setUniform("intensity", m_intensity)) {
70 qCWarning(KWIN_COLORBLINDNESS_CORRECTION) << "Failed to set intensity";
71 }
72
73 for (const auto windows = effects->stackingOrder(); EffectWindow * w : windows) {
74 correctColor(w);
75 }
76 effects->addRepaintFull();
77
78 connect(effects, &EffectsHandler::windowDeleted, this, &ColorBlindnessCorrectionEffect::slotWindowDeleted);
79 connect(effects, &EffectsHandler::windowAdded, this, &ColorBlindnessCorrectionEffect::correctColor);
80}

110void ColorBlindnessCorrectionEffect::reconfigure(ReconfigureFlags flags)
111{
112 if (flags != Effect::ReconfigureAll) {
113 return;
114 }
115
116 ColorBlindnessCorrectionSettings::self()->read();
117 const auto newMode = static_cast<Mode>(ColorBlindnessCorrectionSettings::mode());
118 const auto newIntensity = std::clamp<float>(ColorBlindnessCorrectionSettings::intensity(), 0.0f, 1.0f);
119 if (m_mode == newMode && qFuzzyCompare(m_intensity, newIntensity)) {
120 return;
121 }
122
123 m_mode = newMode;
124 m_intensity = newIntensity;
125
126 disconnect(effects, &EffectsHandler::windowDeleted, this, &ColorBlindnessCorrectionEffect::slotWindowDeleted);
127 disconnect(effects, &EffectsHandler::windowAdded, this, &ColorBlindnessCorrectionEffect::correctColor);
128
129 for (EffectWindow *w : m_windows) {
130 unredirect(w);
131 }
132 m_windows.clear();
133
134 loadData();
135}
