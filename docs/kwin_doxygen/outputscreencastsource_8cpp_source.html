57void OutputScreenCastSource::render(GLFramebuffer *target)
58{
59 const auto [outputTexture, colorDescription] = Compositor::self()->scene()->textureForOutput(m_output);
60 if (!outputTexture) {
61 return;
62 }
63
64 ShaderBinder shaderBinder(ShaderTrait::MapTexture | ShaderTrait::TransformColorspace);
65 QMatrix4x4 projectionMatrix;
66 projectionMatrix.scale(1, -1);
67 projectionMatrix.ortho(QRect(QPoint(), textureSize()));
68 shaderBinder.shader()->setUniform(GLShader::Mat4Uniform::ModelViewProjectionMatrix, projectionMatrix);
69 shaderBinder.shader()->setColorspaceUniformsToSRGB(colorDescription);
70
71 GLFramebuffer::pushFramebuffer(target);
72 outputTexture->render(textureSize());
73 GLFramebuffer::popFramebuffer();
74}
