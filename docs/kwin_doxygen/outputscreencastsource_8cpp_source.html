45void OutputScreenCastSource::render(GLFramebuffer *target)
46{
47 const std::shared_ptr<GLTexture> outputTexture = Compositor::self()->scene()->textureForOutput(m_output);
48 if (!outputTexture) {
49 return;
50 }
51
52 ShaderBinder shaderBinder(ShaderTrait::MapTexture);
53 QMatrix4x4 projectionMatrix;
54 projectionMatrix.scale(1, -1);
55 projectionMatrix.ortho(QRect(QPoint(), textureSize() * m_output->scale()));
56 shaderBinder.shader()->setUniform(GLShader::ModelViewProjectionMatrix, projectionMatrix);
57
58 GLFramebuffer::pushFramebuffer(target);
59 outputTexture->bind();
60 outputTexture->render(textureSize(), m_output->scale());
61 outputTexture->unbind();
62 GLFramebuffer::popFramebuffer();
63}
