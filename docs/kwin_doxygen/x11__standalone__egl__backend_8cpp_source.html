46EglBackend::EglBackend(Display *display, X11StandaloneBackend *backend)
47 : EglOnXBackend(kwinApp()->x11Connection(), display, kwinApp()->x11RootWindow())
48 , m_backend(backend)
49 , m_overlayWindow(std::make_unique<OverlayWindowX11>())
50 , m_layer(std::make_unique<EglLayer>(this))
51{
52 // There is no any way to determine when a buffer swap completes with EGL. Fallback
53 // to software vblank events. Could we use the Present extension to get notified when
54 // the overlay window is actually presented on the screen?
55 m_vsyncMonitor = SoftwareVsyncMonitor::create();
56 connect(backend->renderLoop(), &RenderLoop::refreshRateChanged, this, [this, backend]() {
57 m_vsyncMonitor->setRefreshRate(backend->renderLoop()->refreshRate());
58 });
59 m_vsyncMonitor->setRefreshRate(backend->renderLoop()->refreshRate());
60
61 connect(m_vsyncMonitor.get(), &VsyncMonitor::vblankOccurred, this, &EglBackend::vblank);
62 connect(workspace(), &Workspace::geometryChanged, this, &EglBackend::screenGeometryChanged);
63}

65EglBackend::~EglBackend()
66{
67 // No completion events will be received for in-flight frames, this may lock the
68 // render loop. We need to ensure that the render loop is back to its initial state
69 // if the render backend is about to be destroyed.
70 RenderLoopPrivate::get(m_backend->renderLoop())->invalidate();
71
72 if (isFailed() && m_overlayWindow) {
73 m_overlayWindow->destroy();
74 }
75 cleanup();
76

78 m_overlayWindow->destroy();

87void EglBackend::init()
88{
89 QOpenGLContext *qtShareContext = QOpenGLContext::globalShareContext();
90 EGLDisplay shareDisplay = EGL_NO_DISPLAY;
91 EGLContext shareContext = EGL_NO_CONTEXT;
92 if (qtShareContext) {
93 qDebug(KWIN_X11STANDALONE) << "Global share context format:" << qtShareContext->format();
94#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
95 const QVariant nativeHandle = qtShareContext->nativeHandle();
96 if (!nativeHandle.canConvert<QEGLNativeContext>()) {
97 setFailed(QStringLiteral("Invalid QOpenGLContext::globalShareContext()"));
98 return;
99 } else {
100 QEGLNativeContext handle = qvariant_cast<QEGLNativeContext>(nativeHandle);
101 shareContext = handle.context();
102 shareDisplay = handle.display();
103 }
104#else
105 const auto nativeHandle = qtShareContext->nativeInterface<QNativeInterface::QEGLContext>();
106 if (nativeHandle) {
107 shareContext = nativeHandle->nativeContext();
108 shareDisplay = nativeHandle->display();
109 } else {
110 setFailed(QStringLiteral("Invalid QOpenGLContext::globalShareContext()"));
111 return;
112 }
113#endif
114 }
115 if (shareContext == EGL_NO_CONTEXT) {
116 setFailed(QStringLiteral("QOpenGLContext::globalShareContext() is required"));
117 return;
118 }
119
120 m_fbo = std::make_unique<GLFramebuffer>(0, workspace()->geometry().size());
121
122 kwinApp()->outputBackend()->setSceneEglDisplay(shareDisplay);
123 kwinApp()->outputBackend()->setSceneEglGlobalShareContext(shareContext);
124 EglOnXBackend::init();
125}

183void EglBackend::present(Output *output)
184{
185 // Start the software vsync monitor. There is no any reliable way to determine when
186 // eglSwapBuffers() or eglSwapBuffersWithDamageEXT() completes.
187 m_vsyncMonitor->arm();
188
189 QRegion effectiveRenderedRegion = m_lastRenderedRegion;
190 if (!GLPlatform::instance()->isGLES()) {
191 const QRect displayRect = workspace()->geometry();
192 if (!supportsBufferAge() && options->glPreferBufferSwap() == Options::CopyFrontBuffer && m_lastRenderedRegion != displayRect) {
193 glReadBuffer(GL_FRONT);
194 copyPixels(QRegion(displayRect) - m_lastRenderedRegion, displayRect.size());
195 glReadBuffer(GL_BACK);
196 effectiveRenderedRegion = displayRect;
197 }
198 }
199
200 presentSurface(surface(), effectiveRenderedRegion, workspace()->geometry());
201
202 if (overlayWindow() && overlayWindow()->window()) { // show the window only after the first pass,
203 overlayWindow()->show(); // since that pass may take long
204 }
205}

207void EglBackend::presentSurface(EGLSurface surface, const QRegion &damage, const QRect &screenGeometry)
208{
209 const bool fullRepaint = supportsBufferAge() || (damage == screenGeometry);
210
211 if (fullRepaint || !havePostSubBuffer()) {
212 // the entire screen changed, or we cannot do partial updates (which implies we enabled surface preservation)
213 eglSwapBuffers(eglDisplay(), surface);
214 if (supportsBufferAge()) {
215 eglQuerySurface(eglDisplay(), surface, EGL_BUFFER_AGE_EXT, &m_bufferAge);
216 }
217 } else {
218 // a part of the screen changed, and we can use eglPostSubBufferNV to copy the updated area
219 for (const QRect &r : damage) {
220 eglPostSubBufferNV(eglDisplay(), surface, r.left(), screenGeometry.height() - r.bottom() - 1, r.width(), r.height());
221 }
222 }
223}

288bool EglPixmapTexturePrivate::create(SurfacePixmapX11 *pixmap)
289{
290 const xcb_pixmap_t nativePixmap = pixmap->pixmap();
291 if (nativePixmap == XCB_NONE) {
292 return false;
293 }
294
295 glGenTextures(1, &m_texture);
296 q->setWrapMode(GL_CLAMP_TO_EDGE);
297 q->setFilter(GL_LINEAR);
298 q->bind();
299 const EGLint attribs[] = {
300 EGL_IMAGE_PRESERVED_KHR, EGL_TRUE,
301 EGL_NONE};
302 m_image = eglCreateImageKHR(m_backend->eglDisplay(),
303 EGL_NO_CONTEXT,
304 EGL_NATIVE_PIXMAP_KHR,
305 reinterpret_cast<EGLClientBuffer>(static_cast<uintptr_t>(nativePixmap)),
306 attribs);
307
308 if (EGL_NO_IMAGE_KHR == m_image) {
309 qCDebug(KWIN_X11STANDALONE) << "failed to create egl image";
310 q->unbind();
311 return false;
312 }
313 glEGLImageTargetTexture2DOES(GL_TEXTURE_2D, static_cast<GLeglImageOES>(m_image));
314 q->unbind();
315 q->setYInverted(true);
316 m_size = pixmap->size();
317 updateMatrix();
318 return true;
319}
