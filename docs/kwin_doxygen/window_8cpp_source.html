474QVector<QRectF> Window::shapeRegion() const
475{
476 if (m_shapeRegionIsValid) {
477 return m_shapeRegion;
478 }
479
480 const QRectF bufferGeometry = this->bufferGeometry();
481
482 if (shape()) {
483 auto cookie = xcb_shape_get_rectangles_unchecked(kwinApp()->x11Connection(), frameId(), XCB_SHAPE_SK_BOUNDING);
484 UniqueCPtr<xcb_shape_get_rectangles_reply_t> reply(xcb_shape_get_rectangles_reply(kwinApp()->x11Connection(), cookie, nullptr));
485 if (reply) {
486 m_shapeRegion.clear();
487 const xcb_rectangle_t *rects = xcb_shape_get_rectangles_rectangles(reply.get());
488 const int rectCount = xcb_shape_get_rectangles_rectangles_length(reply.get());
489 for (int i = 0; i < rectCount; ++i) {
490 QRectF region = Xcb::fromXNative(QRect(rects[i].x, rects[i].y, rects[i].width, rects[i].height)).toAlignedRect();
491 // make sure the shape is sane (X is async, maybe even XShape is broken)
492 region = region.intersected(QRectF(QPointF(0, 0), bufferGeometry.size()));
493
494 m_shapeRegion += region;
495 }
496 } else {
497 m_shapeRegion.clear();
498 }
499 } else {
500 m_shapeRegion = {QRectF(0, 0, bufferGeometry.width(), bufferGeometry.height())};
501 }
502
503 m_shapeRegionIsValid = true;
504 return m_shapeRegion;
505}

804void Window::setActive(bool act)
805{
806 if (isZombie()) {
807 return;
808 }
809 if (m_active == act) {
810 return;
811 }
812 m_active = act;
813 const int ruledOpacity = m_active
814 ? rules()->checkOpacityActive(qRound(opacity() * 100.0))
815 : rules()->checkOpacityInactive(qRound(opacity() * 100.0));
816 setOpacity(ruledOpacity / 100.0);
817 workspace()->setActiveWindow(act ? this : nullptr);
818
819 if (!m_active) {
820 cancelAutoRaise();
821 }
822
823 if (!m_active && shadeMode() == ShadeActivated) {
824 setShade(ShadeNormal);
825 }
826
827 StackingUpdatesBlocker blocker(workspace());
828 updateLayer(); // active windows may get different layer
829 auto mainwindows = mainWindows();
830 for (auto it = mainwindows.constBegin(); it != mainwindows.constEnd(); ++it) {
831 if ((*it)->isFullScreen()) { // fullscreens go high even if their transient is active
832 (*it)->updateLayer();
833 }
834 }
835
836 doSetActive();
837 Q_EMIT activeChanged();
838 updateMouseGrab();
839}

881Layer Window::belongsToLayer() const
882{
883 // NOTICE while showingDesktop, desktops move to the AboveLayer
884 // (interchangeable w/ eg. yakuake etc. which will at first remain visible)
885 // and the docks move into the NotificationLayer (which is between Above- and
886 // ActiveLayer, so that active fullscreen windows will still cover everything)
887 // Since the desktop is also activated, nothing should be in the ActiveLayer, though
888 if (isUnmanaged() || isInternal()) {
889 return UnmanagedLayer;
890 }
891 if (isLockScreen() && !waylandServer()) {
892 return UnmanagedLayer;
893 }
894 if (isInputMethod()) {
895 return UnmanagedLayer;
896 }
897 if (isLockScreenOverlay() && waylandServer() && waylandServer()->isScreenLocked()) {
898 return UnmanagedLayer;
899 }
900 if (isDesktop()) {
901 return workspace()->showingDesktop() ? AboveLayer : DesktopLayer;
902 }
903 if (isSplash()) { // no damn annoying splashscreens
904 return NormalLayer; // getting in the way of everything else
905 }
906 if (isDock() || isAppletPopup()) {
907 if (workspace()->showingDesktop()) {
908 return NotificationLayer;
909 }
910 return layerForDock();
911 }
912 if (isPopupWindow()) {
913 return PopupLayer;
914 }
915 if (isOnScreenDisplay()) {
916 return OnScreenDisplayLayer;
917 }
918 if (isNotification()) {
919 return NotificationLayer;
920 }
921 if (isCriticalNotification()) {
922 return CriticalNotificationLayer;
923 }
924 if (workspace()->showingDesktop() && belongsToDesktop()) {
925 return AboveLayer;
926 }
927 if (keepBelow()) {
928 return BelowLayer;
929 }
930 if (isActiveFullScreen()) {
931 return ActiveLayer;
932 }
933 if (keepAbove()) {
934 return AboveLayer;
935 }
936
937 return NormalLayer;
938}

1071void Window::setDesktops(QVector<VirtualDesktop *> desktops)
1072{
1073 // on x11 we can have only one desktop at a time
1074 if (kwinApp()->operationMode() == Application::OperationModeX11 && desktops.size() > 1) {
1075 desktops = QVector<VirtualDesktop *>({desktops.last()});
1076 }
1077
1078 desktops = rules()->checkDesktops(desktops);
1079 if (desktops == m_desktops) {
1080 return;
1081 }
1082
1083 int was_desk = Window::desktop();
1084 const bool wasOnCurrentDesktop = isOnCurrentDesktop() && was_desk >= 0;
1085
1086 m_desktops = desktops;
1087
1088 if (windowManagementInterface()) {
1089 if (m_desktops.isEmpty()) {
1090 windowManagementInterface()->setOnAllDesktops(true);
1091 } else {
1092 windowManagementInterface()->setOnAllDesktops(false);

1094 for (auto desktop : std::as_const(m_desktops)) {
1095 if (!currentDesktops.contains(desktop->id())) {
1096 windowManagementInterface()->addPlasmaVirtualDesktop(desktop->id());
1097 } else {
1098 currentDesktops.removeOne(desktop->id());
1099 }
1100 }
1101 for (const auto &desktopId : std::as_const(currentDesktops)) {
1102 windowManagementInterface()->removePlasmaVirtualDesktop(desktopId);
1103 }
1104 }
1105 }
1106 if (info) {
1107 info->setDesktop(desktop());
1108 }
1109
1110 if ((was_desk == NET::OnAllDesktops) != (desktop() == NET::OnAllDesktops)) {
1111 // onAllDesktops changed
1112 workspace()->updateOnAllDesktopsOfTransients(this);
1113 }
1114
1115 auto transients_stacking_order = workspace()->ensureStackingOrder(transients());
1116 for (auto it = transients_stacking_order.constBegin(); it != transients_stacking_order.constEnd(); ++it) {
1117 (*it)->setDesktops(desktops);
1118 }

1120 if (isModal()) // if a modal dialog is moved, move the mainwindow with it as otherwise
1121 // the (just moved) modal dialog will confusingly return to the mainwindow with
1122 // the next desktop change
1123 {
1124 const auto windows = mainWindows();

1126 other->setDesktops(desktops);

1133 updateWindowRules(Rules::Desktops);

1582bool Window::startInteractiveMoveResize()
1583{
1584 Q_ASSERT(!isInteractiveMoveResize());
1585 Q_ASSERT(QWidget::keyboardGrabber() == nullptr);
1586 Q_ASSERT(QWidget::mouseGrabber() == nullptr);
1587 stopDelayedInteractiveMoveResize();
1588 if (QApplication::activePopupWidget() != nullptr) {
1589 return false; // popups have grab
1590 }
1591 if (isRequestedFullScreen() && (workspace()->outputs().count() < 2 || !isMovableAcrossScreens())) {
1592 return false;
1593 }
1594 if (!doStartInteractiveMoveResize()) {
1595 return false;
1596 }
1597
1598 invalidateDecorationDoubleClickTimer();
1599
1600 setInteractiveMoveResize(true);
1601 workspace()->setMoveResizeWindow(this);
1602
1603 m_interactiveMoveResize.initialGeometry = moveResizeGeometry();
1604 m_interactiveMoveResize.startOutput = moveResizeOutput();
1605 m_interactiveMoveResize.initialMaximizeMode = requestedMaximizeMode();
1606 m_interactiveMoveResize.initialQuickTileMode = quickTileMode();
1607 m_interactiveMoveResize.initialGeometryRestore = geometryRestore();
1608
1609 if (requestedMaximizeMode() != MaximizeRestore) {
1610 switch (interactiveMoveResizeGravity()) {
1611 case Gravity::Left:
1612 case Gravity::Right:
1613 // Quit maximized horizontally state if the window is resized horizontally.
1614 if (requestedMaximizeMode() & MaximizeHorizontal) {
1615 QRectF originalGeometry = geometryRestore();
1616 originalGeometry.setX(moveResizeGeometry().x());
1617 originalGeometry.setWidth(moveResizeGeometry().width());
1618 setGeometryRestore(originalGeometry);
1619 maximize(requestedMaximizeMode() ^ MaximizeHorizontal);
1620 }
1621 break;
1622 case Gravity::Top:
1623 case Gravity::Bottom:
1624 // Quit maximized vertically state if the window is resized vertically.
1625 if (requestedMaximizeMode() & MaximizeVertical) {
1626 QRectF originalGeometry = geometryRestore();
1627 originalGeometry.setY(moveResizeGeometry().y());
1628 originalGeometry.setHeight(moveResizeGeometry().height());
1629 setGeometryRestore(originalGeometry);
1630 maximize(requestedMaximizeMode() ^ MaximizeVertical);
1631 }
1632 break;
1633 case Gravity::TopLeft:

1697// If e.g. the titlebar is already outside of the workarea, there's no point in performing

1700void Window::checkUnrestrictedInteractiveMoveResize()

1703 return;
1704 }

1709 // how much must remain visible when moved away in that direction
1710 left_marge = std::min(100. + borderRight(), moveResizeGeom.width());
1711 right_marge = std::min(100. + borderLeft(), moveResizeGeom.width());

1717 if (moveResizeGeom.bottom() < desktopArea.top() + top_marge) {

1719 }
1720 if (moveResizeGeom.top() > desktopArea.bottom() - bottom_marge) {
1721 setUnrestrictedInteractiveMoveResize(true);
1722 }
1723 if (moveResizeGeom.right() < desktopArea.left() + left_marge) {
1724 setUnrestrictedInteractiveMoveResize(true);
1725 }
1726 if (moveResizeGeom.left() > desktopArea.right() - right_marge) {
1727 setUnrestrictedInteractiveMoveResize(true);
1728 }
1729 if (!isUnrestrictedInteractiveMoveResize() && moveResizeGeom.top() < desktopArea.top()) { // titlebar mustn't go out
1730 setUnrestrictedInteractiveMoveResize(true);
1731 }
1732 }
1733 if (isInteractiveMove()) {
1734 if (moveResizeGeom.bottom() < desktopArea.top() + titlebar_marge) {
1735 setUnrestrictedInteractiveMoveResize(true);
1736 }
1737 // no need to check top_marge, titlebar_marge already handles it
1738 if (moveResizeGeom.top() > desktopArea.bottom() - bottom_marge) { // titlebar mustn't go out
1739 setUnrestrictedInteractiveMoveResize(true);
1740 }
1741 if (moveResizeGeom.right() < desktopArea.left() + left_marge) {
1742 setUnrestrictedInteractiveMoveResize(true);

1820void Window::handleInteractiveMoveResize(qreal x, qreal y, qreal x_root, qreal y_root)
1821{
1822 if (isWaitingForInteractiveMoveResizeSync()) {
1823 return; // we're still waiting for the client or the timeout
1824 }
1825
1826 const Gravity gravity = interactiveMoveResizeGravity();
1827 if ((gravity == Gravity::None && !isMovableAcrossScreens())
1828 || (gravity != Gravity::None && (isShade() || !isResizable()))) {
1829 return;
1830 }
1831
1832 if (!isInteractiveMoveResize()) {
1833 QPointF p(QPointF(x /* - padding_left*/, y /* - padding_top*/) - interactiveMoveOffset());
1834 if (p.manhattanLength() >= QApplication::startDragDistance()) {
1835 if (!startInteractiveMoveResize()) {
1836 setInteractiveMoveResizePointerButtonDown(false);
1837 updateCursor();
1838 return;
1839 }
1840 updateCursor();
1841 } else {
1842 return;
1843 }
1844 }
1845
1846 // ShadeHover or ShadeActive, ShadeNormal was already avoided above
1847 if (gravity != Gravity::None && shadeMode() != ShadeNone) {
1848 setShade(ShadeNone);
1849 }
1850
1851 QPointF globalPos(x_root, y_root);
1852 // these two points limit the geometry rectangle, i.e. if bottomleft resizing is done,
1853 // the bottomleft corner should be at is at (topleft.x(), bottomright().y())
1854 QPointF topleft = globalPos - interactiveMoveOffset();
1855 QPointF bottomright = globalPos + invertedInteractiveMoveOffset();
1856 const QRectF currentMoveResizeGeom = moveResizeGeometry();
1857 QRectF nextMoveResizeGeom = moveResizeGeometry();
1858
1859 // TODO move whole group when moving its leader or when the leader is not mapped?
1860
1861 auto titleBarRect = [this](const QRectF &rect, bool &transposed, int &requiredPixels) -> QRectF {
1862 QRectF titleRect = rect;
1863 titleRect.moveTopLeft(QPointF(0, 0));
1864 switch (titlebarPosition()) {
1865 default:
1866 case Qt::TopEdge:
1867 titleRect.setHeight(borderTop());
1868 break;
1869 case Qt::LeftEdge:
1870 titleRect.setWidth(borderLeft());
1871 transposed = true;
1872 break;
1873 case Qt::BottomEdge:
1874 titleRect.setTop(titleRect.bottom() - borderBottom());
1875 break;
1876 case Qt::RightEdge:
1877 titleRect.setLeft(titleRect.right() - borderRight());
1878 transposed = true;
1879 break;
1880 }
1881 // When doing a restricted move we must always keep 100px of the titlebar
1882 // visible to allow the user to be able to move it again.
1883 requiredPixels = std::min(100 * (transposed ? titleRect.width() : titleRect.height()),
1884 rect.width() * rect.height());
1885 return titleRect;
1886 };
1887
1888 if (isInteractiveResize()) {
1889 if (m_tile && m_tile->supportsResizeGravity(gravity)) {
1890 m_tile->resizeFromGravity(gravity, x_root, y_root);
1891 return;
1892 }
1893
1894 QRectF orig = initialInteractiveMoveResizeGeometry();
1895 SizeMode sizeMode = SizeModeAny;
1896 auto calculateMoveResizeGeom = [&topleft, &bottomright, &orig, &nextMoveResizeGeom, &sizeMode, &gravity]() {
1897 switch (gravity) {
1898 case Gravity::TopLeft:
1899 nextMoveResizeGeom = QRectF(topleft, orig.bottomRight());
1900 break;
1901 case Gravity::BottomRight:
1902 nextMoveResizeGeom = QRectF(orig.topLeft(), bottomright);
1903 break;
1904 case Gravity::BottomLeft:
1905 nextMoveResizeGeom = QRectF(QPointF(topleft.x(), orig.y()), QPointF(orig.right(), bottomright.y()));
1906 break;
1907 case Gravity::TopRight:
1908 nextMoveResizeGeom = QRectF(QPointF(orig.x(), topleft.y()), QPointF(bottomright.x(), orig.bottom()));
1909 break;
1910 case Gravity::Top:
1911 nextMoveResizeGeom = QRectF(QPointF(orig.left(), topleft.y()), orig.bottomRight());
1912 sizeMode = SizeModeFixedH; // try not to affect height
1913 break;
1914 case Gravity::Bottom:
1915 nextMoveResizeGeom = QRectF(orig.topLeft(), QPointF(orig.right(), bottomright.y()));
1916 sizeMode = SizeModeFixedH;
1917 break;
1918 case Gravity::Left:
1919 nextMoveResizeGeom = QRectF(QPointF(topleft.x(), orig.top()), orig.bottomRight());
1920 sizeMode = SizeModeFixedW;
1921 break;
1922 case Gravity::Right:
1923 nextMoveResizeGeom = QRectF(orig.topLeft(), QPointF(bottomright.x(), orig.bottom()));
1924 sizeMode = SizeModeFixedW;
1925 break;
1926 case Gravity::None:
1927 Q_UNREACHABLE();
1928 break;
1929 }
1930 };
1931
1932 // first resize (without checking constrains), then snap, then check bounds, then check constrains
1933 calculateMoveResizeGeom();
1934 // adjust new size to snap to other windows/borders
1935 nextMoveResizeGeom = workspace()->adjustWindowSize(this, nextMoveResizeGeom, gravity);
1936
1937 if (!isUnrestrictedInteractiveMoveResize()) {
1938 // Make sure the titlebar isn't behind a restricted area. We don't need to restrict
1939 // the other directions. If not visible enough, move the window to the closest valid
1940 // point. We bruteforce this by slowly moving the window back to its previous position
1941 const StrutRects strut = workspace()->restrictedMoveArea(VirtualDesktopManager::self()->currentDesktop());
1942 QRegion availableArea(workspace()->clientArea(FullArea, this, workspace()->activeOutput()).toRect());
1943 for (const QRect &rect : strut) {
1944 availableArea -= rect;
1945 }
1946 bool transposed = false;
1947 int requiredPixels;
1948 QRectF bTitleRect = titleBarRect(nextMoveResizeGeom, transposed, requiredPixels);
1949 int lastVisiblePixels = -1;
1950 QRectF lastTry = nextMoveResizeGeom;
1951 bool titleFailed = false;
1952 for (;;) {
1953 const QRect titleRect = bTitleRect.translated(nextMoveResizeGeom.topLeft()).toRect();
1954 int visiblePixels = 0;
1955 int realVisiblePixels = 0;
1956 for (const QRect &rect : availableArea) {
1957 const QRect r = rect & titleRect;
1958 realVisiblePixels += r.width() * r.height();
1959 if ((transposed && r.width() == titleRect.width()) || // Only the full size regions...
1960 (!transposed && r.height() == titleRect.height())) { // ...prevents long slim areas
1961 visiblePixels += r.width() * r.height();
1962 }
1963 }
1964
1965 if (visiblePixels >= requiredPixels) {
1966 break; // We have reached a valid position
1967 }
1968
1969 if (realVisiblePixels <= lastVisiblePixels) {
1970 if (titleFailed && realVisiblePixels < lastVisiblePixels) {
1971 break; // we won't become better
1972 } else {
1973 if (!titleFailed) {
1974 nextMoveResizeGeom = lastTry;
1975 }
1976 titleFailed = true;
1977 }
1978 }
1979 lastVisiblePixels = realVisiblePixels;
1980 QRectF currentTry = nextMoveResizeGeom;
1981 lastTry = currentTry;
1982
1983 // Not visible enough, move the window to the closest valid point. We bruteforce
1984 // this by slowly moving the window back to its previous position.
1985 // The geometry changes at up to two edges, the one with the title (if) shall take
1986 // precedence. The opposing edge has no impact on visiblePixels and only one of
1987 // the adjacent can alter at a time, ie. it's enough to ignore adjacent edges
1988 // if the title edge altered
1989 bool leftChanged = !qFuzzyCompare(currentMoveResizeGeom.left(), currentTry.left());
1990 bool rightChanged = !qFuzzyCompare(currentMoveResizeGeom.right(), currentTry.right());
1991 bool topChanged = !qFuzzyCompare(currentMoveResizeGeom.top(), currentTry.top());
1992 bool btmChanged = !qFuzzyCompare(currentMoveResizeGeom.bottom(), currentTry.bottom());
1993 auto fixChangedState = [titleFailed](bool &major, bool &counter, bool &ad1, bool &ad2) {
1994 counter = false;
1995 if (titleFailed) {
1996 major = false;
1997 }
1998 if (major) {
1999 ad1 = ad2 = false;
2000 }
2001 };
2002 switch (titlebarPosition()) {
2003 default:
2004 case Qt::TopEdge:
2005 fixChangedState(topChanged, btmChanged, leftChanged, rightChanged);
2006 break;
2007 case Qt::LeftEdge:
2008 fixChangedState(leftChanged, rightChanged, topChanged, btmChanged);
2009 break;
2010 case Qt::BottomEdge:
2011 fixChangedState(btmChanged, topChanged, leftChanged, rightChanged);
2012 break;
2013 case Qt::RightEdge:
2014 fixChangedState(rightChanged, leftChanged, topChanged, btmChanged);
2015 break;
2016 }
2017 if (topChanged) {
2018 currentTry.setTop(currentTry.y() + qBound(-1.0, currentMoveResizeGeom.y() - currentTry.y(), 1.0));
2019 } else if (leftChanged) {
2020 currentTry.setLeft(currentTry.x() + qBound(-1.0, currentMoveResizeGeom.x() - currentTry.x(), 1.0));
2021 } else if (btmChanged) {
2022 currentTry.setBottom(currentTry.bottom() + qBound(-1.0, currentMoveResizeGeom.bottom() - currentTry.bottom(), 1.0));
2023 } else if (rightChanged) {
2024 currentTry.setRight(currentTry.right() + qBound(-1.0, currentMoveResizeGeom.right() - currentTry.right(), 1.0));
2025 } else {
2026 break; // no position changed - that's certainly not good
2027 }
2028 nextMoveResizeGeom = currentTry;
2029 }
2030 }
2031
2032 // Always obey size hints, even when in "unrestricted" mode
2033 QSizeF size = constrainFrameSize(nextMoveResizeGeom.size(), sizeMode);
2034 // the new topleft and bottomright corners (after checking size constrains), if they'll be needed
2035 topleft = QPointF(nextMoveResizeGeom.right() - size.width(), nextMoveResizeGeom.bottom() - size.height());
2036 bottomright = QPointF(nextMoveResizeGeom.left() + size.width(), nextMoveResizeGeom.top() + size.height());
2037 orig = nextMoveResizeGeom;
2038
2039 // if aspect ratios are specified, both dimensions may change.
2040 // Therefore grow to the right/bottom if needed.
2041 // TODO it should probably obey gravity rather than always using right/bottom ?
2042 if (sizeMode == SizeModeFixedH) {
2043 orig.setRight(bottomright.x());
2044 } else if (sizeMode == SizeModeFixedW) {
2045 orig.setBottom(bottomright.y());
2046 }
2047
2048 calculateMoveResizeGeom();
2049 } else if (isInteractiveMove()) {
2050 Q_ASSERT(gravity == Gravity::None);
2051 if (!isMovable()) { // isMovableAcrossScreens() must have been true to get here
2052 // Special moving of maximized windows on Xinerama screens
2053 Output *output = workspace()->outputAt(globalPos);
2054 if (isRequestedFullScreen()) {
2055 nextMoveResizeGeom = workspace()->clientArea(FullScreenArea, this, output);
2056 } else {
2057 nextMoveResizeGeom = workspace()->clientArea(MaximizeArea, this, output);
2058 const QSizeF adjSize = constrainFrameSize(nextMoveResizeGeom.size(), SizeModeMax);
2059 if (adjSize != nextMoveResizeGeom.size()) {
2060 QRectF r(nextMoveResizeGeom);
2061 nextMoveResizeGeom.setSize(adjSize);
2062 nextMoveResizeGeom.moveCenter(r.center());
2063 }
2064 }
2065 } else {
2066 // first move, then snap, then check bounds
2067 QRectF geometry = nextMoveResizeGeom;
2068 geometry.moveTopLeft(topleft);
2069 geometry.moveTopLeft(workspace()->adjustWindowPosition(this, geometry.topLeft(),
2070 isUnrestrictedInteractiveMoveResize()));
2071 nextMoveResizeGeom = geometry;
2072
2073 if (!isUnrestrictedInteractiveMoveResize()) {
2074 const StrutRects strut = workspace()->restrictedMoveArea(VirtualDesktopManager::self()->currentDesktop());
2075 QRegion availableArea(workspace()->clientArea(FullArea, this, workspace()->activeOutput()).toRect());
2076 for (const QRect &rect : strut) {
2077 availableArea -= rect; // Strut areas
2078 }
2079 bool transposed = false;
2080 int requiredPixels;
2081 QRectF bTitleRect = titleBarRect(nextMoveResizeGeom, transposed, requiredPixels);
2082 for (;;) {
2083 QRectF currentTry = nextMoveResizeGeom;
2084 const QRectF titleRect(bTitleRect.translated(currentTry.topLeft()));
2085 int visiblePixels = 0;
2086 for (const QRect &rect : availableArea) {
2087 const QRect r = rect & titleRect.toRect();
2088 if ((transposed && r.width() == titleRect.width()) || // Only the full size regions...
2089 (!transposed && r.height() == titleRect.height())) { // ...prevents long slim areas
2090 visiblePixels += r.width() * r.height();
2091 }
2092 }
2093 if (visiblePixels >= requiredPixels) {
2094 break; // We have reached a valid position
2095 }
2096
2097 // (esp.) if there're more screens with different struts (panels) it the titlebar
2098 // will be movable outside the movearea (covering one of the panels) until it
2099 // crosses the panel "too much" (not enough visiblePixels) and then stucks because
2100 // it's usually only pushed by 1px to either direction
2101 // so we first check whether we intersect suc strut and move the window below it
2102 // immediately (it's still possible to hit the visiblePixels >= titlebarArea break
2103 // by moving the window slightly downwards, but it won't stuck)
2104 // see bug #274466
2105 // and bug #301805 for why we can't just match the titlearea against the screen
2106 if (workspace()->outputs().count() > 1) { // optimization
2107 // TODO: could be useful on partial screen struts (half-width panels etc.)
2108 int newTitleTop = -1;
2109 for (const QRect &region : strut) {
2110 QRectF r = region;
2111 if (r.top() == 0 && r.width() > r.height() && // "top panel"
2112 r.intersects(currentTry) && currentTry.top() < r.bottom()) {
2113 newTitleTop = r.bottom();
2114 break;
2115 }
2116 }
2117 if (newTitleTop > -1) {
2118 currentTry.moveTop(newTitleTop); // invalid position, possibly on screen change
2119 nextMoveResizeGeom = currentTry;
2120 break;
2121 }
2122 }
2123
2124 int dx = sign(currentMoveResizeGeom.x() - currentTry.x()),
2125 dy = sign(currentMoveResizeGeom.y() - currentTry.y());
2126 if (visiblePixels && dx) { // means there's no full width cap -> favor horizontally
2127 dy = 0;
2128 } else if (dy) {
2129 dx = 0;
2130 }
2131
2132 // Move it back
2133 currentTry.translate(dx, dy);
2134 nextMoveResizeGeom = currentTry;
2135
2136 if (nextMoveResizeGeom == currentMoveResizeGeom) {
2137 break; // Prevent lockup
2138 }
2139 }
2140 }
2141 }
2142 } else {
2143 Q_UNREACHABLE();
2144 }
2145
2146 if (nextMoveResizeGeom != currentMoveResizeGeom) {
2147 if (isInteractiveMove()) {
2148 move(nextMoveResizeGeom.topLeft());
2149 } else {
2150 doInteractiveResizeSync(nextMoveResizeGeom);
2151 }
2152
2153 Q_EMIT clientStepUserMovedResized(this, nextMoveResizeGeom);
2154 }

2415{
2416 bool replay = false;
2417 switch (cmd) {
2418 case Options::MouseRaise:
2419 workspace()->raiseWindow(this);
2420 break;
2421 case Options::MouseLower: {
2422 workspace()->lowerWindow(this);
2423 // used to be activateNextWindow(this), then topWindowOnDesktop
2424 // since this is a mouseOp it's however safe to use the window under the mouse instead
2425 if (isActive() && options->focusPolicyIsReasonable()) {
2426 Window *next = workspace()->windowUnderMouse(output());
2427 if (next && next != this) {
2428 workspace()->requestFocus(next, false);
2429 }
2430 }
2431 break;
2432 }
2433 case Options::MouseOperationsMenu:
2434 if (isActive() && options->isClickRaise()) {
2435 autoRaise();
2436 }
2437 workspace()->showWindowMenu(QRect(globalPos.toPoint(), globalPos.toPoint()), this);
2438 break;
2439 case Options::MouseToggleRaiseAndLower:
2440 workspace()->raiseOrLowerWindow(this);
2441 break;
2442 case Options::MouseActivateAndRaise: {
2443 replay = isActive(); // for clickraise mode
2444 bool mustReplay = !rules()->checkAcceptFocus(acceptsFocus());
2445 if (mustReplay) {
2446 auto it = workspace()->stackingOrder().constEnd(),
2447 begin = workspace()->stackingOrder().constBegin();

2449 auto c = *it;
2450 if (!c->isClient() || (c->keepAbove() && !keepAbove()) || (keepBelow() && !c->keepBelow())) {
2451 continue; // can never raise above "it"
2452 }
2453 mustReplay = !(c->isOnCurrentDesktop() && c->isOnCurrentActivity() && c->frameGeometry().intersects(frameGeometry()));
2454 }
2455 }
2456 workspace()->takeActivity(this, Workspace::ActivityFocus | Workspace::ActivityRaise);
2457 workspace()->setActiveOutput(globalPos);
2458 replay = replay || mustReplay;
2459 break;
2460 }
2461 case Options::MouseActivateAndLower:
2462 workspace()->requestFocus(this);
2463 workspace()->lowerWindow(this);
2464 workspace()->setActiveOutput(globalPos);
2465 replay = replay || !rules()->checkAcceptFocus(acceptsFocus());
2466 break;
2467 case Options::MouseActivate:
2468 replay = isActive(); // for clickraise mode
2469 workspace()->takeActivity(this, Workspace::ActivityFocus);
2470 workspace()->setActiveOutput(globalPos);
2471 replay = replay || !rules()->checkAcceptFocus(acceptsFocus());
2472 break;
2473 case Options::MouseActivateRaiseAndPassClick:

2475 workspace()->setActiveOutput(globalPos);
2476 replay = true;
2477 break;
2478 case Options::MouseActivateAndPassClick:

2480 workspace()->setActiveOutput(globalPos);
2481 replay = true;
2482 break;
2483 case Options::MouseMaximize:
2484 maximize(MaximizeFull);
2485 break;
2486 case Options::MouseRestore:
2487 maximize(MaximizeRestore);
2488 break;
2489 case Options::MouseMinimize:
2490 minimize();
2491 break;
2492 case Options::MouseAbove: {
2493 StackingUpdatesBlocker blocker(workspace());
2494 if (keepBelow()) {
2495 setKeepBelow(false);
2496 } else {
2497 setKeepAbove(true);
2498 }
2499 break;
2500 }
2501 case Options::MouseBelow: {
2502 StackingUpdatesBlocker blocker(workspace());
2503 if (keepAbove()) {
2504 setKeepAbove(false);
2505 } else {
2506 setKeepBelow(true);
2507 }

2509 }
2510 case Options::MousePreviousDesktop:

2512 break;
2513 case Options::MouseNextDesktop:

2515 break;
2516 case Options::MouseOpacityMore:

2518 setOpacity(std::min(opacity() + 0.1, 1.0));

2520 break;
2521 case Options::MouseOpacityLess:

2600 case Options::MouseNothing:

2604 }

3879 nextOutput = candidateOutput;

4039void Window::checkWorkspacePosition(QRectF oldGeometry, const VirtualDesktop *oldDesktop)
4040{
4041 if (isDock() || isDesktop() || !isPlaceable()) {
4042 return;
4043 }
4044
4045 QRectF newGeom = moveResizeGeometry();
4046
4047 if (!oldGeometry.isValid()) {
4048 oldGeometry = newGeom;
4049 }
4050
4051 VirtualDesktop *desktop = !isOnCurrentDesktop() ? desktops().constLast() : VirtualDesktopManager::self()->currentDesktop();
4052 if (!oldDesktop) {
4053 oldDesktop = desktop;
4054 }
4055
4056 // If the window was touching an edge before but not now move it so it is again.
4057 // Old and new maximums have different starting values so windows on the screen
4058 // edge will move when a new strut is placed on the edge.
4059 QRect oldScreenArea;
4060 QRect screenArea;
4061 if (workspace()->inUpdateClientArea()) {
4062 // check if the window is on an about to be destroyed output
4063 Output *newOutput = moveResizeOutput();
4064 if (!workspace()->outputs().contains(newOutput)) {
4065 newOutput = workspace()->outputAt(newGeom.center());
4066 }
4067 // we need to find the screen area as it was before the change
4068 oldScreenArea = workspace()->previousScreenSizes().value(moveResizeOutput());
4069 if (oldScreenArea.isNull()) {
4070 oldScreenArea = newOutput->geometry();
4071 }
4072 screenArea = newOutput->geometry();
4073 newGeom.translate(screenArea.topLeft() - oldScreenArea.topLeft());
4074 } else {
4075 oldScreenArea = workspace()->clientArea(ScreenArea, workspace()->outputAt(oldGeometry.center()), oldDesktop).toRect();
4076 screenArea = workspace()->clientArea(ScreenArea, this, newGeom.center()).toRect();
4077 }
4078
4079 if (isRequestedFullScreen() || requestedMaximizeMode() != MaximizeRestore || quickTileMode() != QuickTileMode(QuickTileFlag::None)) {
4080 moveResize(ensureSpecialStateGeometry(newGeom));
4081 m_fullscreenGeometryRestore = moveToArea(m_fullscreenGeometryRestore, oldScreenArea, screenArea);
4082 m_maximizeGeometryRestore = moveToArea(m_maximizeGeometryRestore, oldScreenArea, screenArea);

4085
4086 const QRect oldGeomTall = QRect(oldGeometry.x(), oldScreenArea.y(), oldGeometry.width(), oldScreenArea.height()); // Full screen height
4087 const QRect oldGeomWide = QRect(oldScreenArea.x(), oldGeometry.y(), oldScreenArea.width(), oldGeometry.height()); // Full screen width
4088 int oldTopMax = oldScreenArea.y();
4089 int oldRightMax = oldScreenArea.x() + oldScreenArea.width();
4090 int oldBottomMax = oldScreenArea.y() + oldScreenArea.height();
4091 int oldLeftMax = oldScreenArea.x();
4092 int topMax = screenArea.y();
4093 int rightMax = screenArea.x() + screenArea.width();
4094 int bottomMax = screenArea.y() + screenArea.height();
4095 int leftMax = screenArea.x();

4097 const QRect newGeomWide = QRect(screenArea.x(), newGeom.y(), screenArea.width(), newGeom.height()); // Full screen width
4098 // Get the max strut point for each side where the window is (E.g. Highest point for

4100

4213
4214 if (oldGeometry.x() >= oldLeftMax && newGeom.x() < leftMax) {

4226
4227 moveResize(newGeom);
