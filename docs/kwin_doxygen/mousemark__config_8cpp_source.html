38MouseMarkEffectConfig::MouseMarkEffectConfig(QWidget *parent, const QVariantList &args)
39 : KCModule(parent, args)
40 , m_ui(this)
41{
42 QVBoxLayout *layout = new QVBoxLayout(this);
43 layout->addWidget(&m_ui);
44
45 MouseMarkConfig::instance(KWIN_CONFIG);
46 addConfig(MouseMarkConfig::self(), &m_ui);
47
48 // Shortcut config. The shortcut belongs to the component "kwin"!
49 m_actionCollection = new KActionCollection(this, QStringLiteral("kwin"));
50 m_actionCollection->setComponentDisplayName(i18n("KWin"));
51
52 QAction *a = m_actionCollection->addAction(QStringLiteral("ClearMouseMarks"));
53 a->setText(i18n("Clear Mouse Marks"));
54 a->setProperty("isConfigurationAction", true);
55 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::SHIFT | Qt::META | Qt::Key_F11));
56 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::SHIFT | Qt::META | Qt::Key_F11));
57
58 a = m_actionCollection->addAction(QStringLiteral("ClearLastMouseMark"));
59 a->setText(i18n("Clear Last Mouse Mark"));
60 a->setProperty("isConfigurationAction", true);
61 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::SHIFT | Qt::META | Qt::Key_F12));
62 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::SHIFT | Qt::META | Qt::Key_F12));
63
64 m_ui.editor->addCollection(m_actionCollection);
65
66 connect(m_ui.kcfg_LineWidth, qOverload<int>(&QSpinBox::valueChanged), this, [this]() {
67 updateSpinBoxSuffix();
68 });
69}
