77bool QuickSceneEffectPrivate::isItemOnScreen(QQuickItem *item, EffectScreen *screen) const
78{
79 if (!item || !screen || !views.contains(screen)) {
80 return false;
81 }
82
83 const auto &view = views.at(screen);
84 return item->window() == view->window();

187void QuickSceneEffect::checkItemDraggedOutOfScreen(QQuickItem *item)
188{
189 const QRectF globalGeom = QRectF(item->mapToGlobal(QPointF(0, 0)), QSizeF(item->width(), item->height()));
190 QList<EffectScreen *> screens;
191
192 for (const auto &[screen, view] : d->views) {
193 if (!d->isItemOnScreen(item, screen) && screen->geometry().intersects(globalGeom.toRect())) {
194 screens << screen;
195 }
196 }
197
198 Q_EMIT itemDraggedOutOfScreen(item, screens);
199}

201void QuickSceneEffect::checkItemDroppedOutOfScreen(const QPointF &globalPos, QQuickItem *item)
202{
203 const auto it = std::find_if(d->views.begin(), d->views.end(), [this, globalPos, item](const auto &view) {
204 EffectScreen *screen = view.first;
205 return !d->isItemOnScreen(item, screen) && screen->geometry().contains(globalPos.toPoint());
206 });
207 if (it != d->views.end()) {
208 Q_EMIT itemDroppedOutOfScreen(globalPos, item, it->first);
209 }

280KWin::QuickSceneView *QuickSceneEffect::getView(Qt::Edge edge)
281{
282 auto screenView = activeView();
283
284 QuickSceneView *candidate = nullptr;
285
286 for (const auto &[screen, view] : d->views) {
287 switch (edge) {
288 case Qt::LeftEdge:
289 if (view->geometry().left() < screenView->geometry().left()) {
290 // Look for the nearest view from the current
291 if (!candidate || view->geometry().left() > candidate->geometry().left() || (view->geometry().left() == candidate->geometry().left() && view->geometry().top() > candidate->geometry().top())) {
292 candidate = view.get();
293 }
294 }
295 break;
296 case Qt::TopEdge:
297 if (view->geometry().top() < screenView->geometry().top()) {
298 if (!candidate || view->geometry().top() > candidate->geometry().top() || (view->geometry().top() == candidate->geometry().top() && view->geometry().left() > candidate->geometry().left())) {
299 candidate = view.get();
300 }
301 }
302 break;
303 case Qt::RightEdge:
304 if (view->geometry().right() > screenView->geometry().right()) {
305 if (!candidate || view->geometry().right() < candidate->geometry().right() || (view->geometry().right() == candidate->geometry().right() && view->geometry().top() > candidate->geometry().top())) {
306 candidate = view.get();
307 }
308 }
309 break;
310 case Qt::BottomEdge:
311 if (view->geometry().bottom() > screenView->geometry().bottom()) {
312 if (!candidate || view->geometry().bottom() < candidate->geometry().bottom() || (view->geometry().bottom() == candidate->geometry().bottom() && view->geometry().left() > candidate->geometry().left())) {
313 candidate = view.get();
314 }
315 }
316 break;
317 }
318 }
319
320 return candidate;
321}

323void QuickSceneEffect::activateView(QuickSceneView *view)
324{
325 if (!view) {
326 return;
327 }
328
329 auto *av = activeView();
330 // Already properly active?
331 if (view == av && av->window()->activeFocusItem()) {
332 return;
333 }
334
335 for (const auto &[screen, otherView] : d->views) {
336 if (otherView.get() == view && !view->window()->activeFocusItem()) {
337 QFocusEvent focusEvent(QEvent::FocusIn, Qt::ActiveWindowFocusReason);
338 qApp->sendEvent(view->window(), &focusEvent);
339 } else if (otherView.get() != view && otherView->window()->activeFocusItem()) {
340 QFocusEvent focusEvent(QEvent::FocusOut, Qt::ActiveWindowFocusReason);
341 qApp->sendEvent(otherView->window(), &focusEvent);
342 }
343 }
344
345 Q_EMIT activeViewChanged(view);
346}

350{

353 if (it != d->views.end() && it->second->isDirty()) {

402void QuickSceneEffect::addScreen(EffectScreen *screen)
403{
404 auto properties = initialProperties(screen);
405 properties["width"] = screen->geometry().width();
406 properties["height"] = screen->geometry().height();
407
408 auto incubator = new QuickSceneViewIncubator(this, screen, [this, screen](QuickSceneViewIncubator *incubator) {
409 if (incubator->isReady()) {
410 auto view = incubator->result();
411 if (view->contentItem()) {
412 view->contentItem()->setFocus(false);
413 }
414 connect(view.get(), &QuickSceneView::repaintNeeded, this, [screen]() {
415 effects->addRepaint(screen->geometry());
416 });
417 connect(view.get(), &QuickSceneView::renderRequested, view.get(), &QuickSceneView::scheduleRepaint);
418 connect(view.get(), &QuickSceneView::sceneChanged, view.get(), &QuickSceneView::scheduleRepaint);
419 view->scheduleRepaint();
420 d->views[screen] = std::move(view);
421 } else if (incubator->isError()) {
422 qCWarning(LIBKWINEFFECTS) << "Could not create a view for QML file" << d->qmlComponent->url();
423 qCWarning(LIBKWINEFFECTS) << incubator->errors();
424 }
425 });
426
427 incubator->setInitialProperties(properties);
428 d->incubators[screen].reset(incubator);
429 d->qmlComponent->create(*incubator);
430}

432void QuickSceneEffect::startInternal()
433{
434 if (effects->activeFullScreenEffect()) {
435 return;
436 }
437
438 if (Q_UNLIKELY(d->source.isEmpty())) {
439 qWarning() << "QuickSceneEffect.source is empty. Did you forget to call setSource()?";
440 return;
441 }
442
443 if (!d->qmlEngine) {
444 d->qmlEngine = SharedQmlEngine::engine();
445 }
446
447 if (!d->qmlComponent) {
448 d->qmlComponent.reset(new QQmlComponent(d->qmlEngine.get()));
449 d->qmlComponent->loadUrl(d->source);
450 if (d->qmlComponent->isError()) {
451 qWarning().nospace() << "Failed to load " << d->source << ": " << d->qmlComponent->errors();
452 d->qmlComponent.reset();
453 return;
454 }
455 }
456
457 effects->setActiveFullScreenEffect(this);
458 d->running = true;
459
460 // Install an event filter to monitor cursor shape changes.
461 qApp->installEventFilter(this);
462
463 // This is an ugly hack to make hidpi rendering work as expected on wayland until we switch
464 // to Qt 6.3 or newer. See https://codereview.qt-project.org/c/qt/qtdeclarative/+/361506
465 if (effects->waylandDisplay()) {
466 d->dummyWindow.reset(new QWindow());
467 d->dummyWindow->setOpacity(0);
468 d->dummyWindow->resize(1, 1);
469 d->dummyWindow->setFlag(Qt::FramelessWindowHint);
470 d->dummyWindow->setVisible(true);
471 d->dummyWindow->requestActivate();
472 }
473
474 const QList<EffectScreen *> screens = effects->screens();
475 for (EffectScreen *screen : screens) {
476 addScreen(screen);
477 }
478
479 // Ensure one view has an active focus item
480 activateView(activeView());
481
482 connect(effects, &EffectsHandler::screenAdded, this, &QuickSceneEffect::handleScreenAdded);
483 connect(effects, &EffectsHandler::screenRemoved, this, &QuickSceneEffect::handleScreenRemoved);
484
485 effects->grabKeyboard(this);
486 effects->startMouseInterception(this, Qt::ArrowCursor);
487}

489void QuickSceneEffect::stopInternal()
490{
491 disconnect(effects, &EffectsHandler::screenAdded, this, &QuickSceneEffect::handleScreenAdded);
492 disconnect(effects, &EffectsHandler::screenRemoved, this, &QuickSceneEffect::handleScreenRemoved);
493
494 d->incubators.clear();
495 d->views.clear();
496 d->dummyWindow.reset();
497 d->running = false;
498 qApp->removeEventFilter(this);
499 effects->ungrabKeyboard();
500 effects->stopMouseInterception(this);
501 effects->setActiveFullScreenEffect(nullptr);
502 effects->addRepaintFull();
503}

505void QuickSceneEffect::windowInputMouseEvent(QEvent *event)
506{
507 Qt::MouseButtons buttons;
508 QPoint globalPosition;
509 if (QMouseEvent *mouseEvent = dynamic_cast<QMouseEvent *>(event)) {
510 buttons = mouseEvent->buttons();
511 globalPosition = mouseEvent->globalPos();
512 } else if (QWheelEvent *wheelEvent = dynamic_cast<QWheelEvent *>(event)) {
513 buttons = wheelEvent->buttons();
514 globalPosition = wheelEvent->globalPosition().toPoint();
515 } else {
516 return;
517 }
518
519 if (buttons) {
520 if (!d->mouseImplicitGrab) {
521 d->mouseImplicitGrab = viewAt(globalPosition);
522 }
523 }
524
525 QuickSceneView *target = d->mouseImplicitGrab;
526 if (!target) {
527 target = viewAt(globalPosition);
528 }
529
530 if (!buttons) {
531 d->mouseImplicitGrab = nullptr;
532 }
533
534 if (target) {
535 if (buttons) {
536 activateView(target);
537 }
538 target->forwardMouseEvent(event);
539 }
540}
