34void IdleInhibition::registerClient(Window *client)
35{
36 auto updateInhibit = [this, client] {
37 update(client);
38 };
39
40 m_connections[client] = connect(client->surface(), &SurfaceInterface::inhibitsIdleChanged, this, updateInhibit);
41 connect(client, &Window::desktopChanged, this, updateInhibit);
42 connect(client, &Window::clientMinimized, this, updateInhibit);
43 connect(client, &Window::clientUnminimized, this, updateInhibit);
44 connect(client, &Window::windowHidden, this, updateInhibit);
45 connect(client, &Window::windowShown, this, updateInhibit);
46 connect(client, &Window::windowClosed, this, [this, client]() {
47 uninhibit(client);
48 auto it = m_connections.find(client);
49 if (it != m_connections.end()) {
50 disconnect(it.value());
51 m_connections.erase(it);
52 }
53 });
54
55 updateInhibit();
56}

69void IdleInhibition::update(Window *client)
70{
71 if (client->isInternal()) {
72 return;
73 }
74
75 // TODO: Don't honor the idle inhibitor object if the shell client is not
76 // on the current activity (currently, activities are not supported).
77 const bool visible = client->isShown() && client->isOnCurrentDesktop();
78 if (visible && client->surface() && client->surface()->inhibitsIdle()) {
79 inhibit(client);
80 } else {
81 uninhibit(client);
82 }
83}
