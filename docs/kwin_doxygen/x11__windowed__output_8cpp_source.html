47void X11WindowedCursor::update(const QImage &image, const QPointF &hotspot)
48{
49 X11WindowedBackend *backend = m_output->backend();
50
51 xcb_connection_t *connection = backend->connection();
52 xcb_pixmap_t pix = XCB_PIXMAP_NONE;
53 xcb_gcontext_t gc = XCB_NONE;
54 xcb_cursor_t cid = XCB_CURSOR_NONE;
55
56 if (!image.isNull()) {
57 pix = xcb_generate_id(connection);
58 gc = xcb_generate_id(connection);
59 cid = xcb_generate_id(connection);
60
61 // right now on X we only have one scale between all screens, and we know we will have at least one screen
62 const qreal outputScale = 1;
63 const QSize targetSize = image.size() * outputScale / image.devicePixelRatio();
64 const QImage img = image.scaled(targetSize, Qt::KeepAspectRatio);
65
66 xcb_create_pixmap(connection, 32, pix, backend->screen()->root, img.width(), img.height());
67 xcb_create_gc(connection, gc, pix, 0, nullptr);
68
69 xcb_put_image(connection, XCB_IMAGE_FORMAT_Z_PIXMAP, pix, gc, img.width(), img.height(), 0, 0, 0, 32, img.sizeInBytes(), img.constBits());
70
71 XRenderPicture pic(pix, 32);
72 xcb_render_create_cursor(connection, cid, pic, qRound(hotspot.x() * outputScale), qRound(hotspot.y() * outputScale));
73 }
74
75 xcb_change_window_attributes(connection, m_output->window(), XCB_CW_CURSOR, &cid);
76
77 if (pix) {
78 xcb_free_pixmap(connection, pix);
79 }
80 if (gc) {
81 xcb_free_gc(connection, gc);
82 }
83
84 if (m_handle) {
85 xcb_free_cursor(connection, m_handle);
86 }
87 m_handle = cid;
88 xcb_flush(connection);
89}

158void X11WindowedOutput::init(const QSize &pixelSize, qreal scale)
159{
160 const int refreshRate = 60000; // TODO: get refresh rate via randr
161 m_renderLoop->setRefreshRate(refreshRate);
162
163 auto mode = std::make_shared<OutputMode>(pixelSize, m_renderLoop->refreshRate());
164
165 State initialState;
166 initialState.modes = {mode};
167 initialState.currentMode = mode;
168 initialState.scale = scale;
169 setState(initialState);
170
171 const uint32_t eventMask = XCB_EVENT_MASK_KEY_PRESS
172 | XCB_EVENT_MASK_KEY_RELEASE
173 | XCB_EVENT_MASK_BUTTON_PRESS
174 | XCB_EVENT_MASK_BUTTON_RELEASE
175 | XCB_EVENT_MASK_POINTER_MOTION
176 | XCB_EVENT_MASK_ENTER_WINDOW
177 | XCB_EVENT_MASK_LEAVE_WINDOW
178 | XCB_EVENT_MASK_STRUCTURE_NOTIFY
179 | XCB_EVENT_MASK_EXPOSURE;
180
181 const uint32_t values[] = {
182 m_backend->screen()->black_pixel,
183 eventMask,
184 };
185
186 uint32_t valueMask = XCB_CW_BACK_PIXEL | XCB_CW_EVENT_MASK;
187
188 xcb_create_window(m_backend->connection(),
189 XCB_COPY_FROM_PARENT,
190 m_window,
191 m_backend->screen()->root,
192 0, 0,
193 pixelSize.width(), pixelSize.height(),
194 0, XCB_WINDOW_CLASS_INPUT_OUTPUT, XCB_COPY_FROM_PARENT,
195 valueMask, values);
196
197 // select xinput 2 events
198 initXInputForWindow();
199
200 const uint32_t presentEventMask = XCB_PRESENT_EVENT_MASK_COMPLETE_NOTIFY;
201 m_presentEvent = xcb_generate_id(m_backend->connection());
202 xcb_present_select_input(m_backend->connection(), m_presentEvent, m_window, presentEventMask);
203
204 m_winInfo = std::make_unique<NETWinInfo>(m_backend->connection(), m_window, m_backend->screen()->root,
205 NET::WMWindowType, NET::Properties2());
206
207 m_winInfo->setWindowType(NET::Normal);
208 m_winInfo->setPid(QCoreApplication::applicationPid());
209 QIcon windowIcon = QIcon::fromTheme(QStringLiteral("kwin"));
210 auto addIcon = [&windowIcon, this](const QSize &size) {
211 if (windowIcon.actualSize(size) != size) {
212 return;
213 }
214 NETIcon icon;
215 QImage windowImage = windowIcon.pixmap(size).toImage();
216 icon.data = windowImage.bits();
217 icon.size.width = size.width();
218 icon.size.height = size.height();
219 m_winInfo->setIcon(icon, false);
220 };
221 addIcon(QSize(16, 16));
222 addIcon(QSize(32, 32));
223 addIcon(QSize(48, 48));
224
225 m_cursor = std::make_unique<X11WindowedCursor>(this);
226
227 xcb_map_window(m_backend->connection(), m_window);
228}

230void X11WindowedOutput::initXInputForWindow()
231{
232 if (!m_backend->hasXInput()) {
233 return;
234 }
235#if HAVE_X11_XINPUT
236 XIEventMask evmasks[1];
237 unsigned char mask1[XIMaskLen(XI_LASTEVENT)];
238
239 memset(mask1, 0, sizeof(mask1));
240 XISetMask(mask1, XI_TouchBegin);
241 XISetMask(mask1, XI_TouchUpdate);
242 XISetMask(mask1, XI_TouchOwnership);
243 XISetMask(mask1, XI_TouchEnd);
244 evmasks[0].deviceid = XIAllMasterDevices;
245 evmasks[0].mask_len = sizeof(mask1);
246 evmasks[0].mask = mask1;
247 XISelectEvents(m_backend->display(), m_window, evmasks, 1);
248#endif
249}

287bool X11WindowedOutput::setCursor(CursorSource *source)
288{
289 OutputLayer *cursorLayer = Compositor::self()->backend()->cursorLayer(this);
290 if (source) {
291 cursorLayer->setSize(source->size());
292 cursorLayer->setHotspot(source->hotspot());
293 } else {
294 cursorLayer->setSize(QSize());
295 cursorLayer->setHotspot(QPoint());
296 }
297
298 std::optional<OutputLayerBeginFrameInfo> beginInfo = cursorLayer->beginFrame();
299 if (!beginInfo) {
300 return false;
301 }
302
303 RenderTarget *renderTarget = &beginInfo->renderTarget;
304 renderTarget->setDevicePixelRatio(scale());
305
306 RenderLayer renderLayer(m_renderLoop.get());
307 renderLayer.setDelegate(std::make_unique<SceneDelegate>(Compositor::self()->cursorScene()));
308
309 renderLayer.delegate()->prePaint();
310 renderLayer.delegate()->paint(renderTarget, infiniteRegion());
311 renderLayer.delegate()->postPaint();
312
313 cursorLayer->endFrame(infiniteRegion(), infiniteRegion());
314 return true;
315}
