35void EglOnXBackend::init()
36{
37 qputenv("EGL_PLATFORM", "x11");
38 if (!initRenderingContext()) {
39 setFailed(QStringLiteral("Could not initialize rendering context"));
40 return;
41 }
42
43 initKWinGL();
44 if (!hasExtension(QByteArrayLiteral("EGL_KHR_image")) && (!hasExtension(QByteArrayLiteral("EGL_KHR_image_base")) || !hasExtension(QByteArrayLiteral("EGL_KHR_image_pixmap")))) {
45 setFailed(QStringLiteral("Required support for binding pixmaps to EGLImages not found, disabling compositing"));
46 return;
47 }
48 if (!hasGLExtension(QByteArrayLiteral("GL_OES_EGL_image"))) {
49 setFailed(QStringLiteral("Required extension GL_OES_EGL_image not found, disabling compositing"));
50 return;
51 }
52
53 // check for EGL_NV_post_sub_buffer and whether it can be used on the surface
54 if (hasExtension(QByteArrayLiteral("EGL_NV_post_sub_buffer"))) {
55 if (eglQuerySurface(eglDisplay(), surface(), EGL_POST_SUB_BUFFER_SUPPORTED_NV, &surfaceHasSubPost) == EGL_FALSE) {
56 EGLint error = eglGetError();
57 if (error != EGL_SUCCESS && error != EGL_BAD_ATTRIBUTE) {
58 setFailed(QStringLiteral("query surface failed"));
59 return;
60 } else {
61 surfaceHasSubPost = EGL_FALSE;
62 }
63 }
64 }
65
66 if (surfaceHasSubPost) {
67 qCDebug(KWIN_CORE) << "EGL implementation and surface support eglPostSubBufferNV, let's use it";
68
69 // check if swap interval 1 is supported
70 EGLint val;
71 eglGetConfigAttrib(eglDisplay(), config(), EGL_MAX_SWAP_INTERVAL, &val);
72 if (val >= 1) {
73 if (eglSwapInterval(eglDisplay(), 1)) {
74 qCDebug(KWIN_CORE) << "Enabled v-sync";
75 }
76 } else {
77 qCWarning(KWIN_CORE) << "Cannot enable v-sync as max. swap interval is" << val;
78 }
79 } else {
80 /* In the GLX backend, we fall back to using glCopyPixels if we have no extension providing support for partial screen updates.
81 * However, that does not work in EGL - glCopyPixels with glDrawBuffer(GL_FRONT); does nothing.
82 * Hence we need EGL to preserve the backbuffer for us, so that we can draw the partial updates on it and call
83 * eglSwapBuffers() for each frame. eglSwapBuffers() then does the copy (no page flip possible in this mode),
84 * which means it is slow and not synced to the v-blank. */
85 qCWarning(KWIN_CORE) << "eglPostSubBufferNV not supported, have to enable buffer preservation - which breaks v-sync and performance";
86 eglSurfaceAttrib(eglDisplay(), surface(), EGL_SWAP_BEHAVIOR, EGL_BUFFER_PRESERVED);
87 }
88}

90bool EglOnXBackend::initRenderingContext()
91{
92 initClientExtensions();
93 EGLDisplay dpy = kwinApp()->outputBackend()->sceneEglDisplay();
94
95 // Use eglGetPlatformDisplayEXT() to get the display pointer
96 // if the implementation supports it.
97 if (dpy == EGL_NO_DISPLAY) {
98 const bool havePlatformBase = hasClientExtension(QByteArrayLiteral("EGL_EXT_platform_base"));
99 setHavePlatformBase(havePlatformBase);
100 if (havePlatformBase) {
101 // Make sure that the X11 platform is supported
102 if (!hasClientExtension(QByteArrayLiteral("EGL_EXT_platform_x11")) && !hasClientExtension(QByteArrayLiteral("EGL_KHR_platform_x11"))) {
103 qCWarning(KWIN_CORE) << "EGL_EXT_platform_base is supported, but neither EGL_EXT_platform_x11 nor EGL_KHR_platform_x11 is supported."
104 << "Cannot create EGLDisplay on X11";
105 return false;
106 }
107
108 dpy = eglGetPlatformDisplayEXT(EGL_PLATFORM_X11_EXT, m_x11Display, nullptr);
109 } else {
110 dpy = eglGetDisplay(m_x11Display);
111 }
112 }
113
114 if (dpy == EGL_NO_DISPLAY) {
115 qCWarning(KWIN_CORE) << "Failed to get the EGLDisplay";
116 return false;
117 }
118 setEglDisplay(dpy);
119 initEglAPI();
120
121 initBufferConfigs();
122
123 if (!createSurfaces()) {
124 qCCritical(KWIN_CORE) << "Creating egl surface failed";
125 return false;
126 }
127
128 if (!createContext()) {
129 qCCritical(KWIN_CORE) << "Create OpenGL context failed";
130 return false;
131 }
132
133 if (!makeContextCurrent(surface())) {
134 qCCritical(KWIN_CORE) << "Make Context Current failed";
135 return false;
136 }
137
138 EGLint error = eglGetError();
139 if (error != EGL_SUCCESS) {
140 qCWarning(KWIN_CORE) << "Error occurred while creating context " << error;
141 return false;
142 }
143
144 return true;
145}

147EGLSurface EglOnXBackend::createSurface(xcb_window_t window)
148{
149 if (window == XCB_WINDOW_NONE) {
150 return EGL_NO_SURFACE;
151 }
152
153 // Window is 64 bits on a 64-bit architecture whereas xcb_window_t is always 32 bits.
154 ::Window nativeWindow = window;
155
156 EGLSurface surface = EGL_NO_SURFACE;
157 if (havePlatformBase()) {
158 // eglCreatePlatformWindowSurfaceEXT() expects a pointer to the Window.
159 surface = eglCreatePlatformWindowSurfaceEXT(eglDisplay(), config(), (void *)&nativeWindow, nullptr);
160 } else {
161 // eglCreateWindowSurface() expects a Window, not a pointer to the Window. Use
162 // a c style cast as there are (buggy) platforms where the size of the Window
163 // type is not the same as the size of EGLNativeWindowType, reinterpret_cast<>()
164 // may not compile.
165 surface = eglCreateWindowSurface(eglDisplay(), config(), (EGLNativeWindowType)(uintptr_t)nativeWindow, nullptr);
166 }
167
168 return surface;
169}

171bool EglOnXBackend::initBufferConfigs()
172{
173 initBufferAge();
174 const EGLint config_attribs[] = {
175 EGL_SURFACE_TYPE,
176 EGL_WINDOW_BIT | (supportsBufferAge() ? 0 : EGL_SWAP_BEHAVIOR_PRESERVED_BIT),
177 EGL_RED_SIZE,
178 1,
179 EGL_GREEN_SIZE,
180 1,
181 EGL_BLUE_SIZE,
182 1,
183 EGL_ALPHA_SIZE,
184 0,
185 EGL_RENDERABLE_TYPE,
186 isOpenGLES() ? EGL_OPENGL_ES2_BIT : EGL_OPENGL_BIT,
187 EGL_CONFIG_CAVEAT,
188 EGL_NONE,
189 EGL_NONE,
190 };
191
192 EGLint count;
193 EGLConfig configs[1024];
194 if (eglChooseConfig(eglDisplay(), config_attribs, configs, 1024, &count) == EGL_FALSE) {
195 qCCritical(KWIN_CORE) << "choose config failed";
196 return false;
197 }
198
199 UniqueCPtr<xcb_get_window_attributes_reply_t> attribs(xcb_get_window_attributes_reply(m_connection,
200 xcb_get_window_attributes_unchecked(m_connection, m_rootWindow),
201 nullptr));
202 if (!attribs) {
203 qCCritical(KWIN_CORE) << "Failed to get window attributes of root window";
204 return false;
205 }
206
207 setConfig(configs[0]);
208 for (int i = 0; i < count; i++) {
209 EGLint val;
210 if (eglGetConfigAttrib(eglDisplay(), configs[i], EGL_NATIVE_VISUAL_ID, &val) == EGL_FALSE) {
211 qCCritical(KWIN_CORE) << "egl get config attrib failed";
212 }
213 if (uint32_t(val) == attribs->visual) {
214 setConfig(configs[i]);
215 break;
216 }
217 }
218 return true;
219}
