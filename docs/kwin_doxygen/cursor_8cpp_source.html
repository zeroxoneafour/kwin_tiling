90void Cursors::setCurrentCursor(Cursor *cursor)
91{
92 if (m_currentCursor == cursor) {
93 return;
94 }
95
96 Q_ASSERT(m_cursors.contains(cursor) || !cursor);
97
98 if (m_currentCursor) {
99 disconnect(m_currentCursor, &Cursor::cursorChanged, this, &Cursors::emitCurrentCursorChanged);

129void Cursor::loadThemeSettings()
130{
131 QString themeName = QString::fromUtf8(qgetenv("XCURSOR_THEME"));
132 bool ok = false;
133 // XCURSOR_SIZE might not be set (e.g. by startkde)
134 const uint themeSize = qEnvironmentVariableIntValue("XCURSOR_SIZE", &ok);
135 if (!themeName.isEmpty() && ok) {
136 updateTheme(themeName, themeSize);
137 return;
138 }
139 // didn't get from environment variables, read from config file
140 loadThemeFromKConfig();
141}

148 updateTheme(themeName, themeSize);

237xcb_cursor_t Cursor::x11Cursor(const QByteArray &name)
238{
239 Q_ASSERT(kwinApp()->x11Connection());
240 auto it = m_cursors.constFind(name);
241 if (it != m_cursors.constEnd()) {
242 return it.value();
243 }
244
245 if (name.isEmpty()) {
246 return XCB_CURSOR_NONE;
247 }
248
249 xcb_cursor_context_t *ctx;
250 if (xcb_cursor_context_new(kwinApp()->x11Connection(), Xcb::defaultScreen(), &ctx) < 0) {
251 return XCB_CURSOR_NONE;
252 }
253
254 xcb_cursor_t cursor = xcb_cursor_load_cursor(ctx, name.constData());
255 if (cursor == XCB_CURSOR_NONE) {
256 const auto &names = Cursor::cursorAlternativeNames(name);
257 for (const QByteArray &cursorName : names) {
258 cursor = xcb_cursor_load_cursor(ctx, cursorName.constData());
259 if (cursor != XCB_CURSOR_NONE) {
260 break;
261 }
262 }
263 }
264 if (cursor != XCB_CURSOR_NONE) {
265 m_cursors.insert(name, cursor);
266 }
267
268 xcb_cursor_context_free(ctx);
269 return cursor;
270}
