65bool DrmObject::atomicPopulate(drmModeAtomicReq *req) const
66{
67 for (const auto &property : std::as_const(m_props)) {
68 if (property && !property->isImmutable() && !property->isLegacy() && property->needsCommit()) {
69 if (drmModeAtomicAddProperty(req, m_id, property->propId(), property->pending()) <= 0) {
70 qCWarning(KWIN_DRM) << "Adding property" << property->name() << "->" << property->pending()
71 << "to atomic commit failed for object" << this << "with error" << strerror(errno);

116bool DrmObject::updateProperties()
117{
118 DrmUniquePtr<drmModeObjectProperties> properties(drmModeObjectGetProperties(m_gpu->fd(), m_id, m_objectType));
119 if (!properties) {
120 qCWarning(KWIN_DRM) << "Failed to get properties for object" << m_id;
121 return false;
122 }
123 for (int propIndex = 0; propIndex < m_propertyDefinitions.count(); propIndex++) {
124 const PropertyDefinition &def = m_propertyDefinitions[propIndex];
125 bool found = false;
126 for (uint32_t drmPropIndex = 0; drmPropIndex < properties->count_props; drmPropIndex++) {
127 DrmUniquePtr<drmModePropertyRes> prop(drmModeGetProperty(m_gpu->fd(), properties->props[drmPropIndex]));
128 if (!prop) {
129 qCWarning(KWIN_DRM, "Getting property %d of object %d failed!", drmPropIndex, m_id);
130 continue;
131 }
132 if (def.name == prop->name) {
133 if (m_props[propIndex]) {
134 m_props[propIndex]->setCurrent(properties->prop_values[drmPropIndex]);
135 } else {
136 m_props[propIndex] = std::make_unique<DrmProperty>(this, prop.get(), properties->prop_values[drmPropIndex], def.enumNames);
137 }
138 found = true;
139 break;
140 }
141 }
142 if (!found) {
143 m_props[propIndex].reset();
144 }
145 }
146 for (int i = 0; i < m_propertyDefinitions.count(); i++) {
147 bool required = m_gpu->atomicModeSetting() ? m_propertyDefinitions[i].requirement == Requirement::Required
148 : m_propertyDefinitions[i].requirement == Requirement::RequiredForLegacy;
149 if (!m_props[i] && required) {
150 qCWarning(KWIN_DRM, "Required property %s for object %d not found!", qPrintable(m_propertyDefinitions[i].name), m_id);
151 return false;
152 }
153 }
154 return true;
155}

186void DrmObject::printProps(PrintMode mode)
187{
188 bool any = mode == PrintMode::All || std::any_of(m_props.begin(), m_props.end(), [](const auto &prop) {
189 return prop && !prop->isImmutable() && prop->needsCommit();
190 });
191 if (!any) {
192 return;
193 }
194 qCDebug(KWIN_DRM) << typeName() << id();
195 for (const auto &prop : m_props) {
196 if (prop) {
197 uint64_t current = prop->name().startsWith("SRC_") ? prop->current() >> 16 : prop->current();
198 if (prop->isImmutable() || !prop->needsCommit()) {
199 if (mode == PrintMode::All) {
200 qCDebug(KWIN_DRM).nospace() << "\t" << prop->name() << ": " << current;
201 }
202 } else {
203 uint64_t pending = prop->name().startsWith("SRC_") ? prop->pending() >> 16 : prop->pending();
204 qCDebug(KWIN_DRM).nospace() << "\t" << prop->name() << ": " << current << "->" << pending;
205 }
206 }
207 }
208}
