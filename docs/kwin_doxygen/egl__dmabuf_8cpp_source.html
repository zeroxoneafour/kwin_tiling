153KWaylandServer::LinuxDmaBufV1ClientBuffer *EglDmabuf::yuvImport(DmaBufAttributes &&attrs, quint32 flags)
154{
155 YuvFormat yuvFormat;
156 for (YuvFormat f : yuvFormats) {
157 if (f.format == attrs.format) {
158 yuvFormat = f;
159 break;
160 }
161 }
162 if (yuvFormat.format == 0) {
163 return nullptr;
164 }
165 if (attrs.planeCount != yuvFormat.inputPlanes) {
166 return nullptr;
167 }
168
169 QVector<EGLImage> images;
170 for (int i = 0; i < yuvFormat.outputPlanes; i++) {
171 const int planeIndex = yuvFormat.planes[i].planeIndex;
172 const DmaBufAttributes planeAttrs{
173 .planeCount = 1,
174 .width = attrs.width / yuvFormat.planes[i].widthDivisor,
175 .height = attrs.height / yuvFormat.planes[i].heightDivisor,
176 .format = yuvFormat.planes[i].format,
177 .modifier = attrs.modifier,
178 .fd = {attrs.fd[planeIndex].duplicate()},
179 .offset = {attrs.offset[planeIndex], 0, 0, 0},
180 .pitch = {attrs.pitch[planeIndex], 0, 0, 0},
181 };
182 auto *image = m_backend->importDmaBufAsImage(planeAttrs);
183 if (!image) {
184 return nullptr;
185 }
186 images.push_back(image);
187 }
188
189 return new EglDmabufBuffer(images, std::move(attrs), flags, this);
190}

192EglDmabuf *EglDmabuf::factory(AbstractEglBackend *backend)
193{
194 if (!backend->hasExtension(QByteArrayLiteral("EGL_EXT_image_dma_buf_import"))) {
195 return nullptr;
196 }
197
198 if (backend->hasExtension(QByteArrayLiteral("EGL_EXT_image_dma_buf_import_modifiers"))) {
199 eglQueryDmaBufFormatsEXT = (eglQueryDmaBufFormatsEXT_func)eglGetProcAddress("eglQueryDmaBufFormatsEXT");
200 eglQueryDmaBufModifiersEXT = (eglQueryDmaBufModifiersEXT_func)eglGetProcAddress("eglQueryDmaBufModifiersEXT");
201 }
202
203 if (eglQueryDmaBufFormatsEXT == nullptr) {
204 return nullptr;
205 }
206
207 return new EglDmabuf(backend);
208}

259void filterFormatsWithMultiplePlanes(QVector<uint32_t> &formats)
260{
261 QVector<uint32_t>::iterator it = formats.begin();
262 while (it != formats.end()) {
263 for (auto linuxFormat : s_multiPlaneFormats) {
264 if (*it == linuxFormat) {
265 qCDebug(KWIN_OPENGL) << "Filter multi-plane format" << *it;
266 it = formats.erase(it);
267 it--;
268 break;
269 }
270 }
271 it++;
272 }
273}

277 switch (format) {

303void EglDmabuf::setSupportedFormatsAndModifiers()
304{
305 const EGLDisplay eglDisplay = m_backend->eglDisplay();
306 EGLint count = 0;
307 EGLBoolean success = eglQueryDmaBufFormatsEXT(eglDisplay, 0, nullptr, &count);
308
309 if (!success || count == 0) {
310 qCCritical(KWIN_OPENGL) << "eglQueryDmaBufFormatsEXT failed!" << getEglErrorString();
311 return;
312 }
313
314 QVector<uint32_t> formats(count);
315 if (!eglQueryDmaBufFormatsEXT(eglDisplay, count, (EGLint *)formats.data(), &count)) {
316 qCCritical(KWIN_OPENGL) << "eglQueryDmaBufFormatsEXT with count" << count << "failed!" << getEglErrorString();
317 return;
318 }
319
320 filterFormatsWithMultiplePlanes(formats);
321
322 m_supportedFormats.clear();
323 for (auto format : std::as_const(formats)) {
324 if (eglQueryDmaBufModifiersEXT != nullptr) {
325 EGLint count = 0;
326 const EGLBoolean success = eglQueryDmaBufModifiersEXT(eglDisplay, format, 0, nullptr, nullptr, &count);
327 if (success && count > 0) {
328 QVector<uint64_t> modifiers(count);
329 if (eglQueryDmaBufModifiersEXT(eglDisplay, format, count, modifiers.data(), nullptr, &count)) {
330 m_supportedFormats.insert(format, modifiers);
331 continue;
332 }
333 }
334 }
335 m_supportedFormats.insert(format, {DRM_FORMAT_MOD_INVALID});
336 }
337 qCDebug(KWIN_OPENGL) << "EGL driver advertises" << m_supportedFormats.count() << "supported dmabuf formats" << (eglQueryDmaBufModifiersEXT != nullptr ? "with" : "without") << "modifiers";
338
339 auto filterFormats = [this](int bpc) {
340 QHash<uint32_t, QVector<uint64_t>> set;
341 for (auto it = m_supportedFormats.constBegin(); it != m_supportedFormats.constEnd(); it++) {
342 if (bpcForFormat(it.key()) == bpc) {
343 set.insert(it.key(), it.value());
344 }
345 }
346 return set;
347 };
348 if (m_backend->prefer10bpc()) {
349 m_tranches.append({
350 .device = m_backend->deviceId(),
351 .flags = {},
352 .formatTable = filterFormats(10),
353 });
354 }
355 m_tranches.append({
356 .device = m_backend->deviceId(),
357 .flags = {},
358 .formatTable = filterFormats(8),
359 });
360 m_tranches.append({
361 .device = m_backend->deviceId(),
362 .flags = {},
363 .formatTable = filterFormats(-1),
364 });
365 LinuxDmaBufV1RendererInterface::setSupportedFormatsAndModifiers(m_tranches);
366}
