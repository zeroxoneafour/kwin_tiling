21OverviewEffect::OverviewEffect()
22 : m_shutdownTimer(new QTimer(this))
23{
24 m_shutdownTimer->setSingleShot(true);
25 connect(m_shutdownTimer, &QTimer::timeout, this, &OverviewEffect::realDeactivate);
26
27 const QKeySequence defaultToggleShortcut = Qt::META | Qt::Key_W;
28 m_toggleAction = new QAction(this);
29 connect(m_toggleAction, &QAction::triggered, this, &OverviewEffect::toggle);
30 m_toggleAction->setObjectName(QStringLiteral("Overview"));
31 m_toggleAction->setText(i18n("Toggle Overview"));
32 KGlobalAccel::self()->setDefaultShortcut(m_toggleAction, {defaultToggleShortcut});
33 KGlobalAccel::self()->setShortcut(m_toggleAction, {defaultToggleShortcut});
34 m_toggleShortcut = KGlobalAccel::self()->shortcut(m_toggleAction);
35
36 m_realtimeToggleAction = new QAction(this);
37 connect(m_realtimeToggleAction, &QAction::triggered, this, [this]() {
38 if (m_status == Status::Deactivating) {
39 if (m_partialActivationFactor < 0.5) {
40 deactivate();
41 } else {
42 cancelPartialDeactivate();
43 }
44 } else if (m_status == Status::Activating) {
45 if (m_partialActivationFactor > 0.5) {
46 activate();
47 } else {
48 cancelPartialActivate();
49 }
50 }
51 });
52
53 auto progressCallback = [this](qreal progress) {
54 if (!effects->hasActiveFullScreenEffect() || effects->activeFullScreenEffect() == this) {
55 switch (m_status) {
56 case Status::Inactive:
57 case Status::Activating:
58 partialActivate(progress);
59 break;
60 case Status::Active:
61 case Status::Deactivating:
62 partialDeactivate(progress);
63 break;
64 }
65 }
66 };
67
68 effects->registerTouchpadPinchShortcut(PinchDirection::Contracting, 4, m_realtimeToggleAction, progressCallback);
69 effects->registerTouchscreenSwipeShortcut(SwipeDirection::Up, 3, m_realtimeToggleAction, progressCallback);
70
71 connect(effects, &EffectsHandler::screenAboutToLock, this, &OverviewEffect::realDeactivate);
72
73 initConfig<OverviewConfig>();
74 reconfigure(ReconfigureAll);
75
76 setSource(QUrl::fromLocalFile(QStandardPaths::locate(QStandardPaths::GenericDataLocation, QStringLiteral("kwin/effects/overview/qml/ScreenView.qml"))));
77}

83void OverviewEffect::reconfigure(ReconfigureFlags)
84{
85 OverviewConfig::self()->read();
86 setLayout(OverviewConfig::layoutMode());
87 setAnimationDuration(animationTime(300));
88
89 for (const ElectricBorder &border : std::as_const(m_borderActivate)) {
90 effects->unreserveElectricBorder(border, this);
91 }
92
93 for (const ElectricBorder &border : std::as_const(m_touchBorderActivate)) {
94 effects->unregisterTouchBorder(border, m_toggleAction);
95 }
96
97 m_borderActivate.clear();
98 m_touchBorderActivate.clear();
99
100 const QList<int> activateBorders = OverviewConfig::borderActivate();
101 for (const int &border : activateBorders) {
102 m_borderActivate.append(ElectricBorder(border));
103 effects->reserveElectricBorder(ElectricBorder(border), this);
104 }
105
106 const QList<int> touchActivateBorders = OverviewConfig::touchBorderActivate();
107 for (const int &border : touchActivateBorders) {
108 m_touchBorderActivate.append(ElectricBorder(border));
109 effects->registerRealtimeTouchBorder(ElectricBorder(border), m_realtimeToggleAction, [this](ElectricBorder border, const QPointF &deltaProgress, const EffectScreen *screen) {
110 if (m_status == Status::Active) {
111 return;
112 }
113 const int maxDelta = 500; // Arbitrary logical pixels value seems to behave better than scaledScreenSize
114 if (border == ElectricTop || border == ElectricBottom) {
115 partialActivate(std::min(1.0, std::abs(deltaProgress.y()) / maxDelta));
116 } else {
117 partialActivate(std::min(1.0, std::abs(deltaProgress.x()) / maxDelta));
118 }
119 });
120 }
121}

240void OverviewEffect::deactivate()
241{
242 const auto screens = effects->screens();
243 for (const auto screen : screens) {
244 if (QuickSceneView *view = viewForScreen(screen)) {
245 QMetaObject::invokeMethod(view->rootItem(), "stop");
246 }
247 }
248 m_shutdownTimer->start(animationDuration());
249
250 setGestureInProgress(false);
251 setPartialActivationFactor(0.0);
252}
