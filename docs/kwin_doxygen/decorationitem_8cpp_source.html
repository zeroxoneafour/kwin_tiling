90QImage DecorationRenderer::renderToImage(const QRect &geo)
91{
92 Q_ASSERT(m_client);
93
94 // Guess the pixel format of the X pixmap into which the QImage will be copied.
95 QImage::Format format;
96 const int depth = client()->window()->depth();
97 switch (depth) {
98 case 30:
99 format = QImage::Format_A2RGB30_Premultiplied;
100 break;
101 case 24:
102 case 32:
103 format = QImage::Format_ARGB32_Premultiplied;
104 break;
105 default:
106 qCCritical(KWIN_CORE) << "Unsupported client depth" << depth;
107 format = QImage::Format_ARGB32_Premultiplied;
108 break;
109 };
110
111 QImage image(geo.width() * m_devicePixelRatio, geo.height() * m_devicePixelRatio, format);
112 image.setDevicePixelRatio(m_devicePixelRatio);
113 image.fill(Qt::transparent);
114 QPainter p(&image);
115 p.setRenderHint(QPainter::Antialiasing);
116 p.setWindow(QRect(geo.topLeft(), geo.size() * effectiveDevicePixelRatio()));
117 p.setClipRect(geo);
118 renderToPainter(&p, geo);
119 return image;
120}

224WindowQuad buildQuad(const QRectF &partRect, const QPoint &textureOffset,
225 const qreal devicePixelRatio, bool rotated)
226{
227 const QRectF &r = partRect;
228 const int p = DecorationRenderer::TexturePad;
229
230 const int x0 = r.x();
231 const int y0 = r.y();
232 const int x1 = r.x() + r.width();
233 const int y1 = r.y() + r.height();
234
235 WindowQuad quad;
236 if (rotated) {
237 const int u0 = textureOffset.y() + p;
238 const int v0 = textureOffset.x() + p;
239 const int u1 = textureOffset.y() + p + (r.width() * devicePixelRatio);
240 const int v1 = textureOffset.x() + p + (r.height() * devicePixelRatio);
241
242 quad[0] = WindowVertex(x0, y0, v0, u1); // Top-left
243 quad[1] = WindowVertex(x1, y0, v0, u0); // Top-right
244 quad[2] = WindowVertex(x1, y1, v1, u0); // Bottom-right
245 quad[3] = WindowVertex(x0, y1, v1, u1); // Bottom-left
246 } else {
247 const int u0 = textureOffset.x() + p;
248 const int v0 = textureOffset.y() + p;
249 const int u1 = textureOffset.x() + p + (r.width() * devicePixelRatio);
250 const int v1 = textureOffset.y() + p + (r.height() * devicePixelRatio);
251
252 quad[0] = WindowVertex(x0, y0, u0, v0); // Top-left
253 quad[1] = WindowVertex(x1, y0, u1, v0); // Top-right
254 quad[2] = WindowVertex(x1, y1, u1, v1); // Bottom-right
255 quad[3] = WindowVertex(x0, y1, u0, v1); // Bottom-left
256 }
257 return quad;
258}

260WindowQuadList DecorationItem::buildQuads() const
261{
262 if (m_window->frameMargins().isNull()) {
263 return WindowQuadList();
264 }
265
266 QRectF left, top, right, bottom;
267 const qreal devicePixelRatio = m_renderer->effectiveDevicePixelRatio();
268 const int texturePad = DecorationRenderer::TexturePad;
269
270 m_window->layoutDecorationRects(left, top, right, bottom);
271
272 const int topHeight = std::ceil(top.height() * devicePixelRatio);
273 const int bottomHeight = std::ceil(bottom.height() * devicePixelRatio);
274 const int leftWidth = std::ceil(left.width() * devicePixelRatio);
275
276 const QPoint topPosition(0, 0);
277 const QPoint bottomPosition(0, topPosition.y() + topHeight + (2 * texturePad));
278 const QPoint leftPosition(0, bottomPosition.y() + bottomHeight + (2 * texturePad));
279 const QPoint rightPosition(0, leftPosition.y() + leftWidth + (2 * texturePad));
280
281 WindowQuadList list;
282 if (left.isValid()) {
283 list.append(buildQuad(left, leftPosition, devicePixelRatio, true));
284 }
285 if (top.isValid()) {
286 list.append(buildQuad(top, topPosition, devicePixelRatio, false));
287 }
288 if (right.isValid()) {
289 list.append(buildQuad(right, rightPosition, devicePixelRatio, true));
290 }
291 if (bottom.isValid()) {
292 list.append(buildQuad(bottom, bottomPosition, devicePixelRatio, false));
293 }
294 return list;
295}
