42ShadowBuffer::ShadowBuffer(const QSize &size, const GbmFormat &format)
43 : m_size(size)
44 , m_drmFormat(format.drmFormat)
45{
46 m_texture = std::make_unique<GLTexture>(internalFormat(format), size);
47 m_texture->setFilter(GL_NEAREST);
48
49 m_fbo = std::make_unique<GLFramebuffer>(m_texture.get());
50 if (!m_fbo->valid()) {
51 qCCritical(KWIN_DRM) << "Error: framebuffer not complete!";
52 return;
53 }
54
55 m_vbo = std::make_unique<GLVertexBuffer>(KWin::GLVertexBuffer::Static);
56 m_vbo->setData(6, 2, vertices, texCoords);
57}

63void ShadowBuffer::render(DrmPlane::Transformations transform)
64{
65 QMatrix4x4 mvpMatrix;
66 if (transform & DrmPlane::Transformation::Rotate90) {
67 mvpMatrix.rotate(90, 0, 0, 1);
68 } else if (transform & DrmPlane::Transformation::Rotate180) {
69 mvpMatrix.rotate(180, 0, 0, 1);
70 } else if (transform & DrmPlane::Transformation::Rotate270) {
71 mvpMatrix.rotate(270, 0, 0, 1);
72 }
73 if (transform & DrmPlane::Transformation::ReflectX) {
74 mvpMatrix.scale(-1, 1);
75 }
76 if (transform & DrmPlane::Transformation::ReflectY) {
77 mvpMatrix.scale(1, -1);
78 }
79
80 auto shader = ShaderManager::instance()->pushShader(ShaderTrait::MapTexture);
81 shader->setUniform(GLShader::ModelViewProjectionMatrix, mvpMatrix);
82
83 m_texture->bind();
84 m_vbo->render(GL_TRIANGLES);
85 ShaderManager::instance()->popShader();
86}
