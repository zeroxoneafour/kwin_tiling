32QVariant DesktopModel::data(const QModelIndex &index, int role) const
33{
34 if (!index.isValid() || index.column() != 0) {
35 return QVariant();
36 }
37
38 if (index.parent().isValid()) {
39 // parent is valid -> access to Client
40 ClientModel *model = m_clientModels[m_desktopList[index.internalId() - 1]];
41 return model->data(model->index(index.row(), 0), role);
42 }
43
44 const int desktopIndex = index.row();
45 if (desktopIndex >= m_desktopList.count()) {
46 return QVariant();
47 }
48 switch (role) {
49 case Qt::DisplayRole:
50 case DesktopNameRole:
51 return tabBox->desktopName(m_desktopList[desktopIndex]);
52 case DesktopRole:
53 return m_desktopList[desktopIndex];
54 case ClientModelRole:
55 return QVariant::fromValue<void *>(m_clientModels[m_desktopList[desktopIndex]]);
56 default:
57 return QVariant();
58 }
59}

103QModelIndex DesktopModel::index(int row, int column, const QModelIndex &parent) const
104{
105 if (column != 0) {
106 return QModelIndex();
107 }
108 if (row < 0) {
109 return QModelIndex();
110 }
111 if (parent.isValid()) {
112 if (parent.row() < 0 || parent.row() >= m_desktopList.count() || parent.internalId() != 0) {
113 return QModelIndex();
114 }
115 const int desktop = m_desktopList.at(parent.row());
116 const ClientModel *model = m_clientModels.value(desktop);
117 if (row >= model->rowCount()) {
118 return QModelIndex();
119 }
120 return createIndex(row, column, parent.row() + 1);
121 }
122 if (row > m_desktopList.count() || m_desktopList.isEmpty()) {
123 return QModelIndex();
124 }
125 return createIndex(row, column);
126}

146void DesktopModel::createDesktopList()
147{
148 beginResetModel();
149 m_desktopList.clear();
150 qDeleteAll(m_clientModels);
151 m_clientModels.clear();
152
153 switch (tabBox->config().desktopSwitchingMode()) {
154 case TabBoxConfig::MostRecentlyUsedDesktopSwitching: {
155 int desktop = tabBox->currentDesktop();
156 do {
157 m_desktopList.append(desktop);
158 ClientModel *clientModel = new ClientModel(this);
159 clientModel->createClientList(desktop);
160 m_clientModels.insert(desktop, clientModel);
161 desktop = tabBox->nextDesktopFocusChain(desktop);
162 } while (desktop != tabBox->currentDesktop());
163 break;
164 }
165 case TabBoxConfig::StaticDesktopSwitching: {
166 for (int i = 1; i <= tabBox->numberOfDesktops(); i++) {
167 m_desktopList.append(i);
168 ClientModel *clientModel = new ClientModel(this);
169 clientModel->createClientList(i);
170 m_clientModels.insert(i, clientModel);
171 }
172 break;
173 }
174 }
175 endResetModel();
176}
