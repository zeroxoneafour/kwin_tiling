51void ShmClientBufferPrivate::buffer_destroy_callback(wl_listener *listener, void *data)
52{
53 auto bufferPrivate = reinterpret_cast<ShmClientBufferPrivate::DestroyListener *>(listener)->receiver;
54 wl_shm_buffer *buffer = wl_shm_buffer_get(bufferPrivate->q->resource());
55 wl_shm_pool *pool = wl_shm_buffer_ref_pool(buffer);
56
57 wl_list_remove(&bufferPrivate->destroyListener.listener.link);
58 wl_list_init(&bufferPrivate->destroyListener.listener.link);
59
60 bufferPrivate->savedData = QImage(static_cast<const uchar *>(wl_shm_buffer_get_data(buffer)),
61 bufferPrivate->width,
62 bufferPrivate->height,
63 wl_shm_buffer_get_stride(buffer),
64 bufferPrivate->format,
65 cleanupShmPool,
66 pool);
67}

69static bool alphaChannelFromFormat(uint32_t format)
70{
71 switch (format) {
72 case WL_SHM_FORMAT_ABGR16161616:
73 case WL_SHM_FORMAT_ABGR2101010:
74 case WL_SHM_FORMAT_ARGB2101010:
75 case WL_SHM_FORMAT_ARGB8888:
76 return true;
77 case WL_SHM_FORMAT_XBGR16161616:
78 case WL_SHM_FORMAT_XBGR2101010:
79 case WL_SHM_FORMAT_XRGB2101010:
80 case WL_SHM_FORMAT_XRGB8888:
81 default:
82 return false;
83 }
84}

86static QImage::Format imageFormatForShmFormat(uint32_t format)
87{
88 switch (format) {
89#if Q_BYTE_ORDER == Q_LITTLE_ENDIAN
90 case WL_SHM_FORMAT_ABGR16161616:
91 return QImage::Format_RGBA64_Premultiplied;
92 case WL_SHM_FORMAT_XBGR16161616:
93 return QImage::Format_RGBX64;
94 case WL_SHM_FORMAT_ARGB2101010:
95 return QImage::Format_A2RGB30_Premultiplied;
96 case WL_SHM_FORMAT_XRGB2101010:
97 return QImage::Format_RGB30;
98 case WL_SHM_FORMAT_ABGR2101010:
99 return QImage::Format_A2BGR30_Premultiplied;
100 case WL_SHM_FORMAT_XBGR2101010:
101 return QImage::Format_BGR30;
102#endif
103 case WL_SHM_FORMAT_ARGB8888:
104 return QImage::Format_ARGB32_Premultiplied;
105 case WL_SHM_FORMAT_XRGB8888:
106 return QImage::Format_RGB32;
107 default:
108 return QImage::Format_Invalid;
109 }

112ShmClientBuffer::ShmClientBuffer(wl_resource *resource)
113 : ClientBuffer(resource, *new ShmClientBufferPrivate(this))
114{
115 Q_D(ShmClientBuffer);
116
117 wl_shm_buffer *buffer = wl_shm_buffer_get(resource);
118 d->width = wl_shm_buffer_get_width(buffer);
119 d->height = wl_shm_buffer_get_height(buffer);
120 d->hasAlphaChannel = alphaChannelFromFormat(wl_shm_buffer_get_format(buffer));
121 d->format = imageFormatForShmFormat(wl_shm_buffer_get_format(buffer));
122
123 // The underlying shm pool will be referenced if the wl_shm_buffer is destroyed so the
124 // compositor can access buffer data even after the buffer is gone.
125 d->destroyListener.receiver = d;
126 d->destroyListener.listener.notify = ShmClientBufferPrivate::buffer_destroy_callback;
127 wl_resource_add_destroy_listener(resource, &d->destroyListener.listener);
128}

147static void cleanupShmData(void *bufferHandle)
148{
149 Q_ASSERT_X(s_accessCounter > 0, "cleanup", "access counter must be positive");
150 s_accessCounter--;
151 if (s_accessCounter == 0) {
152 s_accessedBuffer = nullptr;
153 }
154 wl_shm_buffer_end_access(static_cast<wl_shm_buffer *>(bufferHandle));

157QImage ShmClientBuffer::data() const
158{
159 if (s_accessedBuffer && s_accessedBuffer != this) {
160 return QImage();
161 }
162
163 Q_D(const ShmClientBuffer);
164 if (wl_shm_buffer *buffer = wl_shm_buffer_get(resource())) {
165 s_accessedBuffer = this;
166 s_accessCounter++;
167 wl_shm_buffer_begin_access(buffer);
168 const uchar *data = static_cast<const uchar *>(wl_shm_buffer_get_data(buffer));
169 const uint32_t stride = wl_shm_buffer_get_stride(buffer);
170 return QImage(data, d->width, d->height, stride, d->format, cleanupShmData, buffer);
171 }
172 return d->savedData;
173}
