62bool OutputBackend::applyOutputChanges(const OutputConfiguration &config)
63{
64 const auto availableOutputs = outputs();
65 QVector<Output *> toBeEnabledOutputs;
66 QVector<Output *> toBeDisabledOutputs;
67 for (const auto &output : availableOutputs) {
68 if (config.constChangeSet(output)->enabled) {
69 toBeEnabledOutputs << output;
70 } else {
71 toBeDisabledOutputs << output;
72 }
73 }
74 for (const auto &output : toBeEnabledOutputs) {
75 output->applyChanges(config);
76 }
77 for (const auto &output : toBeDisabledOutputs) {
78 output->applyChanges(config);
79 }
80 return true;
81}
