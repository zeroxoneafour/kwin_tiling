41TrackMouseEffectConfig::TrackMouseEffectConfig(QWidget *parent, const QVariantList &args)
42 : KCModule(parent, args)
43{
44 TrackMouseConfig::instance(KWIN_CONFIG);
45 m_ui = new TrackMouseEffectConfigForm(this);
46 QVBoxLayout *layout = new QVBoxLayout(this);
47 layout->addWidget(m_ui);
48
49 addConfig(TrackMouseConfig::self(), m_ui);
50
51 m_actionCollection = new KActionCollection(this, QStringLiteral("kwin"));
52 m_actionCollection->setComponentDisplayName(i18n("KWin"));
53 m_actionCollection->setConfigGroup(QStringLiteral("TrackMouse"));
54 m_actionCollection->setConfigGlobal(true);
55
56 QAction *a = m_actionCollection->addAction(s_toggleTrackMouseActionName);
57 a->setText(i18n("Track mouse"));
58 a->setProperty("isConfigurationAction", true);
59
60 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>());
61 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>());
62
63 connect(m_ui->shortcut, &KKeySequenceWidget::keySequenceChanged,
