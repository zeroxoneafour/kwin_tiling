59void TestAppmenu::init()
60{
61 using namespace KWaylandServer;
62 qRegisterMetaType<AppMenuInterface::InterfaceAddress>();
63 delete m_display;
64 m_display = new KWaylandServer::Display(this);
65 m_display->addSocketName(s_socketName);
66 m_display->start();
67 QVERIFY(m_display->isRunning());
68
69 // setup connection
70 m_connection = new KWayland::Client::ConnectionThread;
71 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
72 m_connection->setSocketName(s_socketName);
73
74 m_thread = new QThread(this);
75 m_connection->moveToThread(m_thread);
76 m_thread->start();
77
78 m_connection->initConnection();
79 QVERIFY(connectedSpy.wait());
80
81 m_queue = new KWayland::Client::EventQueue(this);
82 QVERIFY(!m_queue->isValid());
83 m_queue->setup(m_connection);
84 QVERIFY(m_queue->isValid());
85
86 KWayland::Client::Registry registry;
87 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
88
89 QSignalSpy appmenuSpy(&registry, &KWayland::Client::Registry::appMenuAnnounced);
90
91 QVERIFY(!registry.eventQueue());
92 registry.setEventQueue(m_queue);
93 QCOMPARE(registry.eventQueue(), m_queue);
94 registry.create(m_connection->display());
95 QVERIFY(registry.isValid());
96 registry.setup();
97
98 m_compositorInterface = new CompositorInterface(m_display, m_display);
99 QVERIFY(compositorSpy.wait());
100 m_compositor = registry.createCompositor(compositorSpy.first().first().value<quint32>(), compositorSpy.first().last().value<quint32>(), this);
101
102 m_appmenuManagerInterface = new AppMenuManagerInterface(m_display, m_display);
103
104 QVERIFY(appmenuSpy.wait());
105 m_appmenuManager = registry.createAppMenuManager(appmenuSpy.first().first().value<quint32>(), appmenuSpy.first().last().value<quint32>(), this);
106}

134void TestAppmenu::testCreateAndSet()
135{
136 QSignalSpy serverSurfaceCreated(m_compositorInterface, &KWaylandServer::CompositorInterface::surfaceCreated);
137
138 std::unique_ptr<KWayland::Client::Surface> surface(m_compositor->createSurface());
139 QVERIFY(serverSurfaceCreated.wait());
140
141 auto serverSurface = serverSurfaceCreated.first().first().value<KWaylandServer::SurfaceInterface *>();
142 QSignalSpy appMenuCreated(m_appmenuManagerInterface, &KWaylandServer::AppMenuManagerInterface::appMenuCreated);
143
144 QVERIFY(!m_appmenuManagerInterface->appMenuForSurface(serverSurface));
145
146 auto appmenu = m_appmenuManager->create(surface.get(), surface.get());
147 QVERIFY(appMenuCreated.wait());
148 auto appMenuInterface = appMenuCreated.first().first().value<KWaylandServer::AppMenuInterface *>();
149 QCOMPARE(m_appmenuManagerInterface->appMenuForSurface(serverSurface), appMenuInterface);
150
151 QCOMPARE(appMenuInterface->address().serviceName, QString());
152 QCOMPARE(appMenuInterface->address().objectPath, QString());
153
154 QSignalSpy appMenuChangedSpy(appMenuInterface, &KWaylandServer::AppMenuInterface::addressChanged);
155
156 appmenu->setAddress("net.somename", "/test/path");
157
158 QVERIFY(appMenuChangedSpy.wait());
159 QCOMPARE(appMenuInterface->address().serviceName, QString("net.somename"));
160 QCOMPARE(appMenuInterface->address().objectPath, QString("/test/path"));
161
162 // and destroy
163 QSignalSpy destroyedSpy(appMenuInterface, &QObject::destroyed);
164 delete appmenu;
165 QVERIFY(destroyedSpy.wait());
166 QVERIFY(!m_appmenuManagerInterface->appMenuForSurface(serverSurface));
167}
