42void VirtualEglGbmLayer::aboutToStartPainting(const QRegion &damagedRegion)
43{
44 if (m_gbmSurface && m_gbmSurface->bufferAge() > 0 && !damagedRegion.isEmpty() && m_eglBackend->supportsPartialUpdate()) {
45 const QRegion region = damagedRegion & m_output->geometry();
46
47 QVector<EGLint> rects = m_output->regionToRects(region);
48 const bool correct = eglSetDamageRegionKHR(m_eglBackend->eglDisplay(), m_gbmSurface->eglSurface(), rects.data(), rects.count() / 4);
49 if (!correct) {
50 qCWarning(KWIN_DRM) << "eglSetDamageRegionKHR failed:" << getEglErrorString();
51 }
52 }

55std::optional<OutputLayerBeginFrameInfo> VirtualEglGbmLayer::beginFrame()
56{
57 // gbm surface
58 if (doesGbmSurfaceFit(m_gbmSurface.get())) {
59 m_oldGbmSurface.reset();
60 } else {
61 if (doesGbmSurfaceFit(m_oldGbmSurface.get())) {
62 m_gbmSurface = m_oldGbmSurface;

64 if (!createGbmSurface()) {

70 return std::nullopt;
71 }
72 return OutputLayerBeginFrameInfo{
73 .renderTarget = RenderTarget(m_gbmSurface->fbo()),
74 .repaint = m_gbmSurface->repaintRegion(),
75 };

93bool VirtualEglGbmLayer::createGbmSurface()
94{
95 static bool modifiersEnvSet = false;
96 static const bool modifiersEnv = qEnvironmentVariableIntValue("KWIN_DRM_USE_MODIFIERS", &modifiersEnvSet) != 0;
97 const bool allowModifiers = !modifiersEnvSet || modifiersEnv;
98
99 const auto tranches = m_eglBackend->dmabuf()->tranches();
100 for (const auto &tranche : tranches) {
101 for (auto it = tranche.formatTable.constBegin(); it != tranche.formatTable.constEnd(); it++) {
102 const auto size = m_output->pixelSize();
103 const auto config = m_eglBackend->config(it.key());
104 const auto format = it.key();
105 const auto modifiers = it.value();
106
107 if (allowModifiers && !modifiers.isEmpty()) {
108 const auto ret = GbmSurface::createSurface(m_eglBackend, size, format, modifiers, config);
109 if (const auto surface = std::get_if<std::shared_ptr<GbmSurface>>(&ret)) {
110 m_oldGbmSurface = m_gbmSurface;
111 m_gbmSurface = *surface;
112 return true;
113 } else if (std::get<GbmSurface::Error>(ret) != GbmSurface::Error::ModifiersUnsupported) {
114 continue;
115 }
116 }
117 const auto ret = GbmSurface::createSurface(m_eglBackend, size, format, GBM_BO_USE_RENDERING, config);
118 if (const auto surface = std::get_if<std::shared_ptr<GbmSurface>>(&ret)) {
119 m_oldGbmSurface = m_gbmSurface;
120 m_gbmSurface = *surface;
121 return true;
122 }
123 }
124 }
125 return false;
126}

142bool VirtualEglGbmLayer::scanout(SurfaceItem *surfaceItem)
143{
144 static bool valid;
145 static const bool directScanoutDisabled = qEnvironmentVariableIntValue("KWIN_DRM_NO_DIRECT_SCANOUT", &valid) == 1 && valid;
146 if (directScanoutDisabled) {
147 return false;
148 }
149
150 SurfaceItemWayland *item = qobject_cast<SurfaceItemWayland *>(surfaceItem);
151 if (!item || !item->surface()) {
152 return false;
153 }
154 const auto buffer = qobject_cast<KWaylandServer::LinuxDmaBufV1ClientBuffer *>(item->surface()->buffer());
155 if (!buffer || buffer->size() != m_output->pixelSize()) {
156 return false;
157 }
158 const auto scanoutBuffer = GbmBuffer::importBuffer(m_output->gpu(), buffer);
159 if (!scanoutBuffer) {
160 return false;
161 }
162 // damage tracking for screen casting
163 m_currentDamage = m_scanoutSurface == item->surface() ? surfaceItem->damage() : infiniteRegion();
164 surfaceItem->resetDamage();
165 // ensure the pixmap is updated when direct scanout ends
166 surfaceItem->destroyPixmap();
167 m_scanoutSurface = item->surface();
168 m_currentBuffer = scanoutBuffer;
169 return true;
170}
