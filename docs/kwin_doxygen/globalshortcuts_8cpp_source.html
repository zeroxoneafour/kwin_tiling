29GlobalShortcut::GlobalShortcut(Shortcut &&sc, QAction *action)
30 : m_shortcut(sc)
31 , m_action(action)
32{
33 if (auto swipeGesture = std::get_if<RealtimeFeedbackSwipeShortcut>(&sc)) {
34 m_swipeGesture = std::make_unique<SwipeGesture>();
35 m_swipeGesture->setDirection(swipeGesture->direction);
36 m_swipeGesture->setMinimumDelta(QPointF(200, 200));
37 m_swipeGesture->setMaximumFingerCount(swipeGesture->fingerCount);
38 m_swipeGesture->setMinimumFingerCount(swipeGesture->fingerCount);
39 QObject::connect(m_swipeGesture.get(), &SwipeGesture::triggered, m_action, &QAction::trigger, Qt::QueuedConnection);
40 QObject::connect(m_swipeGesture.get(), &SwipeGesture::cancelled, m_action, &QAction::trigger, Qt::QueuedConnection);
41 if (swipeGesture->progressCallback) {
42 QObject::connect(m_swipeGesture.get(), &SwipeGesture::progress, swipeGesture->progressCallback);
43 }
44 } else if (auto pinchGesture = std::get_if<RealtimeFeedbackPinchShortcut>(&sc)) {
45 m_pinchGesture = std::make_unique<PinchGesture>();
46 m_pinchGesture->setDirection(pinchGesture->direction);
47 m_pinchGesture->setMaximumFingerCount(pinchGesture->fingerCount);
48 m_pinchGesture->setMinimumFingerCount(pinchGesture->fingerCount);
49 QObject::connect(m_pinchGesture.get(), &PinchGesture::triggered, m_action, &QAction::trigger, Qt::QueuedConnection);
50 QObject::connect(m_pinchGesture.get(), &PinchGesture::cancelled, m_action, &QAction::trigger, Qt::QueuedConnection);
51 if (pinchGesture->scaleCallback) {
52 QObject::connect(m_pinchGesture.get(), &PinchGesture::progress, pinchGesture->scaleCallback);
53 }
54 }
55}

123bool GlobalShortcutsManager::addIfNotExists(GlobalShortcut sc, DeviceType device)
124{
125 for (const auto &cs : std::as_const(m_shortcuts)) {
126 if (sc.shortcut() == cs.shortcut()) {
127 return false;
128 }
129 }
130
131 const auto &recognizer = device == DeviceType::Touchpad ? m_touchpadGestureRecognizer : m_touchscreenGestureRecognizer;
132 if (std::holds_alternative<RealtimeFeedbackSwipeShortcut>(sc.shortcut())) {
133 recognizer->registerSwipeGesture(sc.swipeGesture());
134 } else if (std::holds_alternative<RealtimeFeedbackPinchShortcut>(sc.shortcut())) {
135 recognizer->registerPinchGesture(sc.pinchGesture());
136 }
137 connect(sc.action(), &QAction::destroyed, this, &GlobalShortcutsManager::objectDeleted);
138 m_shortcuts.push_back(std::move(sc));
139 return true;
140}

167void GlobalShortcutsManager::forceRegisterTouchscreenSwipe(SwipeDirection direction, uint32_t fingerCount, QAction *action, std::function<void(qreal)> progressCallback)
168{
169 GlobalShortcut shortcut{RealtimeFeedbackSwipeShortcut{DeviceType::Touchscreen, direction, progressCallback, fingerCount}, action};
170 const auto it = std::find_if(m_shortcuts.begin(), m_shortcuts.end(), [&shortcut](const auto &s) {
171 return shortcut.shortcut() == s.shortcut();
172 });
173 if (it != m_shortcuts.end()) {
174 m_shortcuts.erase(it);
175 }
176 m_touchscreenGestureRecognizer->registerSwipeGesture(shortcut.swipeGesture());
177 connect(shortcut.action(), &QAction::destroyed, this, &GlobalShortcutsManager::objectDeleted);
178 m_shortcuts.push_back(std::move(shortcut));
179}

181bool GlobalShortcutsManager::processKey(Qt::KeyboardModifiers mods, int keyQt)
182{
183 if (m_kglobalAccelInterface) {
184 if (!keyQt && !mods) {
185 return false;
186 }
187 auto check = [this](Qt::KeyboardModifiers mods, int keyQt) {
188 bool retVal = false;
189 QMetaObject::invokeMethod(m_kglobalAccelInterface,
190 "checkKeyPressed",
191 Qt::DirectConnection,
192 Q_RETURN_ARG(bool, retVal),
193 Q_ARG(int, int(mods) | keyQt));
194 return retVal;
195 };
196 if (check(mods, keyQt)) {
197 return true;
198 } else if (keyQt == Qt::Key_Backtab) {
199 // KGlobalAccel on X11 has some workaround for Backtab
200 // see kglobalaccel/src/runtime/plugins/xcb/kglobalccel_x11.cpp method x11KeyPress
201 // Apparently KKeySequenceWidget captures Shift+Tab instead of Backtab
202 // thus if the key is backtab we should adjust to add shift again and use tab
203 // in addition KWin registers the shortcut incorrectly as Alt+Shift+Backtab
204 // this should be changed to either Alt+Backtab or Alt+Shift+Tab to match KKeySequenceWidget
205 // trying the variants
206 if (check(mods | Qt::ShiftModifier, keyQt)) {
207 return true;
208 }
209 if (check(mods | Qt::ShiftModifier, Qt::Key_Tab)) {
210 return true;
211 }
212 }
213 }
214 return false;
215}

228template<typename ShortcutKind, typename... Args>
229bool match(QVector<GlobalShortcut> &shortcuts, Args... args)
230{
231 for (auto &sc : shortcuts) {
232 if (std::holds_alternative<ShortcutKind>(sc.shortcut())) {
233 if (std::get<ShortcutKind>(sc.shortcut()) == ShortcutKind{args...}) {
234 sc.invoke();
235 return true;
236 }
237 }
238 }
239 return false;
240}
