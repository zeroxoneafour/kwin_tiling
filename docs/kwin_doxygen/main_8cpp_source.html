115void Application::start()
116{
117 // Prevent KWin from synchronously autostarting kactivitymanagerd
118 // Indeed, kactivitymanagerd being a QApplication it will depend
119 // on KWin startup... this is unsatisfactory dependency wise,
120 // and it turns out that it leads to a deadlock in the Wayland case
121 setProperty("org.kde.KActivities.core.disableAutostart", true);
122
123 setQuitOnLastWindowClosed(false);
124
125 if (!m_config->isImmutable() && m_configLock) {
126 // TODO: This shouldn't be necessary
127 // config->setReadOnly( true );
128 m_config->reparseConfiguration();
129 }
130 if (!m_kxkbConfig) {
131 m_kxkbConfig = KSharedConfig::openConfig(QStringLiteral("kxkbrc"), KConfig::NoGlobals);
132 }
133
134 performStartup();
135}

226void Application::setupMalloc()
227{
228#ifdef M_TRIM_THRESHOLD
229 // Prevent fragmentation of the heap by malloc (glibc).
230 //
231 // The default threshold is 128*1024, which can result in a large memory usage
232 // due to fragmentation especially if we use the raster graphicssystem. On the
233 // otherside if the threshold is too low, free() starts to permanently ask the kernel
234 // about shrinking the heap.
235 const int pagesize = sysconf(_SC_PAGESIZE);
236 mallopt(M_TRIM_THRESHOLD, 5 * pagesize);
237#endif // M_TRIM_THRESHOLD
238}

366 } else {
367 m_eventFilters.append(new X11EventFilterContainer(filter));

371static X11EventFilterContainer *takeEventFilter(X11EventFilter *eventFilter,
372 QList<QPointer<X11EventFilterContainer>> &list)
373{
374 for (int i = 0; i < list.count(); ++i) {
375 X11EventFilterContainer *container = list.at(i);
376 if (container->filter() == eventFilter) {
377 return list.takeAt(i);
378 }
379 }
380 return nullptr;
381}

402bool Application::dispatchEvent(xcb_generic_event_t *event)
403{
404 static const QVector<QByteArray> s_xcbEerrors({QByteArrayLiteral("Success"),
405 QByteArrayLiteral("BadRequest"),
406 QByteArrayLiteral("BadValue"),
407 QByteArrayLiteral("BadWindow"),
408 QByteArrayLiteral("BadPixmap"),
409 QByteArrayLiteral("BadAtom"),
410 QByteArrayLiteral("BadCursor"),
411 QByteArrayLiteral("BadFont"),
412 QByteArrayLiteral("BadMatch"),
413 QByteArrayLiteral("BadDrawable"),
414 QByteArrayLiteral("BadAccess"),
415 QByteArrayLiteral("BadAlloc"),
416 QByteArrayLiteral("BadColor"),
417 QByteArrayLiteral("BadGC"),
418 QByteArrayLiteral("BadIDChoice"),
419 QByteArrayLiteral("BadName"),
420 QByteArrayLiteral("BadLength"),
421 QByteArrayLiteral("BadImplementation"),
422 QByteArrayLiteral("Unknown")});
423
424 kwinApp()->updateX11Time(event);
425
426 const uint8_t x11EventType = event->response_type & ~0x80;
427 if (!x11EventType) {
428 // let's check whether it's an error from one of the extensions KWin uses
429 xcb_generic_error_t *error = reinterpret_cast<xcb_generic_error_t *>(event);
430 const QVector<Xcb::ExtensionData> extensions = Xcb::Extensions::self()->extensions();
431 for (const auto &extension : extensions) {
432 if (error->major_code == extension.majorOpcode) {
433 QByteArray errorName;
434 if (error->error_code < s_xcbEerrors.size()) {
435 errorName = s_xcbEerrors.at(error->error_code);
436 } else if (error->error_code >= extension.errorBase) {
437 const int index = error->error_code - extension.errorBase;
438 if (index >= 0 && index < extension.errorCodes.size()) {
439 errorName = extension.errorCodes.at(index);
440 }
441 }
442 if (errorName.isEmpty()) {
443 errorName = QByteArrayLiteral("Unknown");
444 }
445 qCWarning(KWIN_CORE, "XCB error: %d (%s), sequence: %d, resource id: %d, major code: %d (%s), minor code: %d (%s)",
446 int(error->error_code), errorName.constData(),
447 int(error->sequence), int(error->resource_id),
448 int(error->major_code), extension.name.constData(),
449 int(error->minor_code),
450 extension.opCodes.size() > error->minor_code ? extension.opCodes.at(error->minor_code).constData() : "Unknown");
451 return true;
452 }
453 }
454 return false;
455 }
456
457 if (x11EventType == XCB_GE_GENERIC) {
458 xcb_ge_generic_event_t *ge = reinterpret_cast<xcb_ge_generic_event_t *>(event);
459
460 // We need to make a shadow copy of the event filter list because an activated event
461 // filter may mutate it by removing or installing another event filter.
462 const auto eventFilters = m_genericEventFilters;
463
464 for (X11EventFilterContainer *container : eventFilters) {
465 if (!container) {
466 continue;
467 }
468 X11EventFilter *filter = container->filter();
469 if (filter->extension() == ge->extension && filter->genericEventTypes().contains(ge->event_type) && filter->event(event)) {
470 return true;
471 }
472 }
473 } else {
474 // We need to make a shadow copy of the event filter list because an activated event
475 // filter may mutate it by removing or installing another event filter.
476 const auto eventFilters = m_eventFilters;
477
478 for (X11EventFilterContainer *container : eventFilters) {
479 if (!container) {
480 continue;
481 }
482 X11EventFilter *filter = container->filter();
483 if (filter->eventTypes().contains(x11EventType) && filter->event(event)) {
484 return true;
485 }
486 }
487 }
488
489 if (workspace()) {
490 return workspace()->workspaceEvent(event);
491 }
492
493 return false;
494}

496static quint32 monotonicTime()
497{
498 timespec ts;
499
500 const int result = clock_gettime(CLOCK_MONOTONIC, &ts);
501 if (result) {
502 qCWarning(KWIN_CORE, "Failed to query monotonic time: %s", strerror(errno));
503 }
504
505 return ts.tv_sec * 1000 + ts.tv_nsec / 1000000L;
506}

508void Application::updateXTime()
509{
510 switch (operationMode()) {
511 case Application::OperationModeX11:
512 setX11Time(QX11Info::getTimestamp(), TimestampUpdate::Always);
513 break;
514
515 case Application::OperationModeXwayland:
516 setX11Time(monotonicTime(), TimestampUpdate::Always);
517 break;
518
519 default:
520 // Do not update the current X11 time stamp if it's the Wayland only session.
521 break;
522 }
523}

525void Application::updateX11Time(xcb_generic_event_t *event)
526{
527 xcb_timestamp_t time = XCB_TIME_CURRENT_TIME;
528 const uint8_t eventType = event->response_type & ~0x80;
529 switch (eventType) {
530 case XCB_KEY_PRESS:
531 case XCB_KEY_RELEASE:
532 time = reinterpret_cast<xcb_key_press_event_t *>(event)->time;
533 break;
534 case XCB_BUTTON_PRESS:
535 case XCB_BUTTON_RELEASE:
536 time = reinterpret_cast<xcb_button_press_event_t *>(event)->time;
537 break;
538 case XCB_MOTION_NOTIFY:
539 time = reinterpret_cast<xcb_motion_notify_event_t *>(event)->time;
540 break;
541 case XCB_ENTER_NOTIFY:
542 case XCB_LEAVE_NOTIFY:
543 time = reinterpret_cast<xcb_enter_notify_event_t *>(event)->time;
544 break;
545 case XCB_FOCUS_IN:
546 case XCB_FOCUS_OUT:
547 case XCB_KEYMAP_NOTIFY:
548 case XCB_EXPOSE:
549 case XCB_GRAPHICS_EXPOSURE:
550 case XCB_NO_EXPOSURE:
551 case XCB_VISIBILITY_NOTIFY:
552 case XCB_CREATE_NOTIFY:
553 case XCB_DESTROY_NOTIFY:
554 case XCB_UNMAP_NOTIFY:
555 case XCB_MAP_NOTIFY:
556 case XCB_MAP_REQUEST:
557 case XCB_REPARENT_NOTIFY:
558 case XCB_CONFIGURE_NOTIFY:
559 case XCB_CONFIGURE_REQUEST:
560 case XCB_GRAVITY_NOTIFY:
561 case XCB_RESIZE_REQUEST:
562 case XCB_CIRCULATE_NOTIFY:
563 case XCB_CIRCULATE_REQUEST:
564 // no timestamp
565 return;
566 case XCB_PROPERTY_NOTIFY:
567 time = reinterpret_cast<xcb_property_notify_event_t *>(event)->time;
568 break;
569 case XCB_SELECTION_CLEAR:
570 time = reinterpret_cast<xcb_selection_clear_event_t *>(event)->time;
571 break;
572 case XCB_SELECTION_REQUEST:
573 time = reinterpret_cast<xcb_selection_request_event_t *>(event)->time;
574 break;
575 case XCB_SELECTION_NOTIFY:
576 time = reinterpret_cast<xcb_selection_notify_event_t *>(event)->time;
577 break;
578 case XCB_COLORMAP_NOTIFY:
579 case XCB_CLIENT_MESSAGE:
580 case XCB_MAPPING_NOTIFY:
581 case XCB_GE_GENERIC:
582 // no timestamp
583 return;
584 default:
585 // extension handling
586 if (Xcb::Extensions::self()) {
587 if (eventType == Xcb::Extensions::self()->shapeNotifyEvent()) {
588 time = reinterpret_cast<xcb_shape_notify_event_t *>(event)->server_time;
589 }
590 if (eventType == Xcb::Extensions::self()->damageNotifyEvent()) {
591 time = reinterpret_cast<xcb_damage_notify_event_t *>(event)->timestamp;
592 }
593 }
594 break;
595 }
596 setX11Time(time);
597}
