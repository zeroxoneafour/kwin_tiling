382static QSize pickWindowSize(const QImage &image)
383{
384 xcb_connection_t *c = effects->xcbConnection();
385
386 // This will implicitly enable BIG-REQUESTS extension.
387 const uint32_t maximumRequestSize = xcb_get_maximum_request_length(c);
388 const xcb_setup_t *setup = xcb_get_setup(c);
389
390 uint32_t requestSize = sizeof(xcb_put_image_request_t);
391
392 // With BIG-REQUESTS extension an additional 32-bit field is inserted into
393 // the request so we better take it into account.
394 if (setup->maximum_request_length < maximumRequestSize) {
395 requestSize += 4;
396 }
397
398 const uint32_t maximumDataSize = 4 * maximumRequestSize - requestSize;
399 const uint32_t bytesPerPixel = image.depth() >> 3;
400 const uint32_t bytesPerLine = image.bytesPerLine();
401
402 if (image.sizeInBytes() <= maximumDataSize) {
403 return image.size();
404 }
405
406 if (maximumDataSize < bytesPerLine) {
407 return QSize(maximumDataSize / bytesPerPixel, 1);
408 }
409
410 return QSize(image.width(), maximumDataSize / bytesPerLine);
411}

413static xcb_pixmap_t xpixmapFromImage(const QImage &image)
414{
415 xcb_connection_t *c = effects->xcbConnection();
416
417 xcb_pixmap_t pixmap = xcb_generate_id(c);
418 xcb_gcontext_t gc = xcb_generate_id(c);
419
420 xcb_create_pixmap(c, image.depth(), pixmap, effects->x11RootWindow(),
421 image.width(), image.height());
422 xcb_create_gc(c, gc, pixmap, 0, nullptr);
423
424 const int bytesPerPixel = image.depth() >> 3;
425
426 // Figure out how much data we can send with one invocation of xcb_put_image.
427 // In contrast to XPutImage, xcb_put_image doesn't implicitly split the data.
428 const QSize window = pickWindowSize(image);
429
430 for (int i = 0; i < image.height(); i += window.height()) {
431 const int targetHeight = std::min(image.height() - i, window.height());
432 const uint8_t *line = image.scanLine(i);
433
434 for (int j = 0; j < image.width(); j += window.width()) {
435 const int targetWidth = std::min(image.width() - j, window.width());
436 const uint8_t *bytes = line + j * bytesPerPixel;
437 const uint32_t byteCount = targetWidth * targetHeight * bytesPerPixel;
438
439 xcb_put_image(c, XCB_IMAGE_FORMAT_Z_PIXMAP, pixmap, gc,
440 targetWidth, targetHeight, j, i, 0, image.depth(),
441 byteCount, bytes);
442 }
443 }
444
445 xcb_flush(c);
446 xcb_free_gc(c, gc);
447
448 return pixmap;
449}

504bool ScreenShotDBusInterface1::checkCall() const
505{
506 if (!calledFromDBus()) {
507 return false;
508 }
509
510 static bool permissionCheckDisabled = qEnvironmentVariableIntValue("KWIN_SCREENSHOT_NO_PERMISSION_CHECKS") == 1;
511 if (permissionCheckDisabled) {
512 return true;
513 }
514
515 const QDBusReply<uint> reply = connection().interface()->servicePid(message().service());
516 if (reply.isValid()) {
517 const uint pid = reply.value();
518 const auto interfaces = KWin::fetchRestrictedDBusInterfacesFromPid(pid);
519 if (!interfaces.contains(s_dbusInterfaceName)) {
520 sendErrorReply(s_errorNotAuthorized, s_errorNotAuthorizedMsg);
521 qCWarning(KWIN_SCREENSHOT) << "Process" << pid << "tried to take a screenshot without being granted to DBus interface" << s_dbusInterfaceName;
522 return false;
523 }
524 } else {
525 return false;
526 }
527
528 if (isTakingScreenshot()) {
529 sendErrorReply(s_errorAlreadyTaking, s_errorAlreadyTakingMsg);
530 return false;
531 }
532 return true;
533}

535void ScreenShotDBusInterface1::screenshotWindowUnderCursor(int mask)
536{
537 if (isTakingScreenshot()) {
538 sendErrorReply(s_errorAlreadyTaking, s_errorAlreadyTakingMsg);
539 return;
540 }
541
542 EffectWindow *hoveredWindow = nullptr;
543
544 const QPointF cursor = effects->cursorPos();
545 EffectWindowList order = effects->stackingOrder();
546 EffectWindowList::const_iterator it = order.constEnd(), first = order.constBegin();
547 while (it != first) {
548 hoveredWindow = *(--it);
549 if (hoveredWindow->isOnCurrentDesktop()
550 && !hoveredWindow->isMinimized()
551 && !hoveredWindow->isDeleted()
552 && exclusiveContains(hoveredWindow->frameGeometry(), cursor)) {
553 break;
554 }
555 hoveredWindow = nullptr;
556 }
557
558 if (hoveredWindow) {
559 takeScreenShot(hoveredWindow, ScreenShotFlags(mask),
560 new ScreenShotSinkXpixmap1(this, message()));
561 }
562}
