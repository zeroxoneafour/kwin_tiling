63void ColorDevicePrivate::rebuildPipeline()
64{
65 if (dirtyCurves & DirtyCalibrationToneCurve) {
66 updateCalibrationToneCurves();
67 }
68 if (dirtyCurves & DirtyBrightnessToneCurve) {
69 updateBrightnessToneCurves();
70 }
71 if (dirtyCurves & DirtyTemperatureToneCurve) {
72 updateTemperatureToneCurves();
73 }
74 dirtyCurves = DirtyToneCurves();
75
76 std::vector<std::unique_ptr<ColorPipelineStage>> stages;
77 if (calibrationStage) {
78 if (auto s = calibrationStage->dup()) {
79 stages.push_back(std::move(s));
80 } else {
81 return;
82 }
83 }
84 if (brightnessStage) {
85 if (auto s = brightnessStage->dup()) {
86 stages.push_back(std::move(s));
87 } else {
88 return;
89 }
90 }
91 if (temperatureStage) {
92 if (auto s = temperatureStage->dup()) {
93 stages.push_back(std::move(s));
94 } else {
95 return;
96 }
97 }
98
99 const auto tmp = std::make_shared<ColorTransformation>(std::move(stages));
100 if (tmp->valid()) {
101 transformation = tmp;
102 }
103}

115void ColorDevicePrivate::updateTemperatureToneCurves()
116{
117 temperatureStage.reset();
118
119 if (temperature == 6500) {
120 return;
121 }
122
123 // Note that cmsWhitePointFromTemp() returns a slightly green-ish white point.
124 const int blackBodyColorIndex = ((temperature - 1000) / 100) * 3;
125 const qreal blendFactor = (temperature % 100) / 100.0;
126
127 const qreal xWhitePoint = interpolate(blackbodyColor[blackBodyColorIndex + 0],
128 blackbodyColor[blackBodyColorIndex + 3],
129 blendFactor);
130 const qreal yWhitePoint = interpolate(blackbodyColor[blackBodyColorIndex + 1],
131 blackbodyColor[blackBodyColorIndex + 4],
132 blendFactor);
133 const qreal zWhitePoint = interpolate(blackbodyColor[blackBodyColorIndex + 2],
134 blackbodyColor[blackBodyColorIndex + 5],
135 blendFactor);
136
137 const double redCurveParams[] = {1.0, xWhitePoint, 0.0};
138 const double greenCurveParams[] = {1.0, yWhitePoint, 0.0};
139 const double blueCurveParams[] = {1.0, zWhitePoint, 0.0};
140
141 UniqueToneCurvePtr redCurve(cmsBuildParametricToneCurve(nullptr, 2, redCurveParams));
142 if (!redCurve) {
143 qCWarning(KWIN_CORE) << "Failed to build the temperature tone curve for the red channel";
144 return;
145 }
146 UniqueToneCurvePtr greenCurve(cmsBuildParametricToneCurve(nullptr, 2, greenCurveParams));
147 if (!greenCurve) {
148 qCWarning(KWIN_CORE) << "Failed to build the temperature tone curve for the green channel";
149 return;
150 }
151 UniqueToneCurvePtr blueCurve(cmsBuildParametricToneCurve(nullptr, 2, blueCurveParams));
152 if (!blueCurve) {
153 qCWarning(KWIN_CORE) << "Failed to build the temperature tone curve for the blue channel";
154 return;
155 }
156
157 // The ownership of the tone curves will be moved to the pipeline stage.
158 cmsToneCurve *toneCurves[] = {redCurve.release(), greenCurve.release(), blueCurve.release()};
159
160 temperatureStage = std::make_unique<ColorPipelineStage>(cmsStageAllocToneCurves(nullptr, 3, toneCurves));
161 if (!temperatureStage) {
162 qCWarning(KWIN_CORE) << "Failed to create the color temperature pipeline stage";
163 }
164}

168 brightnessStage.reset();

176 UniqueToneCurvePtr redCurve(cmsBuildParametricToneCurve(nullptr, 2, curveParams));

203void ColorDevicePrivate::updateCalibrationToneCurves()
204{
205 calibrationStage.reset();
206
207 if (profile.isNull()) {
208 return;
209 }
210
211 cmsHPROFILE handle = cmsOpenProfileFromFile(profile.toUtf8(), "r");
212 if (!handle) {
213 qCWarning(KWIN_CORE) << "Failed to open color profile file:" << profile;
214 return;
215 }
216
217 cmsToneCurve **vcgt = static_cast<cmsToneCurve **>(cmsReadTag(handle, cmsSigVcgtTag));
218 if (!vcgt || !vcgt[0]) {
219 qCWarning(KWIN_CORE) << "Profile" << profile << "has no VCGT tag";
220 } else {
221 // Need to duplicate the VCGT tone curves as they are owned by the profile.
222 cmsToneCurve *toneCurves[] = {
223 cmsDupToneCurve(vcgt[0]),
224 cmsDupToneCurve(vcgt[1]),
225 cmsDupToneCurve(vcgt[2]),
226 };
227 calibrationStage = std::make_unique<ColorPipelineStage>(cmsStageAllocToneCurves(nullptr, 3, toneCurves));
228 }
229
230 cmsCloseProfile(handle);
231}
