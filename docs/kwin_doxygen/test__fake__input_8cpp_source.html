55void FakeInputTest::init()
56{
57 delete m_display;
58 m_display = new KWaylandServer::Display(this);
59 m_display->addSocketName(s_socketName);
60 m_display->start();
61 QVERIFY(m_display->isRunning());
62 m_display->createShm();
63 m_fakeInputInterface = new FakeInputInterface(m_display);
64 QSignalSpy deviceCreatedSpy(m_fakeInputInterface, &FakeInputInterface::deviceCreated);
65
66 // setup connection
67 m_connection = new KWayland::Client::ConnectionThread;
68 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
69 m_connection->setSocketName(s_socketName);
70
71 m_thread = new QThread(this);
72 m_connection->moveToThread(m_thread);
73 m_thread->start();
74
75 m_connection->initConnection();
76 QVERIFY(connectedSpy.wait());
77
78 m_queue = new KWayland::Client::EventQueue(this);
79 m_queue->setup(m_connection);
80
81 KWayland::Client::Registry registry;
82 QSignalSpy interfacesAnnouncedSpy(&registry, &KWayland::Client::Registry::interfacesAnnounced);
83 registry.setEventQueue(m_queue);
84 registry.create(m_connection);
85 QVERIFY(registry.isValid());
86 registry.setup();
87 QVERIFY(interfacesAnnouncedSpy.wait());
88
89 m_fakeInput =
90 registry.createFakeInput(registry.interface(KWayland::Client::Registry::Interface::FakeInput).name, registry.interface(KWayland::Client::Registry::Interface::FakeInput).version, this);
91 QVERIFY(m_fakeInput->isValid());
92
93 QVERIFY(deviceCreatedSpy.wait());
94 m_device = deviceCreatedSpy.first().first().value<FakeInputDevice *>();
95 QVERIFY(m_device);
96}

98void FakeInputTest::cleanup()
99{
100#define CLEANUP(variable) \
101 if (variable) { \
102 delete variable; \
103 variable = nullptr; \
104 }
105 CLEANUP(m_fakeInput)
106 CLEANUP(m_queue)
107 if (m_connection) {
108 m_connection->deleteLater();

110 }
111 if (m_thread) {
112 m_thread->quit();

116 }
117

120
121 // these are the children of the display
122 m_fakeInputInterface = nullptr;
