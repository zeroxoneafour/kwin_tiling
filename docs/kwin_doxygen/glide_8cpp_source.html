35static QMatrix4x4 createPerspectiveMatrix(const QRectF &rect, const qreal scale)
36{
37 QMatrix4x4 ret;
38
39 const float fovY = std::tan(qDegreesToRadians(60.0f) / 2);
40 const float aspect = 1.0f;
41 const float zNear = 0.1f;
42 const float zFar = 100.0f;
43
44 const float yMax = zNear * fovY;
45 const float yMin = -yMax;
46 const float xMin = yMin * aspect;
47 const float xMax = yMax * aspect;
48
49 ret.frustum(xMin, xMax, yMin, yMax, zNear, zFar);
50
51 const auto deviceRect = scaledRect(rect, scale);
52
53 const float scaleFactor = 1.1 * fovY / yMax;
54 ret.translate(xMin * scaleFactor, yMax * scaleFactor, -1.1);
55 ret.scale((xMax - xMin) * scaleFactor / deviceRect.width(),
56 -(yMax - yMin) * scaleFactor / deviceRect.height(),
57 0.001);
58 ret.translate(-deviceRect.x(), -deviceRect.y());
59
60 return ret;
61}

77{
78 GlideConfig::self()->read();

80
81 m_inParams.edge = static_cast<RotationEdge>(GlideConfig::inRotationEdge());

120void GlideEffect::paintWindow(EffectWindow *w, int mask, QRegion region, WindowPaintData &data)
121{
122 auto animationIt = m_animations.constFind(w);
123 if (animationIt == m_animations.constEnd()) {
124 effects->paintWindow(w, mask, region, data);
125 return;
126 }
127
128 // Perspective projection distorts objects near edges
129 // of the viewport. This is critical because distortions
130 // near edges of the viewport are not desired with this effect.
131 // To fix this, the center of the window will be moved to the origin,
132 // after applying perspective projection, the center is moved back
133 // to its "original" projected position. Overall, this is how the window
134 // will be transformed:
135 // [move to the origin] -> [rotate] -> [translate] ->
136 // -> [perspective projection] -> [reverse "move to the origin"]
137
138 const QMatrix4x4 oldProjMatrix = createPerspectiveMatrix(effects->renderTargetRect(), effects->renderTargetScale());
139 const auto frame = w->frameGeometry();
140 const auto scale = effects->renderTargetScale();
141 const QRectF windowGeo = scaledRect(frame, scale);
142 const QVector3D invOffset = oldProjMatrix.map(QVector3D(windowGeo.center()));
143 QMatrix4x4 invOffsetMatrix;
144 invOffsetMatrix.translate(invOffset.x(), invOffset.y());
145
146 data.setProjectionMatrix(invOffsetMatrix * oldProjMatrix);
147
148 // Move the center of the window to the origin.
149 const QPointF offset = effects->renderTargetRect().center() - w->frameGeometry().center();
150 data.translate(offset.x(), offset.y());
151
152 const GlideParams params = w->isDeleted() ? m_outParams : m_inParams;
153 const qreal t = (*animationIt).timeLine.value();
154
155 switch (params.edge) {
156 case RotationEdge::Top:
157 data.setRotationAxis(Qt::XAxis);
158 data.setRotationOrigin(QVector3D(0, 0, 0));
159 data.setRotationAngle(-interpolate(params.angle.from, params.angle.to, t));
160 break;
161
162 case RotationEdge::Right:
163 data.setRotationAxis(Qt::YAxis);
164 data.setRotationOrigin(QVector3D(w->width(), 0, 0));
165 data.setRotationAngle(-interpolate(params.angle.from, params.angle.to, t));
166 break;
167
168 case RotationEdge::Bottom:
169 data.setRotationAxis(Qt::XAxis);
170 data.setRotationOrigin(QVector3D(0, w->height(), 0));
171 data.setRotationAngle(interpolate(params.angle.from, params.angle.to, t));
172 break;
173
174 case RotationEdge::Left:
175 data.setRotationAxis(Qt::YAxis);
176 data.setRotationOrigin(QVector3D(0, 0, 0));
177 data.setRotationAngle(interpolate(params.angle.from, params.angle.to, t));
178 break;
179
180 default:
181 // Fallback to Top.
182 data.setRotationAxis(Qt::XAxis);
183 data.setRotationOrigin(QVector3D(0, 0, 0));
184 data.setRotationAngle(-interpolate(params.angle.from, params.angle.to, t));
185 break;
186 }
187
188 data.setZTranslation(-interpolate(params.distance.from, params.distance.to, t));
189 data.multiplyOpacity(interpolate(params.opacity.from, params.opacity.to, t));
190
191 effects->paintWindow(w, mask, region, data);
192}

194void GlideEffect::postPaintScreen()
195{
196 auto animationIt = m_animations.begin();
197 while (animationIt != m_animations.end()) {
198 if ((*animationIt).timeLine.done()) {
199 animationIt = m_animations.erase(animationIt);
200 } else {
201 ++animationIt;
202 }
203 }
204
205 effects->addRepaintFull();
206 effects->postPaintScreen();

220void GlideEffect::windowAdded(EffectWindow *w)
221{
222 if (effects->activeFullScreenEffect()) {
223 return;
224 }
225
226 if (!isGlideWindow(w)) {
227 return;
228 }
229
230 if (!w->isVisible()) {
231 return;
232 }
233
234 const void *addGrab = w->data(WindowAddedGrabRole).value<void *>();
235 if (addGrab && addGrab != this) {
236 return;
237 }
238
239 w->setData(WindowAddedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
240
241 GlideAnimation &animation = m_animations[w];
242 animation.timeLine.reset();
243 animation.timeLine.setDirection(TimeLine::Forward);
244 animation.timeLine.setDuration(m_duration);
245 animation.timeLine.setEasingCurve(QEasingCurve::InCurve);
246
247 effects->addRepaintFull();
248}

250void GlideEffect::windowClosed(EffectWindow *w)
251{
252 if (effects->activeFullScreenEffect()) {
253 return;
254 }
255
256 if (!isGlideWindow(w)) {
257 return;
258 }
259
260 if (!w->isVisible() || w->skipsCloseAnimation()) {
261 return;
262 }
263
264 const void *closeGrab = w->data(WindowClosedGrabRole).value<void *>();
265 if (closeGrab && closeGrab != this) {
266 return;
267 }
268
269 w->setData(WindowClosedGrabRole, QVariant::fromValue(static_cast<void *>(this)));
270
271 GlideAnimation &animation = m_animations[w];
272 animation.deletedRef = EffectWindowDeletedRef(w);
273 animation.visibleRef = EffectWindowVisibleRef(w, EffectWindow::PAINT_DISABLED_BY_DELETE);
274 animation.timeLine.reset();
275 animation.timeLine.setDirection(TimeLine::Forward);
276 animation.timeLine.setDuration(m_duration);
277 animation.timeLine.setEasingCurve(QEasingCurve::OutCurve);
278
279 effects->addRepaintFull();
280}

303bool GlideEffect::isGlideWindow(EffectWindow *w) const
304{
305 // We don't want to animate most of plasmashell's windows, yet, some
306 // of them we want to, for example, Task Manager Settings window.
307 // The problem is that all those window share single window class.
308 // So, the only way to decide whether a window should be animated is
309 // to use a heuristic: if a window has decoration, then it's most
310 // likely a dialog or a settings window so we have to animate it.
311 if (w->windowClass() == QLatin1String("plasmashell plasmashell")
312 || w->windowClass() == QLatin1String("plasmashell org.kde.plasmashell")) {
313 return w->hasDecoration();
314 }
315
316 if (s_blacklist.contains(w->windowClass())) {
317 return false;
318 }
319
320 if (w->hasDecoration()) {
321 return true;
322 }
323
324 // Don't animate combobox popups, tooltips, popup menus, etc.
325 if (w->isPopupWindow()) {
326 return false;
327 }
328
329 // Don't animate the outline and the screenlocker as it looks bad.
330 if (w->isLockScreen() || w->isOutline()) {
331 return false;
332 }
333
334 // Override-redirect windows are usually used for user interface
335 // concepts that are not expected to be animated by this effect.
336 if (w->isX11Client() && !w->isManaged()) {
337 return false;
338 }
339
340 return w->isNormalWindow()
341 || w->isDialog();
342}
