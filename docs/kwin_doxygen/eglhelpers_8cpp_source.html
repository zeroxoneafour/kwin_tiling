31EGLConfig configFromFormat(EGLDisplay display, const QSurfaceFormat &surfaceFormat, EGLint surfaceType)
32{
33 // std::max as these values are initialized to -1 by default.
34 const EGLint redSize = std::max(surfaceFormat.redBufferSize(), 0);
35 const EGLint greenSize = std::max(surfaceFormat.greenBufferSize(), 0);
36 const EGLint blueSize = std::max(surfaceFormat.blueBufferSize(), 0);
37 const EGLint alphaSize = std::max(surfaceFormat.alphaBufferSize(), 0);
38 const EGLint depthSize = std::max(surfaceFormat.depthBufferSize(), 0);
39 const EGLint stencilSize = std::max(surfaceFormat.stencilBufferSize(), 0);
40
41 const EGLint renderableType = isOpenGLES() ? EGL_OPENGL_ES2_BIT : EGL_OPENGL_BIT;
42
43 // Not setting samples as QtQuick doesn't need it.
44 const QVector<EGLint> attributes{
45 EGL_SURFACE_TYPE, surfaceType,
46 EGL_RED_SIZE, redSize,
47 EGL_GREEN_SIZE, greenSize,
48 EGL_BLUE_SIZE, blueSize,
49 EGL_ALPHA_SIZE, alphaSize,
50 EGL_DEPTH_SIZE, depthSize,
51 EGL_STENCIL_SIZE, stencilSize,
52 EGL_RENDERABLE_TYPE, renderableType,
53 EGL_NONE};
54
55 EGLint configCount;
56 if (!eglChooseConfig(display, attributes.data(), nullptr, 0, &configCount)) {
57 qCWarning(KWIN_QPA, "eglChooseConfig failed: %x", eglGetError());
58 return EGL_NO_CONFIG_KHR;
59 }
60 if (configCount == 0) {
61 qCWarning(KWIN_QPA, "eglChooseConfig did not return any configs");
62 return EGL_NO_CONFIG_KHR;
63 }
64
65 QVector<EGLConfig> configs(configCount);
66 if (!eglChooseConfig(display, attributes.data(), configs.data(), configCount, &configCount)) {
67 qCWarning(KWIN_QPA, "eglChooseConfig failed: %x", eglGetError());
68 return EGL_NO_CONFIG_KHR;
69 }
70 if (configCount != configs.size()) {
71 qCWarning(KWIN_QPA, "eglChooseConfig did not return requested configs");
72 return EGL_NO_CONFIG_KHR;
73 }
74
75 for (const EGLConfig &config : std::as_const(configs)) {
76 EGLint redConfig, greenConfig, blueConfig, alphaConfig;
77 eglGetConfigAttrib(display, config, EGL_RED_SIZE, &redConfig);
78 eglGetConfigAttrib(display, config, EGL_GREEN_SIZE, &greenConfig);
79 eglGetConfigAttrib(display, config, EGL_BLUE_SIZE, &blueConfig);
80 eglGetConfigAttrib(display, config, EGL_ALPHA_SIZE, &alphaConfig);
81
82 if ((redSize == 0 || redSize == redConfig) && (greenSize == 0 || greenSize == greenConfig) && (blueSize == 0 || blueSize == blueConfig) && (alphaSize == 0 || alphaSize == alphaConfig)) {
83 return config;
84 }
85 }
86
87 // Return first config as a fallback.
88 return configs[0];
89}

91QSurfaceFormat formatFromConfig(EGLDisplay display, EGLConfig config)
92{
93 int redSize = 0;
94 int blueSize = 0;
95 int greenSize = 0;
96 int alphaSize = 0;
97 int stencilSize = 0;
98 int depthSize = 0;
99 int sampleCount = 0;
100
101 eglGetConfigAttrib(display, config, EGL_RED_SIZE, &redSize);
102 eglGetConfigAttrib(display, config, EGL_GREEN_SIZE, &greenSize);
103 eglGetConfigAttrib(display, config, EGL_BLUE_SIZE, &blueSize);
104 eglGetConfigAttrib(display, config, EGL_ALPHA_SIZE, &alphaSize);
105 eglGetConfigAttrib(display, config, EGL_STENCIL_SIZE, &stencilSize);
106 eglGetConfigAttrib(display, config, EGL_DEPTH_SIZE, &depthSize);
107 eglGetConfigAttrib(display, config, EGL_SAMPLES, &sampleCount);
108
109 QSurfaceFormat format;
110 format.setRedBufferSize(redSize);
111 format.setGreenBufferSize(greenSize);
112 format.setBlueBufferSize(blueSize);
113 format.setAlphaBufferSize(alphaSize);
114 format.setStencilBufferSize(stencilSize);
115 format.setDepthBufferSize(depthSize);
116 format.setSamples(sampleCount);
117 format.setRenderableType(isOpenGLES() ? QSurfaceFormat::OpenGLES : QSurfaceFormat::OpenGL);
118 format.setStereo(false);
119
120 return format;
121}
