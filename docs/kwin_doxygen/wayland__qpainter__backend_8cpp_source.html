51void WaylandQPainterPrimaryLayer::remapBuffer()
52{
53 qCDebug(KWIN_WAYLAND_BACKEND) << "Remapped back buffer of surface" << m_waylandOutput->surface();
54
55 const QSize nativeSize(m_waylandOutput->geometry().size() * m_waylandOutput->scale());
56 for (const auto &slot : m_slots) {
57 slot->image = QImage(slot->buffer->address(), nativeSize.width(), nativeSize.height(), QImage::Format_RGB32);
58 }

61void WaylandQPainterPrimaryLayer::present()
62{
63 for (const auto &slot : m_slots) {
64 if (slot.get() == m_back) {
65 slot->age = 1;
66 } else if (slot->age > 0) {
67 slot->age++;
68 }
69 }
70
71 auto s = m_waylandOutput->surface();
72 s->attachBuffer(m_back->buffer);
73 s->damage(m_damageJournal.lastDamage());
74 s->setScale(std::ceil(m_waylandOutput->scale()));
75 s->commit();
76}

83WaylandQPainterBufferSlot *WaylandQPainterPrimaryLayer::acquire()
84{
85 const QSize nativeSize(m_waylandOutput->pixelSize());
86 if (m_swapchainSize != nativeSize) {
87 m_swapchainSize = nativeSize;
88 m_slots.clear();
89 }
90
91 for (const auto &slot : m_slots) {
92 if (slot->buffer->isReleased()) {
93 m_back = slot.get();
94 slot->buffer->setReleased(false);
95 return m_back;
96 }
97 }
98
99 auto buffer = m_pool->getBuffer(nativeSize, nativeSize.width() * 4, KWayland::Client::Buffer::Format::RGB32).toStrongRef();
100 if (!buffer) {
101 qCDebug(KWIN_WAYLAND_BACKEND) << "Did not get a new Buffer from Shm Pool";
102 return nullptr;
103 }
104
105 m_slots.push_back(std::make_unique<WaylandQPainterBufferSlot>(buffer));
106 m_back = m_slots.back().get();
107
108 // qCDebug(KWIN_WAYLAND_BACKEND) << "Created a new back buffer for output surface" << m_waylandOutput->surface();
109 return m_back;
110}
