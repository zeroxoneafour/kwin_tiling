60void HighlightWindowEffect::highlightWindows(const QStringList &windows)
61{
62 QVector<EffectWindow *> effectWindows;
63 effectWindows.reserve(windows.count());
64 for (const auto &window : windows) {
65 if (auto effectWindow = effects->findWindow(QUuid(window)); effectWindow) {
66 effectWindows.append(effectWindow);
67 } else if (auto effectWindow = effects->findWindow(window.toLong()); effectWindow) {
68 effectWindows.append(effectWindow);
69 }
70 }
71 highlightWindows(effectWindows);
72}

74void HighlightWindowEffect::slotWindowAdded(EffectWindow *w)
75{
76 if (!m_highlightedWindows.isEmpty()) {
77 // On X11, the tabbox may ask us to highlight itself before the windowAdded signal
78 // is emitted because override-redirect windows are shown after synthetic 50ms delay.
79 if (m_highlightedWindows.contains(w)) {
80 return;
81 }
82 // This window was demanded to be highlighted before it appeared on the screen.
83 for (const WId &id : std::as_const(m_highlightedIds)) {
84 if (w == effects->findWindow(id)) {
85 const quint64 animationId = startHighlightAnimation(w);
86 complete(animationId);
87 return;
88 }
89 }
90 if (isHighlightWindow(w)) {
91 const quint64 animationId = startGhostAnimation(w); // this window is not currently highlighted
92 complete(animationId);
93 }
94 }
95 slotPropertyNotify(w, m_atom, w); // Check initial value
96}

110void HighlightWindowEffect::slotPropertyNotify(EffectWindow *w, long a, EffectWindow *addedWindow)
111{
112 if (a != m_atom || m_atom == XCB_ATOM_NONE) {
113 return; // Not our atom
114 }
115
116 // if the window is null, the property was set on the root window - see events.cpp
117 QByteArray byteData = w ? w->readProperty(m_atom, m_atom, 32) : effects->readRootProperty(m_atom, m_atom, 32);
118 if (byteData.length() < 1) {
119 // Property was removed, clearing highlight
120 if (!addedWindow || w != addedWindow) {
121 finishHighlighting();
122 }
123 return;
124 }
125 auto *data = reinterpret_cast<uint32_t *>(byteData.data());
126
127 if (!data[0]) {
128 // Purposely clearing highlight by issuing a NULL target
129 finishHighlighting();
130 return;
131 }
132 m_monitorWindow = w;
133 bool found = false;
134 int length = byteData.length() / sizeof(data[0]);
135 // foreach ( EffectWindow* e, m_highlightedWindows )
136 // effects->setElevatedWindow( e, false );
137 m_highlightedWindows.clear();
138 m_highlightedIds.clear();
139 for (int i = 0; i < length; i++) {
140 m_highlightedIds << data[i];
141 EffectWindow *foundWin = effects->findWindow(data[i]);
142 if (!foundWin) {
143 qCDebug(KWIN_HIGHLIGHTWINDOW) << "Invalid window targetted for highlight. Requested:" << data[i];
144 continue; // might come in later.
145 }
146 m_highlightedWindows.append(foundWin);
147 // TODO: We cannot just simply elevate the window as this will elevate it over
148 // Plasma tooltips and other such windows as well
149 // effects->setElevatedWindow( foundWin, true );
150 found = true;
151 }
152 if (!found) {
153 finishHighlighting();
154 return;
155 }
156 prepareHighlighting();
157}

159void HighlightWindowEffect::prepareHighlighting()
160{
161 const EffectWindowList windows = effects->stackingOrder();
162 for (EffectWindow *window : windows) {
163 if (!isHighlightWindow(window)) {
164 continue;
165 }
166 if (isHighlighted(window)) {
167 startHighlightAnimation(window);
168 } else {
169 startGhostAnimation(window);
170 }
171 }
172}

174void HighlightWindowEffect::finishHighlighting()
175{
176 const EffectWindowList windows = effects->stackingOrder();
177 for (EffectWindow *window : windows) {
178 if (isHighlightWindow(window)) {
179 startRevertAnimation(window);
180 }
181 }
182
183 // Sanity check, ideally, this should never happen.
184 if (!m_animations.isEmpty()) {
185 for (quint64 &animationId : m_animations) {
186 cancel(animationId);
187 }
188 m_animations.clear();
189 }
190
191 m_monitorWindow = nullptr;
192 m_highlightedWindows.clear();
193}

195void HighlightWindowEffect::highlightWindows(const QVector<KWin::EffectWindow *> &windows)
196{
197 if (windows.isEmpty()) {
198 finishHighlighting();
199 return;
200 }
201
202 m_monitorWindow = nullptr;
203 m_highlightedWindows.clear();
204 m_highlightedIds.clear();
205 for (auto w : windows) {
206 m_highlightedWindows << w;
207 }
208 prepareHighlighting();
209}

211quint64 HighlightWindowEffect::startGhostAnimation(EffectWindow *window)
212{
213 quint64 &animationId = m_animations[window];
214 if (animationId) {
215 retarget(animationId, FPx2(m_ghostOpacity, m_ghostOpacity), m_fadeDuration);
216 } else {
217 const qreal startOpacity = isInitiallyHidden(window) ? 0 : 1;
218 animationId = set(window, Opacity, 0, m_fadeDuration, FPx2(m_ghostOpacity, m_ghostOpacity),
219 m_easingCurve, 0, FPx2(startOpacity, startOpacity), false, false);
220 }
221 return animationId;
222}

224quint64 HighlightWindowEffect::startHighlightAnimation(EffectWindow *window)
225{
226 quint64 &animationId = m_animations[window];
227 if (animationId) {
228 retarget(animationId, FPx2(1.0, 1.0), m_fadeDuration);
229 } else {
230 const qreal startOpacity = isInitiallyHidden(window) ? 0 : 1;
231 animationId = set(window, Opacity, 0, m_fadeDuration, FPx2(1.0, 1.0),
232 m_easingCurve, 0, FPx2(startOpacity, startOpacity), false, false);
233 }
234 return animationId;

237void HighlightWindowEffect::startRevertAnimation(EffectWindow *window)
238{
239 const quint64 animationId = m_animations.take(window);
240 if (animationId) {
241 const qreal startOpacity = isHighlighted(window) ? 1 : m_ghostOpacity;
242 const qreal endOpacity = isInitiallyHidden(window) ? 0 : 1;
243 animate(window, Opacity, 0, m_fadeDuration, FPx2(endOpacity, endOpacity),
244 m_easingCurve, 0, FPx2(startOpacity, startOpacity), false, false);
245 cancel(animationId);
246 }
