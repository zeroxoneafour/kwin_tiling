72void Deleted::copyToDeleted(Window *window)
73{
74 Q_ASSERT(!window->isDeleted());
75 Window::copyToDeleted(window);
76 m_frameMargins = window->frameMargins();
77 desk = window->desktop();
78 m_desktops = window->desktops();
79 activityList = window->activities();
80 contentsRect = QRectF(window->clientPos(), window->clientSize());
81 m_layer = window->layer();
82 m_frame = window->frameId();
83 m_type = window->windowType();
84 m_windowRole = window->windowRole();
85 m_shade = window->isShade();
86 if (WinInfo *cinfo = dynamic_cast<WinInfo *>(info)) {
87 cinfo->disable();
88 }
89 if (window->isDecorated()) {
90 window->layoutDecorationRects(decoration_left,
91 decoration_top,
92 decoration_right,
93 decoration_bottom);
94 }
95 m_wasClient = true;
96 m_minimized = window->isMinimized();
97 m_modal = window->isModal();
98 m_mainWindows = window->mainWindows();
99 for (Window *w : std::as_const(m_mainWindows)) {
100 connect(w, &Window::windowClosed, this, &Deleted::mainWindowClosed);
101 }
102 m_fullscreen = window->isFullScreen();
103 m_keepAbove = window->keepAbove();
104 m_keepBelow = window->keepBelow();
105 m_caption = window->caption();
106
107 for (auto vd : std::as_const(m_desktops)) {
108 connect(vd, &QObject::destroyed, this, [=, this] {
109 m_desktops.removeOne(vd);
110 });
111 }
112
113 m_wasPopupWindow = window->isPopupWindow();
114 m_wasOutline = window->isOutline();
115 m_wasLockScreen = window->isLockScreen();
116}

118void Deleted::unrefWindow()
119{
120 if (--delete_refcount > 0) {
121 return;
122 }
123 // needs to be delayed
124 // a) when calling from effects, otherwise it'd be rather complicated to handle the case of the
125 // window going away during a painting pass
126 // b) to prevent dangeling pointers in the stacking order, see bug #317765
127 deleteLater();
128}

188QVector<uint> Deleted::x11DesktopIds() const
189{
190 const auto desks = desktops();
191 QVector<uint> x11Ids;
192 x11Ids.reserve(desks.count());
193 std::transform(desks.constBegin(), desks.constEnd(),
194 std::back_inserter(x11Ids),
195 [](const VirtualDesktop *vd) {
196 return vd->x11DesktopNumber();
197 });
198 return x11Ids;
