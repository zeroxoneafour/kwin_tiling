28MagnifierEffect::MagnifierEffect()
29 : m_zoom(1)
30 , m_targetZoom(1)
31 , m_polling(false)
32 , m_lastPresentTime(std::chrono::milliseconds::zero())
33 , m_texture(nullptr)
34 , m_fbo(nullptr)
35{
36 initConfig<MagnifierConfig>();
37 QAction *a;
38 a = KStandardAction::zoomIn(this, &MagnifierEffect::zoomIn, this);
39 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Equal));
40 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Equal));
41
42 a = KStandardAction::zoomOut(this, &MagnifierEffect::zoomOut, this);
43 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Minus));
44 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_Minus));
45
46 a = KStandardAction::actualSize(this, &MagnifierEffect::toggle, this);
47 KGlobalAccel::self()->setDefaultShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_0));
48 KGlobalAccel::self()->setShortcut(a, QList<QKeySequence>() << (Qt::META | Qt::Key_0));
49
50 connect(effects, &EffectsHandler::mouseChanged, this, &MagnifierEffect::slotMouseChanged);
51 connect(effects, &EffectsHandler::windowDamaged, this, &MagnifierEffect::slotWindowDamaged);
52
53 reconfigure(ReconfigureAll);
54}

82void MagnifierEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
83{
84 const int time = m_lastPresentTime.count() ? (presentTime - m_lastPresentTime).count() : 0;
85
86 if (m_zoom != m_targetZoom) {
87 double diff = time / animationTime(500.0);
88 if (m_targetZoom > m_zoom) {
89 m_zoom = std::min(m_zoom * std::max(1 + diff, 1.2), m_targetZoom);
90 } else {
91 m_zoom = std::max(m_zoom * std::min(1 - diff, 0.8), m_targetZoom);
92 if (m_zoom == 1.0) {
93 // zoom ended - delete FBO and texture
94 m_fbo.reset();
95 m_texture.reset();
96 }
97 }
98 }
99
100 if (m_zoom != m_targetZoom) {
101 m_lastPresentTime = presentTime;
102 } else {
103 m_lastPresentTime = std::chrono::milliseconds::zero();
104 }
105
106 effects->prePaintScreen(data, presentTime);
107 if (m_zoom != 1.0) {
108 data.paint |= magnifierArea().adjusted(-FRAME_WIDTH, -FRAME_WIDTH, FRAME_WIDTH, FRAME_WIDTH);
109 }
110}

112void MagnifierEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
113{
114 effects->paintScreen(mask, region, data); // paint normal screen
115 if (m_zoom != 1.0) {
116 // get the right area from the current rendered screen
117 const QRect area = magnifierArea();
118 const QPointF cursor = cursorPos();
119 const auto scale = effects->renderTargetScale();
120
121 QRectF srcArea(cursor.x() - (double)area.width() / (m_zoom * 2),
122 cursor.y() - (double)area.height() / (m_zoom * 2),
123 (double)area.width() / m_zoom, (double)area.height() / m_zoom);
124 if (effects->isOpenGLCompositing()) {
125 m_fbo->blitFromFramebuffer(effects->mapToRenderTarget(srcArea).toRect());
126 // paint magnifier
127 m_texture->bind();
128 auto s = ShaderManager::instance()->pushShader(ShaderTrait::MapTexture);
129 QMatrix4x4 mvp;
130 const QSize size = effects->virtualScreenSize();
131 mvp.ortho(0, size.width() * scale, size.height() * scale, 0, 0, 65535);
132 mvp.translate(area.x() * scale, area.y() * scale);
133 s->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
134 m_texture->render(area.size(), scale);
135 ShaderManager::instance()->popShader();
136 m_texture->unbind();
137 QVector<float> verts;
138 GLVertexBuffer *vbo = GLVertexBuffer::streamingBuffer();
139 vbo->reset();
140 vbo->setColor(QColor(0, 0, 0));
141
142 QRectF areaF = scaledRect(area, scale);
143 const QRectF frame = scaledRect(area.adjusted(-FRAME_WIDTH, -FRAME_WIDTH, FRAME_WIDTH, FRAME_WIDTH), scale);
144 // top frame
145 verts << frame.right() << frame.top();
146 verts << frame.left() << frame.top();
147 verts << frame.left() << areaF.top();
148 verts << frame.left() << areaF.top();
149 verts << frame.right() << areaF.top();
150 verts << frame.right() << frame.top();
151 // left frame
152 verts << areaF.left() << frame.top();
153 verts << frame.left() << frame.top();
154 verts << frame.left() << frame.bottom();
155 verts << frame.left() << frame.bottom();
156 verts << areaF.left() << frame.bottom();
157 verts << areaF.left() << frame.top();
158 // right frame
159 verts << frame.right() << frame.top();
160 verts << areaF.right() << frame.top();
161 verts << areaF.right() << frame.bottom();
162 verts << areaF.right() << frame.bottom();
163 verts << frame.right() << frame.bottom();
164 verts << frame.right() << frame.top();
165 // bottom frame
166 verts << frame.right() << areaF.bottom();
167 verts << frame.left() << areaF.bottom();
168 verts << frame.left() << frame.bottom();
169 verts << frame.left() << frame.bottom();
170 verts << frame.right() << frame.bottom();
171 verts << frame.right() << areaF.bottom();
172 vbo->setData(verts.size() / 2, 2, verts.constData(), nullptr);
173
174 ShaderBinder binder(ShaderTrait::UniformColor);
175 binder.shader()->setUniform(GLShader::ModelViewProjectionMatrix, data.projectionMatrix());
176 vbo->render(GL_TRIANGLES);
177 }
178 }
179}

196void MagnifierEffect::zoomIn()
197{
198 m_targetZoom *= 1.2;
199 if (!m_polling) {
200 m_polling = true;
201 effects->startMousePolling();
202 }
203 if (effects->isOpenGLCompositing() && !m_texture) {
204 effects->makeOpenGLContextCurrent();
205 m_texture = std::make_unique<GLTexture>(GL_RGBA8, m_magnifierSize.width(), m_magnifierSize.height());
206 m_texture->setYInverted(false);
207 m_fbo = std::make_unique<GLFramebuffer>(m_texture.get());
208 }
209 effects->addRepaint(magnifierArea().adjusted(-FRAME_WIDTH, -FRAME_WIDTH, FRAME_WIDTH, FRAME_WIDTH));
210}

212void MagnifierEffect::zoomOut()
213{
214 m_targetZoom /= 1.2;
215 if (m_targetZoom <= 1) {
216 m_targetZoom = 1;
217 if (m_polling) {
218 m_polling = false;
219 effects->stopMousePolling();
220 }
221 if (m_zoom == m_targetZoom) {
222 effects->makeOpenGLContextCurrent();
223 m_fbo.reset();
224 m_texture.reset();
225 }
226 }
227 effects->addRepaint(magnifierArea().adjusted(-FRAME_WIDTH, -FRAME_WIDTH, FRAME_WIDTH, FRAME_WIDTH));
228}

230void MagnifierEffect::toggle()
231{
232 if (m_zoom == 1.0) {
233 if (m_targetZoom == 1.0) {
234 m_targetZoom = 2;
235 }
236 if (!m_polling) {
237 m_polling = true;
238 effects->startMousePolling();
239 }
240 if (effects->isOpenGLCompositing() && !m_texture) {
241 effects->makeOpenGLContextCurrent();
242 m_texture = std::make_unique<GLTexture>(GL_RGBA8, m_magnifierSize.width(), m_magnifierSize.height());
243 m_texture->setYInverted(false);
244 m_fbo = std::make_unique<GLFramebuffer>(m_texture.get());
245 }
246 } else {
247 m_targetZoom = 1;
248 if (m_polling) {
249 m_polling = false;
250 effects->stopMousePolling();
251 }
252 }
253 effects->addRepaint(magnifierArea().adjusted(-FRAME_WIDTH, -FRAME_WIDTH, FRAME_WIDTH, FRAME_WIDTH));
254}
