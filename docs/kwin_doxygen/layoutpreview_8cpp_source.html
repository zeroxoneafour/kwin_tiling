28LayoutPreview::LayoutPreview(const QString &path, bool showDesktopThumbnail, QObject *parent)
29 : QObject(parent)
30 , m_item(nullptr)
31{
32 QQmlEngine *engine = new QQmlEngine(this);
33 QQmlComponent *component = new QQmlComponent(engine, this);
34 qmlRegisterType<WindowThumbnailItem>("org.kde.kwin", 3, 0, "WindowThumbnail");
35 qmlRegisterType<SwitcherItem>("org.kde.kwin", 3, 0, "TabBoxSwitcher");
36 qmlRegisterAnonymousType<QAbstractItemModel>("org.kde.kwin", 3);
37 component->loadUrl(QUrl::fromLocalFile(path));
38 if (component->isError()) {
39 qDebug() << component->errorString();
40 }
41 QObject *item = component->create();
42 auto findSwitcher = [item]() -> SwitcherItem * {
43 if (!item) {
44 return nullptr;
45 }
46 if (SwitcherItem *i = qobject_cast<SwitcherItem *>(item)) {
47 return i;
48 } else if (QQuickWindow *w = qobject_cast<QQuickWindow *>(item)) {
49 return w->contentItem()->findChild<SwitcherItem *>();
50 }
51 return item->findChild<SwitcherItem *>();
52 };
53 if (SwitcherItem *switcher = findSwitcher()) {
54 m_item = switcher;
55 static_cast<ExampleClientModel *>(switcher->model())->showDesktopThumbnail(showDesktopThumbnail);
56 switcher->setVisible(true);
57 }
58 auto findWindow = [item]() -> QQuickWindow * {
59 if (!item) {
60 return nullptr;
61 }
62 if (QQuickWindow *w = qobject_cast<QQuickWindow *>(item)) {
63 return w;
64 }
65 return item->findChild<QQuickWindow *>();
66 };
67 if (QQuickWindow *w = findWindow()) {
68 w->setKeyboardGrabEnabled(true);
69 w->setMouseGrabEnabled(true);
70 w->installEventFilter(this);
71 }
72}

78bool LayoutPreview::eventFilter(QObject *object, QEvent *event)
79{
80 if (event->type() == QEvent::KeyPress) {
81 QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
82 if (keyEvent->key() == Qt::Key_Escape || keyEvent->key() == Qt::Key_Return || keyEvent->key() == Qt::Key_Enter || keyEvent->key() == Qt::Key_Space) {
83 object->deleteLater();
84 deleteLater();
85 }
86 if (m_item && keyEvent->key() == Qt::Key_Tab) {
87 m_item->incrementIndex();
88 }
89 if (m_item && keyEvent->key() == Qt::Key_Backtab) {
90 m_item->decrementIndex();
91 }
92 } else if (event->type() == QEvent::MouseButtonPress) {
93 if (QWindow *w = qobject_cast<QWindow *>(object)) {
94 if (!w->geometry().contains(static_cast<QMouseEvent *>(event)->globalPos())) {
95 object->deleteLater();
96 deleteLater();
97 }
98 }
99 }
100 return QObject::eventFilter(object, event);
101}

129void ExampleClientModel::showDesktopThumbnail(bool showDesktop)
130{
131 const ThumbnailInfo desktopThumbnail = ThumbnailInfo{WindowThumbnailItem::Desktop, i18n("Show Desktop"), QStringLiteral("desktop")};
132 const int desktopIndex = m_thumbnails.indexOf(desktopThumbnail);
133 if (showDesktop == (desktopIndex >= 0)) {
134 return;
135 }
136
137 Q_EMIT beginResetModel();
138 if (showDesktop) {
139 m_thumbnails << desktopThumbnail;
140 } else {
141 m_thumbnails.removeAt(desktopIndex);
142 }
143 Q_EMIT endResetModel();
144}

146QVariant ExampleClientModel::data(const QModelIndex &index, int role) const
147{
148 if (!index.isValid() || index.row() >= rowCount()) {
149 return QVariant();
150 }
151
152 const ThumbnailInfo &item = m_thumbnails.at(index.row());
153
154 switch (role) {
155 case Qt::DisplayRole:
156 case CaptionRole:
157 return item.caption;
158 case MinimizedRole:
159 return false;
160 case DesktopNameRole:
161 return i18nc("An example Desktop Name", "Desktop 1");
162 case IconRole:
163 return item.icon;
164 case WindowIdRole:
165 return item.wId;
166 }
167 return QVariant();
168}
