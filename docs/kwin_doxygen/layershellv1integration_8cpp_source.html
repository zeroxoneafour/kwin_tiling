38void LayerShellV1Integration::createWindow(LayerSurfaceV1Interface *shellSurface)
39{
40 Output *output = shellSurface->output() ? shellSurface->output()->handle() : workspace()->activeOutput();
41 if (!output) {
42 qCWarning(KWIN_CORE) << "Could not find any suitable output for a layer surface";
43 shellSurface->sendClosed();
44 return;
45 }
46
47 Q_EMIT windowCreated(new LayerShellV1Window(shellSurface, output, this));
48}

56void LayerShellV1Integration::destroyWindow(LayerSurfaceV1Interface *shellSurface)
57{
58 const QList<Window *> windows = waylandServer()->windows();
59 for (Window *window : windows) {
60 LayerShellV1Window *layerShellWindow = qobject_cast<LayerShellV1Window *>(window);
61 if (layerShellWindow && layerShellWindow->shellSurface() == shellSurface) {
62 layerShellWindow->destroyWindow();
63 break;
64 }
65 }

68static void adjustWorkArea(const LayerSurfaceV1Interface *shellSurface, QRect *workArea)
69{
70 switch (shellSurface->exclusiveEdge()) {
71 case Qt::LeftEdge:
72 workArea->adjust(shellSurface->leftMargin() + shellSurface->exclusiveZone(), 0, 0, 0);
73 break;
74 case Qt::RightEdge:
75 workArea->adjust(0, 0, -shellSurface->rightMargin() - shellSurface->exclusiveZone(), 0);
76 break;
77 case Qt::TopEdge:
78 workArea->adjust(0, shellSurface->topMargin() + shellSurface->exclusiveZone(), 0, 0);
79 break;
80 case Qt::BottomEdge:
81 workArea->adjust(0, 0, 0, -shellSurface->bottomMargin() - shellSurface->exclusiveZone());
82 break;
83 }
84}

86static void rearrangeLayer(const QList<LayerShellV1Window *> &windows, QRect *workArea,
87 LayerSurfaceV1Interface::Layer layer, bool exclusive)
88{
89 for (LayerShellV1Window *window : windows) {
90 LayerSurfaceV1Interface *shellSurface = window->shellSurface();
91
92 if (shellSurface->layer() != layer) {
93 continue;
94 }
95 if (exclusive != (shellSurface->exclusiveZone() > 0)) {
96 continue;
97 }
98
99 QRect bounds;
100 if (shellSurface->exclusiveZone() == -1) {
101 bounds = window->desiredOutput()->geometry();
102 } else {
103 bounds = *workArea;
104 }
105
106 QRect geometry(QPoint(0, 0), shellSurface->desiredSize());
107
108 if ((shellSurface->anchor() & AnchorHorizontal) && geometry.width() == 0) {
109 geometry.setLeft(bounds.left());
110 geometry.setWidth(bounds.width());
111 } else if (shellSurface->anchor() & Qt::LeftEdge) {
112 geometry.moveLeft(bounds.left());
113 } else if (shellSurface->anchor() & Qt::RightEdge) {
114 geometry.moveRight(bounds.right());
115 } else {
116 geometry.moveLeft(bounds.left() + (bounds.width() - geometry.width()) / 2);
117 }
118
119 if ((shellSurface->anchor() & AnchorVertical) && geometry.height() == 0) {
120 geometry.setTop(bounds.top());
121 geometry.setHeight(bounds.height());
122 } else if (shellSurface->anchor() & Qt::TopEdge) {
123 geometry.moveTop(bounds.top());
124 } else if (shellSurface->anchor() & Qt::BottomEdge) {
125 geometry.moveBottom(bounds.bottom());
126 } else {
127 geometry.moveTop(bounds.top() + (bounds.height() - geometry.height()) / 2);
128 }
129
130 if ((shellSurface->anchor() & AnchorHorizontal) == AnchorHorizontal) {
131 geometry.adjust(shellSurface->leftMargin(), 0, -shellSurface->rightMargin(), 0);
132 } else if (shellSurface->anchor() & Qt::LeftEdge) {
133 geometry.translate(shellSurface->leftMargin(), 0);
134 } else if (shellSurface->anchor() & Qt::RightEdge) {
135 geometry.translate(-shellSurface->rightMargin(), 0);
136 }
137
138 if ((shellSurface->anchor() & AnchorVertical) == AnchorVertical) {
139 geometry.adjust(0, shellSurface->topMargin(), 0, -shellSurface->bottomMargin());
140 } else if (shellSurface->anchor() & Qt::TopEdge) {
141 geometry.translate(0, shellSurface->topMargin());
142 } else if (shellSurface->anchor() & Qt::BottomEdge) {
143 geometry.translate(0, -shellSurface->bottomMargin());
144 }
145
146 // Move the window's bottom if its virtual keyboard is overlapping it
147 if (shellSurface->exclusiveZone() >= 0 && !window->virtualKeyboardGeometry().isEmpty() && geometry.bottom() > window->virtualKeyboardGeometry().top()) {
148 geometry.setBottom(window->virtualKeyboardGeometry().top());
149 }
150
151 if (geometry.isValid()) {
152 window->moveResize(geometry);
153 } else {
154 qCWarning(KWIN_CORE) << "Closing a layer shell window due to invalid geometry";
155 window->closeWindow();
156 continue;
157 }
158
159 if (exclusive && shellSurface->exclusiveZone() > 0) {
160 adjustWorkArea(shellSurface, workArea);
161 }
162 }
163}

165static QList<LayerShellV1Window *> windowsForOutput(Output *output)
166{
167 QList<LayerShellV1Window *> result;
168 const QList<Window *> windows = waylandServer()->windows();
169 for (Window *window : windows) {
170 LayerShellV1Window *layerShellWindow = qobject_cast<LayerShellV1Window *>(window);
171 if (!layerShellWindow || layerShellWindow->desiredOutput() != output) {
172 continue;
173 }
174 if (layerShellWindow->shellSurface()->isCommitted()) {
175 result.append(layerShellWindow);
176 }
177 }
178 return result;
179}

181static void rearrangeOutput(Output *output)
182{
183 const QList<LayerShellV1Window *> windows = windowsForOutput(output);
184 if (!windows.isEmpty()) {
185 QRect workArea = output->geometry();
186
187 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::OverlayLayer, true);
188 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::TopLayer, true);
189 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::BottomLayer, true);
190 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::BackgroundLayer, true);
191
192 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::OverlayLayer, false);
193 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::TopLayer, false);
194 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::BottomLayer, false);
195 rearrangeLayer(windows, &workArea, LayerSurfaceV1Interface::BackgroundLayer, false);
196 }
197}

202
203 const QList<Output *> outputs = workspace()->outputs();
204 for (Output *output : outputs) {
205 rearrangeOutput(output);
206 }
207
208 if (workspace()) {
