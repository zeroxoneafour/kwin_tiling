36void KWin::BlendChanges::start(int delay)
37{
38 int animationDuration = animationTime(400);
39
40 if (!supported() || m_state != Off) {
41 return;
42 }
43 if (effects->hasActiveFullScreenEffect()) {
44 return;
45 }
46
47 const EffectWindowList allWindows = effects->stackingOrder();
48 for (auto window : allWindows) {
49 if (!window->isFullScreen()) {
50 redirect(window);
51 }
52 }
53
54 QTimer::singleShot(delay, this, [this, animationDuration]() {
55 m_timeline.setDuration(std::chrono::milliseconds(animationDuration));
56 effects->addRepaintFull();
57 m_state = Blending;
58 });
59
60 m_state = ShowingCache;
61}

68void BlendChanges::postPaintScreen()
