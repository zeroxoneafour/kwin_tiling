24DrmPipeline::Error DrmPipeline::presentLegacy()
25{
26 if (!m_pending.crtc->current()) {
27 Error err = legacyModeset();
28 if (err != Error::None) {
29 return err;
30 }
31 }
32 const auto buffer = m_pending.layer->currentBuffer();
33 uint32_t flags = DRM_MODE_PAGE_FLIP_EVENT;
34 if (m_pending.syncMode == RenderLoopPrivate::SyncMode::Async || m_pending.syncMode == RenderLoopPrivate::SyncMode::AdaptiveAsync) {
35 flags |= DRM_MODE_PAGE_FLIP_ASYNC;
36 }
37 if (drmModePageFlip(gpu()->fd(), m_pending.crtc->id(), buffer->framebufferId(), flags, gpu()) != 0) {
38 qCWarning(KWIN_DRM) << "Page flip failed:" << strerror(errno);
39 return errnoToError();
40 }
41 m_pageflipPending = true;
42 m_pending.crtc->setNext(buffer);
43 return Error::None;
44}

46DrmPipeline::Error DrmPipeline::legacyModeset()
47{
48 uint32_t connId = m_connector->id();
49 if (!m_pending.layer->checkTestBuffer()) {
50 return Error::TestBufferFailed;
51 }
52 const auto buffer = m_pending.layer->currentBuffer();
53 if (drmModeSetCrtc(gpu()->fd(), m_pending.crtc->id(), buffer->framebufferId(), 0, 0, &connId, 1, m_pending.mode->nativeMode()) != 0) {
54 qCWarning(KWIN_DRM) << "Modeset failed!" << strerror(errno);
55 return errnoToError();
56 }
57 // make sure the buffer gets kept alive, or the modeset gets reverted by the kernel
58 if (m_pending.crtc->current()) {
59 m_pending.crtc->setNext(buffer);
60 } else {
61 m_pending.crtc->setCurrent(buffer);
62 }
63 return Error::None;
64}

66DrmPipeline::Error DrmPipeline::commitPipelinesLegacy(const QVector<DrmPipeline *> &pipelines, CommitMode mode)
67{
68 Error err = Error::None;
69 for (const auto &pipeline : pipelines) {
70 err = pipeline->applyPendingChangesLegacy();
71 if (err != Error::None) {
72 break;
73 }
74 }
75 if (err != Error::None) {
76 // at least try to revert the config
77 for (const auto &pipeline : pipelines) {
78 pipeline->revertPendingChanges();
79 pipeline->applyPendingChangesLegacy();
80 }
81 } else {
82 for (const auto &pipeline : pipelines) {
83 pipeline->applyPendingChanges();
84 pipeline->m_current = pipeline->m_pending;
85 if (mode == CommitMode::CommitModeset && pipeline->activePending()) {
86 pipeline->pageFlipped(std::chrono::steady_clock::now().time_since_epoch());
87 }
88 }
89 }
90 return err;
91}

93DrmPipeline::Error DrmPipeline::applyPendingChangesLegacy()
94{
95 if (!m_pending.active && m_pending.crtc) {
96 drmModeSetCursor(gpu()->fd(), m_pending.crtc->id(), 0, 0, 0);
97 }
98 if (activePending()) {
99 auto vrr = m_pending.crtc->getProp(DrmCrtc::PropertyIndex::VrrEnabled);
100 if (vrr && !vrr->setPropertyLegacy(m_pending.syncMode == RenderLoopPrivate::SyncMode::Adaptive || m_pending.syncMode == RenderLoopPrivate::SyncMode::AdaptiveAsync)) {
101 qCWarning(KWIN_DRM) << "Setting vrr failed!" << strerror(errno);
102 return errnoToError();
103 }
104 if (const auto &rgbRange = m_connector->getProp(DrmConnector::PropertyIndex::Broadcast_RGB)) {
105 rgbRange->setEnumLegacy(m_pending.rgbRange);
106 }
107 if (const auto overscan = m_connector->getProp(DrmConnector::PropertyIndex::Overscan)) {
108 overscan->setPropertyLegacy(m_pending.overscan);
109 } else if (const auto underscan = m_connector->getProp(DrmConnector::PropertyIndex::Underscan)) {
110 const uint32_t hborder = calculateUnderscan();
111 underscan->setEnumLegacy(m_pending.overscan != 0 ? DrmConnector::UnderscanOptions::On : DrmConnector::UnderscanOptions::Off);
112 m_connector->getProp(DrmConnector::PropertyIndex::Underscan_vborder)->setPropertyLegacy(m_pending.overscan);
113 m_connector->getProp(DrmConnector::PropertyIndex::Underscan_hborder)->setPropertyLegacy(hborder);
114 }
115 if (m_pending.crtc != m_current.crtc || m_pending.mode != m_current.mode) {
116 Error err = legacyModeset();
117 if (err != Error::None) {
118 return err;
119 }
120 }
121 if (m_pending.gamma && drmModeCrtcSetGamma(gpu()->fd(), m_pending.crtc->id(), m_pending.gamma->lut().size(), m_pending.gamma->lut().red(), m_pending.gamma->lut().green(), m_pending.gamma->lut().blue()) != 0) {
122 qCWarning(KWIN_DRM) << "Setting gamma failed!" << strerror(errno);
123 return errnoToError();
124 }
125 if (const auto contentType = m_connector->getProp(DrmConnector::PropertyIndex::ContentType)) {
126 contentType->setEnumLegacy(m_pending.contentType);
127 }
128 setCursorLegacy();
129 moveCursorLegacy();
130 }
131 if (!m_connector->getProp(DrmConnector::PropertyIndex::Dpms)->setPropertyLegacy(activePending() ? DRM_MODE_DPMS_ON : DRM_MODE_DPMS_OFF)) {
132 qCWarning(KWIN_DRM) << "Setting legacy dpms failed!" << strerror(errno);
133 return errnoToError();
134 }
135 return Error::None;
136}

138bool DrmPipeline::setCursorLegacy()
139{
140 const auto bo = cursorLayer()->currentBuffer();
141 const uint32_t handle = bo && bo->buffer() && cursorLayer()->isVisible() ? bo->buffer()->handles()[0] : 0;
142
143 struct drm_mode_cursor2 arg = {
144 .flags = DRM_MODE_CURSOR_BO | DRM_MODE_CURSOR_MOVE,
145 .crtc_id = m_pending.crtc->id(),
146 .x = m_pending.cursorLayer->position().x(),
147 .y = m_pending.cursorLayer->position().y(),
148 .width = (uint32_t)gpu()->cursorSize().width(),
149 .height = (uint32_t)gpu()->cursorSize().height(),
150 .handle = handle,
151 .hot_x = m_pending.cursorHotspot.x(),
152 .hot_y = m_pending.cursorHotspot.y(),
153 };
154 return drmIoctl(gpu()->fd(), DRM_IOCTL_MODE_CURSOR2, &arg) == 0;
155}
