30static KConfig *sessionConfig(QString id, QString key)
31{
32 static KConfig *config = nullptr;
33 static QString lastId;
34 static QString lastKey;
35 static QString pattern = QString(QLatin1String("session/%1_%2_%3")).arg(qApp->applicationName());
36 if (id != lastId || key != lastKey) {
37 delete config;
38 config = nullptr;
39 }
40 lastId = id;
41 lastKey = key;
42 if (!config) {
43 config = new KConfig(pattern.arg(id, key), KConfig::SimpleConfig);
44 }
45 return config;
46}

48static const char *const window_type_names[] = {
49 "Unknown", "Normal", "Desktop", "Dock", "Toolbar", "Menu", "Dialog",
50 "Override", "TopMenu", "Utility", "Splash"};
51// change also the two functions below when adding new entries

53static const char *windowTypeToTxt(NET::WindowType type)
54{
55 if (type >= NET::Unknown && type <= NET::Splash) {
56 return window_type_names[type + 1]; // +1 (unknown==-1)
57 }
58 if (type == -2) { // undefined (not really part of NET::WindowType)
59 return "Undefined";
60 }
61 qFatal("Unknown Window Type");
62 return nullptr;
63}

65static NET::WindowType txtToWindowType(const char *txt)
66{
67 for (int i = NET::Unknown;
68 i <= NET::Splash;
69 ++i) {
70 if (qstrcmp(txt, window_type_names[i + 1]) == 0) { // +1
71 return static_cast<NET::WindowType>(i);
72 }
73 }
74 return static_cast<NET::WindowType>(-2); // undefined
75}

82void SessionManager::storeSession(const QString &sessionName, SMSavePhase phase)
83{
84 qCDebug(KWIN_CORE) << "storing session" << sessionName << "in phase" << phase;
85 KConfig *config = sessionConfig(sessionName, QString());
86
87 KConfigGroup cg(config, "Session");
88 int count = 0;
89 int active_client = -1;
90
91 const QList<X11Window *> x11Clients = workspace()->clientList();
92 for (auto it = x11Clients.begin(); it != x11Clients.end(); ++it) {
93 X11Window *c = (*it);
94 if (c->windowType() > NET::Splash) {
95 // window types outside this are not tooltips/menus/OSDs
96 // typically these will be unmanaged and not in this list anyway, but that is not enforced
97 continue;
98 }
99 QByteArray sessionId = c->sessionId();
100 QString wmCommand = c->wmCommand();
101 if (sessionId.isEmpty()) {
102 // remember also applications that are not XSMP capable
103 // and use the obsolete WM_COMMAND / WM_SAVE_YOURSELF
104 if (wmCommand.isEmpty()) {
105 continue;
106 }
107 }
108 count++;
109 if (c->isActive()) {
110 active_client = count;
111 }
112 if (phase == SMSavePhase2 || phase == SMSavePhase2Full) {
113 storeClient(cg, count, c);
114 }
115 }
116 if (phase == SMSavePhase0) {
117 // it would be much simpler to save these values to the config file,
118 // but both Qt and KDE treat phase1 and phase2 separately,
119 // which results in different sessionkey and different config file :(
120 m_sessionActiveClient = active_client;
121 m_sessionDesktop = VirtualDesktopManager::self()->current();
122 } else if (phase == SMSavePhase2) {
123 cg.writeEntry("count", count);
124 cg.writeEntry("active", m_sessionActiveClient);
125 cg.writeEntry("desktop", m_sessionDesktop);
126 } else { // SMSavePhase2Full
127 cg.writeEntry("count", count);
128 cg.writeEntry("active", m_sessionActiveClient);
129 cg.writeEntry("desktop", VirtualDesktopManager::self()->current());
130 }
131 config->sync(); // it previously did some "revert to defaults" stuff for phase1 I think
132}

134void SessionManager::storeClient(KConfigGroup &cg, int num, X11Window *c)
135{
136 c->setSessionActivityOverride(false); // make sure we get the real values
137 QString n = QString::number(num);
138 cg.writeEntry(QLatin1String("sessionId") + n, c->sessionId().constData());
139 cg.writeEntry(QLatin1String("windowRole") + n, c->windowRole());
140 cg.writeEntry(QLatin1String("wmCommand") + n, c->wmCommand());
141 cg.writeEntry(QLatin1String("resourceName") + n, c->resourceName());
142 cg.writeEntry(QLatin1String("resourceClass") + n, c->resourceClass());
143 cg.writeEntry(QLatin1String("geometry") + n, QRectF(c->calculateGravitation(true), c->clientSize()).toRect()); // FRAME
144 cg.writeEntry(QLatin1String("restore") + n, c->geometryRestore());
145 cg.writeEntry(QLatin1String("fsrestore") + n, c->fullscreenGeometryRestore());
146 cg.writeEntry(QLatin1String("maximize") + n, (int)c->maximizeMode());
147 cg.writeEntry(QLatin1String("fullscreen") + n, (int)c->fullScreenMode());
148 cg.writeEntry(QLatin1String("desktop") + n, c->desktop());
149 // the config entry is called "iconified" for back. comp. reasons
150 // (kconf_update script for updating session files would be too complicated)
151 cg.writeEntry(QLatin1String("iconified") + n, c->isMinimized());
152 cg.writeEntry(QLatin1String("opacity") + n, c->opacity());
153 // the config entry is called "sticky" for back. comp. reasons
154 cg.writeEntry(QLatin1String("sticky") + n, c->isOnAllDesktops());
155 cg.writeEntry(QLatin1String("shaded") + n, c->isShade());
156 // the config entry is called "staysOnTop" for back. comp. reasons
157 cg.writeEntry(QLatin1String("staysOnTop") + n, c->keepAbove());
158 cg.writeEntry(QLatin1String("keepBelow") + n, c->keepBelow());
159 cg.writeEntry(QLatin1String("skipTaskbar") + n, c->originalSkipTaskbar());
160 cg.writeEntry(QLatin1String("skipPager") + n, c->skipPager());
161 cg.writeEntry(QLatin1String("skipSwitcher") + n, c->skipSwitcher());
162 // not really just set by user, but name kept for back. comp. reasons
163 cg.writeEntry(QLatin1String("userNoBorder") + n, c->userNoBorder());
164 cg.writeEntry(QLatin1String("windowType") + n, windowTypeToTxt(c->windowType()));
165 cg.writeEntry(QLatin1String("shortcut") + n, c->shortcut().toString());
166 cg.writeEntry(QLatin1String("stackingOrder") + n, workspace()->unconstrainedStackingOrder().indexOf(c));
167 cg.writeEntry(QLatin1String("activities") + n, c->activities());
168}

221void SessionManager::addSessionInfo(KConfigGroup &cg)
222{
223 workspace()->setInitialDesktop(cg.readEntry("desktop", 1));
224 int count = cg.readEntry("count", 0);
225 int active_client = cg.readEntry("active", 0);
226 for (int i = 1; i <= count; i++) {
227 QString n = QString::number(i);
228 SessionInfo *info = new SessionInfo;
229 session.append(info);
230 info->sessionId = cg.readEntry(QLatin1String("sessionId") + n, QString()).toLatin1();
231 info->windowRole = cg.readEntry(QLatin1String("windowRole") + n, QString());
232 info->wmCommand = cg.readEntry(QLatin1String("wmCommand") + n, QString()).toLatin1();
233 info->resourceName = cg.readEntry(QLatin1String("resourceName") + n, QString());
234 info->resourceClass = cg.readEntry(QLatin1String("resourceClass") + n, QString()).toLower();
235 info->geometry = cg.readEntry(QLatin1String("geometry") + n, QRect());
236 info->restore = cg.readEntry(QLatin1String("restore") + n, QRect());
237 info->fsrestore = cg.readEntry(QLatin1String("fsrestore") + n, QRect());
238 info->maximized = cg.readEntry(QLatin1String("maximize") + n, 0);
239 info->fullscreen = cg.readEntry(QLatin1String("fullscreen") + n, 0);
240 info->desktop = cg.readEntry(QLatin1String("desktop") + n, 0);
241 info->minimized = cg.readEntry(QLatin1String("iconified") + n, false);
242 info->opacity = cg.readEntry(QLatin1String("opacity") + n, 1.0);
243 info->onAllDesktops = cg.readEntry(QLatin1String("sticky") + n, false);
244 info->shaded = cg.readEntry(QLatin1String("shaded") + n, false);
245 info->keepAbove = cg.readEntry(QLatin1String("staysOnTop") + n, false);
246 info->keepBelow = cg.readEntry(QLatin1String("keepBelow") + n, false);
247 info->skipTaskbar = cg.readEntry(QLatin1String("skipTaskbar") + n, false);
248 info->skipPager = cg.readEntry(QLatin1String("skipPager") + n, false);
249 info->skipSwitcher = cg.readEntry(QLatin1String("skipSwitcher") + n, false);
250 info->noBorder = cg.readEntry(QLatin1String("userNoBorder") + n, false);
251 info->windowType = txtToWindowType(cg.readEntry(QLatin1String("windowType") + n, QString()).toLatin1().constData());
252 info->shortcut = cg.readEntry(QLatin1String("shortcut") + n, QString());
253 info->active = (active_client == i);
254 info->stackingOrder = cg.readEntry(QLatin1String("stackingOrder") + n, -1);
255 info->activities = cg.readEntry(QLatin1String("activities") + n, QStringList());

283SessionInfo *SessionManager::takeSessionInfo(X11Window *c)
284{
285 SessionInfo *realInfo = nullptr;
286 QByteArray sessionId = c->sessionId();
287 QString windowRole = c->windowRole();
288 QString wmCommand = c->wmCommand();
289 QString resourceName = c->resourceName();
290 QString resourceClass = c->resourceClass();
291
292 // First search ``session''
293 if (!sessionId.isEmpty()) {
294 // look for a real session managed client (algorithm suggested by ICCCM)
295 for (SessionInfo *info : std::as_const(session)) {
296 if (realInfo) {
297 break;
298 }
299 if (info->sessionId == sessionId && sessionInfoWindowTypeMatch(c, info)) {
300 if (!windowRole.isEmpty()) {
301 if (info->windowRole == windowRole) {
302 realInfo = info;
303 session.removeAll(info);
304 }
305 } else {
306 if (info->windowRole.isEmpty()
307 && info->resourceName == resourceName
308 && info->resourceClass == resourceClass) {
309 realInfo = info;
310 session.removeAll(info);
311 }
312 }
313 }
314 }
315 } else {
316 // look for a sessioninfo with matching features.
317 for (SessionInfo *info : std::as_const(session)) {
318 if (realInfo) {
319 break;
320 }
321 if (info->resourceName == resourceName
322 && info->resourceClass == resourceClass
323 && sessionInfoWindowTypeMatch(c, info)) {
324 if (wmCommand.isEmpty() || info->wmCommand == wmCommand) {
325 realInfo = info;
326 session.removeAll(info);
327 }
328 }
329 }
330 }
331 return realInfo;
332}

365// TODO should we rethink this now that we have dedicated start end end save methods?
366void SessionManager::setState(SessionState state)
367{
368 if (state == m_sessionState) {
369 return;
370 }
371 // If we're starting to save a session
372 if (state == SessionState::Saving) {
373 workspace()->rulebook()->setUpdatesDisabled(true);
374 }
375 // If we're ending a save session due to either completion or cancellation
376 if (m_sessionState == SessionState::Saving) {
377 workspace()->rulebook()->setUpdatesDisabled(false);
378 Workspace::self()->forEachClient([](X11Window *client) {
379 client->setSessionActivityOverride(false);
380 });
381 }
382 m_sessionState = state;
383 Q_EMIT stateChanged();
384}
