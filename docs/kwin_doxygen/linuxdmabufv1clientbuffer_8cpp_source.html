33void LinuxDmaBufV1ClientBufferIntegrationPrivate::zwp_linux_dmabuf_v1_bind_resource(Resource *resource)
34{
35 if (resource->version() < ZWP_LINUX_DMABUF_V1_GET_DEFAULT_FEEDBACK_SINCE_VERSION) {
36 for (auto it = supportedModifiers.constBegin(); it != supportedModifiers.constEnd(); ++it) {
37 const uint32_t &format = it.key();
38 const auto &modifiers = it.value();
39 for (const uint64_t &modifier : std::as_const(modifiers)) {
40 if (resource->version() >= ZWP_LINUX_DMABUF_V1_MODIFIER_SINCE_VERSION) {
41 const uint32_t modifier_lo = modifier & 0xffffffff;
42 const uint32_t modifier_hi = modifier >> 32;
43 send_modifier(resource->handle, format, modifier_hi, modifier_lo);
44 } else if (modifier == DRM_FORMAT_MOD_LINEAR || modifier == DRM_FORMAT_MOD_INVALID) {
45 send_format(resource->handle, format);
46 }
47 }
48 }
49 }

102void LinuxDmaBufParamsV1::zwp_linux_buffer_params_v1_add(Resource *resource,
103 int32_t fd,
104 uint32_t plane_idx,
105 uint32_t offset,
106 uint32_t stride,
107 uint32_t modifier_hi,
108 uint32_t modifier_lo)
109{
110 if (Q_UNLIKELY(m_isUsed)) {
111 wl_resource_post_error(resource->handle, error_already_used, "the params object has already been used to create a wl_buffer");
112 close(fd);
113 return;
114 }
115
116 if (Q_UNLIKELY(plane_idx >= 4)) {
117 wl_resource_post_error(resource->handle, error_plane_idx, "plane index %d is out of bounds", plane_idx);
118 close(fd);
119 return;
120 }
121
122 if (Q_UNLIKELY(m_attrs.fd[plane_idx].isValid())) {
123 wl_resource_post_error(resource->handle, error_plane_set, "the plane index %d was already set", plane_idx);
124 close(fd);
125 return;
126 }
127 m_attrs.fd[plane_idx] = KWin::FileDescriptor{fd};
128 m_attrs.offset[plane_idx] = offset;
129 m_attrs.pitch[plane_idx] = stride;
130 m_attrs.modifier = (quint64(modifier_hi) << 32) | modifier_lo;
131 m_attrs.planeCount++;
132}

134void LinuxDmaBufParamsV1::zwp_linux_buffer_params_v1_create(Resource *resource, int32_t width, int32_t height, uint32_t format, uint32_t flags)
135{
136 if (Q_UNLIKELY(m_isUsed)) {
137 wl_resource_post_error(resource->handle, error_already_used, "the params object has already been used to create a wl_buffer");
138 return;
139 }
140
141 if (Q_UNLIKELY(!test(resource, width, height))) {
142 return;
143 }
144
145 m_isUsed = true;
146
147 m_attrs.width = width;
148 m_attrs.height = height;
149 m_attrs.format = format;
150
151 LinuxDmaBufV1ClientBuffer *clientBuffer = m_integration->rendererInterface()->importBuffer(std::move(m_attrs), flags);
152 if (!clientBuffer) {
153 send_failed(resource->handle);
154 return;
155 }
156
157 wl_resource *bufferResource = wl_resource_create(resource->client(), &wl_buffer_interface, 1, 0);
158 if (!bufferResource) {
159 delete clientBuffer;
160 wl_resource_post_no_memory(resource->handle);
161 return;
162 }
163
164 clientBuffer->initialize(bufferResource);
165 send_created(resource->handle, bufferResource);
166
167 DisplayPrivate *displayPrivate = DisplayPrivate::get(m_integration->display());
168 displayPrivate->registerClientBuffer(clientBuffer);
169}

212bool LinuxDmaBufParamsV1::test(Resource *resource, uint32_t width, uint32_t height)
213{
214 if (Q_UNLIKELY(!m_attrs.planeCount)) {
215 wl_resource_post_error(resource->handle, error_incomplete, "no planes have been specified");
216 return false;
217 }
218
219 // Check for holes in the dmabuf set (e.g. [0, 1, 3]).
220 for (int i = 0; i < m_attrs.planeCount; ++i) {
221 if (!m_attrs.fd[i].isValid()) {
222 wl_resource_post_error(resource->handle, error_incomplete, "no dmabuf has been added for plane %d", i);
223 return false;
224 }
225 }
226
227 if (Q_UNLIKELY(width == 0 || height == 0)) {
228 wl_resource_post_error(resource->handle, error_invalid_dimensions, "invalid width %d or height %d", width, height);
229 return false;
230 }
231
232 for (int i = 0; i < m_attrs.planeCount; ++i) {
233 // Check for overflows.
234 if (Q_UNLIKELY(uint64_t(m_attrs.offset[i]) + m_attrs.pitch[i] > UINT32_MAX)) {
235 wl_resource_post_error(resource->handle, error_out_of_bounds, "size overflow for plane %d", i);
236 return false;
237 }
238
239 if (Q_UNLIKELY(i == 0 && uint64_t(m_attrs.offset[i]) + uint64_t(m_attrs.pitch[i]) * height > UINT32_MAX)) {
240 wl_resource_post_error(resource->handle, error_out_of_bounds, "size overflow for plane %d", i);
241 return false;
242 }
243
244 // Don't report an error as it might be caused by the kernel not supporting
245 // seeking on dmabuf.
246 const off_t size = lseek(m_attrs.fd[i].get(), 0, SEEK_END);
247 if (size == -1) {
248 continue;
249 }
250
251 if (Q_UNLIKELY(m_attrs.offset[i] >= size)) {
252 wl_resource_post_error(resource->handle, error_out_of_bounds, "invalid offset %i for plane %d", m_attrs.offset[i], i);
253 return false;
254 }
255
256 if (Q_UNLIKELY(m_attrs.offset[i] + m_attrs.pitch[i] > size)) {
257 wl_resource_post_error(resource->handle, error_out_of_bounds, "invalid stride %i for plane %d", m_attrs.pitch[i], i);
258 return false;
259 }
260
261 // Only valid for first plane as other planes might be sub-sampled according to
262 // fourcc format.
263 if (Q_UNLIKELY(i == 0 && m_attrs.offset[i] + m_attrs.pitch[i] * height > size)) {
264 wl_resource_post_error(resource->handle, error_out_of_bounds, "invalid buffer stride of height for plane %d", i);
265 return false;
266 }
267 }
268
269 return true;
270}

439void LinuxDmaBufV1FeedbackPrivate::send(Resource *resource)
440{
441 send_format_table(resource->handle, m_bufferintegration->table->file.fd(), m_bufferintegration->table->file.size());
442 QByteArray bytes;
443 bytes.append(reinterpret_cast<const char *>(&m_bufferintegration->mainDevice), sizeof(dev_t));
444 send_main_device(resource->handle, bytes);
445 const auto &sendTranche = [this, resource](const LinuxDmaBufV1Feedback::Tranche &tranche) {
446 QByteArray targetDevice;
447 targetDevice.append(reinterpret_cast<const char *>(&tranche.device), sizeof(dev_t));
448 QByteArray indices;
449 for (auto it = tranche.formatTable.begin(); it != tranche.formatTable.end(); it++) {
450 const uint32_t format = it.key();
451 for (const auto &mod : std::as_const(it.value())) {
452 uint16_t index = m_bufferintegration->table->indices[std::pair<uint32_t, uint64_t>(format, mod)];
453 indices.append(reinterpret_cast<const char *>(&index), 2);
454 }
455 }
456 send_tranche_target_device(resource->handle, targetDevice);
457 send_tranche_formats(resource->handle, indices);
458 send_tranche_flags(resource->handle, static_cast<uint32_t>(tranche.flags));
459 send_tranche_done(resource->handle);
460 };
461 for (const auto &tranche : std::as_const(m_tranches)) {
462 sendTranche(tranche);
463 }
464 // send default hints as the last fallback tranche
465 const auto defaultFeedbackPrivate = get(m_bufferintegration->defaultFeedback.get());
466 if (this != defaultFeedbackPrivate) {
467 for (const auto &tranche : std::as_const(defaultFeedbackPrivate->m_tranches)) {
468 sendTranche(tranche);
469 }
470 }
471 send_done(resource->handle);
472}

491LinuxDmaBufV1FormatTable::LinuxDmaBufV1FormatTable(const QHash<uint32_t, QVector<uint64_t>> &supportedModifiers)
492{
493 QVector<linux_dmabuf_feedback_v1_table_entry> data;
494 for (auto it = supportedModifiers.begin(); it != supportedModifiers.end(); it++) {
495 const uint32_t format = it.key();
496 for (const uint64_t &mod : *it) {
497 indices.insert({format, mod}, data.size());
498 data.append({format, 0, mod});
499 }
500 }
501
502 const auto size = data.size() * sizeof(linux_dmabuf_feedback_v1_table_entry);
503 file = KWin::RamFile("kwin-dmabuf-feedback-table", data.constData(), size, KWin::RamFile::Flag::SealWrite);
504 if (!file.isValid()) {
505 qCCritical(KWIN_CORE) << "Failed to create RamFile for LinuxDmaBufV1FormatTable";
506 return;
507 }
