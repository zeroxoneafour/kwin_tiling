33TrackMouseEffect::TrackMouseEffect()
34 : m_angle(0)
35{
36 initConfig<TrackMouseConfig>();
37 if (effects->isOpenGLCompositing() || effects->compositingType() == QPainterCompositing) {
38 m_angleBase = 90.0;
39 }
40 m_mousePolling = false;
41
42 m_action = new QAction(this);
43 m_action->setObjectName(QStringLiteral("TrackMouse"));
44 m_action->setText(i18n("Track mouse"));
45 KGlobalAccel::self()->setDefaultShortcut(m_action, QList<QKeySequence>());
46 KGlobalAccel::self()->setShortcut(m_action, QList<QKeySequence>());
47
48 connect(m_action, &QAction::triggered, this, &TrackMouseEffect::toggle);
49
50 connect(effects, &EffectsHandler::mouseChanged, this, &TrackMouseEffect::slotMouseChanged);
51 reconfigure(ReconfigureAll);
52}

100void TrackMouseEffect::paintScreen(int mask, const QRegion &region, ScreenPaintData &data)
101{
102 effects->paintScreen(mask, region, data); // paint normal screen
103
104 if (effects->isOpenGLCompositing() && m_texture[0] && m_texture[1]) {
105 ShaderBinder binder(ShaderTrait::MapTexture);
106 GLShader *shader(binder.shader());
107 if (!shader) {
108 return;
109 }
110 glEnable(GL_BLEND);
111 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
112 QMatrix4x4 matrix(data.projectionMatrix());
113 const QPointF p = m_lastRect[0].topLeft() + QPoint(m_lastRect[0].width() / 2.0, m_lastRect[0].height() / 2.0);
114 const float x = p.x();
115 const float y = p.y();
116 const auto scale = effects->renderTargetScale();
117 for (int i = 0; i < 2; ++i) {
118 matrix.translate(x * scale, y * scale, 0.0);
119 matrix.rotate(i ? -2 * m_angle : m_angle, 0, 0, 1.0);
120 matrix.translate(-x * scale, -y * scale, 0.0);
121 QMatrix4x4 mvp(matrix);
122 mvp.translate(m_lastRect[i].x() * scale, m_lastRect[i].y() * scale);
123 shader->setUniform(GLShader::ModelViewProjectionMatrix, mvp);
124 m_texture[i]->bind();
125 m_texture[i]->render(m_lastRect[i].size(), scale);
126 m_texture[i]->unbind();
127 }
128 glDisable(GL_BLEND);
129 } else if (effects->compositingType() == QPainterCompositing && !m_image[0].isNull() && !m_image[1].isNull()) {
130 QPainter *painter = effects->scenePainter();
131 const QPointF p = m_lastRect[0].topLeft() + QPoint(m_lastRect[0].width() / 2.0, m_lastRect[0].height() / 2.0);
132 for (int i = 0; i < 2; ++i) {
133 painter->save();
134 painter->translate(p.x(), p.y());
135 painter->rotate(i ? -2 * m_angle : m_angle);
136 painter->translate(-p.x(), -p.y());
137 painter->drawImage(m_lastRect[i], m_image[i]);
138 painter->restore();
139 }
140 }
141}

164void TrackMouseEffect::toggle()
165{
166 switch (m_state) {
167 case State::ActivatedByModifiers:
168 m_state = State::ActivatedByShortcut;
169 break;
170
171 case State::ActivatedByShortcut:
172 m_state = State::Inactive;
173 break;
174
175 case State::Inactive:
176 if (!init()) {
177 return;
178 }
179 m_state = State::ActivatedByShortcut;
180 break;
181
182 default:
183 Q_UNREACHABLE();
184 break;
185 }
186
187 effects->addRepaint(m_lastRect[0].adjusted(-1, -1, 1, 1));
188}

190void TrackMouseEffect::slotMouseChanged(const QPointF &, const QPointF &,
191 Qt::MouseButtons, Qt::MouseButtons,
192 Qt::KeyboardModifiers modifiers, Qt::KeyboardModifiers)
193{
194 if (!m_mousePolling) { // we didn't ask for it but maybe someone else did...
195 return;
196 }
197
198 switch (m_state) {
199 case State::ActivatedByModifiers:
200 if (modifiers == m_modifiers) {
201 return;
202 }
203 m_state = State::Inactive;
204 break;
205
206 case State::ActivatedByShortcut:
207 return;
208
209 case State::Inactive:
210 if (modifiers != m_modifiers) {
211 return;
212 }
213 if (!init()) {
214 return;
215 }
216 m_state = State::ActivatedByModifiers;
217 break;
218
219 default:
220 Q_UNREACHABLE();
221 break;
222 }
223
224 effects->addRepaint(m_lastRect[0].adjusted(-1, -1, 1, 1));
225}

227void TrackMouseEffect::loadTexture()
228{
229 QString f[2] = {QStandardPaths::locate(QStandardPaths::AppDataLocation, QStringLiteral("tm_outer.png")),
230 QStandardPaths::locate(QStandardPaths::AppDataLocation, QStringLiteral("tm_inner.png"))};
231 if (f[0].isEmpty() || f[1].isEmpty()) {
232 return;
233 }
234
235 for (int i = 0; i < 2; ++i) {
236 if (effects->isOpenGLCompositing()) {
237 QImage img(f[i]);
238 m_texture[i] = std::make_unique<GLTexture>(img);
239 m_lastRect[i].setSize(img.size());
240 }
241 if (effects->compositingType() == QPainterCompositing) {
242 m_image[i] = QImage(f[i]);
243 m_lastRect[i].setSize(m_image[i].size());
244 }
245 }
246}
