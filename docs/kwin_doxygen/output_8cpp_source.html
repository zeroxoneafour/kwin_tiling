20QDebug operator<<(QDebug debug, const Output *output)
21{
22 QDebugStateSaver saver(debug);
23 debug.nospace();
24 if (output) {
25 debug << output->metaObject()->className() << '(' << static_cast<const void *>(output);

27 debug << ", geometry=" << output->geometry();
28 debug << ", scale=" << output->scale();
29 if (debug.verbosity() > 2) {
30 debug << ", manufacturer=" << output->manufacturer();
31 debug << ", model=" << output->model();

222void Output::applyChanges(const OutputConfiguration &config)
223{
224 auto props = config.constChangeSet(this);
225 Q_EMIT aboutToChange();
226
227 State next = m_state;
228 next.enabled = props->enabled;
229 next.transform = props->transform;
230 next.position = props->pos;
231 next.scale = props->scale;
232 next.rgbRange = props->rgbRange;
233
234 setState(next);
235 setVrrPolicy(props->vrrPolicy);
236
237 Q_EMIT changed();
238}

250static QUuid generateOutputId(const QString &eisaId, const QString &model,
251 const QString &serialNumber, const QString &name)
252{
253 static const QUuid urlNs = QUuid("6ba7b811-9dad-11d1-80b4-00c04fd430c8"); // NameSpace_URL
254 static const QUuid kwinNs = QUuid::createUuidV5(urlNs, QStringLiteral("https://kwin.kde.org/o/"));
255
256 const QString payload = QStringList{name, eisaId, model, serialNumber}.join(':');
257 return QUuid::createUuidV5(kwinNs, payload);
258}

266void Output::setState(const State &state)
267{
268 const QRect oldGeometry = geometry();
269 const State oldState = m_state;
270
271 m_state = state;
272
273 if (oldGeometry != geometry()) {
274 Q_EMIT geometryChanged();
275 }
276 if (oldState.scale != state.scale) {
277 Q_EMIT scaleChanged();
278 }
279 if (oldState.modes != state.modes) {
280 Q_EMIT modesChanged();
281 }
282 if (oldState.currentMode != state.currentMode) {
283 Q_EMIT currentModeChanged();
284 }
285 if (oldState.transform != state.transform) {
286 Q_EMIT transformChanged();
287 }
288 if (oldState.overscan != state.overscan) {
289 Q_EMIT overscanChanged();
290 }
291 if (oldState.dpmsMode != state.dpmsMode) {
292 Q_EMIT dpmsModeChanged();
293 }
294 if (oldState.rgbRange != state.rgbRange) {
295 Q_EMIT rgbRangeChanged();
296 }
297 if (oldState.enabled != state.enabled) {
298 Q_EMIT enabledChanged();
299 }
300}

324QMatrix4x4 Output::logicalToNativeMatrix(const QRectF &rect, qreal scale, Transform transform)
325{
326 QMatrix4x4 matrix;
327 matrix.scale(scale);
328
329 switch (transform) {
330 case Transform::Normal:
331 case Transform::Flipped:
332 break;
333 case Transform::Rotated90:
334 case Transform::Flipped90:
335 matrix.translate(0, rect.width());
336 matrix.rotate(-90, 0, 0, 1);
337 break;
338 case Transform::Rotated180:
339 case Transform::Flipped180:
340 matrix.translate(rect.width(), rect.height());
341 matrix.rotate(-180, 0, 0, 1);
342 break;
343 case Transform::Rotated270:
344 case Transform::Flipped270:
345 matrix.translate(rect.height(), 0);
346 matrix.rotate(-270, 0, 0, 1);
347 break;
348 }
349
350 switch (transform) {
351 case Transform::Flipped:
352 case Transform::Flipped90:
353 case Transform::Flipped180:
354 case Transform::Flipped270:
355 matrix.translate(rect.width(), 0);
356 matrix.scale(-1, 1);
357 break;
358 default:
359 break;
360 }
361
362 matrix.translate(-rect.x(), -rect.y());
363
364 return matrix;
365}
