30bool X11Filter::event(xcb_generic_event_t *event)
31{
32 const auto tab = workspace()->tabbox();
33 if (!tab->isGrabbed()) {
34 return false;
35 }
36 const uint8_t eventType = event->response_type & ~0x80;
37 switch (eventType) {
38 case XCB_BUTTON_PRESS:
39 case XCB_BUTTON_RELEASE: {
40 auto e = reinterpret_cast<xcb_button_press_event_t *>(event);
41 xcb_allow_events(connection(), XCB_ALLOW_ASYNC_POINTER, XCB_CURRENT_TIME);
42 if (!tab->isShown() && tab->isDisplayed()) {
43 if (effects && static_cast<EffectsHandlerImpl *>(effects)->isMouseInterception()) {
44 // pass on to effects, effects will filter out the event
45 return false;
46 }
47 }
48 if (eventType == XCB_BUTTON_PRESS) {
49 return buttonPress(e);
50 }
51 return false;
52 }
53 case XCB_MOTION_NOTIFY: {
54 motion(event);
55 break;
56 }
57 case XCB_KEY_PRESS: {
58 keyPress(event);
59 return true;
60 }
61 case XCB_KEY_RELEASE:
62 keyRelease(event);
63 return true;
64 }
65 return false;
66}
67bool X11Filter::buttonPress(xcb_button_press_event_t *event)
68{
69 // press outside Tabbox?
70 const auto tab = workspace()->tabbox();
71 QPoint pos(event->root_x, event->root_y);
72 if ((!tab->isShown() && tab->isDisplayed())
73 || (!tabBox->containsPos(pos) && (event->detail == XCB_BUTTON_INDEX_1 || event->detail == XCB_BUTTON_INDEX_2 || event->detail == XCB_BUTTON_INDEX_3))) {
74 tab->close(); // click outside closes tab
75 return true;
76 }
77 if (event->detail == XCB_BUTTON_INDEX_5 || event->detail == XCB_BUTTON_INDEX_4) {
78 // mouse wheel event
79 const QModelIndex index = tabBox->nextPrev(event->detail == XCB_BUTTON_INDEX_5);
80 if (index.isValid()) {
81 tab->setCurrentIndex(index);
82 }
83 return true;
84 }
85 return false;
86}

105void X11Filter::keyRelease(xcb_generic_event_t *event)
106{
107 const auto ev = reinterpret_cast<xcb_key_release_event_t *>(event);
108 unsigned int mk = ev->state & (KKeyServer::modXShift() | KKeyServer::modXCtrl() | KKeyServer::modXAlt() | KKeyServer::modXMeta());
109 // ev.state is state before the key release, so just checking mk being 0 isn't enough
110 // using XQueryPointer() also doesn't seem to work well, so the check that all
111 // modifiers are released: only one modifier is active and the currently released
112 // key is this modifier - if yes, release the grab
113 int mod_index = -1;
114 for (int i = XCB_MAP_INDEX_SHIFT;
115 i <= XCB_MAP_INDEX_5;
116 ++i) {
117 if ((mk & (1 << i)) != 0) {
118 if (mod_index >= 0) {
119 return;
120 }
121 mod_index = i;
122 }
123 }
124 bool release = false;
125 if (mod_index == -1) {
126 release = true;
127 } else {
128 Xcb::ModifierMapping xmk;
129 if (xmk) {
130 xcb_keycode_t *keycodes = xmk.keycodes();
131 const int maxIndex = xmk.size();
132 for (int i = 0; i < xmk->keycodes_per_modifier; ++i) {
133 const int index = xmk->keycodes_per_modifier * mod_index + i;
134 if (index >= maxIndex) {
135 continue;
136 }
137 if (keycodes[index] == ev->detail) {
138 release = true;
139 }
140 }
141 }
142 }
143 if (release) {
144 workspace()->tabbox()->modifiersReleased();
145 }
146}
