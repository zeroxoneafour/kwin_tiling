109void SurfaceItemWayland::handleChildSubSurfacesChanged()
110{
111 const QList<KWaylandServer::SubSurfaceInterface *> below = m_surface->below();
112 const QList<KWaylandServer::SubSurfaceInterface *> above = m_surface->above();
113
114 for (int i = 0; i < below.count(); ++i) {
115 SurfaceItemWayland *subsurfaceItem = getOrCreateSubSurfaceItem(below[i]);
116 subsurfaceItem->setZ(i - below.count());
117 }
118
119 for (int i = 0; i < above.count(); ++i) {
120 SurfaceItemWayland *subsurfaceItem = getOrCreateSubSurfaceItem(above[i]);
121 subsurfaceItem->setZ(i);
122 }
123}
