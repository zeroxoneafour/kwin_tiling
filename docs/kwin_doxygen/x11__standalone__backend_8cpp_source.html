76bool XrandrEventFilter::event(xcb_generic_event_t *event)
77{
78 Q_ASSERT((event->response_type & ~0x80) == Xcb::Extensions::self()->randrNotifyEvent());
79 // let's try to gather a few XRandR events, unlikely that there is just one
80 m_backend->scheduleUpdateOutputs();
81
82 // update default screen
83 auto *xrrEvent = reinterpret_cast<xcb_randr_screen_change_notify_event_t *>(event);
84 xcb_screen_t *screen = Xcb::defaultScreen();
85 if (xrrEvent->rotation & (XCB_RANDR_ROTATION_ROTATE_90 | XCB_RANDR_ROTATION_ROTATE_270)) {
86 screen->width_in_pixels = xrrEvent->height;
87 screen->height_in_pixels = xrrEvent->width;
88 screen->width_in_millimeters = xrrEvent->mheight;
89 screen->height_in_millimeters = xrrEvent->mwidth;
90 } else {
91 screen->width_in_pixels = xrrEvent->width;
92 screen->height_in_pixels = xrrEvent->height;
93 screen->width_in_millimeters = xrrEvent->mwidth;
94 screen->height_in_millimeters = xrrEvent->mheight;
95 }
96
97 return false;
98}

276template<typename T>
277void X11StandaloneBackend::doUpdateOutputs()
278{
279 QVector<Output *> changed;
280 QVector<Output *> added;
281 QVector<Output *> removed = m_outputs;
282
283 if (Xcb::Extensions::self()->isRandrAvailable()) {
284 T resources(rootWindow());
285 if (!resources.isNull()) {
286
287 std::span crtcs(resources.crtcs(), resources->num_crtcs);
288 for (auto crtc : crtcs) {
289 Xcb::RandR::CrtcInfo info(crtc, resources->config_timestamp);
290
291 const QRect geometry = info.rect();
292 if (!geometry.isValid()) {
293 continue;
294 }
295
296 float refreshRate = -1.0f;
297
298 for (auto mode : std::span(resources.modes(), resources->num_modes)) {
299 if (info->mode == mode.id) {
300 if (mode.htotal != 0 && mode.vtotal != 0) { // BUG 313996
301 // refresh rate calculation - WTF was wikipedia 1998 when I needed it?
302 int dotclock = mode.dot_clock,
303 vtotal = mode.vtotal;
304 if (mode.mode_flags & XCB_RANDR_MODE_FLAG_INTERLACE) {
305 dotclock *= 2;
306 }
307 if (mode.mode_flags & XCB_RANDR_MODE_FLAG_DOUBLE_SCAN) {
308 vtotal *= 2;
309 }
310 refreshRate = dotclock / float(mode.htotal * vtotal);
311 }
312 break; // found mode
313 }
314 }
315
316 for (auto xcbOutput : std::span(info.outputs(), info->num_outputs)) {
317 Xcb::RandR::OutputInfo outputInfo(xcbOutput, resources->config_timestamp);
318 if (outputInfo->crtc != crtc) {
319 continue;
320 }
321
322 X11Output *output = findX11Output(outputInfo.name());
323 if (output) {
324 changed.append(output);
325 removed.removeOne(output);
326 } else {
327 output = new X11Output(this);
328 added.append(output);
329 }
330
331 // TODO: Perhaps the output has to save the inherited gamma ramp and
332 // restore it during tear down. Currently neither standalone x11 nor
333 // drm platform do this.
334 Xcb::RandR::CrtcGamma gamma(crtc);
335
336 output->setRenderLoop(m_renderLoop.get());
337 output->setCrtc(crtc);
338 output->setGammaRampSize(gamma.isNull() ? 0 : gamma->size);
339 auto it = std::find(crtcs.begin(), crtcs.end(), crtc);
340 int crtcIndex = std::distance(crtcs.begin(), it);
341 output->setXineramaNumber(crtcIndex);
342
343 QSize physicalSize(outputInfo->mm_width, outputInfo->mm_height);
344 switch (info->rotation) {
345 case XCB_RANDR_ROTATION_ROTATE_0:
346 case XCB_RANDR_ROTATION_ROTATE_180:
347 break;
348 case XCB_RANDR_ROTATION_ROTATE_90:
349 case XCB_RANDR_ROTATION_ROTATE_270:
350 physicalSize.transpose();
351 break;
352 case XCB_RANDR_ROTATION_REFLECT_X:
353 case XCB_RANDR_ROTATION_REFLECT_Y:
354 break;
355 }
356
357 X11Output::Information information{
358 .name = outputInfo.name(),
359 .physicalSize = physicalSize,
360 };
361
362 auto edidProperty = Xcb::RandR::OutputProperty(xcbOutput, atoms->edid, XCB_ATOM_INTEGER, 0, 100, false, false);
363 bool ok;
364 if (auto data = edidProperty.toByteArray(&ok); ok && !data.isEmpty()) {
365 if (auto edid = Edid(data, edidProperty.data()->num_items); edid.isValid()) {
366 information.manufacturer = edid.manufacturerString();
367 information.model = edid.monitorName();
368 information.serialNumber = edid.serialNumber();
369 information.edid = data;
370 }
371 }
372
373 auto mode = std::make_shared<OutputMode>(geometry.size(), refreshRate * 1000);
374
375 X11Output::State state = output->m_state;
376 state.modes = {mode};
377 state.currentMode = mode;
378 state.position = geometry.topLeft();
379
380 output->setInformation(information);
381 output->setState(state);
382 break;
383 }
384 }
385 }
386 }
387
388 // The workspace handles having no outputs poorly. If the last output is about to be
389 // removed, create a dummy output to avoid crashing.
390 if (changed.isEmpty() && added.isEmpty()) {
391 auto dummyOutput = new X11PlaceholderOutput(this);
392 m_outputs << dummyOutput;
393 Q_EMIT outputAdded(dummyOutput);
394 dummyOutput->updateEnabled(true);
395 }
396
397 // Process new outputs. Note new outputs must be introduced before removing any other outputs.
398 for (Output *output : std::as_const(added)) {
399 m_outputs.append(output);
400 Q_EMIT outputAdded(output);
401 if (auto placeholderOutput = qobject_cast<X11PlaceholderOutput *>(output)) {
402 placeholderOutput->updateEnabled(true);
403 } else if (auto nativeOutput = qobject_cast<X11Output *>(output)) {
404 nativeOutput->updateEnabled(true);
405 }
406 }
407
408 // Outputs have to be removed last to avoid the case where there are no enabled outputs.
409 for (Output *output : std::as_const(removed)) {
410 m_outputs.removeOne(output);
411 if (auto placeholderOutput = qobject_cast<X11PlaceholderOutput *>(output)) {
412 placeholderOutput->updateEnabled(false);
413 } else if (auto nativeOutput = qobject_cast<X11Output *>(output)) {
414 nativeOutput->updateEnabled(false);
415 }
416 Q_EMIT outputRemoved(output);
417 output->unref();
418 }
419
420 // Make sure that the position of an output in m_outputs matches its xinerama index, there
421 // are X11 protocols that use xinerama indices to identify outputs.
422 std::sort(m_outputs.begin(), m_outputs.end(), [](const Output *a, const Output *b) {
423 const auto xa = qobject_cast<const X11Output *>(a);
424 if (!xa) {
425 return false;
426 }
427 const auto xb = qobject_cast<const X11Output *>(b);
428 if (!xb) {
429 return true;
430 }
431 return xa->xineramaNumber() < xb->xineramaNumber();
432 });
433
434 Q_EMIT outputsQueried();
435}

462static bool refreshRate_compare(const Output *first, const Output *smallest)
463{
464 return first->refreshRate() < smallest->refreshRate();
465}
466
467static int currentRefreshRate()
468{
469 static const int refreshRate = qEnvironmentVariableIntValue("KWIN_X11_REFRESH_RATE");
470 if (refreshRate) {
471 return refreshRate;
472 }
473
474 const QVector<Output *> outputs = kwinApp()->outputBackend()->outputs();
475 if (outputs.isEmpty()) {
476 return 60000;
477 }
478
479 static const QString syncDisplayDevice = qEnvironmentVariable("__GL_SYNC_DISPLAY_DEVICE");
480 if (!syncDisplayDevice.isEmpty()) {
481 for (const Output *output : outputs) {
482 if (output->name() == syncDisplayDevice) {
483 return output->refreshRate();
484 }
485 }
486 }
487
488 auto syncIt = std::min_element(outputs.begin(), outputs.end(), refreshRate_compare);
489 return (*syncIt)->refreshRate();
490}

492void X11StandaloneBackend::updateRefreshRate()
493{
494 int refreshRate = currentRefreshRate();
495 if (refreshRate <= 0) {
496 qCWarning(KWIN_X11STANDALONE) << "Bogus refresh rate" << refreshRate;
497 refreshRate = 60000;
498 }
499
500 m_renderLoop->setRefreshRate(refreshRate);
501}
