100void PreviewBridge::createFactory()
101{
102 m_factory.clear();
103
104 if (m_plugin.isNull()) {
105 setValid(false);
106 qWarning() << "Plugin not set";
107 return;
108 }
109
110 const auto offers = KPluginMetaData::findPlugins(s_pluginName);
111 auto item = std::find_if(offers.constBegin(), offers.constEnd(), [this](const auto &plugin) {
112 return plugin.pluginId() == m_plugin;
113 });
114 if (item != offers.constEnd()) {
115 m_factory = KPluginFactory::loadFactory(*item).plugin;
116 }
117
118 setValid(!m_factory.isNull());
119}

155void PreviewBridge::configure(QQuickItem *ctx)
156{
157 if (!m_valid) {
158 return;
159 }
160 // setup the UI
161 QDialog *dialog = new QDialog();
162 dialog->setAttribute(Qt::WA_DeleteOnClose);
163 if (m_lastCreatedClient) {
164 dialog->setWindowTitle(m_lastCreatedClient->caption());
165 }
166
167 // create the KCModule through the plugintrader
168 QVariantMap args;
169 if (!m_theme.isNull()) {
170 args.insert(QStringLiteral("theme"), m_theme);
171 }
172
173 KCModule *kcm = m_factory->create<KCModule>(dialog, QVariantList({args}));
174
175 if (!kcm) {
176 return;
177 }
178
179 auto save = [this, kcm] {
180 kcm->save();
181 if (m_lastCreatedSettings) {
182 Q_EMIT m_lastCreatedSettings->decorationSettings()->reconfigured();
183 }
184 // Send signal to all kwin instances
185 QDBusMessage message = QDBusMessage::createSignal(QStringLiteral("/KWin"),
186 QStringLiteral("org.kde.KWin"),
187 QStringLiteral("reloadConfig"));
188 QDBusConnection::sessionBus().send(message);
189 };
190 connect(dialog, &QDialog::accepted, this, save);
191
192 QDialogButtonBox *buttons = new QDialogButtonBox(QDialogButtonBox::Ok | QDialogButtonBox::Cancel | QDialogButtonBox::RestoreDefaults | QDialogButtonBox::Reset,
193 dialog);
194
195 QPushButton *reset = buttons->button(QDialogButtonBox::Reset);
196 reset->setEnabled(false);
197 // Here we connect our buttons with the dialog
198 connect(buttons, &QDialogButtonBox::accepted, dialog, &QDialog::accept);
199 connect(buttons, &QDialogButtonBox::rejected, dialog, &QDialog::reject);
200 connect(reset, &QPushButton::clicked, kcm, &KCModule::load);
201 auto changedSignal = static_cast<void (KCModule::*)(bool)>(&KCModule::changed);
202 connect(kcm, changedSignal, reset, &QPushButton::setEnabled);
203 connect(buttons->button(QDialogButtonBox::RestoreDefaults), &QPushButton::clicked, kcm, &KCModule::defaults);
204
205 QVBoxLayout *layout = new QVBoxLayout(dialog);
206 layout->addWidget(kcm);
207 layout->addWidget(buttons);
208
209 if (ctx->window()) {
210 dialog->winId(); // so it creates windowHandle
211 dialog->windowHandle()->setTransientParent(QQuickRenderControl::renderWindowFor(ctx->window()));
212 dialog->setModal(true);
213 }
214
215 dialog->show();
216}
