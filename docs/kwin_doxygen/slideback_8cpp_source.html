27void SlideBackEffect::slotStackingOrderChanged()
28{
29 if (effects->activeFullScreenEffect() || m_tabboxActive) {
30 oldStackingOrder = effects->stackingOrder();
31 usableOldStackingOrder = usableWindows(oldStackingOrder);
32 return;
33 }
34
35 EffectWindowList newStackingOrder = effects->stackingOrder(),
36 usableNewStackingOrder = usableWindows(newStackingOrder);
37 if (usableNewStackingOrder == usableOldStackingOrder || usableNewStackingOrder.isEmpty()) {
38 oldStackingOrder = newStackingOrder;
39 usableOldStackingOrder = usableNewStackingOrder;
40 return;
41 }
42
43 m_upmostWindow = usableNewStackingOrder.last();
44
45 if (m_upmostWindow == m_justMapped) { // a window was added, got on top, stacking changed. Nothing impressive
46 m_justMapped = nullptr;
47 } else if (!usableOldStackingOrder.isEmpty() && m_upmostWindow != usableOldStackingOrder.last()) {
48 windowRaised(m_upmostWindow);
49 }
50
51 oldStackingOrder = newStackingOrder;
52 usableOldStackingOrder = usableNewStackingOrder;
53}

55void SlideBackEffect::windowRaised(EffectWindow *w)
56{
57 // Determine all windows on top of the activated one
58 bool currentFound = false;
59 for (EffectWindow *tmp : std::as_const(oldStackingOrder)) {
60 if (!currentFound) {
61 if (tmp == w) {
62 currentFound = true;
63 }
64 } else {
65 if (isWindowUsable(tmp) && tmp->isOnCurrentDesktop() && w->isOnCurrentDesktop()) {
66 // Do we have to move it?
67 if (intersects(w, tmp->frameGeometry().toRect())) {
68 QRect slideRect;
69 slideRect = getSlideDestination(getModalGroupGeometry(w), tmp->frameGeometry().toRect());
70 effects->setElevatedWindow(tmp, true);
71 elevatedList.append(tmp);
72 motionManager.manage(tmp);
73 motionManager.moveWindow(tmp, slideRect);
74 destinationList.insert(tmp, slideRect);
75 coveringWindows.append(tmp);
76 } else {
77 // Does it intersect with a moved (elevated) window and do we have to elevate it too?
78 for (EffectWindow *elevatedWindow : std::as_const(elevatedList)) {
79 if (tmp->frameGeometry().intersects(elevatedWindow->frameGeometry())) {
80 effects->setElevatedWindow(tmp, true);
81 elevatedList.append(tmp);
82 break;
83 }
84 }
85 }
86 }
87 if (tmp->isDock() || tmp->keepAbove()) {
88 effects->setElevatedWindow(tmp, true);
89 elevatedList.append(tmp);
90 }
91 }
92 }
93 // If a window is minimized it could happen that the panels stay elevated without any windows sliding.
94 // clear all elevation settings
95 if (!motionManager.managingWindows()) {
96 for (EffectWindow *tmp : std::as_const(elevatedList)) {
97 effects->setElevatedWindow(tmp, false);
98 }
99 }
100}

102QRect SlideBackEffect::getSlideDestination(const QRect &windowUnderGeometry, const QRect &windowOverGeometry)
103{
104 // Determine the shortest way:
105 int leftSlide = windowUnderGeometry.left() - windowOverGeometry.right() - 20;
106 int rightSlide = windowUnderGeometry.right() - windowOverGeometry.left() + 20;
107 int upSlide = windowUnderGeometry.top() - windowOverGeometry.bottom() - 20;
108 int downSlide = windowUnderGeometry.bottom() - windowOverGeometry.top() + 20;
109
110 int horizSlide = leftSlide;
111 if (std::abs(horizSlide) > std::abs(rightSlide)) {
112 horizSlide = rightSlide;
113 }
114 int vertSlide = upSlide;
115 if (std::abs(vertSlide) > std::abs(downSlide)) {
116 vertSlide = downSlide;
117 }
118
119 QRect slideRect = windowOverGeometry;
120 if (std::abs(horizSlide) < std::abs(vertSlide)) {
121 slideRect.moveLeft(slideRect.x() + horizSlide);
122 } else {
123 slideRect.moveTop(slideRect.y() + vertSlide);
124 }
125 return slideRect;
126}

128void SlideBackEffect::prePaintScreen(ScreenPrePaintData &data, std::chrono::milliseconds presentTime)
129{
130 int time = 0;
131 if (m_lastPresentTime.count()) {
132 time = (presentTime - m_lastPresentTime).count();
133 }
134 m_lastPresentTime = presentTime;
135
136 if (motionManager.managingWindows()) {
137 motionManager.calculate(time);
138 data.mask |= Effect::PAINT_SCREEN_WITH_TRANSFORMED_WINDOWS;
139 }
140
141 const EffectWindowList windows = effects->stackingOrder();
142 for (auto *w : windows) {
143 w->setData(WindowForceBlurRole, QVariant(true));
144 }
145
146 effects->prePaintScreen(data, presentTime);
147}

183void SlideBackEffect::postPaintWindow(EffectWindow *w)
184{
185 if (motionManager.isManaging(w)) {
186 if (destinationList.contains(w)) {
187 if (!motionManager.isWindowMoving(w)) { // has window reched its destination?
188 // If we are still intersecting with the upmostWindow it is moving. slide to somewhere else
189 // restore the stacking order of all windows not intersecting any more except panels
190 if (coveringWindows.contains(w)) {
191 EffectWindowList tmpList;
192 for (EffectWindow *tmp : std::as_const(elevatedList)) {
193 QRect elevatedGeometry = tmp->frameGeometry().toRect();
194 if (motionManager.isManaging(tmp)) {
195 elevatedGeometry = motionManager.transformedGeometry(tmp).toAlignedRect();
196 }
197 if (m_upmostWindow && !tmp->isDock() && !tmp->keepAbove() && m_upmostWindow->frameGeometry().intersects(elevatedGeometry)) {
198 QRect newDestination;
199 newDestination = getSlideDestination(getModalGroupGeometry(m_upmostWindow), elevatedGeometry);
200 if (!motionManager.isManaging(tmp)) {
201 motionManager.manage(tmp);
202 }
203 motionManager.moveWindow(tmp, newDestination);
204 destinationList[tmp] = newDestination;
205 } else {
206 if (!tmp->isDock()) {
207 bool keepElevated = false;
208 for (EffectWindow *elevatedWindow : std::as_const(tmpList)) {
209 if (tmp->frameGeometry().intersects(elevatedWindow->frameGeometry())) {
210 keepElevated = true;
211 }
212 }
213 if (!keepElevated) {
214 effects->setElevatedWindow(tmp, false);
215 elevatedList.removeAll(tmp);
216 }
217 }
218 }
219 tmpList.append(tmp);
220 }
221 } else {
222 // Move the window back where it belongs
223 motionManager.moveWindow(w, w->frameGeometry().toRect());
224 destinationList.remove(w);
225 }
226 }
227 } else {
228 // is window back at its original position?
229 if (!motionManager.isWindowMoving(w)) {
230 motionManager.unmanage(w);
231 effects->addRepaintFull();
232 }
233 }
234 if (coveringWindows.contains(w)) {
235 // It could happen that there is no aciveWindow() here if the user clicks the close-button on an inactive window.
236 // Just skip... the window will be removed in windowDeleted() later
237 if (m_upmostWindow && !intersects(m_upmostWindow, motionManager.transformedGeometry(w).toAlignedRect())) {
238 coveringWindows.removeAll(w);
239 if (coveringWindows.isEmpty()) {
240 // Restore correct stacking order
241 for (EffectWindow *tmp : std::as_const(elevatedList)) {
242 effects->setElevatedWindow(tmp, false);
243 }
244 elevatedList.clear();
245 }
246 }
247 }
248 }
249 if (!isActive()) {
250 m_lastPresentTime = std::chrono::milliseconds::zero();
251 }
252 effects->postPaintWindow(w);
253}

255void SlideBackEffect::slotWindowDeleted(EffectWindow *w)
256{
257 if (w == m_upmostWindow) {
258 m_upmostWindow = nullptr;
259 }
260 if (w == m_justMapped) {
261 m_justMapped = nullptr;
262 }
263 usableOldStackingOrder.removeAll(w);
264 oldStackingOrder.removeAll(w);
265 coveringWindows.removeAll(w);
266 elevatedList.removeAll(w);
267 if (motionManager.isManaging(w)) {
268 motionManager.unmanage(w);
269 }
270}

277void SlideBackEffect::slotWindowUnminimized(EffectWindow *w)
278{
279 // SlideBack should not be triggered on an unminimized window. For this we need to store the last unminimized window.
280 m_justMapped = w;
281 // the stackingOrderChanged() signal came before the window turned an effect window
282 // usually this is no problem as the change shall not be caught anyway, but
283 // the window may have changed its stack position, bug #353745
284 slotStackingOrderChanged();

308EffectWindowList SlideBackEffect::usableWindows(const EffectWindowList &allWindows)
309{
310 EffectWindowList retList;
311 auto isWindowVisible = [](const EffectWindow *window) {
312 return window && effects->virtualScreenGeometry().intersects(window->frameGeometry().toAlignedRect());
313 };
314 for (EffectWindow *tmp : std::as_const(allWindows)) {
315 if (isWindowUsable(tmp) && isWindowVisible(tmp)) {
