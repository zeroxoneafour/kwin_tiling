106void VirtualDesktops::showAboutAnimation()
107{
108 const QModelIndex index = m_data->animationsModel()->index(m_data->animationsModel()->animationIndex(), 0);
109 if (!index.isValid()) {
110 return;
111 }
112
113 const QString name = index.data(AnimationsModel::NameRole).toString();
114 const QString comment = index.data(AnimationsModel::DescriptionRole).toString();
115 const QString author = index.data(AnimationsModel::AuthorNameRole).toString();
116 const QString email = index.data(AnimationsModel::AuthorEmailRole).toString();
117 const QString website = index.data(AnimationsModel::WebsiteRole).toString();
118 const QString version = index.data(AnimationsModel::VersionRole).toString();
119 const QString license = index.data(AnimationsModel::LicenseRole).toString();
120 const QString icon = index.data(AnimationsModel::IconNameRole).toString();
121
122 const KAboutLicense::LicenseKey licenseType = KAboutLicense::byKeyword(license).key();
123
124 KAboutData aboutData(
125 name, // Plugin name
126 name, // Display name
127 version, // Version
128 comment, // Short description
129 licenseType, // License
130 QString(), // Copyright statement
131 QString(), // Other text
132 website.toLatin1() // Home page
133 );
134 aboutData.setProgramLogo(icon);
135
136 const QStringList authors = author.split(',');
137 const QStringList emails = email.split(',');
138
139 if (authors.count() == emails.count()) {
140 int i = 0;
141 for (const QString &author : authors) {
142 if (!author.isEmpty()) {
143 aboutData.addAuthor(i18n(author.toUtf8()), QString(), emails[i]);
144 }
145 i++;
146 }
147 }
148
149 QPointer<KAboutApplicationDialog> aboutPlugin = new KAboutApplicationDialog(aboutData);
150 aboutPlugin->exec();
151
152 delete aboutPlugin;
153}
