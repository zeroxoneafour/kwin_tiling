27static QString getHostName()
28{
29#ifdef HOST_NAME_MAX
30 char hostnamebuf[HOST_NAME_MAX];
31#else
32 char hostnamebuf[256];
33#endif
34 if (gethostname(hostnamebuf, sizeof hostnamebuf) >= 0) {
35 hostnamebuf[sizeof(hostnamebuf) - 1] = 0;
36 return QString::fromLocal8Bit(hostnamebuf);
37 }
38 return QString();
39}

41GetAddrInfo::GetAddrInfo(const QString &hostName, QObject *parent)

79void GetAddrInfo::resolve()
80{
81 if (m_resolving) {
82 return;
83 }
84 m_resolving = true;
85 *m_addressHints = {};
86 m_addressHints->ai_family = PF_UNSPEC;
87 m_addressHints->ai_socktype = SOCK_STREAM;
88 m_addressHints->ai_flags |= AI_CANONNAME;
89
90 m_watcher->setFuture(QtConcurrent::run([this]() {
91 return getaddrinfo(m_hostName.toLocal8Bit().constData(), nullptr, m_addressHints.get(), &m_address);
92 }));
93 m_ownAddressWatcher->setFuture(QtConcurrent::run([this] {
94 // needs to be performed in a lambda as getHostName() returns a temporary value which would
95 // get destroyed in the main thread before the getaddrinfo thread is able to read it
96 return getaddrinfo(getHostName().toLocal8Bit().constData(), nullptr, m_addressHints.get(), &m_ownAddress);
97 }));
98}

116bool GetAddrInfo::resolved(QFutureWatcher<int> *watcher)
117{
118 if (!watcher->isFinished()) {
119 return false;
120 }
121 if (watcher->result() != 0) {
122 qCDebug(KWIN_CORE) << "getaddrinfo failed with error:" << gai_strerror(watcher->result());
123 // call failed;
124 deleteLater();
125 return false;
126 }
127 return true;
128}

130void GetAddrInfo::compare()
131{
132 if (!m_resolved || !m_ownResolved) {
133 return;
134 }
135 addrinfo *address = m_address;
136 while (address) {
137 if (address->ai_canonname && m_hostName == QByteArray(address->ai_canonname).toLower()) {
138 addrinfo *ownAddress = m_ownAddress;
139 bool localFound = false;
140 while (ownAddress) {
141 if (ownAddress->ai_canonname && QByteArray(ownAddress->ai_canonname).toLower() == m_hostName) {
142 localFound = true;
143 break;
144 }
145 ownAddress = ownAddress->ai_next;
146 }
147 if (localFound) {
148 Q_EMIT local();
149 break;
150 }
151 }
152 address = address->ai_next;
153 }
154 deleteLater();
155}

169void ClientMachine::resolve(xcb_window_t window, xcb_window_t clientLeader)
170{
171 if (m_resolved) {
172 return;
173 }
174 QString name = NETWinInfo(connection(), window, rootWindow(), NET::Properties(), NET::WM2ClientMachine).clientMachine();
175 if (name.isEmpty() && clientLeader && clientLeader != window) {
176 name = NETWinInfo(connection(), clientLeader, rootWindow(), NET::Properties(), NET::WM2ClientMachine).clientMachine();
177 }
178 if (name.isEmpty()) {
179 name = localhost();
180 }
181 if (name == localhost()) {
182 setLocal();
183 }
184 m_hostName = name;
185 checkForLocalhost();
186 m_resolved = true;
187}

189void ClientMachine::checkForLocalhost()
190{
191 if (isLocal()) {
192 // nothing to do
193 return;
194 }
195 QString host = getHostName();
196
197 if (!host.isEmpty()) {
198 host = host.toLower();
199 const QString lowerHostName(m_hostName.toLower());
200 if (host == lowerHostName) {
201 setLocal();
202 return;
203 }
204 if (int index = host.indexOf('.'); index != -1) {
205 if (QStringView(host).left(index) == lowerHostName) {
206 setLocal();
207 return;
208 }
209 } else {
210 m_resolving = true;
211 // check using information from get addr info
212 // GetAddrInfo gets automatically destroyed once it finished or not
213 GetAddrInfo *info = new GetAddrInfo(lowerHostName, this);
214 connect(info, &GetAddrInfo::local, this, &ClientMachine::setLocal);
215 connect(info, &GetAddrInfo::destroyed, this, &ClientMachine::resolveFinished);
216 info->resolve();
217 }
218 }
219}
