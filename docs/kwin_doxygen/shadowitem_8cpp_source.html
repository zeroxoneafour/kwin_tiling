76static inline void distributeHorizontally(QRectF &leftRect, QRectF &rightRect)
77{
78 if (leftRect.right() > rightRect.left()) {
79 const qreal boundedRight = std::min(leftRect.right(), rightRect.right());
80 const qreal boundedLeft = std::max(leftRect.left(), rightRect.left());
81 const qreal halfOverlap = (boundedRight - boundedLeft) / 2.0;
82 leftRect.setRight(boundedRight - halfOverlap);
83 rightRect.setLeft(boundedLeft + halfOverlap);
84 }
85}
86
87static inline void distributeVertically(QRectF &topRect, QRectF &bottomRect)
88{
89 if (topRect.bottom() > bottomRect.top()) {
90 const qreal boundedBottom = std::min(topRect.bottom(), bottomRect.bottom());
91 const qreal boundedTop = std::max(topRect.top(), bottomRect.top());
92 const qreal halfOverlap = (boundedBottom - boundedTop) / 2.0;
93 topRect.setBottom(boundedBottom - halfOverlap);
94 bottomRect.setTop(boundedTop + halfOverlap);
95 }
96}

98WindowQuadList ShadowItem::buildQuads() const
99{
100 // Do not draw shadows if window width or window height is less than 5 px. 5 is an arbitrary choice.
101 if (!m_window->wantsShadowToBeRendered() || m_window->width() < 5 || m_window->height() < 5) {
102 return WindowQuadList();
103 }
104
105 const QSizeF top(m_shadow->elementSize(Shadow::ShadowElementTop));
106 const QSizeF topRight(m_shadow->elementSize(Shadow::ShadowElementTopRight));
107 const QSizeF right(m_shadow->elementSize(Shadow::ShadowElementRight));
108 const QSizeF bottomRight(m_shadow->elementSize(Shadow::ShadowElementBottomRight));
109 const QSizeF bottom(m_shadow->elementSize(Shadow::ShadowElementBottom));
110 const QSizeF bottomLeft(m_shadow->elementSize(Shadow::ShadowElementBottomLeft));
111 const QSizeF left(m_shadow->elementSize(Shadow::ShadowElementLeft));
112 const QSizeF topLeft(m_shadow->elementSize(Shadow::ShadowElementTopLeft));
113
114 const QMarginsF shadowMargins(
115 std::max({topLeft.width(), left.width(), bottomLeft.width()}),
116 std::max({topLeft.height(), top.height(), topRight.height()}),
117 std::max({topRight.width(), right.width(), bottomRight.width()}),
118 std::max({bottomRight.height(), bottom.height(), bottomLeft.height()}));
119
120 const QRectF outerRect = rect();
121
122 const int width = shadowMargins.left() + std::max(top.width(), bottom.width()) + shadowMargins.right();
123 const int height = shadowMargins.top() + std::max(left.height(), right.height()) + shadowMargins.bottom();
124
125 QRectF topLeftRect;
126 if (!topLeft.isEmpty()) {
127 topLeftRect = QRectF(outerRect.topLeft(), topLeft);
128 } else {
129 topLeftRect = QRectF(outerRect.left() + shadowMargins.left(),
130 outerRect.top() + shadowMargins.top(), 0, 0);
131 }
132
133 QRectF topRightRect;
134 if (!topRight.isEmpty()) {
135 topRightRect = QRectF(outerRect.right() - topRight.width(), outerRect.top(),
136 topRight.width(), topRight.height());
137 } else {
138 topRightRect = QRectF(outerRect.right() - shadowMargins.right(),
139 outerRect.top() + shadowMargins.top(), 0, 0);
140 }
141
142 QRectF bottomRightRect;
143 if (!bottomRight.isEmpty()) {
144 bottomRightRect = QRectF(outerRect.right() - bottomRight.width(),
145 outerRect.bottom() - bottomRight.height(),
146 bottomRight.width(), bottomRight.height());
147 } else {
148 bottomRightRect = QRectF(outerRect.right() - shadowMargins.right(),
149 outerRect.bottom() - shadowMargins.bottom(), 0, 0);
150 }
151
152 QRectF bottomLeftRect;
153 if (!bottomLeft.isEmpty()) {
154 bottomLeftRect = QRectF(outerRect.left(), outerRect.bottom() - bottomLeft.height(),
155 bottomLeft.width(), bottomLeft.height());
156 } else {
157 bottomLeftRect = QRectF(outerRect.left() + shadowMargins.left(),
158 outerRect.bottom() - shadowMargins.bottom(), 0, 0);
159 }
160
161 // Re-distribute the corner tiles so no one of them is overlapping with others.
162 // By doing this, we assume that shadow's corner tiles are symmetric
163 // and it is OK to not draw top/right/bottom/left tile between corners.
164 // For example, let's say top-left and top-right tiles are overlapping.
165 // In that case, the right side of the top-left tile will be shifted to left,
166 // the left side of the top-right tile will shifted to right, and the top
167 // tile won't be rendered.
168 distributeHorizontally(topLeftRect, topRightRect);
169 distributeHorizontally(bottomLeftRect, bottomRightRect);
170 distributeVertically(topLeftRect, bottomLeftRect);
171 distributeVertically(topRightRect, bottomRightRect);
172
173 qreal tx1 = 0.0,
174 tx2 = 0.0,
175 ty1 = 0.0,
176 ty2 = 0.0;
177
178 WindowQuadList quads;
179 quads.reserve(8);
180
181 if (topLeftRect.isValid()) {
182 tx1 = 0.0;
183 ty1 = 0.0;
184 tx2 = topLeftRect.width();
185 ty2 = topLeftRect.height();
186 WindowQuad topLeftQuad;
187 topLeftQuad[0] = WindowVertex(topLeftRect.left(), topLeftRect.top(), tx1, ty1);
188 topLeftQuad[1] = WindowVertex(topLeftRect.right(), topLeftRect.top(), tx2, ty1);
189 topLeftQuad[2] = WindowVertex(topLeftRect.right(), topLeftRect.bottom(), tx2, ty2);
190 topLeftQuad[3] = WindowVertex(topLeftRect.left(), topLeftRect.bottom(), tx1, ty2);
191 quads.append(topLeftQuad);
192 }
193
194 if (topRightRect.isValid()) {
195 tx1 = width - topRightRect.width();
196 ty1 = 0.0;
197 tx2 = width;
198 ty2 = topRightRect.height();
199 WindowQuad topRightQuad;
200 topRightQuad[0] = WindowVertex(topRightRect.left(), topRightRect.top(), tx1, ty1);
201 topRightQuad[1] = WindowVertex(topRightRect.right(), topRightRect.top(), tx2, ty1);
202 topRightQuad[2] = WindowVertex(topRightRect.right(), topRightRect.bottom(), tx2, ty2);
203 topRightQuad[3] = WindowVertex(topRightRect.left(), topRightRect.bottom(), tx1, ty2);
204 quads.append(topRightQuad);
205 }
206
207 if (bottomRightRect.isValid()) {
208 tx1 = width - bottomRightRect.width();
209 tx2 = width;
210 ty1 = height - bottomRightRect.height();
211 ty2 = height;
212 WindowQuad bottomRightQuad;
213 bottomRightQuad[0] = WindowVertex(bottomRightRect.left(), bottomRightRect.top(), tx1, ty1);
214 bottomRightQuad[1] = WindowVertex(bottomRightRect.right(), bottomRightRect.top(), tx2, ty1);
215 bottomRightQuad[2] = WindowVertex(bottomRightRect.right(), bottomRightRect.bottom(), tx2, ty2);
216 bottomRightQuad[3] = WindowVertex(bottomRightRect.left(), bottomRightRect.bottom(), tx1, ty2);
217 quads.append(bottomRightQuad);
218 }
219
220 if (bottomLeftRect.isValid()) {
221 tx1 = 0.0;
222 tx2 = bottomLeftRect.width();
223 ty1 = height - bottomLeftRect.height();
224 ty2 = height;
225 WindowQuad bottomLeftQuad;
226 bottomLeftQuad[0] = WindowVertex(bottomLeftRect.left(), bottomLeftRect.top(), tx1, ty1);
227 bottomLeftQuad[1] = WindowVertex(bottomLeftRect.right(), bottomLeftRect.top(), tx2, ty1);
228 bottomLeftQuad[2] = WindowVertex(bottomLeftRect.right(), bottomLeftRect.bottom(), tx2, ty2);
229 bottomLeftQuad[3] = WindowVertex(bottomLeftRect.left(), bottomLeftRect.bottom(), tx1, ty2);
230 quads.append(bottomLeftQuad);
231 }
232
233 QRectF topRect(QPointF(topLeftRect.right(), outerRect.top()),
234 QPointF(topRightRect.left(), outerRect.top() + top.height()));
235
236 QRectF rightRect(QPointF(outerRect.right() - right.width(), topRightRect.bottom()),
237 QPointF(outerRect.right(), bottomRightRect.top()));
238
239 QRectF bottomRect(QPointF(bottomLeftRect.right(), outerRect.bottom() - bottom.height()),
240 QPointF(bottomRightRect.left(), outerRect.bottom()));
241
242 QRectF leftRect(QPointF(outerRect.left(), topLeftRect.bottom()),
243 QPointF(outerRect.left() + left.width(), bottomLeftRect.top()));
244
245 // Re-distribute left/right and top/bottom shadow tiles so they don't
246 // overlap when the window is too small. Please notice that we don't
247 // fix overlaps between left/top(left/bottom, right/top, and so on)
248 // corner tiles because corresponding counter parts won't be valid when
249 // the window is too small, which means they won't be rendered.
250 distributeHorizontally(leftRect, rightRect);
251 distributeVertically(topRect, bottomRect);
252
253 if (topRect.isValid()) {
254 tx1 = shadowMargins.left();
255 ty1 = 0.0;
256 tx2 = tx1 + top.width();
257 ty2 = topRect.height();
258 WindowQuad topQuad;
259 topQuad[0] = WindowVertex(topRect.left(), topRect.top(), tx1, ty1);
260 topQuad[1] = WindowVertex(topRect.right(), topRect.top(), tx2, ty1);
261 topQuad[2] = WindowVertex(topRect.right(), topRect.bottom(), tx2, ty2);
262 topQuad[3] = WindowVertex(topRect.left(), topRect.bottom(), tx1, ty2);
263 quads.append(topQuad);
264 }
265
266 if (rightRect.isValid()) {
267 tx1 = width - rightRect.width();
268 ty1 = shadowMargins.top();
269 tx2 = width;
270 ty2 = ty1 + right.height();
271 WindowQuad rightQuad;
272 rightQuad[0] = WindowVertex(rightRect.left(), rightRect.top(), tx1, ty1);
273 rightQuad[1] = WindowVertex(rightRect.right(), rightRect.top(), tx2, ty1);
274 rightQuad[2] = WindowVertex(rightRect.right(), rightRect.bottom(), tx2, ty2);
275 rightQuad[3] = WindowVertex(rightRect.left(), rightRect.bottom(), tx1, ty2);
276 quads.append(rightQuad);
277 }
278
279 if (bottomRect.isValid()) {
280 tx1 = shadowMargins.left();
281 ty1 = height - bottomRect.height();
282 tx2 = tx1 + bottom.width();
283 ty2 = height;
284 WindowQuad bottomQuad;
285 bottomQuad[0] = WindowVertex(bottomRect.left(), bottomRect.top(), tx1, ty1);
286 bottomQuad[1] = WindowVertex(bottomRect.right(), bottomRect.top(), tx2, ty1);
287 bottomQuad[2] = WindowVertex(bottomRect.right(), bottomRect.bottom(), tx2, ty2);
288 bottomQuad[3] = WindowVertex(bottomRect.left(), bottomRect.bottom(), tx1, ty2);
289 quads.append(bottomQuad);
290 }
291
292 if (leftRect.isValid()) {
293 tx1 = 0.0;
294 ty1 = shadowMargins.top();
295 tx2 = leftRect.width();
296 ty2 = ty1 + left.height();
297 WindowQuad leftQuad;
298 leftQuad[0] = WindowVertex(leftRect.left(), leftRect.top(), tx1, ty1);
299 leftQuad[1] = WindowVertex(leftRect.right(), leftRect.top(), tx2, ty1);
300 leftQuad[2] = WindowVertex(leftRect.right(), leftRect.bottom(), tx2, ty2);
301 leftQuad[3] = WindowVertex(leftRect.left(), leftRect.bottom(), tx1, ty2);
302 quads.append(leftQuad);
303 }
304
305 return quads;
306}
