159std::shared_ptr<GbmBuffer> GbmBuffer::importBuffer(DrmGpu *gpu, KWaylandServer::LinuxDmaBufV1ClientBuffer *clientBuffer)
160{
161 const auto &attrs = clientBuffer->attributes();
162 gbm_bo *bo;
163 if (attrs.modifier != DRM_FORMAT_MOD_INVALID || attrs.offset[0] > 0 || attrs.planeCount > 1) {
164 gbm_import_fd_modifier_data data = {};
165 data.format = attrs.format;
166 data.width = static_cast<uint32_t>(attrs.width);
167 data.height = static_cast<uint32_t>(attrs.height);
168 data.num_fds = attrs.planeCount;
169 data.modifier = attrs.modifier;
170 for (int i = 0; i < attrs.planeCount; i++) {
171 data.fds[i] = attrs.fd[i].get();
172 data.offsets[i] = attrs.offset[i];
173 data.strides[i] = attrs.pitch[i];
174 }
175 bo = gbm_bo_import(gpu->gbmDevice(), GBM_BO_IMPORT_FD_MODIFIER, &data, GBM_BO_USE_SCANOUT);
176 } else {
177 gbm_import_fd_data data = {};
178 data.fd = attrs.fd[0].get();
179 data.width = static_cast<uint32_t>(attrs.width);
180 data.height = static_cast<uint32_t>(attrs.height);
181 data.stride = attrs.pitch[0];
182 data.format = attrs.format;
183 bo = gbm_bo_import(gpu->gbmDevice(), GBM_BO_IMPORT_FD, &data, GBM_BO_USE_SCANOUT);
184 }
185 if (bo) {
186 return std::make_shared<GbmBuffer>(gpu, bo, clientBuffer, GBM_BO_USE_SCANOUT);
187 } else {
188 return nullptr;
189 }
190}

192std::shared_ptr<GbmBuffer> GbmBuffer::importBuffer(DrmGpu *gpu, GbmBuffer *buffer, uint32_t flags)
193{
194 const auto &fds = buffer->fds();
195 if (!fds[0].isValid()) {
196 return nullptr;
197 }
198 const auto strides = buffer->strides();
199 const auto offsets = buffer->offsets();
200 gbm_import_fd_modifier_data data = {
201 .width = (uint32_t)buffer->size().width(),
202 .height = (uint32_t)buffer->size().height(),
203 .format = buffer->format(),
204 .num_fds = (uint32_t)buffer->planeCount(),
205 .fds = {},
206 .strides = {},
207 .offsets = {},
208 .modifier = buffer->modifier(),
209 };
210 for (uint32_t i = 0; i < data.num_fds; i++) {
211 data.fds[i] = fds[i].get();
212 data.strides[i] = strides[i];
213 data.offsets[i] = offsets[i];
214 }
215 gbm_bo *bo = gbm_bo_import(gpu->gbmDevice(), GBM_BO_IMPORT_FD_MODIFIER, &data, flags);
216 if (bo) {
217 return std::make_shared<GbmBuffer>(gpu, bo, flags);
218 } else {
219 return nullptr;
220 }
