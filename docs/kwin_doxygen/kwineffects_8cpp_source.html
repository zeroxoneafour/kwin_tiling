204QMatrix4x4 PaintData::toMatrix(qreal deviceScale) const
205{
206 QMatrix4x4 ret;
207 if (d->translation != QVector3D(0, 0, 0)) {
208 ret.translate(d->translation * deviceScale);
209 }
210 if (d->scale != QVector3D(1, 1, 1)) {
211 ret.scale(d->scale);
212 }
213
214 if (d->rotationAngle != 0) {
215 ret.translate(d->rotationOrigin * deviceScale);
216 ret.rotate(d->rotationAngle, d->rotationAxis);
217 ret.translate(-d->rotationOrigin * deviceScale);
218 }
219
220 return ret;
221}

812WindowQuad WindowQuad::makeSubQuad(double x1, double y1, double x2, double y2) const
813{
814 Q_ASSERT(x1 < x2 && y1 < y2 && x1 >= left() && x2 <= right() && y1 >= top() && y2 <= bottom());
815 WindowQuad ret(*this);
816 // vertices are clockwise starting from topleft
817 ret.verts[0].px = x1;
818 ret.verts[3].px = x1;
819 ret.verts[1].px = x2;
820 ret.verts[2].px = x2;
821 ret.verts[0].py = y1;
822 ret.verts[1].py = y1;
823 ret.verts[2].py = y2;
824 ret.verts[3].py = y2;
825
826 const double xOrigin = left();
827 const double yOrigin = top();
828
829 const double widthReciprocal = 1 / (right() - xOrigin);
830 const double heightReciprocal = 1 / (bottom() - yOrigin);
831
832 for (int i = 0; i < 4; ++i) {
833 const double w1 = (ret.verts[i].px - xOrigin) * widthReciprocal;
834 const double w2 = (ret.verts[i].py - yOrigin) * heightReciprocal;
835
836 // Use bilinear interpolation to compute the texture coords.
837 ret.verts[i].tx = (1 - w1) * (1 - w2) * verts[0].tx + w1 * (1 - w2) * verts[1].tx + w1 * w2 * verts[2].tx + (1 - w1) * w2 * verts[3].tx;
838 ret.verts[i].ty = (1 - w1) * (1 - w2) * verts[0].ty + w1 * (1 - w2) * verts[1].ty + w1 * w2 * verts[2].ty + (1 - w1) * w2 * verts[3].ty;
839 }
840
841 return ret;
842}

848WindowQuadList WindowQuadList::splitAtX(double x) const
849{
850 WindowQuadList ret;
851 ret.reserve(count());
852 for (const WindowQuad &quad : *this) {
853 bool wholeleft = true;
854 bool wholeright = true;
855 for (int i = 0; i < 4; ++i) {
856 if (quad[i].x() < x) {
857 wholeright = false;
858 }
859 if (quad[i].x() > x) {
860 wholeleft = false;
861 }
862 }
863 if (wholeleft || wholeright) { // is whole in one split part
864 ret.append(quad);
865 continue;
866 }
867 if (quad.top() == quad.bottom() || quad.left() == quad.right()) { // quad has no size
868 ret.append(quad);
869 continue;
870 }
871 ret.append(quad.makeSubQuad(quad.left(), quad.top(), x, quad.bottom()));
872 ret.append(quad.makeSubQuad(x, quad.top(), quad.right(), quad.bottom()));
873 }
874 return ret;
875}

882 bool wholetop = true;
883 bool wholebottom = true;
884 for (int i = 0; i < 4; ++i) {
885 if (quad[i].y() < y) {
886 wholebottom = false;

893 ret.append(quad);

895 }

901 ret.append(quad.makeSubQuad(quad.left(), y, quad.right(), quad.bottom()));

906WindowQuadList WindowQuadList::makeGrid(int maxQuadSize) const
907{
908 if (empty()) {
909 return *this;
910 }
911
912 // Find the bounding rectangle
913 double left = first().left();
914 double right = first().right();
915 double top = first().top();
916 double bottom = first().bottom();
917
918 for (const WindowQuad &quad : std::as_const(*this)) {
919 left = std::min(left, quad.left());
920 right = std::max(right, quad.right());
921 top = std::min(top, quad.top());
