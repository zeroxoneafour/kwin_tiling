65bool EGLPlatformContext::makeCurrent(QPlatformSurface *surface)
66{
67 const EGLSurface eglSurface = eglSurfaceForPlatformSurface(surface);
68
69 const bool ok = eglMakeCurrent(eglDisplay(), eglSurface, eglSurface, eglContext());
70 if (!ok) {
71 qCWarning(KWIN_QPA, "eglMakeCurrent failed: %x", eglGetError());
72 return false;
73 }
74
75 if (surface->surface()->surfaceClass() == QSurface::Window) {
76 // QOpenGLContextPrivate::setCurrentContext will be called after this
77 // method returns, but that's too late, as we need a current context in
78 // order to bind the content framebuffer object.
79 QOpenGLContextPrivate::setCurrentContext(context());
80
81 Window *window = static_cast<Window *>(surface);
82 window->bindContentFBO();
83 }
84
85 return true;
86}

113void EGLPlatformContext::swapBuffers(QPlatformSurface *surface)
114{
115 if (surface->surface()->surfaceClass() == QSurface::Window) {
116 Window *window = static_cast<Window *>(surface);
117 InternalWindow *internalWindow = window->internalWindow();
118 if (!internalWindow) {
119 return;
120 }
121 glFlush();
122 auto fbo = window->swapFBO();
123 window->bindContentFBO();
124 internalWindow->present(fbo);
125 }
126}

141void EGLPlatformContext::create(const QSurfaceFormat &format, EGLContext shareContext)
142{
143 if (!eglBindAPI(isOpenGLES() ? EGL_OPENGL_ES_API : EGL_OPENGL_API)) {
144 qCWarning(KWIN_QPA, "eglBindAPI failed: 0x%x", eglGetError());
145 return;
146 }
147
148 m_config = configFromFormat(m_eglDisplay, format);
149 if (m_config == EGL_NO_CONFIG_KHR) {
150 qCWarning(KWIN_QPA) << "Could not find suitable EGLConfig for" << format;
151 return;
152 }
153
154 m_format = formatFromConfig(m_eglDisplay, m_config);
155
156 const QByteArray eglExtensions = eglQueryString(eglDisplay(), EGL_EXTENSIONS);
157 const QList<QByteArray> extensions = eglExtensions.split(' ');
158 const bool haveRobustness = extensions.contains(QByteArrayLiteral("EGL_EXT_create_context_robustness"));
159 const bool haveCreateContext = extensions.contains(QByteArrayLiteral("EGL_KHR_create_context"));
160 const bool haveContextPriority = extensions.contains(QByteArrayLiteral("EGL_IMG_context_priority"));
161
162 std::vector<std::unique_ptr<AbstractOpenGLContextAttributeBuilder>> candidates;
163 if (isOpenGLES()) {
164 if (haveCreateContext && haveRobustness && haveContextPriority) {
165 auto glesRobustPriority = std::make_unique<EglOpenGLESContextAttributeBuilder>();
166 glesRobustPriority->setVersion(2);
167 glesRobustPriority->setRobust(true);
168 glesRobustPriority->setHighPriority(true);
169 candidates.push_back(std::move(glesRobustPriority));
170 }
171 if (haveCreateContext && haveRobustness) {
172 auto glesRobust = std::make_unique<EglOpenGLESContextAttributeBuilder>();
173 glesRobust->setVersion(2);
174 glesRobust->setRobust(true);
175 candidates.push_back(std::move(glesRobust));
176 }
177 if (haveContextPriority) {
178 auto glesPriority = std::make_unique<EglOpenGLESContextAttributeBuilder>();
179 glesPriority->setVersion(2);
180 glesPriority->setHighPriority(true);
181 candidates.push_back(std::move(glesPriority));
182 }
183 auto gles = std::make_unique<EglOpenGLESContextAttributeBuilder>();
184 gles->setVersion(2);
185 candidates.push_back(std::move(gles));
186 } else {
187 // Try to create a 3.1 core context
188 if (m_format.majorVersion() >= 3 && haveCreateContext) {
189 if (haveRobustness && haveContextPriority) {
190 auto robustCorePriority = std::make_unique<EglContextAttributeBuilder>();
191 robustCorePriority->setVersion(m_format.majorVersion(), m_format.minorVersion());
192 robustCorePriority->setRobust(true);
193 robustCorePriority->setForwardCompatible(true);
194 if (m_format.profile() == QSurfaceFormat::CoreProfile) {
195 robustCorePriority->setCoreProfile(true);
196 } else if (m_format.profile() == QSurfaceFormat::CompatibilityProfile) {
197 robustCorePriority->setCompatibilityProfile(true);
198 }
199 robustCorePriority->setHighPriority(true);
200 candidates.push_back(std::move(robustCorePriority));
201 }
202 if (haveRobustness) {
203 auto robustCore = std::make_unique<EglContextAttributeBuilder>();
204 robustCore->setVersion(m_format.majorVersion(), m_format.minorVersion());
205 robustCore->setRobust(true);
206 robustCore->setForwardCompatible(true);
207 if (m_format.profile() == QSurfaceFormat::CoreProfile) {
208 robustCore->setCoreProfile(true);
209 } else if (m_format.profile() == QSurfaceFormat::CompatibilityProfile) {
210 robustCore->setCompatibilityProfile(true);
211 }
212 candidates.push_back(std::move(robustCore));
213 }
214 if (haveContextPriority) {
215 auto corePriority = std::make_unique<EglContextAttributeBuilder>();
216 corePriority->setVersion(m_format.majorVersion(), m_format.minorVersion());
217 corePriority->setForwardCompatible(true);
218 if (m_format.profile() == QSurfaceFormat::CoreProfile) {
219 corePriority->setCoreProfile(true);
220 } else if (m_format.profile() == QSurfaceFormat::CompatibilityProfile) {
221 corePriority->setCompatibilityProfile(true);
222 }
223 corePriority->setHighPriority(true);
224 candidates.push_back(std::move(corePriority));
225 }
226 auto core = std::make_unique<EglContextAttributeBuilder>();
227 core->setVersion(m_format.majorVersion(), m_format.minorVersion());
228 core->setForwardCompatible(true);
229 if (m_format.profile() == QSurfaceFormat::CoreProfile) {
230 core->setCoreProfile(true);
231 } else if (m_format.profile() == QSurfaceFormat::CompatibilityProfile) {
232 core->setCompatibilityProfile(true);
233 }
234 candidates.push_back(std::move(core));
235 }
236 if (haveRobustness && haveCreateContext && haveContextPriority) {
237 auto robustPriority = std::make_unique<EglContextAttributeBuilder>();
238 robustPriority->setRobust(true);
239 robustPriority->setHighPriority(true);
240 candidates.push_back(std::move(robustPriority));
241 }
242 if (haveRobustness && haveCreateContext) {
243 auto robust = std::make_unique<EglContextAttributeBuilder>();
244 robust->setRobust(true);
245 candidates.push_back(std::move(robust));
246 }
247 candidates.emplace_back(new EglContextAttributeBuilder);
248 }
249
250 EGLContext context = EGL_NO_CONTEXT;
251 for (auto it = candidates.begin(); it != candidates.end(); it++) {
252 const auto attribs = (*it)->build();
253 context = eglCreateContext(eglDisplay(), m_config, shareContext, attribs.data());
254 if (context != EGL_NO_CONTEXT) {
255 qCDebug(KWIN_QPA) << "Created EGL context with attributes:" << (*it).get();
256 break;
257 }
258 }
259
260 if (context == EGL_NO_CONTEXT) {
261 qCWarning(KWIN_QPA) << "Failed to create EGL context";
262 return;
263 }
264 m_context = context;
265 updateFormatFromContext();
266}

268void EGLPlatformContext::updateFormatFromContext()
269{
270 const EGLSurface oldDrawSurface = eglGetCurrentSurface(EGL_DRAW);
271 const EGLSurface oldReadSurface = eglGetCurrentSurface(EGL_READ);
272 const EGLContext oldContext = eglGetCurrentContext();
273
274 eglMakeCurrent(m_eglDisplay, EGL_NO_SURFACE, EGL_NO_SURFACE, m_context);
275
276 const char *version = reinterpret_cast<const char *>(glGetString(GL_VERSION));
277 int major, minor;
278 if (parseOpenGLVersion(version, major, minor)) {
279 m_format.setMajorVersion(major);
280 m_format.setMinorVersion(minor);
281 } else {
282 qCWarning(KWIN_QPA) << "Unrecognized OpenGL version:" << version;
283 }
284
285 GLint value;
286
287 if (m_format.version() >= qMakePair(3, 0)) {
288 glGetIntegerv(GL_CONTEXT_FLAGS, &value);
289 if (value & GL_CONTEXT_FLAG_FORWARD_COMPATIBLE_BIT) {
290 m_format.setOption(QSurfaceFormat::DeprecatedFunctions);
291 }
292 if (value & GL_CONTEXT_FLAG_DEBUG_BIT) {
293 m_format.setOption(QSurfaceFormat::DebugContext);
294 }
295 } else {
296 m_format.setOption(QSurfaceFormat::DeprecatedFunctions);
297 }
298
299 if (m_format.version() >= qMakePair(3, 2)) {
300 glGetIntegerv(GL_CONTEXT_PROFILE_MASK, &value);
301 if (value & GL_CONTEXT_CORE_PROFILE_BIT) {
302 m_format.setProfile(QSurfaceFormat::CoreProfile);
303 } else if (value & GL_CONTEXT_COMPATIBILITY_PROFILE_BIT) {
304 m_format.setProfile(QSurfaceFormat::CompatibilityProfile);
305 } else {
306 m_format.setProfile(QSurfaceFormat::NoProfile);
307 }
308 } else {
309 m_format.setProfile(QSurfaceFormat::NoProfile);
310 }
311
312 eglMakeCurrent(m_eglDisplay, oldDrawSurface, oldReadSurface, oldContext);
313}
