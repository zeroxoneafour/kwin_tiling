329inline bool Rules::checkSetRule(SetRule rule, bool init)
330{
331 if (rule > (SetRule)DontAffect) { // Unused or DontAffect
332 if (rule == (SetRule)Force || rule == (SetRule)ApplyNow
333 || rule == (SetRule)ForceTemporarily || init) {
334 return true;
335 }
336 }
337 return false;
338}
