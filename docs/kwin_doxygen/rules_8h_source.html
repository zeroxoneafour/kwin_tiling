334inline bool Rules::checkSetRule(SetRule rule, bool init)
335{
336 if (rule > (SetRule)DontAffect) { // Unused or DontAffect
337 if (rule == (SetRule)Force || rule == (SetRule)ApplyNow
338 || rule == (SetRule)ForceTemporarily || init) {
339 return true;
340 }
341 }
342 return false;
343}

346{

348}
