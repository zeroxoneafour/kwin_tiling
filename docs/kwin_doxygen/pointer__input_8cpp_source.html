77void PointerInputRedirection::init()
78{
79 Q_ASSERT(!inited());
80 waylandServer()->seat()->setHasPointer(input()->hasPointer());
81 connect(input(), &InputRedirection::hasPointerChanged,
82 waylandServer()->seat(), &KWaylandServer::SeatInterface::setHasPointer);
83
84 m_cursor = new CursorImage(this);
85 setInited(true);
86 InputDeviceHandler::init();
87
88 if (!input()->hasPointer()) {
89 Cursors::self()->hideCursor();
90 }
91 connect(input(), &InputRedirection::hasPointerChanged, this, []() {
92 if (input()->hasPointer()) {
93 Cursors::self()->showCursor();
94 } else {
95 Cursors::self()->hideCursor();
96 }
97 });
98
99 connect(Cursors::self()->mouse(), &Cursor::rendered, m_cursor, &CursorImage::markAsRendered);
100 connect(m_cursor, &CursorImage::changed, Cursors::self()->mouse(), [this] {
101 Cursors::self()->mouse()->setSource(m_cursor->source());
102 updateCursorOutputs();
103 });
104 Q_EMIT m_cursor->changed();
105
106 connect(workspace(), &Workspace::outputsChanged, this, &PointerInputRedirection::updateAfterScreenChange);
107#if KWIN_BUILD_SCREENLOCKER
108 if (waylandServer()->hasScreenLockerIntegration()) {
109 connect(ScreenLocker::KSldApp::self(), &ScreenLocker::KSldApp::lockStateChanged, this, [this]() {
110 if (waylandServer()->seat()->hasPointer()) {
111 waylandServer()->seat()->cancelPointerPinchGesture();
112 waylandServer()->seat()->cancelPointerSwipeGesture();
113 }
114 update();
115 });
116 }
117#endif
118 connect(workspace(), &QObject::destroyed, this, [this] {
119 setInited(false);
120 });
121 connect(waylandServer(), &QObject::destroyed, this, [this] {
122 setInited(false);
123 });
124 connect(waylandServer()->seat(), &KWaylandServer::SeatInterface::dragEnded, this, [this]() {
125 // need to force a focused pointer change
126 setFocus(nullptr);
127 update();
128 });
129 // connect the move resize of all window
130 auto setupMoveResizeConnection = [this](Window *window) {
131 connect(window, &Window::clientStartUserMovedResized, this, &PointerInputRedirection::updateOnStartMoveResize);
132 connect(window, &Window::clientFinishUserMovedResized, this, &PointerInputRedirection::update);
133 };
134 const auto clients = workspace()->allClientList();
135 std::for_each(clients.begin(), clients.end(), setupMoveResizeConnection);
136 connect(workspace(), &Workspace::windowAdded, this, setupMoveResizeConnection);
137
138 // warp the cursor to center of screen containing the workspace center
139 if (const Output *output = workspace()->outputAt(workspace()->geometry().center())) {
140 warp(output->geometry().center());
141 }
142 updateAfterScreenChange();
143}

171class PositionUpdateBlocker
172{
173public:
174 PositionUpdateBlocker(PointerInputRedirection *pointer)
175 : m_pointer(pointer)
176 {
177 s_counter++;
178 }
179 ~PositionUpdateBlocker()
180 {
181 s_counter--;
182 if (s_counter == 0) {
183 if (!s_scheduledPositions.isEmpty()) {
184 const auto pos = s_scheduledPositions.takeFirst();
185 m_pointer->processMotionInternal(pos.pos, pos.delta, pos.deltaNonAccelerated, pos.time, nullptr);
186 }
187 }
188 }
189
190 static bool isPositionBlocked()
191 {
192 return s_counter > 0;
193 }

196 {
197 s_scheduledPositions.append({pos, delta, deltaNonAccelerated, time});
198 }

200private:
201 static int s_counter;

204 QPointF pos;
205 QPointF delta;
206 QPointF deltaNonAccelerated;
207 std::chrono::microseconds time;
208 };
209 static QVector<ScheduledPosition> s_scheduledPositions;
210
211 PointerInputRedirection *m_pointer;
212};

227void PointerInputRedirection::processMotionInternal(const QPointF &pos, const QPointF &delta, const QPointF &deltaNonAccelerated, std::chrono::microseconds time, InputDevice *device)
228{
229 input()->setLastInputHandler(this);
230 if (!inited()) {
231 return;
232 }
233 if (PositionUpdateBlocker::isPositionBlocked()) {
234 PositionUpdateBlocker::schedulePosition(pos, delta, deltaNonAccelerated, time);
235 return;
236 }
237
238 PositionUpdateBlocker blocker(this);
239 updatePosition(pos);
240 MouseEvent event(QEvent::MouseMove, m_pos, Qt::NoButton, m_qtButtons,
241 input()->keyboardModifiers(), time,
242 delta, deltaNonAccelerated, device);
243 event.setModifiersRelevantForGlobalShortcuts(input()->modifiersRelevantForGlobalShortcuts());
244
245 update();
246 input()->processSpies(std::bind(&InputEventSpy::pointerEvent, std::placeholders::_1, &event));
247 input()->processFilters(std::bind(&InputEventFilter::pointerEvent, std::placeholders::_1, &event, 0));
248}

250void PointerInputRedirection::processButton(uint32_t button, InputRedirection::PointerButtonState state, std::chrono::microseconds time, InputDevice *device)
251{
252 input()->setLastInputHandler(this);
253 QEvent::Type type;
254 switch (state) {
255 case InputRedirection::PointerButtonReleased:
256 type = QEvent::MouseButtonRelease;
257 break;
258 case InputRedirection::PointerButtonPressed:
259 type = QEvent::MouseButtonPress;
260 update();
261 break;
262 default:
263 Q_UNREACHABLE();
264 return;
265 }
266
267 updateButton(button, state);
268
269 MouseEvent event(type, m_pos, buttonToQtMouseButton(button), m_qtButtons,
270 input()->keyboardModifiers(), time, QPointF(), QPointF(), device);
271 event.setModifiersRelevantForGlobalShortcuts(input()->modifiersRelevantForGlobalShortcuts());
272 event.setNativeButton(button);
273
274 input()->processSpies(std::bind(&InputEventSpy::pointerEvent, std::placeholders::_1, &event));
275
276 if (!inited()) {
277 return;
278 }
279
280 input()->processFilters(std::bind(&InputEventFilter::pointerEvent, std::placeholders::_1, &event, button));
281
282 if (state == InputRedirection::PointerButtonReleased) {
283 update();
284 }
285}

465void PointerInputRedirection::cleanupDecoration(Decoration::DecoratedClientImpl *old, Decoration::DecoratedClientImpl *now)
466{
467 disconnect(m_decorationGeometryConnection);
468 m_decorationGeometryConnection = QMetaObject::Connection();
469
470 disconnect(m_decorationDestroyedConnection);
471 m_decorationDestroyedConnection = QMetaObject::Connection();
472
473 if (old) {
474 // send leave event to old decoration
475 QHoverEvent event(QEvent::HoverLeave, QPointF(), QPointF());
476 QCoreApplication::instance()->sendEvent(old->decoration(), &event);
477 }
478 if (!now) {
479 // left decoration
480 return;
481 }
482
483 auto pos = m_pos - now->window()->pos();
484 QHoverEvent event(QEvent::HoverEnter, pos, pos);
485 QCoreApplication::instance()->sendEvent(now->decoration(), &event);
486 now->window()->processDecorationMove(pos, m_pos);
487
488 m_decorationGeometryConnection = connect(
489 decoration()->window(), &Window::frameGeometryChanged, this, [this]() {
490 // ensure maximize button gets the leave event when maximizing/restore a window, see BUG 385140
491 const auto oldDeco = decoration();
492 update();
493 if (oldDeco && oldDeco == decoration() && !decoration()->window()->isInteractiveMove() && !decoration()->window()->isInteractiveResize() && !areButtonsPressed()) {
494 // position of window did not change, we need to send HoverMotion manually
495 const QPointF p = m_pos - decoration()->window()->pos();
496 QHoverEvent event(QEvent::HoverMove, p, p);
497 QCoreApplication::instance()->sendEvent(decoration()->decoration(), &event);
498 }
499 },
500 Qt::QueuedConnection);
501
502 // if our decoration gets destroyed whilst it has focus, we pass focus on to the same window
503 m_decorationDestroyedConnection = connect(now, &QObject::destroyed, this, &PointerInputRedirection::update, Qt::QueuedConnection);
504}

506void PointerInputRedirection::focusUpdate(Window *focusOld, Window *focusNow)
507{
508 if (focusOld && focusOld->isClient()) {
509 focusOld->pointerLeaveEvent();
510 breakPointerConstraints(focusOld->surface());
511 disconnectPointerConstraintsConnection();
512 }
513 disconnect(m_focusGeometryConnection);
514 m_focusGeometryConnection = QMetaObject::Connection();
515
516 if (focusNow && focusNow->isClient()) {
517 focusNow->pointerEnterEvent(m_pos);
518 }
519
520 auto seat = waylandServer()->seat();
521 if (!focusNow || !focusNow->surface()) {
522 seat->notifyPointerLeave();
523 return;
524 }
525
526 seat->notifyPointerEnter(focusNow->surface(), m_pos, focusNow->inputTransformation());
527
528 m_focusGeometryConnection = connect(focusNow, &Window::inputTransformationChanged, this, [this]() {
529 // TODO: why no assert possible?
530 if (!focus()) {
531 return;
532 }
533 // TODO: can we check on the window instead?
534 if (workspace()->moveResizeWindow()) {
535 // don't update while moving
536 return;
537 }
538 auto seat = waylandServer()->seat();
539 if (focus()->surface() != seat->focusedPointerSurface()) {
540 return;
541 }
542 seat->setFocusedPointerSurfaceTransformation(focus()->inputTransformation());
543 });
544
545 m_constraintsConnection = connect(focusNow->surface(), &KWaylandServer::SurfaceInterface::pointerConstraintsChanged,
546 this, &PointerInputRedirection::updatePointerConstraints);
547 m_constraintsActivatedConnection = connect(workspace(), &Workspace::windowActivated,
548 this, &PointerInputRedirection::updatePointerConstraints);
549 updatePointerConstraints();
550}

608void PointerInputRedirection::updatePointerConstraints()
609{
610 if (!focus()) {
611 return;
612 }
613 const auto s = focus()->surface();
614 if (!s) {
615 return;
616 }
617 if (s != waylandServer()->seat()->focusedPointerSurface()) {
618 return;
619 }
620 if (!supportsWarping()) {
621 return;
622 }
623 const bool canConstrain = m_enableConstraints && focus() == workspace()->activeWindow();
624 const auto cf = s->confinedPointer();
625 if (cf) {
626 if (cf->isConfined()) {
627 if (!canConstrain) {
628 cf->setConfined(false);
629 m_confined = false;
630 disconnectConfinedPointerRegionConnection();
631 }
632 return;
633 }
634 const QRegion r = getConstraintRegion(focus(), cf);
635 if (canConstrain && r.contains(m_pos.toPoint())) {
636 cf->setConfined(true);
637 m_confined = true;
638 m_confinedPointerRegionConnection = connect(cf, &KWaylandServer::ConfinedPointerV1Interface::regionChanged, this, [this]() {
639 if (!focus()) {
640 return;
641 }
642 const auto s = focus()->surface();
643 if (!s) {
644 return;
645 }
646 const auto cf = s->confinedPointer();
647 if (!getConstraintRegion(focus(), cf).contains(m_pos.toPoint())) {
648 // pointer no longer in confined region, break the confinement
649 cf->setConfined(false);
650 m_confined = false;
651 } else {
652 if (!cf->isConfined()) {
653 cf->setConfined(true);
654 m_confined = true;
655 }
656 }
657 });
658 return;
659 }
660 } else {
661 m_confined = false;
662 disconnectConfinedPointerRegionConnection();
663 }
664 const auto lock = s->lockedPointer();
665 if (lock) {
666 if (lock->isLocked()) {
667 if (!canConstrain) {
668 const auto hint = lock->cursorPositionHint();
669 lock->setLocked(false);
670 m_locked = false;
671 disconnectLockedPointerAboutToBeUnboundConnection();
672 if (!(hint.x() < 0 || hint.y() < 0) && focus()) {
673 processMotionAbsolute(focus()->mapFromLocal(hint), waylandServer()->seat()->timestamp());
674 }
675 }
676 return;
677 }
678 const QRegion r = getConstraintRegion(focus(), lock);
679 if (canConstrain && r.contains(m_pos.toPoint())) {
680 lock->setLocked(true);
681 m_locked = true;
682
683 // The client might cancel pointer locking from its side by unbinding the LockedPointerInterface.
684 // In this case the cached cursor position hint must be fetched before the resource goes away
685 m_lockedPointerAboutToBeUnboundConnection = connect(lock, &KWaylandServer::LockedPointerV1Interface::aboutToBeDestroyed, this, [this, lock]() {
686 const auto hint = lock->cursorPositionHint();
687 if (hint.x() < 0 || hint.y() < 0 || !focus()) {
688 return;
689 }
690 auto globalHint = focus()->mapFromLocal(hint);
691
692 // When the resource finally goes away, reposition the cursor according to the hint
693 connect(lock, &KWaylandServer::LockedPointerV1Interface::destroyed, this, [this, globalHint]() {
694 processMotionAbsolute(globalHint, waylandServer()->seat()->timestamp());
695 });
696 });
697 // TODO: connect to region change - is it needed at all? If the pointer is locked it's always in the region
698 }
699 } else {
700 m_locked = false;
701 disconnectLockedPointerAboutToBeUnboundConnection();
702 }
703}

705QPointF PointerInputRedirection::applyPointerConfinement(const QPointF &pos) const
706{
707 if (!focus()) {
708 return pos;
709 }
710 auto s = focus()->surface();
711 if (!s) {
712 return pos;
713 }
714 auto cf = s->confinedPointer();
715 if (!cf) {
716 return pos;
717 }
718 if (!cf->isConfined()) {
719 return pos;
720 }
721
722 const QRegion confinementRegion = getConstraintRegion(focus(), cf);
