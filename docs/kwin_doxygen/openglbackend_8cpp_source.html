46void OpenGLBackend::copyPixels(const QRegion &region, const QSize &screenSize)
47{
48 const int height = screenSize.height();
49 for (const QRect &r : region) {
50 const int x0 = r.x();
51 const int y0 = height - r.y() - r.height();
52 const int x1 = r.x() + r.width();
53 const int y1 = height - r.y();
54
55 glBlitFramebuffer(x0, y0, x1, y1, x0, y0, x1, y1, GL_COLOR_BUFFER_BIT, GL_NEAREST);

64bool OpenGLBackend::checkGraphicsReset()
65{
66 const GLenum status = KWin::glGetGraphicsResetStatus();
67 if (Q_LIKELY(status == GL_NO_ERROR)) {
68 return false;
69 }
70
71 switch (status) {
72 case GL_GUILTY_CONTEXT_RESET:
73 qCWarning(KWIN_OPENGL) << "A graphics reset attributable to the current GL context occurred.";
74 break;
75 case GL_INNOCENT_CONTEXT_RESET:
76 qCWarning(KWIN_OPENGL) << "A graphics reset not attributable to the current GL context occurred.";
77 break;
78 case GL_UNKNOWN_CONTEXT_RESET:
79 qCWarning(KWIN_OPENGL) << "A graphics reset of an unknown cause occurred.";
80 break;
81 default:
82 break;
83 }
84
85 QElapsedTimer timer;
86 timer.start();
87
88 // Wait until the reset is completed or max one second
89 while (timer.elapsed() < 1000 && KWin::glGetGraphicsResetStatus() != GL_NO_ERROR) {
90 usleep(50);
91 }
92 if (timer.elapsed() >= 1000) {
93 qCWarning(KWIN_OPENGL) << "Waiting for glGetGraphicsResetStatus to return GL_NO_ERROR timed out!";
94 }
95
96 return true;
97}
