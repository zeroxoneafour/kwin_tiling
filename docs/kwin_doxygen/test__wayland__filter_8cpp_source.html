99void TestFilter::testFilter()
100{
101 QFETCH(bool, accessAllowed);
102
103 // setup connection
104 std::unique_ptr<KWayland::Client::ConnectionThread> connection(new KWayland::Client::ConnectionThread());
105 QSignalSpy connectedSpy(connection.get(), &KWayland::Client::ConnectionThread::connected);
106 connection->setSocketName(s_socketName);
107
108 std::unique_ptr<QThread> thread(new QThread(this));
109 connection->moveToThread(thread.get());
110 thread->start();
111
112 connection->initConnection();
113 QVERIFY(connectedSpy.wait());
114
115 // use low level API as Server::Display::connections only lists connections which have
116 // been previous fetched via getConnection()
117 if (accessAllowed) {
118 wl_client *clientConnection;
119 wl_client_for_each(clientConnection, wl_display_get_client_list(*m_display))
120 {
121 m_display->m_allowedClients << clientConnection;
122 }
123 }
124
125 KWayland::Client::EventQueue queue;
126 queue.setup(connection.get());
127
128 KWayland::Client::Registry registry;
129 QSignalSpy registryDoneSpy(&registry, &KWayland::Client::Registry::interfacesAnnounced);
130 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
131 QSignalSpy blurSpy(&registry, &KWayland::Client::Registry::blurAnnounced);
132
133 registry.setEventQueue(&queue);
134 registry.create(connection->display());
135 QVERIFY(registry.isValid());
136 registry.setup();
137
138 QVERIFY(registryDoneSpy.wait());
139 QVERIFY(compositorSpy.count() == 1);
140 QVERIFY(blurSpy.count() == accessAllowed ? 1 : 0);
141
142 thread->quit();
143 thread->wait();
144}
