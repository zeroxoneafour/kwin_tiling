101void TestFilter::testFilter()
102{
103 QFETCH(bool, accessAllowed);
104
105 // setup connection
106 std::unique_ptr<KWayland::Client::ConnectionThread> connection(new KWayland::Client::ConnectionThread());
107 QSignalSpy connectedSpy(connection.get(), &KWayland::Client::ConnectionThread::connected);
108 connection->setSocketName(s_socketName);
109
110 std::unique_ptr<QThread> thread(new QThread(this));
111 connection->moveToThread(thread.get());
112 thread->start();
113
114 connection->initConnection();
115 QVERIFY(connectedSpy.wait());
116
117 // use low level API as Server::Display::connections only lists connections which have
118 // been previous fetched via getConnection()
119 if (accessAllowed) {
120 wl_client *clientConnection;
121 wl_client_for_each(clientConnection, wl_display_get_client_list(*m_display))
122 {
123 m_display->m_allowedClients << clientConnection;
124 }
125 }
126
127 KWayland::Client::EventQueue queue;
128 queue.setup(connection.get());
129
130 KWayland::Client::Registry registry;
131 QSignalSpy registryDoneSpy(&registry, &KWayland::Client::Registry::interfacesAnnounced);
132 QSignalSpy compositorSpy(&registry, &KWayland::Client::Registry::compositorAnnounced);
133 QSignalSpy blurSpy(&registry, &KWayland::Client::Registry::blurAnnounced);
134
135 registry.setEventQueue(&queue);
136 registry.create(connection->display());
137 QVERIFY(registry.isValid());
138 registry.setup();
139
140 QVERIFY(registryDoneSpy.wait());
141 QVERIFY(compositorSpy.count() == 1);
142 QVERIFY(blurSpy.count() == accessAllowed ? 1 : 0);
143
144 thread->quit();
145 thread->wait();
146}
