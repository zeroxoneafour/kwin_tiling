61void Window::setGeometry(const QRect &rect)
62{
63 const QRect oldGeometry = geometry();
64 QPlatformWindow::setGeometry(rect);
65
66 if (window()->isVisible() && rect.isValid()) {
67 const QSize nativeSize = rect.size() * m_scale;
68 if (m_contentFBO) {
69 if (m_contentFBO->size() != nativeSize) {
70 m_resized = true;
71 }
72 }
73 QWindowSystemInterface::handleGeometryChange(window(), geometry());
74 }
75
76 if (isExposed() && oldGeometry.size() != rect.size()) {
77 QWindowSystemInterface::handleExposeEvent(window(), QRect(QPoint(), rect.size()));
78 }
79}

116void Window::createFBO()
117{
118 const QRect &r = geometry();
119 if (m_contentFBO && r.size().isEmpty()) {
120 return;
121 }
122 const QSize nativeSize = r.size() * m_scale;
123 m_contentFBO.reset(new QOpenGLFramebufferObject(nativeSize.width(), nativeSize.height(), QOpenGLFramebufferObject::CombinedDepthStencil));
124 if (!m_contentFBO->isValid()) {
125 qCWarning(KWIN_QPA) << "Content FBO is not valid";
126 }
127 m_resized = false;
128}
