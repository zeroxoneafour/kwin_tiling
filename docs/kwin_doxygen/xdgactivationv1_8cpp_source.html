28static bool isPrivilegedInWindowManagement(const ClientConnection *client)
29{
30 Q_ASSERT(client);
31 auto requestedInterfaces = client->property("requestedInterfaces").toStringList();
32 return requestedInterfaces.contains(QLatin1String("org_kde_plasma_window_management")) || requestedInterfaces.contains(QLatin1String("kde_lockscreen_overlay_v1"));
33}

35static const QString windowDesktopFileName(Window *window)
36{
37 QString ret = window->desktopFileName();
38 if (!ret.isEmpty()) {
39 return ret;
40 }
41
42 // Fallback to StartupWMClass for legacy apps
43 const auto resourceName = window->resourceName();
44 const auto service = KApplicationTrader::query([&resourceName](const KService::Ptr &service) {
45 return service->property("StartupWMClass").toString().compare(resourceName, Qt::CaseInsensitive) == 0;
46 });
47
48 if (!service.isEmpty()) {
49 ret = service.constFirst()->desktopEntryName();
50 }
51 return ret;
52}

54XdgActivationV1Integration::XdgActivationV1Integration(XdgActivationV1Interface *activation, QObject *parent)
55 : QObject(parent)
56{
57 Workspace *ws = Workspace::self();
58 connect(ws, &Workspace::windowActivated, this, [this](Window *window) {
59 if (!m_currentActivationToken || !window || window->property("token").toString() == m_currentActivationToken->token) {
60 return;
61 }
62
63 // We check that it's not the app that we are trying to activate
64 if (windowDesktopFileName(window) != m_currentActivationToken->applicationId) {
65 // But also that the new one has been requested after the token was requested
66 if (window->lastUsageSerial() < m_currentActivationToken->serial) {
67 return;
68 }
69 }
70
71 clear();
72 });
73 activation->setActivationTokenCreator([this](ClientConnection *client, SurfaceInterface *surface, uint serial, SeatInterface *seat, const QString &appId) -> QString {
74 Workspace *ws = Workspace::self();
75 Q_ASSERT(client); // Should always be available as it's coming straight from the wayland implementation
76 const bool isPrivileged = isPrivilegedInWindowManagement(client);
77 if (!isPrivileged && ws->activeWindow() && ws->activeWindow()->surface() != surface) {
78 qCWarning(KWIN_CORE) << "Cannot grant a token to" << client;
79 return QStringLiteral("not-granted-666");
80 }
81
82 return requestToken(isPrivileged, surface, serial, seat, appId);
83 });
84
85 connect(activation, &XdgActivationV1Interface::activateRequested, this, &XdgActivationV1Integration::activateSurface);
86}

88QString XdgActivationV1Integration::requestToken(bool isPrivileged, SurfaceInterface *surface, uint serial, SeatInterface *seat, const QString &appId)
89{
90 static int i = 0;
91 const auto newToken = QStringLiteral("kwin-%1").arg(++i);
92
93 if (m_currentActivationToken) {
94 clear();
95 }
96 bool showNotify = false;
97 QIcon icon = QIcon::fromTheme(QStringLiteral("system-run"));
98 if (const QString desktopFilePath = Window::findDesktopFile(appId); !desktopFilePath.isEmpty()) {
99 KDesktopFile df(desktopFilePath);
100 Window *window = Workspace::self()->activeWindow();
101 if (!window || appId != window->desktopFileName()) {
102 const auto desktop = df.desktopGroup();
103 showNotify = desktop.readEntry("X-KDE-StartupNotify", desktop.readEntry("StartupNotify", true));
104 }
105 icon = QIcon::fromTheme(df.readIcon(), icon);
106 }
107 m_currentActivationToken.reset(new ActivationToken{newToken, isPrivileged, surface, serial, seat, appId, showNotify, waylandServer()->plasmaActivationFeedback()->createActivation(appId)});
108 if (showNotify) {
109 Q_EMIT effects->startupAdded(m_currentActivationToken->token, icon);
110 }
111 return newToken;
112}

114void XdgActivationV1Integration::activateSurface(SurfaceInterface *surface, const QString &token)
115{
116 Workspace *ws = Workspace::self();
117 auto window = waylandServer()->findWindow(surface);
118 if (!window) {
119 qCWarning(KWIN_CORE) << "could not find the toplevel to activate" << surface;
120 return;
121 }
122
123 if (!m_currentActivationToken || m_currentActivationToken->token != token) {
124 qCDebug(KWIN_CORE) << "Refusing to activate " << window << " (provided token: " << token << ", current token:" << (m_currentActivationToken ? m_currentActivationToken->token : QStringLiteral("null")) << ")";
125 window->demandAttention();
126 return;
127 }
128
129 auto ownerWindow = waylandServer()->findWindow(m_currentActivationToken->surface);
130 qCDebug(KWIN_CORE) << "activating" << window << surface << "on behalf of" << m_currentActivationToken->surface << "into" << ownerWindow;
131 if (!ws->activeWindow() || ws->activeWindow() == ownerWindow || ws->activeWindow()->lastUsageSerial() < m_currentActivationToken->serial || m_currentActivationToken->isPrivileged) {
132 ws->activateWindow(window);
133 } else {
134 qCWarning(KWIN_CORE) << "Activation requested while owner isn't active" << (ownerWindow ? ownerWindow->desktopFileName() : "null")
135 << m_currentActivationToken->applicationId;
136 window->demandAttention();
137 clear();
138 }
139}
