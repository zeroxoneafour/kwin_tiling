24void XwaylandWindow::associate()
25{
26 if (surface()->isMapped()) {
27 initialize();
28 } else {
29 // Queued connection because we want to mark the window ready for painting after
30 // the associated surface item has processed the new surface state.
31 connect(surface(), &SurfaceInterface::mapped, this, &XwaylandWindow::initialize, Qt::QueuedConnection);
32 }

35void XwaylandWindow::initialize()
36{
37 if (!readyForPainting()) { // avoid "setReadyForPainting()" function calling overhead
38 if (syncRequest().counter == XCB_NONE) { // cannot detect complete redraw, consider done now
39 setReadyForPainting();
40 }
41 }

44bool XwaylandWindow::wantsSyncCounter() const
45{
46 // When the frame window is resized, the attached buffer will be destroyed by
47 // Xwayland, causing unexpected invalid previous and current window pixmaps.
48 // With the addition of multiple window buffers in Xwayland 1.21, X11 clients
49 // are no longer able to destroy the buffer after it's been committed and not
50 // released by the compositor yet.
51 static const quint32 xwaylandVersion = xcb_get_setup(kwinApp()->x11Connection())->release_number;
52 return xwaylandVersion >= 12100000;
53}
