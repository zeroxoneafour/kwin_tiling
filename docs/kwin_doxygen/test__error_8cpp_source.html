50void ErrorTest::init()
51{
52 delete m_display;
53 m_display = new KWaylandServer::Display(this);
54 m_display->addSocketName(s_socketName);
55 m_display->start();
56 QVERIFY(m_display->isRunning());
57 m_display->createShm();
58 m_ci = new CompositorInterface(m_display, m_display);
59 m_psi = new PlasmaShellInterface(m_display, m_display);
60
61 // setup connection
62 m_connection = new KWayland::Client::ConnectionThread;
63 QSignalSpy connectedSpy(m_connection, &KWayland::Client::ConnectionThread::connected);
64 m_connection->setSocketName(s_socketName);
65
66 m_thread = new QThread(this);
67 m_connection->moveToThread(m_thread);
68 m_thread->start();
69
70 m_connection->initConnection();
71 QVERIFY(connectedSpy.wait());
72
73 m_queue = new KWayland::Client::EventQueue(this);
74 m_queue->setup(m_connection);
75
76 KWayland::Client::Registry registry;
77 QSignalSpy interfacesAnnouncedSpy(&registry, &KWayland::Client::Registry::interfacesAnnounced);
78 registry.setEventQueue(m_queue);
79 registry.create(m_connection);
80 QVERIFY(registry.isValid());
81 registry.setup();
82 QVERIFY(interfacesAnnouncedSpy.wait());
83
84 m_compositor =
85 registry.createCompositor(registry.interface(KWayland::Client::Registry::Interface::Compositor).name, registry.interface(KWayland::Client::Registry::Interface::Compositor).version, this);
86 QVERIFY(m_compositor);
87 m_plasmaShell = registry.createPlasmaShell(registry.interface(KWayland::Client::Registry::Interface::PlasmaShell).name,
88 registry.interface(KWayland::Client::Registry::Interface::PlasmaShell).version,
89 this);
90 QVERIFY(m_plasmaShell);
91}

120void ErrorTest::testMultiplePlasmaShellSurfacesForSurface()
121{
122 // this test verifies that creating two ShellSurfaces for the same Surface triggers a protocol error
123 QSignalSpy errorSpy(m_connection, &KWayland::Client::ConnectionThread::errorOccurred);
124 // PlasmaShell is too smart and doesn't allow us to create a second PlasmaShellSurface
125 // thus we need to cheat by creating a surface manually
126 auto surface = wl_compositor_create_surface(*m_compositor);
127 std::unique_ptr<KWayland::Client::PlasmaShellSurface> shellSurface1(m_plasmaShell->createSurface(surface));
128 std::unique_ptr<KWayland::Client::PlasmaShellSurface> shellSurface2(m_plasmaShell->createSurface(surface));
129 QVERIFY(!m_connection->hasError());
130 QVERIFY(errorSpy.wait());
131 QVERIFY(m_connection->hasError());
132 QCOMPARE(m_connection->errorCode(), EPROTO);
133 wl_surface_destroy(surface);
134}
