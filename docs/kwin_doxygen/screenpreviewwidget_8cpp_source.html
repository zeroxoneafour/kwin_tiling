37 void updateScreenGraphics()
38 {
39 int bottomElements = screenGraphics->elementSize("base").height() + screenGraphics->marginSize(Plasma::Types::BottomMargin);
40 QRect bounds(QPoint(0, 0), QSize(q->size().width(), q->height() - bottomElements));
41
42 QSize monitorSize(q->size().width(), q->size().width() / ratio);
43 monitorSize.scale(bounds.size(), Qt::KeepAspectRatio);
44
45 if (monitorSize.isEmpty()) {
46 return;
47 }
48
49 monitorRect = QRect(QPoint(0, 0), monitorSize);
50 monitorRect.moveCenter(bounds.center());
51
52 screenGraphics->resizeFrame(monitorRect.size());
53
54 previewRect = screenGraphics->contentsRect().toRect();
55 previewRect.moveCenter(bounds.center());
56 }

110void ScreenPreviewWidget::paintEvent(QPaintEvent *event)
111{
112 if (d->monitorRect.size().isEmpty()) {
113 return;
114 }
115
116 QPainter painter(this);
117 QPoint standPosition(d->monitorRect.center().x() - d->screenGraphics->elementSize("base").width() / 2, d->previewRect.bottom());
118
119 d->screenGraphics->paint(&painter, QRect(standPosition, d->screenGraphics->elementSize("base")), "base");
120 d->screenGraphics->paintFrame(&painter, d->monitorRect.topLeft());
121
122 painter.save();
123 if (!d->preview.isNull()) {
124 painter.setRenderHint(QPainter::SmoothPixmapTransform);
125 painter.drawPixmap(d->previewRect, d->preview, d->preview.rect());
126 }
127 painter.restore();
128
129 d->screenGraphics->paint(&painter, d->previewRect, "glass");
130}
